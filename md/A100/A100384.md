from sympy import factorint
def A100384(n):
    k, a = 2, [max(factorint(m+2)) for m in range(n)]
    while True:
        for i in range(1, n):
            if a[i-1] >= a[i]:
                break
        else:
            return k
        a = a[i:] + [max(factorint(k+j+n)) for j in range(i)]
        k += i # _Chai Wah Wu_, Jul 24 2017
