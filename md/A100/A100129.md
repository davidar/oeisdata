from itertools import count, islice
def A100129_gen(startvalue=1): # generator of terms
    a = 1<<(m:=max(startvalue,1))
    for n in count(m):
        if (s:=str(n))==str(a)[:len(s)]:
            yield n
        a <<= 1
A100129_list = list(islice(A100129_gen(),4)) # _Chai Wah Wu_, Apr 10 2023
