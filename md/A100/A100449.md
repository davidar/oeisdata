## Python
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def A100449(n):
    if n == 0:
        return 1
    c, j = 0, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*((A100449(k1)-3)//2)
        j, k1 = j2, n//j2
    return 2*(n*(n-1)-c+j)+1 # _Chai Wah Wu_, Mar 29 2021
```
