## Python
```Python
from math import factorial
from sympy import isprime, prime
def afind(limit, startat=1):
    for k in range(startat, limit+1):
        s = str(k)
        pk = prime(k)
        if isprime( factorial(pk-1) + pk**10 ):
            print(k, end=", ")
afind(100) # _Michael S. Branicky_, Nov 30 2021
```
