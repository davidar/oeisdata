## Python
```Python
from itertools import count, islice
def A100083_gen(startvalue=1): # generator of terms >= startvalue
    for n in count(max(startvalue,1)):
        a, c = 0, 1
        for m in range(2,n):
            c = c*m%n
            if c==0:
                break
            a = (a+c)%n
        if not a:
            yield n
A100083_list = list(islice(A100083_gen(),20)) # _Chai Wah Wu_, Apr 16 2024
```
