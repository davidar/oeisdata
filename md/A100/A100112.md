## Python
```Python
from sympy import integer_nthroot, mobius
def A100112(n):
    b = 0
    k, r = integer_nthroot(n,2)
    a, c = (mobius(k),k-1) if r else (0,k)
    for i in range(1,c+1):
        m, j = mobius(i), i**2
        a += m*(n//j)
        b += m*((n-1)//j)
    return a if a>b else 0 # _Chai Wah Wu_, May 11 2024
```
