## Python
```Python
from mpmath import *
mp.dps=20
def a007814(n):
    return 1 + bin(n - 1)[2:].count('1') - bin(n)[2:].count('1')
for n in range(11):
    y=2**sum(a007814(i) for i in range(2, n+1))
    l=chop(taylor(lambda x: legendre(n, x), 0, n))
    print([int(i*y) for i in l]) # _Indranil Ghosh_, Jul 02 2017
```
