## Python
```Python
from itertools import count, islice
from sympy import divisor_count as d
def agen():
    seen = set()
    for n in count(1):
        dn = d(n)
        m = dn
        while m in seen: m += dn
        yield m
        seen.add(m)
print(list(islice(agen(), 68))) # _Michael S. Branicky_, Dec 08 2022
```
