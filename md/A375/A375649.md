def a(n):
    p,c = 1,0
    while p < n:
        p2, k = p << 1, p
        while k >= 1:
            for j in range(k % p, n - k, k << 1):
                for i in range(min(k, n - j - k)):
                    ij = i+j
                    if ij // p2 == (ij + k) // p2: c += 1
            k >>= 1
        p <<= 1
    return c
print([a(n) for n in range(1, 61)])
