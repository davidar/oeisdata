## Python

from collections import Counter
from sympy import factorint
def A375849(n): return max(filter(lambda x: x&1,sum((Counter(factorint(i)) for i in range(2,n+1)),start=Counter()).values())) # _Chai Wah Wu_, Aug 31 2024
