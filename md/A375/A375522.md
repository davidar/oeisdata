from itertools import islice
from math import gcd
from sympy import nextprime
def A375522_gen(): # generator of terms
    p, q, k = 0, 1, 1
    while (k:=nextprime(k)):
        m=q//(k*(q-p))+1
        p, q = p*k*m+q, k*m*q
        p //= (r:=gcd(p,q))
        q //= r
        yield q
A375522_list = list(islice(A375522_gen(),11)) # _Chai Wah Wu_, Aug 30 2024
