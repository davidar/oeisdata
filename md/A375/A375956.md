from itertools import count, islice
from sympy import isprime
def A375956_gen(): # generator of terms
    a, l, r = 2, 1, 10
    yield 2
    for n in count(1):
        b = 10**n
        c = b*r
        for i in count(10**(n-1)):
            m = c*i+a*b+int(str(i)[::-1])
            if isprime(m):
                yield m
                a = m
                l += n<<1
                r *= 10**(n<<1)
                break
A375956_list = list(islice(A375956_gen(),20)) # _Chai Wah Wu_, Sep 27 2024
