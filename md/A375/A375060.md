from itertools import count, islice
def agen(): # generator of terms
    seen, an = {0}, 0
    for n in count(1):
        if an == 0: yield n-1
        an = 0 if n in seen else an + n
        seen.add(an)
print(list(islice(agen(), 64))) # _Michael S. Branicky_, Jul 29 2024
