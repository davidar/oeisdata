def a(n):
    a = [0]
    for i in range(1,n):
        a+=[-1]
    countA = 1
    seq = [0]
    m = 1
    while(countA < n):
        x = seq[m-1]-m
        if(x<0):
            if(-x < n and a[-x]==-1):
                a[-x] = m
                countA+=1
            seq+=[seq[m-1]+m]
        else:
            if(x not in seq):
                seq+=[x]
            else:
                seq+=[seq[m-1]+m]
        m+=1
    return a #find all terms in range(0,n)
