## Python
```Python
from sympy import primepi
def A375926(n):
    def bisection(f,kmin=0,kmax=1):
        while f(kmax) > kmax: kmax <<= 1
        while kmax-kmin > 1:
            kmid = kmax+kmin>>1
            if f(kmid) <= kmid:
                kmax = kmid
            else:
                kmin = kmid
        return kmax
    def f(x): return n+bisection(lambda y:primepi(x+1+y))-1
    return bisection(f,n,n) # _Chai Wah Wu_, Sep 15 2024
```
