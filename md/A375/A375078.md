P:=proc(q) local a,b,c,n; a:=[1]; b:=[1];
for n from 2 to q do c:=numboccur(n mod 10,b); a:=[op(a),c];
if c=0 then b:=[op(b),0]; else b:=[op(b),op(convert(c,base,10))]; fi; od;
print(op(a)); end: P(10^3);

## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    s = "1"
    yield 1
    for n in count(2):
        an = s.count(str(n%10))
        s += str(an)
        yield an
print(list(islice(agen(), 82))) # _Michael S. Branicky_, Aug 02 2024
```

from collections import Counter
def A375078_gen(): # generator of terms
    c, n = Counter({1:1}), 1
    yield 1
    while True:
        n = (n+1)%10
        yield c[n]
        c += Counter(map(int,str(c[n])))
A375078_list = list(islice(A375078_gen(),30)) # _Chai Wah Wu_, Aug 03 2024
