## Python
```Python
from sympy import *
SHCN = [2, 6, 12, 60, 120, 360, 2520, 5040, 55440, 720720]
for x in range(3, 16000, 2):
    waysFound = 0
    if isprime(x):
        iterC = 0
        while iterC < len(SHCN) and SHCN[iterC] < x:
            if isprime(x - SHCN[iterC]):
                waysFound += 1
            iterC += 1
        if waysFound == 0:
            print(x)
```
