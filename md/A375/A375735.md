from itertools import count
from sympy import primepi, integer_nthroot, primefactors
def A375735(n):
    def f(x): return int(n+1+sum(primepi(integer_nthroot(x,k)[0]) for k in range(1,x.bit_length())))
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
