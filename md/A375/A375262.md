## Python
```Python
from itertools import count
from math import prod
from sympy import factorint
def A375262(n):
    for m in count(1):
        f = factorint(m)
        if ((n<<1)+1)*m*prod((p-1)**2 for p in f)==prod(p**(e+2)-p for p,e in f.items())<<1:
            return m # _Chai Wah Wu_, Aug 11 2024
```
