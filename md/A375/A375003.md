## Python
```Python
from sympy import divisors, integer_nthroot
def A375003(n):
    k = 0
    for c in range(1,n-1):
        for d in divisors(c,generator=True):
            for x in range(1,d):
                xy = x*(d-x)
                a = (c//d)**2
                b = a-(n-c-xy<<2)
                if b>=0:
                    q,r = integer_nthroot(b,2)
                    if r:
                        w = (c//d+q)//2
                        if 1<=w<c//d:
                            k += 1+bool(q)
    return k # _Chai Wah Wu_, Jul 27 2024
```
