from itertools import count, islice
from math import gcd, lcm
def A375520_gen(): # generator of terms
    p, q, c = 0, 1, 1
    for k in count(1):
        m = q//(k*(q-p))+1
        p, q = p*k*m+q, k*m*q
        p //= (r:=gcd(p,q))
        q //= r
        c = lcm(c,k)
        yield q//c
A375520_list = list(islice(A375520_gen(),11)) # _Chai Wah Wu_, Aug 28 2024
