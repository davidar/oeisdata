## Python
```Python
def a(n):
    T, S = [n], set()
    while n > 1:
        if n & 1 == 0: n >>= 1
        else: n = 3 * n + 1
        T.append(n)
    for i in range(len(T)):
        for j in range(i, len(T)):
            S.add(T[i] + T[j])
    return len(S)
print([a(n) for n in range(1, 60)]) # _Dar√≠o Clavijo_, Aug 24 2024
```
