## Python
```Python
from itertools import count, islice
from sympy import perfect_power
def A375714_gen(): # generator of terms
    a, b = -1, 0
    for n in count(1):
        c = not perfect_power(n)
        if c:
            a += 1
        if b&(c^1):
            yield a
        b = c
A375714_list = list(islice(A375714_gen(),52)) # _Chai Wah Wu_, Sep 11 2024
```
