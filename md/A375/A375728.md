## Python
```Python
from itertools import count
def a(n):
    s, strs = sum(range(1, n+1)), list(map(str, range(1, n+1)))
    for k in count(1):
        if int("".join(strs))%s == 0: return k
        s, strs = s + n, strs[1:] + [str(k+n)]
print([a(n) for n in range(1, 47)]) # _Michael S. Branicky_, Aug 25 2024
```
