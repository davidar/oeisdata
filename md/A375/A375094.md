## Python
```Python
# output in b-file format
from itertools import count
n = 0
for k in count():
    m = k
    s = 0
    while m > 1:
        m = m // 2 if m % 2 == 0 else 3*m+1
        s += 1
    while n < s:
        print(n, k, flush=True)
        n += 1
```
