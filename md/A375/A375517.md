from itertools import count, islice
from math import gcd
def A375517_gen(): # generator of terms
    p, q = 0, 1
    for k in count(1):
        m = q//(k*(q-p))+1
        p, q = p*k*m+q, k*m*q
        p //= (r:=gcd(p,q))
        q //= r
        yield q//k
A375517_list = list(islice(A375517_gen(),11)) # _Chai Wah Wu_, Aug 28 2024
