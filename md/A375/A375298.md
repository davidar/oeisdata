def a(n):
  if n == 1:
    return 1
  elif n == 2:
    return 2
  elif n == 3:
    return 5
  elif n == 4:
    return 8
  elif n == 9:
    return 37
  elif n % 8 == 3:
    return (2*n**2 - n + 1) // 4 - 1
  else:
    return (2*n**2 - n + 1) // 4
