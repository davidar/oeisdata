from itertools import count
def A375507_list(nmax):
    a = [1]
    def digits():
        for i in count():
            for d in str(a[i]):
                yield int(d)
    diff = 0
    for n,d in enumerate(digits(),1):
        if n==nmax: return a
        diff = 10*diff+d
        a.append(a[-1]+diff) # _Pontus von Br√∂mssen_, Aug 18 2024
