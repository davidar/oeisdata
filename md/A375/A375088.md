## Python
```Python
from itertools import islice
def mountain(h):
    return list(range(h, 2*h + 1)) + list(range(2*h-1, h-1, -1))
def agen():
    a = [1, 2, 1]
    yield 1
    i = 1
    while 1:
       a += mountain(a[i])
       yield a[i]
       i += 1
print(islice(agen(), 104))
```
