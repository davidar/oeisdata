## Python
```Python
import zlib
def g():
  max_m,i=0,1
  while True:
    if (m := len(zlib.compress(b"A"*i))) > max_m:
      max_m = m
      yield i
    i+=1
from itertools import islice
print(list(islice(g(),45)))
```
