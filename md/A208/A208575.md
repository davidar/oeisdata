## Python
```Python
from operator import mul
def A(n, p=2):
    return n if n<p else A(n//p, p+1)*10 + n%p
def a(n):
    x=str(A(n - 1))
    return 0 if "0" in x else reduce(mul, [int(i) for i in x])
print([a(n) for n in range(1, 101)]) # _Indranil Ghosh_, Jun 19 2017
```
