from sympy.ntheory.factor_ import digits
def A208279(n):
    if n == 0: return 1
    s = digits(n,5)[1:]
    return 0 if any(x>2 for x in s) else ((6,2,4,8)[a&3] if (a:=s.count(1)) else 6) # _Chai Wah Wu_, Dec 08 2023
