## Python
```Python
def A007953(n):
    a=0
    sh=n
    while sh > 0:
        a += sh % 10
        sh //= 10
    return a
def isA135027(n):
    if n % 10 == 0:
        return False
    else:
        return A007953(n**2) == 10
for n in range(70000):
    if isA135027(n):
        print(n)
```
