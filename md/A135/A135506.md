## Python
```Python
from itertools import count, islice
from math import lcm
def A135506_gen(): # generator of terms
    x = 1
    for n in count(2):
        y = x+lcm(x,n)
        yield y//x-1
        x = y
A135506_list = list(islice(A135506_gen(),20)) # _Chai Wah Wu_, Mar 13 2023
```
