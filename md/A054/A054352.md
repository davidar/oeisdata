## Python
```Python
def aupton(nn):
  alst, A054353, idx = [1], 0, 1
  K = Kolakoski()  # using Kolakoski() in A000002
  for n in range(2, nn+1):
    target = alst[-1]
    while idx <= target:
      A054353 += next(K)
      idx += 1
    alst.append(A054353 + 1)  # a(n) = A054353(a(n-1))+1
  return alst
print(aupton(36))  # _Michael S. Branicky_, Jan 12 2021
```
