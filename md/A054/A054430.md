## Python
```Python
from sympy import totient
def A(u):
    a=[]
    k=0
    for n in range(2, u + 1):
        m=k + totient(n)
        for i in range(1, totient(n) + 1):
            a+=[m,]
            m-=1
            k+=1
    return a
print(A(30)) # _Indranil Ghosh_, May 23 2017, translated from MAPLE code
```
