## Python
```Python
from sympy import isprime
def ok(n):
    if not isprime(n): return False
    s = str(n); srev = s[::-1]
    return isprime(int(srev)) and isprime(int(s[:-1] + srev))
print([k for k in range(10**4) if ok(k)]) # _Michael S. Branicky_, Nov 17 2023
```
