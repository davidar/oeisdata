from sympy import nextprime
from itertools import count, islice
def agen(): # generator of terms
    p, q, n, rl, argp = 2, 3, 1, 0, 2
    for k in count(1):
        if q - p >= 4: rl += 1
        else:
            if rl >= n: yield argp; n = rl + 1
            rl, argp = 0, q
        p, q = q, nextprime(q)
print(list(islice(agen(), 14))) # _Michael S. Branicky_, Aug 23 2022
