## Python
```Python
from sympy import divisors
from sympy.ntheory.primetest import is_square
def A054495(n): return next(d for d in divisors(n) if is_square(m:=5*(n//d)**2-4) or is_square(m+8)) # _Chai Wah Wu_, May 06 2024
```
