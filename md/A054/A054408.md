## Python
```Python
from sympy import isprime
def aupton(terms):
    alst, aset, asum = [], set(), 0
    while len(alst) < terms:
        an = 1
        while True:
            while an in aset: an += 1
            if isprime(asum + an):
                alst, aset, asum = alst + [an], aset | {an}, asum + an
                break
            an += 1
    return alst
print(aupton(66)) # _Michael S. Branicky_, Jun 05 2021
```
