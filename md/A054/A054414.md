## Python
```Python
from operator import sub
from sympy import integer_log
def A054414(n):
    if n == 0: return 1
    def f(x): return n+sub(*integer_log(1<<x,3))+1
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m # _Chai Wah Wu_, Oct 09 2024
```
