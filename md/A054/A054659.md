## Python
```Python
def ok(s, t): return len(set(t)) == len(t) and len(set(s+t)) == len(s+t)
def agen(): # generator of complete sequence of terms
    an, MAX = 0, 987654321
    while True:
        if an < MAX: yield an
        else: return
        an, s = an+1, str(an)
        MAX = 10**(10-len(s))
        while an < MAX and not ok(s, str(an)): an += 1
print(list(agen())) # _Michael S. Branicky_, Jun 30 2022
```
