from collections import Counter
def aupton(terms):
    num, alst, inventory = 0, [0, 0], Counter([0])
    for n in range(2, terms+1):
        c = inventory[num]
        num = 0 if c == 0 else num + 1
        alst.append(c)
        inventory.update([alst[-2] + alst[-1]])
    return alst
print(aupton(97)) # _Michael S. Branicky_, Nov 05 2021
