def combine(a,b):
  c = ''
  for i in range(max(len(a),len(b))*2):
   if i%2 == 0:
    if len(a) > i/2:
     c += (a[int(i/2)])
   else:
    if len(b) > i/2:
     c += (b[int(i/2)])
  return c
x = '0'
while True:
  x = combine(combine(len(x)*'0',len(x)*'1')[:len(x)],x)
