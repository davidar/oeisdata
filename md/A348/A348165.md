## Python
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def b(n, i):
    if n > i*(i+1)*(2*i+1)//6: return 0
    if i == 0: return 1
    return b(n+i**2, i-1) + b(abs(n-i**2), i-1)
def a(n): return b(n, n)
print([a(n) for n in range(50)]) # _Michael S. Branicky_, Jan 28 2022
```
