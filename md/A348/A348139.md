## Python
```Python
from math import isqrt
def ok(n):
    s = str(n)
    if len(s) != 3: return False
    a, b, c = list(map(int, s))
    D = b**2 - 4*a*c
    return D >= 0 and isqrt(D)**2 == D
def afull(): return [m for m in range(100, 1000) if ok(m)]
print(afull()) # _Michael S. Branicky_, Oct 02 2021
```
