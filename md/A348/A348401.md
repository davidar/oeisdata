## Python
```Python
from math import factorial
def a(n):
    f = [factorial(i) for i in range(1, n+1)]
    for m, fm in enumerate(f, start=1):
        for fk in f[:m]:
            if n == fm // fk:
                return m
print([a(n) for n in range(1, 70)]) # _Michael S. Branicky_, Oct 17 2021
```
