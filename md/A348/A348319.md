## Python
```Python
def ispal(n): s = str(n); return s == s[::-1]
def aupto(limit):
    aset, m, mm = set(), 10, 100
    while mm <= limit:
        if not ispal(m):
            mk = mm
            while mk <= limit:
                if ispal(mk): aset.add(mk)
                mk *= m
        mm += 2*m + 1
        m += 1
    return sorted(aset)
print(aupto(10**13)) # _Michael S. Branicky_, Oct 12 2021
```
