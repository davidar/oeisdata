## Python
```Python
from gmpy2 import digits, is_prime, next_prime
def A348226(n): # code assumes n <= 63 or n is prime
    if is_prime(n):
        return 2
    p = 2
    while True:
        for i in range(n-1,1,-1):
            s = digits(p,i)
            if not is_prime(int(s,n)):
                break
        else:
            return p
        p = next_prime(p) # _Chai Wah Wu_, Nov 19 2021
```
