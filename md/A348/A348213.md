## Python

from sympy import totient, divisors
def A348213(n):
    c, k = 0, n
    m = sum(set(map(totient,divisors(k,generator=True))))
    while m != k:
        k = m
        m = sum(set(map(totient,divisors(k,generator=True))))
        c += 1
    return c # _Chai Wah Wu_, Nov 15 2021
