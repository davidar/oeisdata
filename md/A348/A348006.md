## Python
```Python
def A348006(n):
    c, mr = n, 0
    while c > 1:
        if c % 2:
            mr = max(mr, 2*c+1)
            c = 3*c+1
        else:
            c //= 2
    return mr
print([A348006(n) for n in range(1, 100)])
```
