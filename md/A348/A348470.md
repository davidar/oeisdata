## Python
```Python
from itertools import islice, count
from math import gcd
from sympy import primefactors
def A064413gen(): # generator of terms
    yield 1
    yield 2
    l, s, b = 2, 3, set()
    for _ in count(0):
        i = s
        while True:
            if not i in b and gcd(i,l) > 1:
                yield i
                l = i
                b.add(i)
                while s in b:
                    b.remove(s)
                    s += 1
                break
            i += 1
def A348470(n): return 1 if n == 1 else min(primefactors(next(islice(A064413gen(),n-1,None)))) # _Chai Wah Wu_, Dec 07 2021
```
