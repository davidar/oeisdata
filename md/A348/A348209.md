from labmath import multord
def w(r,n):
    z,s=r,set()
    while z%n not in s:
        s.add(z%n)
        z=z//n
    return len(s)
def a(n):
    if n==2: return 1
    else:
        v=n
        while v%2==0: v=v//2
        if v==1: return 0
        else:
            k,r,m=0,1,n**(n-1)
            while r<m: k,r=k+1,2*r
            M,e=m*n,k+multord(2,v**n)
            while (w(r,n)<n) and (k<e): k,r=k+1,2*r%M
            if w(r,n)==n: return k
            else: return 0
