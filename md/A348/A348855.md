## Python
```Python
from sympy import isprime, nextprime
def aupton(terms):
    alst, aset = [1], {1}
    while len(alst) < terms:
        if isprime(alst[-1]):
            an = 2*alst[-1] + 1
        else:
            p = 2
            while p in aset: p = nextprime(p)
            an = p
        alst.append(an); aset.add(an)
    return alst
print(aupton(60)) # _Michael S. Branicky_, Nov 02 2021
```
