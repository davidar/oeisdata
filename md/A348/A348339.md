## Python
```Python
def tail(m):
    global n; s = str(m)
    return m if len(s) <= n else int(s[-n:])
for n in range(1, 10):
    M = []
    for i in range(10**n):
        t = i; L = [t]
        while i >= 0:
            t = tail(t*t)
            if t not in L: L.append(t)
            else: break
        d = len(L)
        if d not in M: M.append(d)
    print(len(M), end = ', ')
```

def A348339(n):
    m, s = 10**n, set()
    for k in range(m):
        c, k2, kset = 0, k, set()
        while k2 not in kset:
            kset.add(k2)
            c += 1
            k2 = k2*k2 % m
        s.add(c)
    return len(s) # _Chai Wah Wu_, Oct 19 2021
