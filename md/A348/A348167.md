## Python
```Python
def A348167():
    x = -1
    while True:
        x = x + 1
        if x & (x>>1): continue
        if (x & 3) == 0: continue
        negx = ~x
        gaps = negx & (negx >> 1) & (negx >> 2)
        if (gaps-1) & x != x: continue
        yield x
```
