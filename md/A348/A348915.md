from math import prod
from sympy import factorint
def A348915(n):
    f = factorint(n>>(m:=(~n&n-1).bit_length())).items()
    d = prod(e+1 for p,e in f)
    s = prod((p**(e+1)-1)//(p-1) for p, e in f)
    return s+d*m # _Chai Wah Wu_, Jul 16 2022
