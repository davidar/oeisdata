## Python
```Python
from itertools import groupby
def ok(n):
    if n == 0: return True
    b = bin(n)[2:]
    c = "".join(bin(len(list(g)))[2:] for k, g in groupby(b))
    return b.startswith(c)
print(list(filter(ok, range(2**17)))) # _Michael S. Branicky_, Oct 02 2021
```
