def aupton(terms):
    alst, astr, n = [1], "1", 1
    while len(alst) < terms:
        an, n = alst[-1], len(alst)
        target, pow10 = int(astr[:n]), 1
        while len(alst) == n:
            i = 0
            while i < pow10:
                q, r = divmod(target+i, an)
                if r == 0 and q not in alst:
                    alst.append(q)
                    astr += str(q)
                    break
                i += an - r
            pow10 *= 10
            target *= 10
    return alst
print(aupton(21)) # _Michael S. Branicky_, Jan 07 2022
