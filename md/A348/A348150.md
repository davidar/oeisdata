## Python
```Python
def niven(n):
    s = str(n)
    return '0' not in s and n%sum(map(int, s)) == 0
def a(n):
    k = int("1"*n)
    while not niven(k): k += 1
    return k
print([a(n) for n in range(1, 22)]) # _Michael S. Branicky_, Oct 09 2021
```
