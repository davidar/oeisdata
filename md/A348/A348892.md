from functools import lru_cache
@lru_cache(maxsize=None)
def b(n, i):
    if n > (i*(i+1)//2)**2: return 0
    if i == 0: return 1
    return b(n+i**3, i-1) + b(abs(n-i**3), i-1)
def a(n): return b(n, n)
print([a(n) for n in range(54)]) # _Michael S. Branicky_, Jan 28 2022
