## Python
```Python
from math import prod
from itertools import product
def a(n):
    maxr = 0
    for p in product("123456789", repeat=n):
        maxr = max(maxr, int("".join(p))%prod(map(int, p)))
    return maxr
print([a(n) for n in range(1, 7)]) # _Michael S. Branicky_, Nov 01 2021
```
