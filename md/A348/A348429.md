## Python
```Python
# see link for faster version
def ispal(n): s = str(n); return s == s[::-1]
def aupto(limit):
    aset, m, mm = {1}, 2, 4
    while mm <= limit:
        if ispal(m):
            mk = mm
            while mk <= limit:
                if ispal(mk): aset.add(mk)
                mk *= m
        mm += 2*m + 1
        m += 1
    return sorted(aset)
print(aupto(10**11)) # _Michael S. Branicky_, Oct 18 2021
```
