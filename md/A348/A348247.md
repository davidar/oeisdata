from collections import Counter
from sympy import prime
def A348247(n):
    c, b, p, i = Counter(), 1, prime(n), 1
    while True:
        k, kb = 1, b
        while c[kb] >= kb:
            k += 1
            kb += b
        if kb == p:
            return i
        c[kb] += 1
        b = k
        i += 1 # _Chai Wah Wu_, Oct 21 2021
