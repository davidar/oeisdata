## Python

x = 1
n = 0
runs = [[n]]
while x < 30:
    print(x - 1, n)
    runs2 = []
    n = 0
    while runs:
        for pmx in (x, -x):
            if runs[0][ -1] + pmx not in runs[0]:
                runs2.append(runs[0] + [runs[0][ -1] + pmx])
                n += 1
        del runs[0]
    runs = runs2
    x += 1
