## Python
```Python
from sympy import divisors
def A000120(n): return bin(n).count('1')
def aupto(limit):
  alst = []
  for m in range(1, limit+1):
    if A000120(m) == sum(A000120(d) for d in divisors(m)[:-1]): alst += [m]
  return alst
print(aupto(1563)) # _Michael S. Branicky_, Feb 25 2021
```
