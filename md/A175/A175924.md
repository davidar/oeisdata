## Python
```Python
import math
for N in range(1, 10):
    repdigits = 1
    n = 0
    while repdigits < N:
        n += 1
        s = str(2 ** n)
        prev = ""
        repdigits = maxrepdigits = 1
        for d in s:
            if d == prev: repdigits += 1
            else:
                maxrepdigits = max(maxrepdigits, repdigits)
                repdigits = 1
            prev = d
        repdigits = max(maxrepdigits, repdigits)
    print(N, 2 ** n)
```
