## Python
```Python
from sympy import isprime
out = []
run = 1
for n in range(4, 10000):
    isrun = True
    for o in range(run): isrun *= not isprime(n + o - run)
    if isrun:
        for o in range(run): out.append(n + o - run)
        run += 2
print(out)
```
