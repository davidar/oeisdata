from sympy import factorint, primepi
def ok(n): return n > 1 and n%sum(e*10**(primepi(p)-1) for p, e in factorint(n).items()) == 0
print([k for k in range(10**4) if ok(k)]) # _Michael S. Branicky_, Mar 17 2024
