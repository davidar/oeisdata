## Python
```Python
from itertools import islice
def agen(): # generator of terms
    pow2 = 1
    while True:
        s = str(pow2)
        c = sum(1 for i in range(len(s)-2) if s[i:i+3] == "666")
        if c > 0: yield c
        pow2 <<= 1
print(list(islice(agen(), 88))) # _Michael S. Branicky_, Apr 07 2024
```
