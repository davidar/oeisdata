```Python
from math import gcd
from sympy import nextprime
from sympy.ntheory import digits
def A371194(n):
    m, j = 1, 0
    if n > 3:
        for j in range(1,n):
            if gcd((n*(n-1)>>1)+j,n-1) == 1:
                 break
    if j == 0:
        for i in range(2,n):
            m = n*m+i
    elif j == 1:
        for i in range(1,n):
            m = n*m+i
    else:
        for i in range(2,1+j):
            m = n*m+i
        for i in range(j,n):
            m = n*m+i
    m -= 1
    while True:
        s = digits(m:=nextprime(m),n)[1:]
        if 0 not in s and len(set(s))==n-1:
            return m
```
