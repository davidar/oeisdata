## Python
```Python
from sympy.ntheory.residue_ntheory import n_order
from sympy import nextprime
def a(n):
  if n == 1: return 1
  if n & 1 == 1: return n_order(2, n)
  p = 2
  while n % p == 0:
    p = nextprime(p)
  return n_order(p, n)
print([a(n) for n in range(1, 76)])
```
