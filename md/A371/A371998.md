## Python
```Python
from functools import cache
@cache
def sf(n):
    if n == 0: return 1
    return n * sf(n - 1) + (-1 if n % 2 else 1)
def a(n):
    h, r = divmod(n, 2)
    return sf(h) * (h + 1) if r else sf(h)
print([a(n) for n in range(33)])
```
