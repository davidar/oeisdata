## Python
```Python
from sympy import isprime, nextprime
from itertools import count, islice
def f(p):
    k = 1
    while isprime(p+4*k*(k+1)): k += 1
    return k
def agen(verbose=False): # generator of terms
    adict, n, p = dict(), 1, 1
    while True:
        p = nextprime(p)
        v = f(p)
        if v not in adict:
            adict[v] = p
            if verbose: print("FOUND", v, p)
        while n in adict:
            yield adict[n]; n += 1
print(list(islice(agen(), 14))) # _Michael S. Branicky_, Apr 12 2024
```
