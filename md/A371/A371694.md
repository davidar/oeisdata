from sympy import primerange, prevprime, nextprime; p = 2; b = 0
for n in range(1, 57): print(p, end = ", "); q = max(nextprime(p), prevprime(p + n - b + 1)); m = len(list(primerange(p+1, q))); p = q; b += m
