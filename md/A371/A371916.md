## Python

```Python
def a(n):
    a, b, c, d = 1, 1, 1, 1
    for _ in range(n):
        a, b, c, d = b, c, d, int(str(a+b+c+d).replace('0', ''))
    return a
```

```Python
# faster for initial segment of sequence
from itertools import islice
def agen(): # generator of terms
    a, b, c, d = 1, 1, 1, 1
    while True:
        yield a
        a, b, c, d = b, c, d, int(str(a+b+c+d).replace("0", ""))
```
