## Python
```Python
from itertools import islice
def agen(): # generator of terms
    pow2 = 1
    while True:
        s = str(pow2)
        if (c := s.count("666")) > 0: yield c
        pow2 <<= 1
print(list(islice(agen(), 88))) # _Michael S. Branicky_, Apr 07 2024
```
