from sympy import prime
def A371497_row(n):
    q = prime(n)
    res = {i*i % q for i in range(1, q//2 + 1)}
    if q % 4 == 1:
        res = {a for a in res if 2*a < q}
    else:
        res = {((a % 4 - 1) * q + a) % (4*q) for a in res}
        res = {a if a < 2*q else 4*q - a for a in res}
    return sorted(res)
