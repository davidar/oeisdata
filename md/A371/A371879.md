## Python
```Python
from sympy import factorint, isprime
def ok(n):
    if n < 10 or not isprime(n): return False
    s = str(n)
    for i in range(len(s)):
        ti = int(s[:i] + s[i+1:])
        f = factorint(ti)
        if not len(f) == 2 == sum(f.values()):
            return False
    return True
print([k for k in range(16000) if ok(k)]) # _Michael S. Branicky_, May 25 2024
```
