```Python
from sympy import isprime
l = []
for i in range(2,185):
    if isprime(i):
        sum = 0
        reminder = 1
        for j in range(2, i):
            reminder = (reminder * j) % i
            sum += reminder
        l.append(sum)
print(l)
```

```Python
from sympy import prime
def A371035(n):
    a, c, p = 0, 1, prime(n)
    for m in range(2,p):
        c = c*m%p
        a += c
    return a # _Chai Wah Wu_, Apr 16 2024
```
