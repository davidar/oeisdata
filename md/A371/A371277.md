## Python
```Python
def T_Lah(n, k):
    if k < 2 or k > n:
        return 0
    elif n == k == 2:
        return 1
    else:
        return T_Lah(n-1, k-1) + ((n+k-1)**3) * T_Lah(n-1, k)
def print_triangle(rows):
    for n in range(rows):
        row_values = [T_Lah(n, k) for k in range(n+1)]
        print(' '.join(map(str, row_values)).center(rows*10))
rows = 10
print_triangle(rows)
```
