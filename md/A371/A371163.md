from itertools import count, islice
def A371163_gen(): # generator of terms
    c = 0
    for n in count(0):
        if not (n<<1)&n:
            if  int(bin(n)[2:].replace('01','1'),2) == c:
                yield c
            c += 1
A371163_list = list(islice(A371163_gen(),6)) # _Chai Wah Wu_, Mar 18 2024
