## Python
```Python
from itertools import permutations as per
a=[]
for n in range(1,7):
    for k in [int(''.join(d)) for d in per('0123456789', 10)]:
        if all(str(k**n).count(d) ==n for d in '0123456789'):
            a.append(k)
            break
print(a)
```
