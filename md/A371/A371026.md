## Python
```Python
from functools import cache
@cache
def T(n, k):   # After _Werner Schulte_
    if k == 0: return 0**n
    if k == n: return n * T(n-1, n-1)
    return k * T(n-1, k-1) + (4*n - 4 + k) * T(n-1, k)
for n in range(8): print([T(n, k) for k in range(n + 1)])
```
