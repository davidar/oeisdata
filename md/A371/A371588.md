## Python

from itertools import count
from sympy import integer_nthroot
def A371588(n):
    a, b = 1, 2
    while True:
        s = sorted(str(b))
        l = len(s)
        m = int(''.join(s[::-1]))
        u = int(''.join(s))
        for i in count(max(2,integer_nthroot(u,n)[0])):
            if (k:=i**n) > m:
                break
            t = sorted(str(k))
            if ['0']*(l-len(t))+t == s:
                return b
                break
        a, b = b, a+b
