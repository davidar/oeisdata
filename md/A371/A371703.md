## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    an, aset = 1, {1}
    while True:
        yield an
        an = next(k for k in count(2) if k not in aset and set(str(an+k)) == set(str(an*k)))
        aset.add(an)
print(list(islice(agen(), 66))) # _Michael S. Branicky_, Apr 03 2024
```
