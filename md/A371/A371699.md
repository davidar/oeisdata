## Python
```Python
from itertools import count
from sympy import factorint, integer_log
def A371699(n):
    for m in count(4):
        f = factorint(m)
        if sum(f.values()) > 1:
            c = 0
            for p in sorted(f,reverse=True):
                a = pow(n,integer_log(p,n)[0]+1,m)
                for _ in range(f[p]):
                    c = (c*a+p)%m
            if not c:
                return m
```
