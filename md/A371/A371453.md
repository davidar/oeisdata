```Python
from math import isqrt
from sympy import primepi, primerange
def A371453(n):
    def f(x,n): return int(n+x+(t:=primepi(s:=isqrt(x)))+(t*(t-1)>>1)-sum(primepi(x//k) for k in primerange(1, s+1)))
    def A006881(n):
        m, k = n, f(n,n)
        while m != k:
            m, k = k, f(k,n)
        return m
    return sum(1<<A006881(i)-1 for i, j in enumerate(bin(n)[:1:-1],1) if j=='1') # _Chai Wah Wu_, Aug 16 2024
```
