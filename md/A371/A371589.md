## Python
```Python
from itertools import count
from sympy import integer_nthroot
def A371589(n):
    a, b = 1, 2
    while True:
        s = sorted(str(b))
        m = int(''.join(s[::-1]))
        u = int(''.join(s))
        for i in count(max(2,integer_nthroot(u,n)[0])):
            if (k:=i**n) > m:
                break
            t = sorted(str(k))
            if t == s:
                return i
                break
        a, b = b, a+b
```
