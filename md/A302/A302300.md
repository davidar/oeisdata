def frequencies(partition, n):
    tot = 0
    freq_list = []
    i = 0
    for p in partition:
        freq = [0 for i in range(n+1)]
        for i in p:
            freq[i] += 1
        for f in freq:
            if f == 0:
                tot += 1
        freq_list.append(freq)
    return freq_list
def sum_square_freqs_of_one(freq_part):
    tot = 0
    for f in freq_part:
        count = 0
        for i in f:
            if i == 1:
                count += 1
        tot += count*count
    return tot
