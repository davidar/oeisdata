## Python
```Python
from math import log2, ceil
def cyclic_gray(nb, n, a):
    if len(a) == n:
        if bin(a[-1]).count('1') == 1:
            return 1
        return 0
    r = 0
    for i in range(nb):
        x = a[-1] ^ (1<<i)
        if x not in a:
            r += cyclic_gray(nb, n, a+[x])
    return r
print([cyclic_gray(ceil(log2(n))+1, n*2, [0]) for n in range(1, 9)])
```
