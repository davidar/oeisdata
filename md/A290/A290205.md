    f = lambda n:sum((1,3,3,2,1,4,2,4,1,8,5,1,3,1,1,3,10,1,1,1,1,4,4,8,4,10)[a] for d in num2words(n).replace(' and ','') if 0<=(a:=ord(d)-97)<=25)
