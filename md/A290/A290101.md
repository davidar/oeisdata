from sympy import flatten
def ok(n, L):
    return any(i < n for i in L)
def a(n):
    if n==1: return 0
    L=[n]
    i = 0
    while not ok(n, L):
        L=set(flatten([[3*k + 1, k//2] if k%2==0 else 3*k + 1 for k in L]))
        i+=1
    return i
print([a(n) for n in range(1, 121)]) # _Indranil Ghosh_, Aug 31 2017
