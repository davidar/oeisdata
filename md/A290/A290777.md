## Python
```Python
from sympy.core.cache import cacheit
@cacheit
def b(n, k):
    if n == 0:
        return 1
    return sum(b(j, k) * b(n - j - 1, k) * k**j for j in range(n))
def a(n): return b(n, n)
print([a(n) for n in range(16)]) # _Indranil Ghosh_, Aug 10 2017
```
