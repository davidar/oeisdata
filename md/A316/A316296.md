## JavaScript
```JavaScript
function f(n,m){
    var count = 0;
    for(var i=m+1; i<2*m; i++){
        if(i%n === 0) count++;
    }
    return count;
}
function sf(n){
    var sum = 0;
    for(var i=1; i<=n; i++){
        sum += f(i, n);
    }
    return sum;
}
```
