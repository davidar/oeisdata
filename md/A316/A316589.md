def length_collatz_chain(start):
    i=0
    while start != 1:
        if (start % 2 == 0):
            start = start / 2
        else:
            start = 3 * start + 1
        i = i+1
    return i
def is_prime(num):
    if num == 1: return(0)
    for k in range(2, num):
       if (num % k) == 0:
           return(0)
    return(1)
