## Python
```Python
from gmpy2 import is_prime, is_square
for p in range(3, 10 ** 4, 2):
    flag = 0
    if is_prime(p):
        for q in range(3, p, 2):
            if is_square(p * q - 1) and is_prime(q):
                flag = 1
                break
    if flag:
        print(p, end=", ")
```
