def A122971(n): return n**30
from sympy import nextprime
def is_A122971(N, k=30): # 2nd opt. arg to check for powers other than 30
    p = 2
    while N >= p**k:
        for e in range(N):
            if N % p: break
            N //= p
        if e % k: return False
        p = nextprime(p)
    return N < 2  #  _M. F. Hasler_, Jul 24 2022
