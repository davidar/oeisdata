## Python
```Python
from sympy import Matrix,eye
def A122771_row(n):
  if n==0: return [1]
  A=Matrix(n,n,lambda i,j:int(i==n-1 or i==j-1))
  p=(eye(n)+A.inv()).charpoly()
  return [(-1)**n*c for c in p.all_coeffs()[::-1]] # _Pontus von BrÃ¶mssen_, May 01 2021
```
