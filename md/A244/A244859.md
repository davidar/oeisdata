A244859:= proc(n) local m,d2,d5;
d2:= padic:-ordp(n,2);
d5:= padic:-ordp(n,5);
m:= n/2^d2/5^d5;
10^max(d2,d5)*(10^numtheory:-order(10,9*m)-1)/9
end proc:

## Python
```Python
def a(n):
    if n == 0: return 0
    moddict = {0: 0}
    for e in range(1, n+2):
        repe = (10**e-1)//9
        r = repe%n
        if r in moddict:
            return repe - moddict[r]
        else:
            moddict[r] = repe
print([a(n) for n in range(29)]) # _Michael S. Branicky_, Feb 22 2024
```
