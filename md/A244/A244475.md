## Maple
```Maple
A002487 := proc(n,k)
    option remember;
    if k =0 then
        1;
    elif k = 2^n-1 then
        n+1 ;
    elif type(k,'even') then
        procname(n-1,k/2) ;
    else
        procname(n-1,(k-1)/2)+procname(n-1,(k+1)/2) ;
    end if;
end proc:
A244475 := proc(n)
    {seq(A002487(n,k),k=0..2^n-1)} ;
    sort(%) ;
    op(-5,%) ;
end proc:
for n from 3 do
    print(A244475(n)) ;
od: # _R. J. Mathar_, Oct 25 2014
```
