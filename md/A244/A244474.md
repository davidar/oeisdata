A002487 := proc(n,k)
    option remember;
    if k =0 then
        1;
    elif k = 2^n-1 then
        n+1 ;
    elif type(k,'even') then
        procname(n-1,k/2) ;
    else
        procname(n-1,(k-1)/2)+procname(n-1,(k+1)/2) ;
    end if;
