from sympy import isprime
A244369 = []
for n in range(1,10):
....for m in range(n-1,-1,-1):
........l = ''.join([str(d) for d in range(n,m-1,-1)])
........p = int(l+l[-2::-1])
........if isprime(p):
............A244369.append(p)
....for m in range(n+1,10):
........l = ''.join([str(d) for d in range(n,m+1)])
........p = int(l+l[-2::-1])
