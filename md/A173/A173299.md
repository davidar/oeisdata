## Python

from fractions import Fraction
def a173299_gen(a, b):
    while True:
        yield a.numerator
        b, a = b + Fraction(a, 2), b
g = a173299_gen(1, 2)
print([next(g) for _ in range(34)])  # _Nick Hobson_, Feb 20 2024
