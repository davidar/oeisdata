## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    sl, sr, sk = "", "", "1"
    for k in count(1):
        sk = bin(k)[2:]
        sl += sk
        yield int(sl + sr, 2)
        sr = sk + sr
print(list(islice(agen(), 16))) # _Michael S. Branicky_, Feb 18 2023
```
