from itertools import accumulate
def A173253(n):
    if n<=1:
        return n+1
    c, blist = 2, (0,1)
    for _ in range(n-1):
        c += (blist := tuple(accumulate(reversed(blist),initial=0)))[-1]
    return c # _Chai Wah Wu_, Apr 16 2023
