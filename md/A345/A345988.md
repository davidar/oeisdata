## Python
```Python
from itertools import combinations
from math import prod
from sympy import factorint
from sympy.ntheory.modular import crt
def A345988(n):
    if n == 1:
        return 2
    plist = tuple(p**q for p, q in factorint(n).items())
    return n*(n-1) if len(plist) == 1 else (s:= int(min(min(crt((m, n//m), (0, -1))[0], crt((n//m, m), (0, -1))[0]) for m in (prod(d) for l in range(1, len(plist)//2+1) for d in combinations(plist, l)))))*(s+1) # _Chai Wah Wu_, May 31 2022
```
