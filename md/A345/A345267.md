from sympy import bernoulli
def a(n):
    n_ = n % 8
    k = n//4 + 1
    b = bernoulli(2*k)/(4*k)
    d = {0:1, 1:2, 4:1, 5:1}
    if n_ == 2:
        return 2 * b.numerator()
    elif n_ == 3:
        return 2 * b.denominator()
    elif n_ == 6:
        return -1 * b.numerator()
    elif n_ == 7:
        return b.denominator()
    else:
        return d[n_]
