## Python
```Python
from num2words import num2words
def n2w(n):
    return num2words(n).replace(" and", "") .replace(",", "")
def afind(startfrom=1, limit=float('inf')):
    last, t = startfrom, startfrom + 1 + startfrom%2
    if startfrom%2 == 1:
        print(startfrom, end=", ")
    while t <= limit:
        target = n2w(last)
        while n2w(t) >= target:
            t += 2
            if t > limit: return
        last = t
        print(t, end=", ")
afind(limit=10**6) # _Michael S. Branicky_, Jan 04 2022
```
