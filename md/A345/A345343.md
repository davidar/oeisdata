## Python
```Python
from sympy import isprime
def comp(d, i): return d[:i] + str((10-int(d[i]))%10) + d[i+1:]
def ok(n):
    d = str(n)
    return all(isprime(int(comp(d, i))) for i in range(len(d)))
print(list(filter(ok, range(1, 11000)))) # _Michael S. Branicky_, Jun 14 2021
```
