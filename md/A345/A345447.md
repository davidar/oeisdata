def aupto(limit):
    aset = set()
    for i in range(1, limit//3):
        for j in range(i, limit//3):
            t = i + j + 2*i*j
            if t > limit: break
            aset.update([t, t+2])
    return sorted(an for an in aset if an <= limit)
print(aupto(80)) # _Michael S. Branicky_, Jul 05 2021
