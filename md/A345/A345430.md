from sympy.core.numbers import igcdex
def A345430(n): return sum(abs(v) for u, v, w in (igcdex(x,y) for x in range(1,n+1) for y in range(1,n+1)) if w == 1) # _Chai Wah Wu_, Jun 22 2021
