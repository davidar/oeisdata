## Python
```Python
from sympy import isprime
def a(n, d=1):
    k, bread = 0, str(d)*n
    while not isprime(int(bread + str(k) + bread)): k += 1
    return k
print([a(n) for n in range(1, 67)]) # _Michael S. Branicky_, Jun 11 2021
```
