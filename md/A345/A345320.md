from math import prod
from sympy import factorint
def A345320(n):
    f = factorint(n).items()
    return (prod(p**(q+1)-1 for p, q in f) - prod(p**(q//2+1)-1 for p, q in f))//prod(p-1 for p, q in f) # _Chai Wah Wu_, Jun 14 2021
