T:= proc(x,y) option remember; local g,u0,v0,t0,t1,t2;
   g:= igcd(x,y);
   if g > 1 then return procname(x/g,y/g) fi;
   v0:= y^(-1) mod x;
   u0:= (1-y*v0)/x;
   t0:= (v0*x-u0*y)/(x^2+y^2);
   t1:= floor(t0);
   if t0 < t1 + 1/2 then u0+v0 + t1*(y-x)
   else u0+v0 + (t1+1)*(y-x)
   fi
end proc:
