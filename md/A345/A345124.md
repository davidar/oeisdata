from sympy import isprime
def A345124(n):
    C = [[1]]
    for i in range(n-1):
        C.append(sum(([6*c-1,6*c+1] for c in C[-1]),[]))
    k = 1
    while 1:
        k6 = 6*k
        for i in range(n):
            if any(isprime(k6-c) or isprime(k6+c) for c in C[i]):
                break
            k6 *= 6
        else:
            return k
        k += 1 # _Pontus von BrÃ¶mssen_, Nov 01 2021
