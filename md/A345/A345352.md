## Python
```Python
def a(n):
    b = bin(n)[2:]
    bb = bin(len(b))[2:]
    if bb != '1' + '0'*(len(bb)-1): b = '0'*(2**len(bb) - len(b)) + b
    return int(b[:len(b)//2][::-1] + b[len(b)//2:][::-1], 2)
print([a(n) for n in range(60)]) # _Michael S. Branicky_, Jun 15 2021
```
