## Python
```Python
from sympy import divisors
def a(n):
    if n%2 == 0: return n
    divs = divisors(n)
    for i, d in enumerate(divs[:len(divs)//2]):
        p = lastj = d
        for j in divs[i+1:]:
            if p >= n or j - lastj > 1: break
            p, lastj = p*j, j
        if p == n: return d
    return n
print([a(n) for n in range(1, 70)]) # _Michael S. Branicky_, Jun 29 2021
```
