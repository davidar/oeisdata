from itertools import count, islice
def A133743_gen(): # generator of terms
    aset2, alist = set(), []
    for k in map(lambda x:x**2, count(1)):
        bset2 = set()
        for a in alist:
            if (b:=a+k) in aset2:
                break
            bset2.add(b)
        else:
            yield k
            alist.append(k)
            aset2.update(bset2)
A133743_list = list(islice(A133743_gen(),30)) # _Chai Wah Wu_, Sep 11 2023
