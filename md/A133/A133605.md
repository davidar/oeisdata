from itertools import count, islice
def A133605_gen(): # generator of terms
    aset2, alist = set(), []
    for k in count(1):
        bset2 = set()
        for a in alist:
            if (b:=a+k) in aset2:
                break
            bset2.add(b)
        else:
            if k in aset2:
                yield k
            alist.append(k)
            aset2.update(bset2)
A133605_list = list(islice(A133605_gen(),30)) # _Chai Wah Wu_, Sep 11 2023
