## Python
```Python
from sympy import isprime
A321210_list = []
for i in range(2**20):
    s = bin(i)[2:]
    s += s[-2::-1]
    p = int(s) + int('02'*(len(s)//2)+'0')
    q = 6*p+1
    t = str(q)
    if t == t[::-1] and isprime(p) and isprime(q):
        A321210_list.append(q) # _Chai Wah Wu_, Dec 06 2019
```
