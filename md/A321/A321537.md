read("transforms"):
A321537 := proc(n)
    local dgsin,dgsout,pos ;
    dgsin := convert(n,base,10) ;
    dgsout := [] ;
    for pos from 2 to nops(dgsin) do
        if op(pos,dgsin) = op(pos-1,dgsin) then
            dgsout := [op(pos,dgsin),op(dgsout)] ;
        end if;
    end do:
    digcatL(dgsout) ;
