## Python
```Python
from __future__ import division
def A321441(n):
    if n == 0:
        return 1
    c = 0
    for i in range(n):
        mi = n + i*(i+1)//2
        for j in range(i,n):
            mj = mi + j*(j+1)//2
            for k in range(j+1,n+1):
                r = mj - k*k
                if r < 0:
                    break
                if not r % k:
                    c += 1
    return c # _Chai Wah Wu_, Nov 11 2018
```
