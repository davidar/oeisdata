## Python
```Python
from sympy import divisors
def a(n):
    if n == 1: return 1
    divisorsn, m = len(divisors(n)), n + 1
    while len(divisors(m)) != divisorsn: m += 1
    return m
print([a(n) for n in range(1, 72)]) # _Michael S. Branicky_, Sep 14 2021
```
