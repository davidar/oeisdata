def A079475(n):
    s = str(n)
    l = len(s)
    if l % 2:
        s = s[:-1]+'1'+s[-1]
    return int(''.join(s[i+1]*int(s[i]) for i in range(0,l,2))) # _Chai Wah Wu_, Sep 01 2021
