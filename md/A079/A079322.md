## Python
```Python
from sympy import isprime
from itertools import count
def agen():
    p = 1
    for k in count(1):
        p *= k**k
        if not isprime(p + 11):
            yield p + 11
g = agen()
print([next(g) for n in range(1, 12)]) # _Michael S. Branicky_, Dec 15 2021
```
