## Maple

```Maple
# Implementation by _R. J. Mathar_, May 08 2019:
A045918 := proc(n)
    local a,f,pd,dgs,i ;
    a := [] ;
    f := 0 ;
    pd := -1 ;
    dgs := convert(n,base,10) ;
    for i from 1 to nops(dgs) do
        if op(-i,dgs) <> pd then
            if pd >= 0 then
                a := [op(a),f,pd] ;
            end if;
            pd := op(-i,dgs) ;
            f := 1 ;
        else
            f:= f+1 ;
        end if;
    end do:
    a := [op(a),f,pd] ;
    digcatL(%) ;
end proc:
isA079342 := proc(n)
    simplify( modp(A045918(n) ,n) = 0 ) ;
end proc:
for n from 1 to 30000 do
    if isA079342(n) then
        print(n) ;
    end if;
end do:
```

## Python
```Python
def LS(n): return int(''.join(str(len(list(g)))+k for k, g in groupby(str(n))))
def ok(n): return LS(n)%n == 0
print([k for k in range(1, 10**4) if ok(k)]) # _Michael S. Branicky_, Aug 28 2024
```
