## Python

from sympy import factorint, prod
def a072438(n):
    f = factorint(n)
    return 1 if n == 1 else prod(i**f[i] for i in f if i % 4 != 1)
def a(n): return n//a072438(n) # _Indranil Ghosh_, May 08 2017
