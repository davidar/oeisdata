## Maple
```Maple
isA251603 := proc(n)
    if modp(n &^ n-2,n+2) = 0 then
        true;
    else
        false;
    end if;
end proc:
A251603 := proc(n)
    option remember;
    local a;
    if n = 1 then
        3;
    else
        for a from procname(n-1)+1 do
            if isA251603(a) then
                return a;
            end if;
        end do:
    end if;
end proc: # _R. J. Mathar_, Jan 09 2015
```
