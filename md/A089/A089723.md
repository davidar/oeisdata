## Python
```Python
from math import gcd
from sympy import factorint, divisor_sigma
def a(n):
    if n == 1: return 1
    e = list(factorint(n).values())
    g = e[0]
    for ei in e[1:]: g = gcd(g, ei)
    return divisor_sigma(g, 0)
print([a(n) for n in range(1, 105)]) # _Michael S. Branicky_, Jul 15 2021
```
