## Maple
```Maple
with(numtheory);
A007814 := n -> padic[ordp](n, 2):
T:= n -> add(2^(3*m),m=0..A007814(n));
f := proc(n) local t2,i,d;
if n=0 then return(1); fi;
t2:=0; for d in divisors(n) do if (d mod 2) = 1 then t2:=t2+d^3; fi; od:
48*(1+12*T(n))*t2;
end;
[seq(f(n),n=0..50)];
```
