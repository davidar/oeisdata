A318157_list = []
for n in range(2,10**6):
    if not (isprime(n) or isprime(n+1) or isprime(n+2) or isprime(n+3)):
        if isprime(4*n+5):
            A318157_list.append(4*n+5)
        if isprime(4*n+7):
