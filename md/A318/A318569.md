## Python
```Python
def comp(s): z, o = ord('0'), ord('1'); return s.translate({z:o, o:z})
def BCR(n): return int(comp(bin(n)[2:])[::-1], 2)
def a(n):
    kn = n
    while BCR(kn) != kn: kn += n
    return kn//n
print([a(n) for n in range(1, 72)]) # _Michael S. Branicky_, Mar 20 2022
```
