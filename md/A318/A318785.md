## Python
```Python
nmax=500
def is_prime(num):
    if num == 0 or num == 1: return(0)
    for k in range(2, num):
       if (num % k) == 0:
           return(0)
    return(1)
def c9(num):
    s=str(num)
    l=len(str(num))
    n=""
    for k in range(l):
        n = n+str(9-int(s[k]))
    return(int(n))
ris = ""
for i in range(2,nmax):
    if is_prime(c9(i)):
       ris = ris+str(i)+","
print(ris)
```
