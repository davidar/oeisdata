## Python

from sympy import divisors
def A146208(n):
    ds = divisors(n)
    c, s = 0, [-d for d in ds[::-1]]+ds
    for x in s:
        d2 = [d//x for d in ds if d%x==0]
        for y in (f-x for f in [-d for d in d2[::-1]]+d2):
            m, k = x*(z:=x+y), 1
            while n >= abs(m) and k<=n:
                if n == m:
                    c += 1
                z += y
                m *= z
                k += 1
    return c # _Chai Wah Wu_, May 11 2023
