## Python
```Python
from sympy import nextprime, Matrix
k,A117345_list, plist = 1,[], [2, 3, 5, 7, 11, 13, 17, 19, 23]
while k < 10**7:
    if Matrix(plist).reshape(3,3).det() == 0:
        A117345_list.append(k)
    k,plist = k+1,plist[1:]+[nextprime(plist[-1])] # _Chai Wah Wu_, Jan 25 2021
```
