## Python
```Python
from sympy import isprime
from itertools import product
def a(n):
    for p in product("123456789", repeat=n):
        s = "".join(p)
        if isprime(sum(int(s[:i])**int(s[i:]) for i in range(1, n))):
            return int(s)
print([a(n) for n in range(2, 6)]) # _Michael S. Branicky_, Jun 27 2022
```
