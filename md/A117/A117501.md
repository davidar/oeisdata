## Python

from sympy.core.cache import cacheit
@cacheit
def a(n, k):
    return 1 if k==1 else n if k==2 else a(n, k - 1) + a(n, k - 2)
for n in range(1, 21): print([a(n - k + 1, k) for k in range(1, n + 1)]) # _Indranil Ghosh_, Aug 19 2017
