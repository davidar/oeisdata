## Maple
```Maple
A117818 := proc(n)
    local a,d;
    if isprime(n) or n =1 then
        return n;
    end if;
    a := -1 ;
    for d in numtheory[divisors](n) do
        if d < n and d> a then
            a := d ;
        end if;
    end do:
    a ;
end proc:
```

```Python
import sympy
def A117818(n):
    if n == 1:
        return 1
    else:
        _=sympy.ntheory.factor_.primefactors(n)
        return _[-1]
```
