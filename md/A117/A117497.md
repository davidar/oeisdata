## Maple
```Maple
A117497 := proc(n)
    option remember ;
    local prev,d,a ;
    if n <= 2 then
        n-1 ;
    else
        a := n ;
        for prev from n-1 to 1 by -1 do
            for d in numtheory[divisors](prev) do
                if d+prev = n then
                    a := min(a,procname(prev)+1) ;
                end if;
            end do:
        end do:
        a ;
    end if;
end proc:
seq(A117497(n),n=1..100) ; # _R. J. Mathar_, Mar 02 2022
```
