## Python

from sympy import isprime
A117790_list, a, b = [1], 1, 3
for _ in range(10**6):
    if isprime(sum(int(d) for d in str(b))):
        A117790_list.append(b)
    a, b = b, a+b # _Chai Wah Wu_, Dec 25 2015
