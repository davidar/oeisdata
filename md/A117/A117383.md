## Python
```Python
from num2words import num2words as n2w
def letters(n):
    return "".join(c for c in n2w(n, lang='es') if c.isalpha())
def ok(n):
    w = letters(n)
    return len(w) == len(set(w))
print([k for k in range(1, 100000) if ok(k)]) # _Michael S. Branicky_, Jun 08 2022
```
