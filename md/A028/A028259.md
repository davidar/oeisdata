j = 0
while j<3100000:
# to obtain n correct Engel expansion terms about n^2/2 continued fraction steps are needed; 3100000 is a safe bound
    if j == 0:
        p0, q0 = 1, 1
    elif j == 1:
        p1, q1 = p0+1, 1
    else:
        p0, p1 = p1, p1+p0
        q0, q1 = q1, q1+q0
    j = j+1
