## Python
```Python
from sympy import totient, n_order, divisors
def A023138(n):
    m = n>>(~n & n-1).bit_length()
    a, b = divmod(m,3)
    while not b:
        m = a
        a, b = divmod(m,3)
    return sum(totient(d)//n_order(6,d) for d in divisors(m,generator=True) if d>1)+1 # _Chai Wah Wu_, Apr 09 2024
```
