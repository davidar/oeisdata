Res:= 0:
for d from 1 to 7 do
  for n from 10^(d-1)+8 to 5*10^(d-1)-1 by 9 do
    if sort(convert(n,base,10)) = sort(convert(2*n,base,10)) then
      Res:= Res, n
    fi
od od:
Res; # _Robert Israel_, Mar 20 2017
