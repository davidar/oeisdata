## Python
```Python
def A023143(end):
    primes=[2,3]
    a023143_list=[1]
    num=3
    while len(primes)<=end:
        num+=1
        prime=False
        length=len(primes)
        for y in range(0,length):
            if num % primes[y]!=0:
                prime=True
            else:
                prime=False
                break
        if (prime):
            primes.append(num)
    for x in range(2, len(primes)):
        if (primes[x-1]%(x))==1:
            a023143_list.append(x)
    return a023143_list
```
