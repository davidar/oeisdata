## Python
```Python
from collections import Counter
def passes(x): return set(Counter(str(x**2)).values()) == {2}
def afull():
    c = 0
    for n in range(1, 11):
        c += sum(1 for x in range(10**(n-1), 10**n) if passes(x))
        print(c, end=", ")
afull() # _Michael S. Branicky_, May 12 2023
```
