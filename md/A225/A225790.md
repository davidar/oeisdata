## GAP
```GAP
A225790 := n -> Size(grp(n));
grp := n -> Group(Concatenation(List([1,n+1..n^2-n+1], s->flip(s, n, 1)), List([1..n], s->flip(s, n, n))));
flip := function(start, nterms, skip) return Product([1..Int(nterms/2)], m->(start + skip*(m - 1), start + skip*(nterms - m)), ()); end; # _Eric M. Schmidt_, Nov 05 2013
```

## Python
```Python
a1,n = 1,1
print(n,a1)
while n < 12:
    n = n+1
    if n%2 == 0:
        nn = n//2
        a = 2**(2*nn*nn+2*nn-1)*3**(nn*nn)
        a1 = a
    else:
        a = 4*a1
    print(n,a) # _A.H.M. Smeets_, Aug 15 2019
```
