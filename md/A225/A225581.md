## Python
```Python
from sympy import isprime, nextprime, prime
def a(n):
    q, p = 3, prime(n)
    while not isprime(p*q + p + q): q = nextprime(q)
    return q
print([a(n) for n in range(1, 102)]) # _Michael S. Branicky_, Sep 06 2021
```
