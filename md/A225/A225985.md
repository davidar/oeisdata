from itertools import count, islice
def agen(): # generator of terms
    for n in count(1):
        removed  = "".join(d if d in "13579" else "" for d in str(n))
        if removed != "": yield int(removed)
print(list(islice(agen(), 80))) # _Michael S. Branicky_, Aug 05 2022
