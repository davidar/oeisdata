## Python
```Python
def isTriangular(a):
    sr = 1L << (long.bit_length(long(a)) >> 1)
    a += a
    while a < sr*(sr+1):  sr>>=1
    b = sr>>1
    while b:
      s = sr+b
      if a >= s*(s+1):  sr = s
      b>>=1
    return (a==sr*(sr+1))
n = tn = 0
while 1:
for m in range(1, 1000000000):
    if isTriangular(tn + m*(m+1)): break
  print str(m)+',',
  n += 1
  tn += n
```
