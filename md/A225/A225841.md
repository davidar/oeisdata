## Python
```Python
primes = []
n = 1
sum = 2
primorial = 6
def addPrime(k):
  global n, sum, primorial
  for p in primes:
    if k%p==0:  return
    if p*p > k:  break
  primes.append(k)
  sum += primorial
  primorial *= k
  n += 1
  if sum % n == 0:  print(n, end=',')
print(1, end=',')
for p in range(5, 100000, 6):
  addPrime(p)
  addPrime(p+2)
```
