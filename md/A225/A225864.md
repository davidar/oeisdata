## Python
```Python
from __future__ import division
from sympy import isprime
A225864_list = []
for l in range(1,20):
    plist, q = [p for p in [2,3,5,7] if isprime(l-1+p)], (10**l-1)//9
    for i in range(l):
        for p in plist:
            r = q+(p-1)*10**i
            if not isprime(r):
                A225864_list.append(r) # _Chai Wah Wu_, Aug 15 2017
```
