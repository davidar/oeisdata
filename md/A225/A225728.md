primes = [2]*2
primes[1] = 3
def addPrime(k):
  for p in primes:
    if k%p==0:  return
    if p*p > k:  break
  primes.append(k)
for n in range(5,100000000,6):
  addPrime(n)
  addPrime(n+2)
sum = 0
primorial = 1
for p in primes:
  sum += primorial
