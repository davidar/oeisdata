def isTriangular(a):
    sr = 1L << (long.bit_length(a) >> 1)
    a += a
    while a < sr*(sr+1):  sr>>=1
    b = sr>>1
    while b:
      s = sr+b
      if a >= s*(s+1):  sr = s
      b>>=1
    return (a==sr*(sr+1))
