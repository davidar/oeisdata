## Maple
```Maple
with(numtheory);
P:=proc(i)
local f,n,p,pfs;
for n from 1 to i do
    pfs:=ifactors(n)[2];
    f:=n*add(op(2,p)/op(1,p),p=pfs);
    if sigma(n)=sigma(f) then print(n); fi;
od;
end:
P(1000);
```
