## Python

from operator import mul
from sympy import primefactors
def rad(n): return 1 if n<2 else reduce(mul, primefactors(n))
def a(n): return int(round(rad(n**2 + n)/(n + 1))) # _Indranil Ghosh_, May 24 2017
