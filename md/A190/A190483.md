from sympy import sqrt, floor
r=sqrt(2)
def a(n): return floor((2*n + 1)*r) - 2*floor(n*r) - floor(r)
print([a(n) for n in range(1, 501)]) # _Indranil Ghosh_, Jul 02 2017
