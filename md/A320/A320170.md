## Python
```Python
CurrentNum=10
PreviousNum=1
NewNum=0
def NextPalindrome(StartNum):
    FoundNext=0
    t=0
    n=0
    Number=0
    Lastdigit=0
    while FoundNext<=PreviousNum:
        n=n+1
        t=StartNum+n
        Number=t
        Reverse=0
        while Number>0:
            Lastdigit=Number%10
            Reverse=(Reverse*10)+Lastdigit
            Number=Number//10
        if t==Reverse:
            FoundNext=n
    return n
print(1)
print(10)
for x in range(50):
    NewNum=NextPalindrome(CurrentNum)
    print(NewNum)
    PreviousNum=CurrentNum
    CurrentNum=NewNum
```
