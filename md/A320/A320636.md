## Python
```Python
def A073785(n): # after _Reinhard Zumkeller_
    if n == 0: return 0
    (q, r) = divmod(n, -3)
    (nn, m) = (q, r) if r >= 0 else (q+1, r+3)
    return A073785(nn)*10 + m
def a(n): return A073785(-n)
print([a(n) for n in range(1, 47)]) # _Michael S. Branicky_, Dec 11 2021
```
