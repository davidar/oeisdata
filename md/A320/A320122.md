## Python
```Python
def digits(n, b):
    r = []
    m = n
    while m > 0:
        r = [m % b] + r
        m = m // b
    return r
def fibo(n, b):
    L = digits(n, b)
    m = len(L) - 1
    while L[m] < n:
        L.append(sum(k for k in L))
        L.pop(0)
    return L[m] == n
def test(n):
    for b in range(2, n + 1):
        if fibo(n, b):
            return True
    return False
print([n for n in range(2, 2001) if not test(n)])
```
