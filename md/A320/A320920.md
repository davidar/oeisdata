## Python
```Python
from sympy import factorial, binomial
def A320920(n):
    w, m = int(factorial(n)), n
    bc = [int(binomial(n-1,i)) % w for i in range(n+1)]
    while True:
        bc[n] = (bc[n-1]+bc[n]) % w
        if bc[n] == 0:
            return m
        for i in range(n-1,0,-1):
            bc[i] = (bc[i-1]+bc[i]) % w
        m += 1 # _Chai Wah Wu_, Oct 25 2018
```
