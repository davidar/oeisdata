## Python
```Python
def A320572(n):
    n = int(str(n).strip('0'))
    if n != 1:
        s = set(range(1, 10))
        a = 0
        m = 1
        while s:
            a += 1
            m *= n
            s.difference_update(int(z) for z in str(m))
        return a
    else:
        return 0
```
