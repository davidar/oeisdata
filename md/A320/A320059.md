## Python
```Python
from __future__ import division
from sympy import factorint
def A320059(n):
    c1, c2 = 1, 1
    for p, a in factorint(n).items():
        c1 *= (p**(2*a+1)-1)//(p-1)
        c2 *= (p**(a+1)-1)//(p-1)
    return c1-c2 # _Chai Wah Wu_, Oct 05 2018
```
