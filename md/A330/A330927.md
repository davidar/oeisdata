## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    h1, h2 = 1, 2
    while True:
        if h2 - h1 == 1: yield h1
        h1, h2 = h2, next(k for k in count(h2+1) if k%sum(map(int, str(k))) == 0)
print(list(islice(agen(), 52))) # _Michael S. Branicky_, Mar 17 2024
```
