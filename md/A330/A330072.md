def bitlist(n):
    output = []
    while n != 0:
        output.append(n % 2)
        n /= 2
    return output
#converts a number into a list of the digits in binary reversed
def bitsum(bitlist):
    output = 0
    for bit in bitlist:
        if bit == 1:
            output += 1
    return output
#gives the cross sum of a bitlist
def a(bitlist):
    output = 0
    l = len(bitlist)
    for x in range(l):
        output += bitlist[x] * (l - x) * (2**(x + 1) - 1)
    return output
