## Python
```Python
from sympy import primepi, integer_nthroot, primefactors
def A330669(n):
    if n == 1: return 0
    def f(x): return int(n-2+x-sum(primepi(integer_nthroot(x,k)[0]) for k in range(1,x.bit_length())))
    kmin, kmax = 1,2
    while f(kmax) >= kmax:
        kmax <<= 1
    while True:
        kmid = kmax+kmin>>1
        if f(kmid) < kmid:
            kmax = kmid
        else:
            kmin = kmid
        if kmax-kmin <= 1:
            break
    return int(primepi(primefactors(kmax)[0])) # _Chai Wah Wu_, Aug 20 2024
```
