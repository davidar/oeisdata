## Python
```Python
def BaseUp(n,b):
    up, b1 = 0, 1
    while n > 0:
        up, b1, n = up+(n%b)*b1, b1*(b+1), n//b
    return up
n = 2
while n < 20:
    k = n
    while BaseUp(k,n)%k != 0:
        k = k+1
    print(n,k)
    n = n+1 # _A.H.M. Smeets_, Mar 31 2020
```
