## Python
```Python
def primes(n):
    divisors = [ d for d in range(2,n//2+1) if n % d == 0 ]
    return [ d for d in divisors if \
             all( d % od != 0 for od in divisors if od != d ) ]
pprimes = {}
for i in range(1, 10000):
    res = len(primes(i))
    if res == 0:
        res = 1
    pprimes[i] = res
for k in range(1, 10000):
    s = 0
    for i in range(1, k):
        if pprimes[i] <= pprimes[k]:
            s+=1
    print(s)
```
