## Python
```Python
from sympy import isprime
def afind(limit, startk=0):
    pow2, pow3 = 2**startk, 3**startk
    for k in range(startk, limit+1):
        if isprime(int(str(pow2) + str(pow3))): print(k, end=", ")
        pow2 *= 2; pow3 *= 3
afind(300) # _Michael S. Branicky_, Sep 08 2021
```
