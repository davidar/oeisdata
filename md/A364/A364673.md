from collections import Counter
def A364673_list(maxn):
    count = Counter()
    for i in range(maxn//3):
        A,f,i = [[(i+1, )]],0,0
        while f == 0:
            A.append([])
            for j in A[i]:
                for k in j:
                    x = j + (j[-1] + k, )
                    y = sum(x)
                    if y <= maxn:
                        A[i+1].append(x)
                        count.update({y})
            if len(A[i+1]) < 1: f += 1
            i += 1
    return [count[z]+1 for z in range(maxn+1)] # _John Tyler Rascoe_, Mar 09 2024
