from math import gcd
from sympy import nextprime
def A364493(n):
    c, p, k = 1, 1, n
    while k:
        c *= (p:=nextprime(p))**(s:=(~k&k-1).bit_length())
        k >>= s+1
    return n*c*p//gcd(c*p,n)**2 # _Chai Wah Wu_, Jul 26 2023
