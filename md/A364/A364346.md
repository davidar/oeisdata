## Python
```Python
from collections import Counter
from itertools import combinations_with_replacement
from sympy.utilities.iterables import partitions
def A364346(n): return sum(1 for p in partitions(n) if max(p.values(),default=1)==1 and not any(q[0]+q[1]==q[2] for q in combinations_with_replacement(sorted(Counter(p).elements()),3))) # _Chai Wah Wu_, Sep 20 2023
```
