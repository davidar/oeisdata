from collections import Counter
def A364935_list(max_n):
    A, C = [1], Counter()
    for n in range(2,max_n+1):
        x = str(A[n-2])
        C.update({x})
        z = str(int(x[0])+C[x]) + x + str(int(x[-1])+C[x])
        A.append(sum(int(z[i]) for i in range (0,len(z))))
    return(A) # _John Tyler Rascoe_, Oct 22 2023
