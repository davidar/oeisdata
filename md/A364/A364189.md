from itertools import count
from sympy import divisor_count
def A364189(n):
    b = {a:=tuple(divisor_count(i) for i in range(1,n+1)),}
    for m in count(n+1):
        if (a:=a[1:]+(divisor_count(m),)) in b:
            return m-n+1
