## Python
```Python
def A364801(n):
    if n<4: return 0
    a, b, s = 1, 2, 0
    for i in bin(n)[-1:1:-1]:
        if int(i):
            s += a
        a, b = b, a+b
    return A364801(s)+1 # _Chai Wah Wu_, Aug 10 2023
```
