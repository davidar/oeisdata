## Python
```Python
def a(n):
  paths = [[0]*(n+1) for _ in range(n+1)]
  paths[0][0] = 1
  result = paths[0][n]
  for turn in range(1,n+1):
    for value in range(n+1):
      if paths[turn-1][value] > 0:
        if value-turn >= 0:
          paths[turn][value-turn] += paths[turn-1][value]
        if value+turn <= n:
          paths[turn][value+turn] += paths[turn-1][value]
    result += paths[turn][n]
  return result
```
