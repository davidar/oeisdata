using Nemo
function bigger_than_prime_bound(len)
    R, z = PolynomialRing(ZZ, "z")
    exp = eta_qexp(24, len, z)
    t = BigInt.([coeff(exp, j) for j in 0:len - 1])
    n_vals = []
    for i in eachindex(t)
        if abs(t[i]) > 2 * i ^ (11/2)
            push!(n_vals, i)
        end
    end
    n_vals
end
