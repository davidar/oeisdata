```Python
def g(L):
  D = {}
  SQ = [x*x for x in range(0, 3*L+1)]
  for j in range(1, L):
    for k in range(1, L):
      q = SQ[j + k * 2] - SQ[j + k] - SQ[j]
      if (0 < q < L):
        if q not in D:
          D[q] = 1
        else:
          D[q] += 1
  return [x for x in sorted(D.keys()) if D[x] > 1]
print(g(256))
```

```Python
from sympy import divisors
def isok(A_n):
  s = 0
  for d in divisors(A_n):
    t = A_n // d + d
    q, r = divmod(t, 4)
    if r == 0 and q < d: s += 1
  return s > 1
```
