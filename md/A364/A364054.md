## Python
```Python
from itertools import count, islice
from sympy import nextprime
def A364054_gen(): # generator of terms
    a, aset, p = 1, {0,1}, 2
    while True:
        yield a
        for b in count(a%p,p):
            if b not in aset:
                aset.add(b)
                a, p = b, nextprime(p)
                break
A364054_list = list(islice(A364054_gen(),30)) # _Chai Wah Wu_, Oct 22 2023
```
