## Python
```Python
from itertools import count
from sympy import isprime
def A364879(n):
    c, cn, m = 0, 0, n+1<<1
    for k in count(2):
        if isprime(k):
            c += k
            cn += k*m
        if k*(k+1)-1 >= cn:
            return k # _Chai Wah Wu_, Sep 10 2023
```
