## Python
```Python
from collections import Counter
from sympy.utilities.iterables import partitions
def A364464(n): return sum(1 for s,p in map(lambda x: (x[0],tuple(sorted(Counter(x[1]).elements()))), filter(lambda p:max(p[1].values(),default=1)==1,partitions(n,size=True))) if set(p).isdisjoint({p[i+1]-p[i] for i in range(s-1)})) # _Chai Wah Wu_, Sep 26 2023
```
