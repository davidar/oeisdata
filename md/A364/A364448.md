## Python
```Python
a = []
ban = [0 for n in range(160)]
for i in range(1000):
    can = ban.index(0,1)
    ban = [max(b-1,0) for b in ban]
    a.append(can)
    ban[can] = can**2
```

## C
```C
#include<stdlib.h>
int main(void){
    int N = 1000;
    int *a = (int*)malloc((N+1)*sizeof(int));
    int *ban = (int*)malloc(160*sizeof(int));
    for(int n = 0; n < N; ++n){
        for(int can = 1; can < 160; ++can){
            if(ban[can] == 0){
                a[n] = can;
                ban[can] = can*can+1;
                break;
            }
        }
        for(int can = 1; can < 160; ++can) if(ban[can]) --ban[can];
    }
    free(a); free(ban);
    return 0;
}
```
