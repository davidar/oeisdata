## Python
```Python
def a_list(n):
    if n <= 2:
        return 1
    sequence = [1, 1]
    target_number_index = 0
    times_to_add = sequence[target_number_index]
    for _ in range(n - 2):
        if times_to_add == 0:
            target_number_index += 1
            times_to_add = sequence[target_number_index]
        last_term = sequence[-1]
        sequence.append(last_term + sequence[target_number_index])
        times_to_add -= 1
    return sequence
```
