## Sage
```Sage
def A006232_list(len):
    f, R, C = 1, [1], [1]+[0]*(len-1)
    for n in (1..len-1):
        for k in range(n, 0, -1):
            C[k] = -C[k-1] * k / (k + 1)
        C[0] = -sum(C[k] for k in (1..n))
        R.append((C[0]*f).numerator())
        f *= n
    return R
print(A006232_list(20)) # _Peter Luschny_, Feb 19 2016
```
