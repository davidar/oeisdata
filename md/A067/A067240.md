A067240(n)=
{
    local(f=factor(n), r=0, p, e);
    for (i=1, matsize(f)[1],
        p=f[i,1];  e=f[i,2];
        r += (p-1)*p^(e-1);
    );
    return(r);
} /* _Joerg Arndt_, Jun 10 2011 */
