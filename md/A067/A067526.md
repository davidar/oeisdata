## Python
```Python
from sympy import isprime
def ok(n):
  k, pow2 = 1, 2
  while pow2 < n - 1:
    if not isprime(n-pow2): return False
    pow2 *= 2
  return (2 < n)
print([m for m in range(1, 200) if ok(m)]) # _Michael S. Branicky_, Mar 04 2021
```
