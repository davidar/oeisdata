from sympy import primepi, isprime
def A067970(n):
    if n == 0: return 8
    m, k = n, primepi(n+1) + n + (n+1>>1)
    while m != k:
        m, k = k, primepi(k) + n + (k>>1)
    for d in range(2,7,2):
        if not isprime(m+d):
            return d # _Chai Wah Wu_, Jul 31 2024
