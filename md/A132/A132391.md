```Maple
    m:= nops(X);
    if m > 0 then
      count:= count+nops(X);
      R:= R, op(X);
    fi
```

## Python
```Python
def aupto(limit):
  alst = []
  for m in range(34, limit+1):
    if len(set(str(m*m)[:4])) == 1: alst.append(m)
  return alst
print(aupto(33333)) # _Michael S. Branicky_, Mar 12 2021
```
