def row(n):
    perm = tuple(range(1, n+1)); reach = {perm}; frontier = {perm}
    out = [1] if n == 0 else []
    for k in range(n):
        r1 = set()
        out.append(len(frontier))
        while len(frontier) > 0:
            q = frontier.pop()
            for i in range(n):
                for j in range(i+1, n+1):
                    qp = list(q)
                    qp[i:j] = qp[i:j][::-1]
                    qp = tuple(qp)
                    if qp not in reach:
                        reach.add(qp)
                        r1.add(qp)
        frontier = r1
    return out
print([an for n in range(9) for an in row(n)]) # _Michael S. Branicky_, Jan 26 2023
