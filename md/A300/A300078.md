## Python

from fractions import Fraction
def A300078(n):
    zr, zc, c = Fraction(0,1), Fraction(0,1), 0
    cr, cc = Fraction(-5,4)-Fraction(1,10**(2*n)), Fraction(1,10**n)
    zr2, zc2 = zr**2, zc**2
    while zr2 + zc2 <= 4:
        zr, zc = zr2 - zc2 + cr, 2*zr*zc + cc
        zr2, zc2 = zr**2, zc**2
        c += 1
    return c # _Chai Wah Wu_, Mar 03 2018
