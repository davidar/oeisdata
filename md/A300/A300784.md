## Python
```Python
# see A159842 for the definition of dc, fin, per, u, N, N2
def a(n):
    return (dc(u, N, N2)(n) + 2*dc(fin(1, -1, 0, 4), u, u, N)(n)
      + 3*dc(fin(1, 3), u, u, N)(n)
      + 2*dc(fin(1, 1), u, u, per(0, 1, 0, -1))(n)) // 8
print([a(n) for n in range(1, 300)])
```
