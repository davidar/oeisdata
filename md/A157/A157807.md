## Python

from math import gcd
for s in range(2, 100, 2):
  for i in range(1, s):
    if gcd(i, s - i) != 1: continue
    print(i)
  for i in range(s, 0, -1):
    if gcd(i, s + 1 - i) != 1: continue
    print(i)
