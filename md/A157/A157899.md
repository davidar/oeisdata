## Python
```Python
# see linked program for going to large numbers
from num2words import num2words
def consonants(n):
    w = num2words(n).replace(" and", "")
    return sum(1 for c in w if c in "bcdfghjklmnpqrstvwxz")
def aupton(nn):
    k, adict, alst = 1, {}, []
    while len(alst) < nn:
        c = consonants(k)
        if c not in adict:
            adict[c] = k
            alst = []
            for i in range(1, max(adict)+1):
                if i not in adict:
                    break
                alst.append(adict[i])
        k += 1
    return alst
print(aupton(27)) # _Michael S. Branicky_, Jan 29 2022
```
