## Python
```Python
from math import prod, factorial
from itertools import combinations
def A038667(n):
    m = factorial(n)
    return 0 if n == 0 else min(abs((p:=prod(d))-m//p) for l in range(n,n//2,-1) for d in combinations(range(1,n+1),l)) # _Chai Wah Wu_, Apr 06 2022
```
