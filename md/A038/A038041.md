import math
def a(n):
    count = 0
    for k in range(1, n + 1):
        if n % k == 0:
            count += math.factorial(n) // (math.factorial(k) ** (n // k) * math.factorial(n // k))
    return count # _Paul Muljadi_, Sep 25 2024
