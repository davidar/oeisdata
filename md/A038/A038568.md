## Python

from sympy import totient, gcd
def a(n):
    s=1
    k=2
    while s<n:
        s+=2*totient(k)
        k+=1
    s-=2*totient(k - 1)
    j=1
    while s<n:
        if gcd(j, k - 1)==1: s+=2
        j+=1
    if s>n: return j - 1
    return k - 1 # _Indranil Ghosh_, May 23 2017, translated from Mathematica
