from itertools import count, islice
def a_gen():
    A = [1]
    yield(1)
    for m in count(0):
        for k in range(2**m):
            x = A[(2**m)-k-1]*A[k] + 1
            A.append(x)
            yield(x)
A119614_list = list(islice(a_gen(), 2**6)) # _John Tyler Rascoe_, May 10 2024
