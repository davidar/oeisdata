from sympy import factorint
A119908_list, c, s = [], {}, 3
for n in range(2,2**16):
....for p,e in factorint(4*n-2).items():
........if p in c:
............c[p] += e
........else:
............c[p] = e
....for p,e in factorint(n+1).items():
........if c[p] == e:
............del c[p]
........else:
............c[p] -= e
....if n == s:
........c2 = [p for p,e in c.items() if e >= 2]
........A119908_list.append(1 if c2 == [] else max(c2))
