## Python
```Python
def isrep(n, b):
  while n >= b:
    n, r = divmod(n, b)
    if r != 1: return False
  return n == 1
def agen():
  yield 1
  n = 2
  while True:
    reps = 2 # n is a repunit in bases 1 and n-1
    for b in range(2, n-1):
      if isrep(n, b): reps += 1
      if reps == 4: yield n; break
    n += 1
for m in agen(): print(m) # _Michael S. Branicky_, Jan 31 2021
```
