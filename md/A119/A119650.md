## Python
```Python
from sympy import isprime
def afind(limit, startk=0):
    c1, c2, pow10 = 2**67-1, 2**257-1, 10**startk
    for k in range(startk, limit+1):
        if isprime(c1*pow10 + c2): print(k, end=", ")
        pow10 *= 10
afind(1500) # _Michael S. Branicky_, Sep 12 2021
```
