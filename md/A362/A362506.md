## Python
```Python
from sympy import divisors
from itertools import count
def agen(): # generator of terms
    yield 0
    for n in count(1):
        for x in divisors(n):
            if set(str(x)) == set(str(n//x)):
                yield x
                break
print(list(islice(agen(), 71))) # _Michael S. Branicky_, Apr 23 2023
```
