## Python
```Python
from sympy import nextprime
from itertools import count, islice
def agen(): # generator of terms
    an = 1
    for n in count(2):
        yield an
        an = nextprime(an + n**2) - n**2
print(list(islice(agen(), 62))) # _Michael S. Branicky_, Apr 16 2023
```
