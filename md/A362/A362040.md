## Python
```Python
from itertools import islice
def gen_a():
    seen = set()
    sums = []
    new = 0
    while True:
        for v in sums: seen.add(v + new)
        sums = [v + new for v in sums]
        sums.append(0)
        new = len(seen)
        yield new
print(list(islice(gen_a(), 60))) # _Winston de Greef_, Apr 15 2023
```
