from sympy import isprime
binary_string = ""
for i in range(3, 300):
    if isprime(i):
        num_to_append = str(int(((i % 4)-1)/2))
        binary_string += num_to_append
        num_in_base_10 = int(binary_string, 2)
        print(i, num_to_append, binary_string, num_in_base_10)
