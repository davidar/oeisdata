from sympy import factorint
from itertools import islice
from collections import Counter
def A362031gen(): # generator of terms
    an, c, d = 1, Counter(), dict()
    while True:
        yield an
        pf = d[an] if an in d else sum(factorint(an).values())
        c[pf] += 1
        an = c[pf]
