from sympy import sieve
def okA362060(n):
    p = sieve[n]
    while n and p:
        if n%10 == p%10:
            n //= 10
        p //= 10
    return n == 0
print([sieve[k] for k in range(1, 10**6) if okA362060(k)]) # _Michael S. Branicky_, Apr 07 2023
