## Python

from itertools import count, islice
from sympy import primenu
def A362178_gen(): # generator of terms
    a, b = {1,2}, 2
    yield from (1,2)
    while True:
        for b in count(p:=primenu(b),p):
            if b not in a:
                yield b
                a.add(b)
                break
A362178_list = list(islice(A362178_gen(),20)) # _Chai Wah Wu_, Apr 12 2023
