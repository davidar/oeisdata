## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    s, lastd = "0", [(k for k in count(i, 10)) for i in range(10)]
    for n in count(0):
        an = next(lastd[int(s[0])])
        s = s[1:] + str(an)
        yield an
print(list(islice(agen(), 65))) # _Michael S. Branicky_, Apr 25 2023
```
