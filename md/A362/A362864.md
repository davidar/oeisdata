from itertools import count, islice
from sympy import divisor_count
def A362864_gen(): # generator of terms
    c = 0
    for k in count(1):
        if not (c:=c+k-divisor_count(k))%k:
            yield k
A362864_list = list(islice(A362864_gen(),15)) # _Chai Wah Wu_, May 20 2023
