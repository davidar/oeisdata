## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    m = 1
    for n in count(1):
        while not all(m%(2*k) < k for k in range(1, n+1)): m += 1
        yield m
print(list(islice(agen(), 37))) # _Michael S. Branicky_, Apr 24 2023
```
