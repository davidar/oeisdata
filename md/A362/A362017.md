from sympy import nextprime; R = [2]; L = [2]
for n in range(2, 57):
    p0 = L[-1]; p1 = nextprime(p0); M = [p1]; g0 = p1-p0; p = nextprime(p1); g1 = p-p1
    while g1 < g0 and p-p1 <= g1: M.append(p); p1 = p; p = nextprime(p)
    L = M; R.append(L[0])
