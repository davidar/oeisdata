## Python
```Python
from itertools import count
from num2words import num2words
def A362120(n): return next(filter(lambda k:len(s:=num2words(k).replace('-','').replace(',','').replace(' and ','').replace(' ',''))>=n and s[n-1]=='e',count(1))) # _Chai Wah Wu_, Apr 21 2023
```
