```Python
from sympy import divisors
from sympy.solvers.diophantine.diophantine import cornacchia
def A362961(n):
    c = 0
    for d in divisors(n):
        if (k:=d**2)>n:
            break
        q, r = divmod(n,k)
        if not r:
            c += sum(d*(a[0]+(a[1] if a[0]!=a[1] else 0)) for a in cornacchia(1,1,q) or [])
    return c # _Chai Wah Wu_, May 15 2023
```
