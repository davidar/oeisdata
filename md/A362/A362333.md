## Python
```Python
from sympy import factorint
def A362333(n):
    f = factorint(n)
    gpf = max(f,default=None)
    a = 0
    for p in f:
        m = gpf
        v = 0
        while m >= p:
            m //= p
            v += m
        a = max(a,-(-f[p]//v))
    return a
```
