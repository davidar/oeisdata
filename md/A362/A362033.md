from itertools import count, islice
from sympy import primeomega
def A362033_gen(): # generator of terms
    a, b, c = {}, {}, 1
    for n in count(1):
        if c == 1: yield n
        d = b[c] = b.get(c,primeomega(c))
        c = a[d] = a.get(d,0)+1
A362033_list = list(islice(A362033_gen(),10)) # _Chai Wah Wu_, Apr 06 2023
