## Python
```Python
from sympy.ntheory.primetest import is_square
from sympy import factorint
def A362068(n):
    if is_square(n):
        return 1
    if all(map(lambda x:x[0]&3<3 or x[1]&1^1, factorint(k:=n>>(m:=(~n&n-1).bit_length())).items())):
        return 2
    if m&1 or 3*k&7<7:
        return 3
    return 4 # _Chai Wah Wu_, Apr 27 2023
```
