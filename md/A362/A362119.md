## Python
```Python
from sympy.ntheory import digits
from itertools import count, islice
def agen(s="", base=6): yield from (int(s:=s+"".join(map(str, digits(n, base)[1:]))) for n in count(1))
print(list(islice(agen(), 20)))
```
