## Python
```Python
from num2words import num2words
def rankcat(n):
    return int("".join(str(ord(c)-96) for c in num2words(n).replace(" and", "") if c.isalpha()))
def ok(n):
    return n and rankcat(n)%n == 0
print([k for k in range(1000) if ok(k)]) # _Michael S. Branicky_, Apr 08 2023
```
