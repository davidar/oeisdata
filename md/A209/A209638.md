from sympy import prime
def a(n):
    n = 2*n
    m = 1
    if n<2: return 1
    while n>1:
        if n%2==0:
            n//=2
            m*=2
        else:
            n=(n - 1)//2
            m=prime(m)
    return m
print(sorted([a(n) for n in range(101)])) # _Indranil Ghosh_, May 26 2017
