```Maple
A177025 := proc(p)
    local ii,a,n,s,m ;
    ii := 2*p ;
    a := 0 ;
    for n in numtheory[divisors](ii) do
        if n > 2 then
            s := ii/n ;
            if (s-2) mod (n-1) = 0 then
                a := a+1 ;
            end if;
        end if;
    end do:
    return a;
end proc: # _R. J. Mathar_, Jan 10 2013
```

## Python
```Python
from sympy import divisors
def a(n):
    i=2*n
    x=0
    for d in divisors(i):
        if d>2:
            s=i/d
            if (s - 2)%(d - 1)==0: x+=1
    return x # _Indranil Ghosh_, Apr 28 2017, translated from Maple code by _R. J. Mathar_
```
