## Python

def A177316(n):
    if n == 0: return 1
    m, g = 1, 0
    for k in range(n+1):
        g += m*n**2//(n+k)**2
        m *= ((n+k+1)*(n-k))**2
        m //= (k+1)**4
    return g # _Chai Wah Wu_, Oct 03 2022
