# Alternative:
nc:= proc(a,b,c)
  local t;
  if c=0 and (a=0 or b=0) then return 0 fi;
  t:= (a mod 2) + (b mod 2) + c;
  if t < 2 then  procname(floor(a/2),floor(b/2),0)
  else  1 + procname(floor(a/2),floor(b/2),1)
  fi
end proc:
seq(nc(n,n^2-n,0),n=0..100); # _Robert Israel_, Oct 23 2019
