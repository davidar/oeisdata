from sympy import factorint
def a(n):
    if n < 2: return -1
    klst, f = [n], sorted(factorint(n, multiple=True))
    while len(f) > 1:
        klst.append(int("".join(map(str, f))))
        f = sorted(factorint(klst[-1], multiple=True))
    return len(klst) - 1
print([a(n) for n in range(1, 49)]) # _Michael S. Branicky_, Aug 02 2021
