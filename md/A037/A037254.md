## Python

def T(n, k):
    if k==1:
        if n==1: return 1
        else: return T(n - 1, (n + 1)//2)
    return T(n, 1) + T(n - 1, k - 1)
for n in range(1, 12): print([T(n, k) for k in range(1, n + 1)]) # _Indranil Ghosh_, Jun 03 2017
