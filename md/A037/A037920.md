## Python
```Python
from sympy import factorint
def iterate(n):
    flst, f = [n], sorted(factorint(n, multiple=True))
    while len(f) > 1:
        flst += f
        f = sorted(factorint(int("".join(map(str, f))), multiple=True))
    return flst
print(iterate(8)) # _Michael S. Branicky_, Aug 02 2021
```
