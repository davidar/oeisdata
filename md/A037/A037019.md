## Python
```Python
from math import prod
from sympy import factorint, prime
def a(n):
    pf = factorint(n, multiple=True)
    return prod(prime(i)**(pi-1) for i, pi in enumerate(pf[::-1], 1))
print([a(n) for n in range(1, 42)]) # _Michael S. Branicky_, Jul 24 2022
```
