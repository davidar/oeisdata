## Python
```Python
from sympy import divisors
def a(n):
  divisors_gt1 = divisors(n)[1:]
  if len(divisors_gt1) == 0: return 0
  else: return int("".join(str(d) for d in divisors_gt1))
print([a(n) for n in range(1, 40)]) # _Michael S. Branicky_, Dec 31 2020
```
