def A154810(n):
    def f(x): return n+(x>>(l:=x.bit_length())-(k:=l+1>>1))-(int(bin(x)[k+1:1:-1],2)>(x&(1<<k)-1))+(1<<k-1+(l&1^1))-1
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    return int(bin(m)[2:]) # _Chai Wah Wu_, Jul 24 2024
