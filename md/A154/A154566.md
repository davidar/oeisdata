Function Flag(ByVal s As String, ByVal num As Long) As Long
Dim b&(9), t&, i&
Flag = 1
If Len(s) <> 10 * num Then
    Flag = 0
    Exit Function
End If
For i = 1 To Len(s)
    t = Val(Mid(s, i, 1))
    b(t) = b(t) + 1
    If b(t) > num Then
        Flag = 0
        Exit Function
    End If
Next
End Function

Function Mypower(ByVal num As Currency, ByVal power As Long) As String
Dim b(), temp, i&, j&
ReDim b(1 To 2 * power)
ReDim s(1 To 2 * power)
b(2 * power - 1) = Val(Left(num, 5))
b(2 * power) = Val(Right(num, 5))
For i = 2 To power
    temp = 0
    For j = 2 * power To 1 Step -1
        temp = b(j) * num + temp
        b(j) = Format(Val(Right(temp, 5)), "00000")
        temp = Int(temp / 10 ^ 5)
    Next
Next
Mypower = Join(b, "")
End Function

## Python

def flag(p, n):
    for i in range(10):
        if not p.count(str(i)) == n:
            return False
    return True
def a(n):
    for i in range(3*int(10**(10-1/n)/3), 10**10, 3):
        if flag(str(i**n), n):
            return i
for i in range(1, 41):
    print(a(i), end=", ") # _Zhining Yang_, Oct 05 2022
