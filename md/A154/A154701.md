Res:= NULL: count:= 0:

L:= [1]:
for n from 2 while count < 100 do
  L[1]:=L[1]+1;
  for k from 1 while L[k]=10 do L[k]:= 0;
    if k = nops(L) then L:= [0$nops(L),1]; break
    else L[k+1]:= L[k+1]+1 fi
  od:
  s:= convert(L,`+`);
  if n mod s = 0 then
     state:= min(state+1,3);

  i=n;

    count=count+i%10;
    i=i/10;
