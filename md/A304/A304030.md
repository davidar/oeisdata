## Python

def A304030(n):
    prevc = c = n
    h = 0
    while c > 1:
        if c % 2:
            c = 3*c+1
            if prevc < n and c > n: h += 1
        else:
            c //= 2
            if prevc > n and c < n: h += 1
        prevc = c
    return h
print([A304030(n) for n in range(1, 100)]) # _Paolo Xausa_, Feb 22 2022
