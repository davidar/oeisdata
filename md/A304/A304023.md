## Python

def f(n): return 0 if n < 1 else f(n//3*2)*10 + n%3
def a(n):
  k = 0
  while len(str(f(k))) != n: k += 1
  return f(k)
print([a(n) for n in range(1, 22)]) # _Michael S. Branicky_, Feb 12 2021 after _Michel Marcus_
