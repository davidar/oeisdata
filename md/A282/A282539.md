## Python
```Python
from sympy import isprime
a = [1]
found = True
while found:
  found = False
  prev = a[len(a)-1]
  for k in range(2,10001):
    if isprime(prev*k-1) and isprime(prev*k+1) and (k not in a):
      print str(k)+',',
      a.append(k)
      found = True
      break
```
