```Maple
with(numtheory):
Ql:=[]; Qu:=[]; Q:=[]; Nl:=[]; Nu:=[]; N:=[]; Th:=[];
for i1 from 1 to 300 do
p:=ithprime(i1);
if (p mod 8) = 3 then
ql:=0; qu:=0; q:=0; nl:=0; nu:=0; n:=0;
for j from 1 to p-1 do
if legendre(j,p)=1 then
  q:=q+j;
  if j<p/2 then ql:=ql+j; else qu:=qu+j; fi;
else
  n:=n+j;
  if j<p/2 then nl:=nl+j; else nu:=nu+j; fi;
fi;
                    od;
Ql:=[op(Ql),ql];
Qu:=[op(Qu),qu];
Q:=[op(Q),q];
Nl:=[op(Nl),nl];
Nu:=[op(Nu),nu];
N:=[op(N),n];
Th:=[op(Th),q+ql];
fi;
od:
Ql; Qu; Q; Nl; Nu; N; Th; # A282721 - A282727
# Alternative
f:= proc(p) local q,r,t,j;
  r:= (p-1)/2; t:= 0;
  for j from 1 to r do
    q:= j^2 mod p;
    if q <= r then t:= t+q fi;
od:
t
end proc:
```

## Python
```Python
from sympy import isprime
def a(p):
    r=(p - 1)//2
    t=0
    for j in range(1, r + 1):
        q=(j**2)%p
        if q<=r:t+=q
    return t
print([a(p) for p in range(3, 2001, 8) if isprime(p)]) # _Indranil Ghosh_, Mar 27 2017, translated from Maple code
```
