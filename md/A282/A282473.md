## Python
```Python
from math import ceil
def a(n): #Outputs all numbers less than n which are in the sequence
    def s(n):
        r = 0
        while n:
            r, n = r + n% 10, n//10
        return r
    mult9=[]
    if n%9==0:
        for x in range(1, ceil(n/9)+1):
            mult9.append(9*x)
    else:
        for x in range(1, ceil(n/9)):
            mult9.append(9*x)
    for y in range(1, ceil(n/10)+1):
        mult9.remove(10*y-s(10*y))
    return mult9
```
