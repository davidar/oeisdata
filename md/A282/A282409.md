## Python
```Python
from sympy import isprime
def a(n):
    l=[n]
    while True:
        if n%2==0: n//=2
        else: n = 3*n + 1
        l.append(n)
        if n<2: break
    o=list(filter(lambda i: i%2==1, l))
    e=list(filter(lambda i: i%2==0, l))
    return [o, e]
print([n for n in range(2, 1001) if isprime(len(a(n)[0])) and isprime(len(a(n)[1]))]) # _Indranil Ghosh_, Apr 14 2017
```
