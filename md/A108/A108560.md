```Python
from sympy import nextprime
def aupton(terms):
  p, p_idx, alst = 1, 0, []
  while len(alst) < terms:
    p, p_idx = nextprime(p), p_idx+1
    if (p_idx + p)%11 == 0: alst.append((p_idx + p)//11)
  return alst
print(aupton(55)) # _Michael S. Branicky_, Apr 30 2021
```
