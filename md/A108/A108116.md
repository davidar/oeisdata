def SL(d, s):
  for i1 in range(int(d[0]=="0"), len(s)-int(d[0])-1):
    i2 = i1 + int(d[0]) + 1
    if not (s[i1] or s[i2]):
      s[i1] = s[i2] = d[0]
      r = d[1:]
      if r: yield from SL(r, s)
      else: yield int("".join(s))
      s[i1] = s[i2] = 0
