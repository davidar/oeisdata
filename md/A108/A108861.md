## Python
```Python
from itertools import islice
def A108861(): # generator of terms
    k, k2, kf = 1, 2, 1
    while True:
        c = sum(int(d) for d in str(k2*kf))
        if not c % k: yield k
        k += 1
        k2 *= 2
        kf *= k
A108861_list = list(islice(A108861(),10)) # _Chai Wah Wu_, Oct 26 2021
```
