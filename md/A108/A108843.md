## Python

from sympy import isprime
from itertools import count, islice
def agen(): yield from (t for i in count(1) for f in "17" if isprime(t:=int(f + "6"*i + f)))
print(list(islice(agen(), 10))) # _Michael S. Branicky_, Jan 27 2023
