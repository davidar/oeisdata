## Python
```Python
def a(n):
    if n<2: return 0
    b=bin(n)[2:]
    s=0
    while "0" in b:
        N=int(b[-1] + b[:-1], 2)
        s+=1
        b=bin(N)[2:]
    return s
print([a(n) for n in range(105)]) # _Indranil Ghosh_, May 25 2017
```
