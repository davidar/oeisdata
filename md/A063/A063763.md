## Python
```Python
from math import isqrt
from sympy import primepi
def A063763(n):
    def f(x): return int(n+x-sum(primepi(x//i)-primepi(i) for i in range(2,isqrt(x)+1)))
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m # _Chai Wah Wu_, Oct 05 2024
```
