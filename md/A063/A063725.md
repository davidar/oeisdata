from math import prod
from sympy import factorint
def A063725(n):
    f = factorint(n)
    return prod(1 if p==2 else (e+1 if p&3==1 else (e+1)&1) for p, e in f.items())-(not any(e&1 for e in f.values())) if n else 0 # _Chai Wah Wu_, May 17 2023
