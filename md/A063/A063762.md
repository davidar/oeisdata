## Python
```Python
from math import isqrt
from sympy import primepi
def A063762(n):
    def f(x): return int(n+(x if x<=3 else x-primepi(x//(y:=isqrt(x)))-sum(primepi(x//i)-primepi(i) for i in range(2,y))))
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m # _Chai Wah Wu_, Oct 05 2024
```
