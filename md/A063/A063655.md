## Maple
```Maple
A063655 := proc(n)
    local i,j;
    for i from floor(sqrt(n)) to 1 by -1 do
        j := floor(n/i) ;
        if i*j = n then
            return i+j;
        end if;
    end do:
end proc:
seq(A063655(n), n=1..80); # Winston C. Yang, Feb 03 2002
```

## Python

```Python
from sympy import divisors
def A063655(n):
    d = divisors(n)
    l = len(d)
    return d[(l-1)//2] + d[l//2] # _Chai Wah Wu_, Jun 14 2019
```
