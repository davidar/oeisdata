## Python
```Python
from math import isqrt
from sympy import primepi
def A063538(n):
    def f(x): return int(n+x-primepi(x//(y:=isqrt(x)))-sum(primepi(x//i)-primepi(i) for i in range(1,y)))
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m # _Chai Wah Wu_, Oct 05 2024
```
