## Ruby

def a(n)
  seq = [1]
  (2..Float::INFINITY).each do |i|
    return seq.last[0...n].last if seq.length > n
    indices = seq.each_index.select { |j| i % seq[j] == 0 }
    seq << indices.map(&:next).reduce(:+)
  end
end # _Peter Kagey_, Feb 25 2016
