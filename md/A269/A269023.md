## Python
```Python
from itertools import count
def A269023(n):
    def bisection(f,kmin=0,kmax=1):
        while f(kmax) > kmax: kmax <<= 1
        while kmax-kmin > 1:
            kmid = kmax+kmin>>1
            if f(kmid) <= kmid:
                kmax = kmid
            else:
                kmin = kmid
        return kmax
    def f(x):
        if x==1: return n+1
        z = x**x
        for y in count(x,-1):
            if y**(y+1) <= z:
                return n+y
            z //= x
    return bisection(f,n,n) # _Chai Wah Wu_, Sep 10 2024
```
