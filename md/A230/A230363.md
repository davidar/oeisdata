import math
f=1
for n in range(1, 1000000):
    f *= n
    t = b = 0
    while t<=f:
        x = f-t
        a = int(math.sqrt(x))
        if a*a==x:
            print str(f)+',',
            break
        b += 1
        t = b*(b+1)/2
