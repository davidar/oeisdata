# take ifsSorted from A080670
A230625 := proc(n)
    local Ldgs, p,eb,pb,b ;
    b := 2;
    if n = 1 then
        return 1;
    end if;
    Ldgs := [] ;
    for p in ifsSorted(n) do
        pb := convert(op(1,p),base,b) ;
        Ldgs := [op(pb),op(Ldgs)] ;
        if op(2, p) > 1 then
            eb := convert(op(2,p),base,b) ;
            Ldgs := [op(eb),op(Ldgs)] ;
        end if;
    end do:
    add( op(e,Ldgs)*b^(e-1),e=1..nops(Ldgs)) ;
end proc:
