## Python
```Python
from sympy import primepi, integer_nthroot
def A230078(n):
    if n == 1: return 1
    def f(x): return int(n+1+sum(primepi(integer_nthroot(x>>1,k)[0]) for k in range(1,(x>>1).bit_length())))
    kmin, kmax = 0,1
    while f(kmax) > kmax:
        kmax <<= 1
    while kmax-kmin > 1:
        kmid = kmax+kmin>>1
        if f(kmid) <= kmid:
            kmax = kmid
        else:
            kmin = kmid
    return kmax # _Chai Wah Wu_, Aug 29 2024
```
