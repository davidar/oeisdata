from collections import Counter

for _ in range(30001):
    i = s
    while True:
        if i not in b:
            li, o = Counter(bin(i)[2:]), 0
            for d in (l+li).values():
                if d % 2:
                    if o > 0:
                        break
                    o += 1
            else:
                A230892_list.append(i)
                l = li
                b.add(i)
                while s in b:
                    b.remove(s)
                    s += 1
