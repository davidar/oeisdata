## MATLAB
```MATLAB
% Input:
%  n: an integer
% Output:
%  m: max power of 3 such that 3^m divides n
%  M: 1-by-K matrix where M(i) is the max power of 3 such that 3^M(i) divides n
function [m,M] = Omega3(n)
  M = NaN*zeros(1,n);
  M(1)=0; M(2)=0; M(3)=0;
    for k=4:n
      if M(k-3)~=0
        M(k)=M(k-k/3)+1;
      else
        M(k)=0;
      end
    end
    m=M(end);
end
```
