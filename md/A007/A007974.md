def a(n):
    perm = tuple(range(1, n+1)); reach = {perm}; frontier = {perm}
    for k in range(n-3):
        r1 = set()
        while len(frontier) > 0:
            q = frontier.pop()
            for i in range(n):
                for j in range(i+1, n+1):
                    qp = list(q)
                    qp[i:j] = qp[i:j][::-1]
                    qp = tuple(qp)
                    if qp not in reach:
                        reach.add(qp)
                        r1.add(qp)
        frontier = r1
    return len(frontier)
print([a(n) for n in range(3, 10)]) # _Michael S. Branicky_, Jan 26 2023
