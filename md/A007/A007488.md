## Maple
```Maple
revdigs:= proc(n)
local L,nL,j;
L:= convert(n,base,10);
nL:= nops(L);
add(L[i]*10^(nL-i),i=1..nL);
end:
map(proc(i) local r; r:= revdigs(i^2); if isprime(r) then r else NULL fi end proc, {$1..9999}); # _Robert Israel_, Aug 14 2014
```
