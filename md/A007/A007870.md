b:= proc(n, i) option remember; `if`(n=0 or i=1, [1$2], ((f, g)->
      [f[1]+g[1], f[2]*g[2]*i^g[1]])(b(n, i-1), b(n-i, min(n-i, i))))
    end:
a:= n-> b(n$2)[2]:
seq(a(n), n=0..12);  # _Alois P. Heinz_, Jul 30 2013

a = lambda n: prod(map(prod, ordered_partitions(n))) if n > 0 else 1
