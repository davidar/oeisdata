A007089 := proc(n) option remember;
if n <= 0 then 0
else
  if (n mod 3) = 0 then 10*procname(n/3) else procname(n-1) + 1 fi
fi end:

def A007089(n):
  n,s = divmod(n,3); t = 1
  while n: n,r = divmod(n,3); t *= 10; s += r*t
  return s # _M. F. Hasler_, Feb 15 2023
