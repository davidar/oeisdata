isA007629 := proc(n)
    local L,t,a ;
    if n < 10 then
        return false;
    end if;
    L := ListTools[Reverse](convert(n,base,10)) ;
    t :=  nops(L) ;
    while true do
        a := add(op(-i,L),i=1..t) ;
        L := [op(L),a] ;
        if a > n then
            return false;
        elif a = n then
            return true;
        end if;
    end do:

      repfigit ns = s == n || s < n && (repfigit $ s : init ns) where
         s = sum ns
