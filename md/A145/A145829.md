## Python

from itertools import count, islice
from sympy import integer_nthroot
def A145829gen(): # generator of terms
    m = 0
    for n in count(1):
        m ^= n**2
        a, b = integer_nthroot(m,2)
        if b: yield a
A145829_list = list(islice(A145829gen(),20)) # _Chai Wah Wu_, Dec 16 2021
