## Python
```Python
import numpy as np
def isA145204(n):
    if n == 0: return True
    c = 0
    d = int(np.base_repr(n, base = 3))
    while d % 10 == 0:
        c += 1
        d //= 10
    return c % 2 == 1
print([n for n in range(231) if isA145204(n)]) # _Peter Luschny_, Aug 05 2020
```
