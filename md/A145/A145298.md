## Python

from itertools import islice
from sympy import nextprime, sqrt_mod_iter
def A145298_gen(): # generator of terms
    p = 1
    while (p:=nextprime(p)):
        if p&3==1:
            yield min(sqrt_mod_iter(-1,p**5))
A145298_list = list(islice(A145298_gen(),20)) # _Chai Wah Wu_, May 04 2024
