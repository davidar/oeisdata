# Gives 15 correct digits
from mpmath import stieltjes,fac,quad
def limgen(n):
    terms = []
    for y in range(3, n):
        for x in range(y, n):
            terms.append((((-1)**y)*stieltjes(x)*(x-(y-1))**(y-2))/(fac(x-(y-2))*fac(y-2)))
    return terms
f = lambda x: x**(1/x)
int01 = quad(f, [0,1])
limit = sum(limgen(60)) + 1.5 - stieltjes(0) - int01
print(limit)
