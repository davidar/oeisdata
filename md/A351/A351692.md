## Python
```Python
from sympy import isprime, sieve
def a(n):
    pn = sieve[n]
    for k in range(1, n):
        if isprime(pn + 2*sieve[n-k]) and isprime(pn + 2*sieve[n+k]):
            return k
    return 0
print([a(n) for n in range(1, 86)]) # _Michael S. Branicky_, May 10 2022
```
