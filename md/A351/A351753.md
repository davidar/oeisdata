## Python
```Python
from itertools import count
def A351753(n):
    s1, s2 = tuple(), tuple()
    for i, s in enumerate(int(d) for n in count(1) for d in bin(n)[2:]):
        if i < n:
            s1 += (s,)
            s2 += (s,)
        else:
            s2 = s2[1:]+(s,)
            if s1 == s2:
                return i-n+2 # _Chai Wah Wu_, Feb 18 2022
```
