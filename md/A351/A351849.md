## Python
```Python
def A351849_row(n):
    s = "1" * n
    row = [int(s)]
    while s != "1":
        if s[0] == "1": s += "23"
        elif s[0] == "2": s += "1"
        else: s += "111"
        s = s[2:]
        row.append(int(s))
    return row
nrows = 4
print([A351849_row(n) for n in range(1, nrows + 1)])
```
