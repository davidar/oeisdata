## Python
```Python
from itertools import count, islice
def isdb(n): b = bin(n)[2:]; return b.count("0") == b.count("1")
def agen(): yield from (b for b in count(1) if isdb(b) and isdb(b**b))
print(list(islice(agen(), 2))) # _Michael S. Branicky_, Jun 12 2022
```
