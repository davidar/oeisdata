## Python
```Python
from sympy import nextprime
from itertools import islice
def agen(): # generator of terms
    p, q, r, s = 2, 3, 5, 7
    while True:
        if q-p == 10 and r-q == 20 and s-r == 30:
            yield p
        p, q, r, s = q, r, s, nextprime(s)
print(list(islice(agen(), 30))) # _Michael S. Branicky_, Feb 19 2022
```
