from itertools import count, islice
def A351964_gen(): # generator of terms
    yield 0
    l1, s, b = 1, 2, set()
    while True:
        for i in count(s):
            if not (i in b or i & l1):
                yield (~i&i-1).bit_length()
                l1 = i
                b.add(i)
                while s in b:
                    b.remove(s)
                    s += 1
                break
A351964_list = list(islice(A351964_gen(),30)) # _Chai Wah Wu_, Jul 16 2022
