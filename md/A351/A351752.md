## Python
```Python
from math import floor, ceil
from sympy import divisors
def a(n):
    arr = divisors(n)
    length = ceil(len(arr)/2)
    total = 0
    for i in range(length):
        total += arr[-(i+1)] - arr[i]
    avg = total / length
    return floor(avg)
def main():
    for i in range(1, 250):
        print("{}, ".format(a(i)), end="")
```
