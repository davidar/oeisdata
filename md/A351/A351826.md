## Python
```Python
from itertools import count
from sympy import isprime
def A351826(n):
    for k in count(1,2):
        c = 0
        for j in count(1):
            if k-2**j < 2:
                break
            if isprime(k-2**j) and isprime(k+2**j):
                c += 1
            if c > n:
                break
        if c == n: return k # _Chai Wah Wu_, Mar 24 2022
```
