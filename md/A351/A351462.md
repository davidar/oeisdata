## Python
```Python
from math import prod
from sympy import factorint
from functools import cache
@cache
def a(n):
    if n == 1: return 1
    return prod(a(p**k-1)+2 for p, k in factorint(n).items())
print([a(n) for n in range(1, 59)]) # _Michael S. Branicky_, Feb 13 2022
```
