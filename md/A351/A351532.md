from sympy.abc import i, j
from sympy.solvers.diophantine.diophantine import diop_quadratic
def A351532(n):
    return sum(1 for d in diop_quadratic(n**2+3*i*j-2*n*(i+j)) if 0 < d[0] < n and 0 < d[1] < n) # _Chai Wah Wu_, Feb 15 2022
