## Python
```Python
from itertools import islice
def c(s, t): return sum(t.count(si) for si in s)
def agen(): # generator of terms
    an, target = 1, "1"
    while True:
        yield an
        k = an + 1
        while c(str(k), target) != 3: k += 1
        an, target = k, str(k)
print(list(islice(agen(), 52))) # _Michael S. Branicky_, Jan 30 2022
```
