## Python
```Python
from itertools import islice
from sympy import isprime, primepi
def A087865_gen(): # generator of terms
    k = 1
    while True:
        k *= 3
        if isprime(p:=primepi(k)):
            yield p
A087865_list = list(islice(A087865_gen(),5)) # _Chai Wah Wu_, Jun 06 2024
```
