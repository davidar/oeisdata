## Perl

sub a {
  my $nv= ((shift() << 1) | 1);
  my $bp= 1;
  while (($nv & 1) xor ($nv & 2)) {
    $nv>>= 1;
    $bp++;
  }
  return $bp;
} # _Ruud H.G. van Tol_, Nov 16 2021
