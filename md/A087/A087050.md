## Python
```Python
from math import isqrt, prod
from sympy import mobius, factorint
def A087050(n):
    def f(x): return n+sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1))
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    return prod(p**(e>>1) for p, e in factorint(m).items() if e>1) # _Chai Wah Wu_, Jul 22 2024
```
