## Python
```Python
def ispal(n):
    w=str(n)
    return w==w[::-1]
def A087991(n):
    s = 0
    for i in range(1, n+1):
        if n%i==0 and not ispal(i):
            s+=1
    return s
print([A087991(n) for n in range(1,60)]) # _Indranil Ghosh_, Feb 10 2017
```
