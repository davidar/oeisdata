## Python

from fractions import Fraction
def A087704(n):
    x, c = Fraction(n,1), 0
    while x.denominator > 1 or x<=n:
        x = Fraction(5*x.__floor__(),3)
        c += 1
    return c # _Chai Wah Wu_, Sep 01 2023
