## Python
```Python
def lunar_add(n,m):
    sn, sm = str(n), str(m)
    l = max(len(sn),len(sm))
    return int(''.join(max(i,j) for i,j in zip(sn.rjust(l,'0'),sm.rjust(l,'0'))))
def lunar_mul(n,m):
    sn, sm, y = str(n), str(m), 0
    for i in range(len(sm)):
        c = sm[-i-1]
        y = lunar_add(y,int(''.join(min(j,c) for j in sn))*10**i)
    return y # _Chai Wah Wu_, Sep 06 2015
```
