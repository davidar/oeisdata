## Python
```Python
# for N < 100
import math
for N in range (1, 100):
   ANZ = 0
   for M in range(1, 195312500):
      Z = int(str(int(N)) + str(int(M)))
      if ((Z % M == 0) and (M % 10 > 0)):
         ANZ = ANZ + 1
   print(N, ANZ)
```
