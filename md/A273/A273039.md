## Python
```Python
def ROR(n):                # returns A038572(n)
    BL = len(bin(n))-2
    return (n>>1) + ((n&1) << (BL-1))
def ROL(n):                # returns A006257(n)
    BL = len(bin(n))-2
    return (n*2) - (1<<BL) + 1
for n in range(1000):
    X = n
    Xs = []
    while not (X in Xs):
        Xs.append(X)
        if X==0:
            print str(n)+',',
            break
        X = X ^ ROR(X) ^ ROL(X)
```
