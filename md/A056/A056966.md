def A056966(n):
    s = bin(n)[2:]
    s = '0'*(len(s)&1)+s
    return int('0'+''.join(s[i+1]*int(s[i])for i in range(0,len(s),2)),2) # _Chai Wah Wu_, Feb 12 2023
