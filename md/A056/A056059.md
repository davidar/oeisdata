## Python
```Python
from sympy import binomial, gcd
from sympy.ntheory.factor_ import core
def a001405(n): return binomial(n, n//2)
def a055229(n):
    c = core(n)
    return gcd(c, n//c)
def a(n): return a055229(a001405(n))
print([a(n) for n in range(1, 151)]) # _Indranil Ghosh_, Jul 20 2017
```
