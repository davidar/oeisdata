## Python
```Python
from sympy import divisors, mobius
def a000048(n): return 1 if n<1 else sum([mobius(d)*2**(n/d) for d in divisors(n) if d%2 == 1])/(2*n)
def a(n): return a000048(n) - 0**(n - 1) # _Indranil Ghosh_, Apr 28 2017
```
