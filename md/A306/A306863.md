## Python
```Python
from itertools import count
from sympy import primepi, isprime
def A306863(n):
    if n == 1: return 2
    m, k = n, primepi(n) + n + (n>>1)
    while m != k:
        m, k = k, primepi(k) + n + (k>>1)
    return next(c for c in count(0) if not isprime(m+(c<<1)+2)) # _Chai Wah Wu_, Aug 02 2024
```
