## Maple
```Maple
# Maple from _N. J. A. Sloane_, Mar 04 2020, starting at n=1:  First define z(n) = A115004
z := proc(n)
    local a, b, r ;
    r := 0 ;
    for a from 1 to n do
    for b from 1 to n do
        if igcd(a, b) = 1 then
            r := r+(n+1-a)*(n+1-b);
        end if;
    end do:
    end do:
    r ;
end proc:
a := n-> z(n)+n^2+2*n;
[seq(a(n), n=1..50)];
```
