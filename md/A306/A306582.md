## Python
```Python
from sympy import prime
def A306582(n):
    plist, rlist, x = [prime(i) for i in range(1,n+1)], [0]*n, 0
    while True:
        for i in range(n-1):
            if rlist[i] >= rlist[i+1]:
                break
        else:
            return x
        for i in range(n):
            rlist[i] = (rlist[i] + 1) % plist[i]
        x += 1 # _Chai Wah Wu_, Jun 15 2019
```
