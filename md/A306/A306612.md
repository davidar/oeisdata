## Python
```Python
from sympy import prime
def A306612(n):
    plist, x = [prime(i) for i in range(1,n+1)], 3
    rlist = [-x % p for p in plist]
    while True:
        for i in range(n-1):
            if rlist[i] >= rlist[i+1]:
                break
        else:
            return x
        for i in range(n):
            rlist[i] = (rlist[i] - 1) % plist[i]
        x += 1 # _Chai Wah Wu_, Jun 15 2019
```
