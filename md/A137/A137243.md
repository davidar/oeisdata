## Python
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def A137243(n):
    if n == 0:
        return 0
    c, j = 0, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*(A137243(k1)//4-1)
        j, k1 = j2, n//j2
    return 4*(n*(n-1)-c+j) # _Chai Wah Wu_, Mar 29 2021
```
