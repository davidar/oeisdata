def process(s,n,catalog,cache):
....l=len(s)
....if l==n:
........catalog.add(s)
........return
....if s in cache:
........return
....cache.add(s)
....for x in range(l):
........for y in range(x+1,min(x+n-l,l)+1):
............process(s[:y]+s[x:],n,catalog,cache)
def A137744(n):
....catalog=set()
....cache=set()
