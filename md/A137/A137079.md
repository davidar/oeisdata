## Maple
```Maple
Good := {2, 3, 5, 6}: R:= 5, 6:
G[1]:= {5, 6}:
for d from 2 to 26 do
  G[d]:= select(t -> member(floor((t^2 mod 10^d)/10^(d-1)), Good), map(t -> seq(10^(d-1)*i+t, i=Good), G[d-1]));
  for t in G[d] do
     if convert(convert(t^2, base, 10), set) subset Good then  R:= R, t fi
od od:
sort([R]); # _Robert Israel_, Nov 24 2023
```
