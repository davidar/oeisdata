## Python
```Python
def A137401(n):
    ndict = {}
    for i in range(1,n):
        m = pow(i,3,n)
        if m in ndict:
            ndict[m] += 1
        else:
            ndict[m] = 1
    count = 0
    for i in ndict:
        ni = ndict[i]
        for j in ndict:
            k = (i+j) % n
            if k in ndict:
                count += ni*ndict[j]*ndict[k]
    return count # _Chai Wah Wu_, Jun 06 2017
```
