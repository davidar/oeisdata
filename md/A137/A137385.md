```Python
for cle in range(1,100):
    n=0
    for r1 in range(1,14):
        for r2 in range(r1,14):
            ur1=cle-(10**(13-r1))%cle
            ur2=cle-(10**(13-r2))%cle
            for c1 in range(10):
                for c2 in range(10):
                    if c1==c2:
                        dif=1
                    else:
                        dif=(c1*ur1-c2*ur2)%cle
                    if dif==0:
                        n=n+1
    print(cle,':',n)
```
