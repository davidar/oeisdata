## Python
```Python
from sympy import nextprime
def auptodigs(maxdigits):
  alst = []
  for n in range(1, maxdigits+1):
    p1 = nextprime(10**(n-1))
    p2 = nextprime(p1)
    p3 = nextprime(p2)
    alst.extend([p1, p2, p3])
  return alst
print(auptodigs(11)) # _Michael S. Branicky_, May 07 2021
```
