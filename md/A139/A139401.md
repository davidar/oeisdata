## Python

import math
def A139401(n):
    if int(math.sqrt(n)) == math.sqrt(n):
        return 0
    for pp in range(2, n + 2):  # only really need to check prime powers
        residues = frozenset(pow(k, 2, pp) for k in range(pp))
        if n % pp not in residues:
            return pp  # _Dan Uznanski_, Jun 22 2021
