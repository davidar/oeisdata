## Python

from itertools import islice
def agen(): # generator of terms
    an, y = 2, 1
    while y < 10:
        yield an
        an, y = an + 10*(an%10), 1
        while y < 10:
            if str(an+y)[0] == str(y):
                an += y
                break
            y += 1
print(list(islice(agen(), 47))) # _Michael S. Branicky_, Apr 08 2022
