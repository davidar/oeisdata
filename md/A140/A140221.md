from math import gcd, isqrt
def A140221_gen(startvalue=1): # generator of terms >= startvalue
    return filter(lambda n: gcd(n,-(s:=isqrt(n))**2+(sum(n//k for k in range(1,s+1))<<1))==1,count(max(startvalue,1)))
