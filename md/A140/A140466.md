```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def A140466(n): # based on second formula in A018805
    if n == 0:
        return 0
    c, j = 0, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*(A140466(k1)//2-1)
        j, k1 = j2, n//j2
    return 2*(n*(n-1)-c+j) # _Chai Wah Wu_, Mar 25 2021
```
