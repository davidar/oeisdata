## Python

from itertools import count
from sympy import mobius
def A124301(n):
    c, m = 1, 1
    for i in count(2):
        k = mobius(i)
        if m!=k:
            if c>n:
                return i-c
            c = 0
        c += 1
        m = k # _Chai Wah Wu_, Oct 08 2024
