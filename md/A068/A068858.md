from itertools import islice
from sympy import sqrt_mod_iter
def A068858_gen(): # generator of terms
    a = 3
    while True:
        yield a
        b = a+1
        for d in sqrt_mod_iter(1,a):
            if d**2-1 == a:
                d += a
            if d&1:
                d += a
            if d < b:
                b = d
        a = b**2-1
A068858_list = list(islice(A068858_gen(),11)) # _Chai Wah Wu_, May 05 2024
