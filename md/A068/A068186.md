## Python
```Python
from sympy import factorint
def A068186(n):
    if n == 1:
        return 1
    pf = factorint(n)
    ps = sorted(pf.keys(),reverse=True)
    if ps[0] > 7:
        return 0
    s = ''
    for p in ps:
        s += str(p)*(n*pf[p])
    return int(s) # _Chai Wah Wu_, Aug 12 2017
```
