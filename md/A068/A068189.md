## Python
```Python
def convert(n):
    if n == 1:
        return 1
    result = 0
    cur = 1
    while n > 1:
        found = False
        for i in range(9, 1, -1):
            if n % i == 0:
                result += cur * i
                cur *= 10
                n //= i
                found = True
                break
        if not found:
            return 0
    return result
N = 256
for n in range(1, N):
    print(n, convert(n))
# _Dmitry Kamenetsky_, Oct 20 2008
```
