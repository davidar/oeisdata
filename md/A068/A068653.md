## Python
```Python
from itertools import product
from sympy import isprime
A068653_list = []
for l in range(1,9):
    for m in product(('1379' if l > 1 else '123579'),repeat=l):
        for d in '0123456789':
            s = ''.join(m)+d
            n = int(s)
            if not isprime(n):
                for k in range(len(s)-1):
                    s = s[1:]+s[0]
                    if not isprime(int(s)):
                        break
                else:
                    A068653_list.append(n) # _Chai Wah Wu_, May 06 2017
```
