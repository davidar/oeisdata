## Python
```Python
from sympy import primerange
import math
primes = list(primerange(2,10000))
for n in range(1,10000):
    d = n
    sum = 0
    product = 1
    for p in primes:
        if d%p==0:
            sum += p
            product *= p
            while d%p==0:
                d//=p
            if d==1:
                break
    if sum*product==n:
        print(n, end=',')
# _Alex Ratushnyak_, Aug 18 2012
```
