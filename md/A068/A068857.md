## Python

from itertools import islice
from sympy import sqrt_mod_iter
def A068857_gen(): # generator of terms
    yield 0
    a = 8
    while True:
        yield a
        b = a+1
        for d in sqrt_mod_iter(1,a):
            if d==1 or d**2-1 == a:
                d += a
            if d&1 and d < b:
                b = d
        a = b**2-1
A068857_list = list(islice(A068857_gen(),11)) # _Chai Wah Wu_, May 05 2024
