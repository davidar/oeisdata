## Python
```Python
from itertools import islice
from sympy import sqrt_mod_iter
def A068513_gen(): # generator of terms
    a = 120
    while True:
        yield a>>3
        b = a+1
        for d in sqrt_mod_iter(1,a):
            if d==1 or d**2-1 == a:
                d += a
            if d&1 and d < b:
                b = d
        a = b**2-1
A068513_list = list(islice(A068513_gen(),10)) # _Chai Wah Wu_, May 05 2024
```
