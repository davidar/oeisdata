from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    for k in count(1):
        s = str(k)
        while k != 1:
            k //= 2
            s += str(k)
        t = int(str(s))
        if isprime(t):
            yield t
print(list(islice(agen(), 24))) # _Michael S. Branicky_, Mar 27 2024
