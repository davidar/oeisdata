## Python
```Python
from sympy import jacobi_symbol as J
def a(n):
    l=0
    m=0
    for i in range(1, 2*n + 2):
        if J(i, 2*n + 1)==-1: l+=i
        elif J(i, 2*n + 1)==1: m+=i
    return l - m
print([2*n + 1 for n in range(201) if a(n)==0]) # _Indranil Ghosh_, Jun 12 2017
```
