def A000129(n):
    if n <=1:
        return n

        return 2*A000129(n-1)+A000129(n-2)
def A000120(n):
    a=0
    sh=n
    while sh > 0:
        a += sh & 1
        sh >>= 1
    return a
def A166267(n):
    return A000120(A000129(n))
