from sympy import factorint, totient
A166374 = [n for n in range(1,10**6) if sum([int(n*e/p) for p,e in factorint(n).items()]) == totient(n)] # _Chai Wah Wu_, Aug 22 2014, edited by _Antti Karttunen_, Mar 13 2021
