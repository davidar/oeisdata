## Python

from math import isqrt
def A166248(n): return abs(n*(n-1)+((s:=isqrt(n))**2*(s+1)-sum((q:=n//k)*((k<<1)+q+1) for k in range(1,s+1))>>1)) # _Chai Wah Wu_, Nov 01 2023
