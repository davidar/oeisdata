## Python
```Python
from __future__ import division
from sympy import divisors
def A082183(n):
    t = n*(n+1)
    ds = divisors(t)
    for i in range(len(ds)//2-2,-1,-1):
        x = ds[i]
        y = t//x
        a, b = divmod(y-x,2)
        if b:
            return a
    return -1 # _Chai Wah Wu_, Sep 12 2017
```
