def A082410(n):
    if n == 1:
        return 0
    s = bin(n-1)[2:]
    m = len(s)
    i = s[::-1].find('1')
    return 1-int(s[m-i-2]) if m-i-2 >= 0 else 1 # _Chai Wah Wu_, Apr 08 2021
