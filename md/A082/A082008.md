def A082008( n ):
    if n == 1: return 1
    y = 2 ** int( log( n, 2 ) )
    yc = 2 ** 2 ** int( log( log( y, 2 ), 2 ) )
    yr = y // yc
    return (yc-1) * int( (n-y) / yr + 1 ) + A082008( yr + n % yr )
