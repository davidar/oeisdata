## Python
```Python
from sympy import isprime, nextprime
def ok(p): return sum(isprime(int(c+str(p)+c)) for c in "1379") >= 2
def aupto(limit): # only test primes
  alst, p = [], 2
  while p <= limit:
    if ok(p): alst.append(p)
    p = nextprime(p)
  return alst
print(aupto(1277)) #_Michael S. Branicky_, Feb 02 2021
```
