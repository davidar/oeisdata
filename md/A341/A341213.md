def tau(n): # A000005
    d, t = 1, 0
    while d*d < n:
        if n%d == 0:
            t = t+2
        d = d+1
    if d*d == n:
        t = t+1
    return t
n, a = 1, 1 # corrected by _Martin Ehrenstein_, Apr 14 2021
while n > 0:
    nn, t1 = 1, tau(a)
    while nn < n and tau(a-nn) == (nn+1)*t1:
        nn = nn+1
    if nn == n:
        print(n,a)
        n = n+1
    a = a+1 # _A.H.M. Smeets_, Feb 07 2021
