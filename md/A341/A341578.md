from sympy import divisors
def A341578(n):
    c = min((d//2+1)*(n**2//(2*d)+1) for d in divisors(n**2,generator=True) if d<=n)
    return c if n % 2 else min(c,(n//2+1)**2-1) # _Chai Wah Wu_, Mar 05 2021
