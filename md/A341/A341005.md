## Python

def ok(n):
  sums = [0, 0]
  for d in str(n): sums[d in "13579"] += int(d)
  return abs(sums[0] - sums[1]) == 4
print(list(filter(ok, range(759)))) # _Michael S. Branicky_, Apr 13 2021
