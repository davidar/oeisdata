def aupton(terms):
  A308092, bstr, rl_lst, rl_idx, n = [1, 2], "110", [2], 2, 3

    an = int(bstr[:n], 2) - int(bstr[:n-1], 2)
    A308092, bstr, n = A308092 + [an], bstr + bin(an)[2:], n+1
    new_runs = [len(list(g)) for k, g in groupby(bstr[rl_idx:])]
