def A084019(n):
    if n == 1: return 9
    y = 10*(x:=10**(len(str(n>>1))-1))
    if n<x+y:
        c = n-x
        c = 10**len(str(c))-1-c
        return c*x+int(str(c)[-2::-1]or 0)
    else:
        c = n-y
        c = 10**len(str(c))-1-c
        return c*y+int(str(c)[-1::-1]or 0) # _Chai Wah Wu_, Jun 13 2024
