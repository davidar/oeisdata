## Python
```Python
from math import isqrt
from itertools import count
def A084187(n):
    a, b = 2, (1<<n+2)-1
    c = (b+1>>1)|1
    for k in count(1-n):
        if isqrt(a)&b==c:
            return k
        a<<=2 # _Chai Wah Wu_, Jan 25 2024
```
