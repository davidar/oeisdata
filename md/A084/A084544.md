def A084544(n):
    m = (3*n+1).bit_length()-1>>1
    return int(''.join((str(((3*n+1-(1<<(m<<1)))//(3<<((m-1-j)<<1))&3)+1) for j in range(m)))) # _Chai Wah Wu_, Feb 08 2023
