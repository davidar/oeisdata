## Python
```Python
from itertools import islice
from sympy import isprime, sieve
def agen(): yield from (pn for n, pn in enumerate(sieve, 1) if isprime(int(str(pn)+str(n))))
print(list(islice(agen(), 52))) # _Michael S. Branicky_, Dec 22 2022
```
