## Python
```Python
from math import lcm
from sympy import sieve
def a(n):
    pn = sieve[n]
    return lcm(*[q+1 for q in sieve.primerange(2, pn) if (pn+1)%(q+1)==0])
print([a(n) for n in range(1, 100)]) # _Michael S. Branicky_, Aug 13 2024
```
