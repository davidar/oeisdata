from sympy import isprime
def aupton(terms):
    alst, astr = [9], "9"
    for n in range(2, terms+1):
        an = 1
        while not isprime(int(str(an) + astr + str(an))): an += 1
        alst.append(an); astr = str(an) + astr + str(an)
    return alst
print(aupton(60)) # _Michael S. Branicky_, Sep 01 2021
