```Maple
A000111 := proc(n)
    option remember;
    sec(x)+tan(x) ;
    coeftayl(%,x=0,n)*n! ;
```

## Python
```Python
from itertools import accumulate, count, islice
def A231894_gen(): # generator of terms
    blist, c = tuple(), 1
    for i in count(1):
        yield (blist := tuple(accumulate(reversed(blist),initial=c)))[-1]
        c = c*(4*i+2)//(i+2)
A231894_list = list(islice(A231894_gen(),40)) # _Chai Wah Wu_, Jun 12 2022
```
