maxrow <- 8 # by choice
b01 <- 0 # b01 is going to be A010059
a <- 1
for(m in 0:maxrow) for(k in 0:(2^m-1)){
   b01[2^(m+1)+    k] <-     b01[2^m+k]
     a[2^(m+1)+    k] <-       a[2^m+k]  + 2^(m+b01[2^(m+1)+    k])
   b01[2^(m+1)+2^m+k] <- 1 - b01[2^m+k]
     a[2^(m+1)+2^m+k] <-       a[2^m+k]  + 2^(m+b01[2^(m+1)+2^m+k])
}
(a <- c(0,a))
