## Python
```Python
from sympy import factorint
def is_A064901(n):
    f = factorint(n)
    return (sum([f[i] for i in f]) == 2) and (max(f) % min(f) == 3)
def first_A064901(n):
    x = 1
    an = []
    while len(an) < n:
        if is_A064901(x): an.append(x)
        x += 2
    return an # _John Cerkan_, Apr 14 2018
```
