from itertools import count, islice
def A064310_gen(): # generator of terms
    yield from (1,1,0)
    a, c = 0, 1
    for n in count(1):
        yield (a:=(c:=c*((n<<2)+2)//(n+2))-a>>1)*(1 if n&1 else -1)
A064310_list = list(islice(A064310_gen(),20)) # _Chai Wah Wu_, Apr 27 2023
