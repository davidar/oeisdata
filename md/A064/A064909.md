## Python
```Python
from sympy import factorint
def is_A064909(n):
    f = factorint(n)
    return (sum([f[i] for i in f]) == 2) and (max(f) % min(f) == 11)
def first_A064909(n):
    x = 1
    an = []
    while len(an) < n:
        if is_A064909(x): an.append(x)
        x += 2
    return an # _John Cerkan_, Apr 14 2018
```
