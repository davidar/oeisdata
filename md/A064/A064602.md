## Python
```Python
from math import isqrt
def f(n): return n*(n+1)*(2*n+1)//6
def a(n):
    s = isqrt(n)
    return sum(f(n//k) + k*k*(n//k) for k in range(1, s+1)) - s*f(s)
print([a(k) for k in range(1, 43)]) # _Michael S. Branicky_, Oct 01 2022 after _Daniel Suteu_
```
