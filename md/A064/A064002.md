## Python
```Python
from itertools import count, islice
def bgen(): yield from ((i, j) for j in count(1) for i in range(1, j+1))
def agen():
    a, g = [None, 1], bgen()
    for n in count(2):
        yield a[-1];
        i, j = next(g)
        a.append(a[i] + a[j] + 1)
print(list(islice(agen(), 72))) # _Michael S. Branicky_, Jul 25 2022
```
