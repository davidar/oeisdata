## Python
```Python
from sympy import factorint
def is_A064907(n):
    f = factorint(n)
    return (sum([f[i] for i in f]) == 2) and (max(f) % min(f) == 9)
def list_A064907(cnt):
    inx = 0
    n = 1
    an = []
    while inx < cnt:
        if is_A064907(n):
            an.append(n)
            inx += 1
        n += 2
    return an #_John Cerkan_, Apr 14 2018
```
