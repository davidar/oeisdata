## Python
```Python
from sympy import floor, ceiling
def A064834(n):
    x, y = str(n), 0
    lx2 = len(x)/2
    for a,b in zip(x[:floor(lx2)],x[:ceiling(lx2)-1:-1]):
        y += abs(int(a)-int(b))
    return y
# _Chai Wah Wu_, Aug 09 2014
```
