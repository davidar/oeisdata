## Python
```Python
from sympy import primerange, prime
def aupton(terms):
  alst = [0]
  for n, pn in enumerate(primerange(1, prime(terms)+1), start=1):
    x = alst[-1] - pn
    alst += [x if x > 0 and x not in alst else alst[-1] + pn]
  return alst
print(aupton(60)) # _Michael S. Branicky_, May 30 2021
```
