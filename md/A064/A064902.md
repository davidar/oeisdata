## Python
```Python
from sympy import factorint
def is_A064902(n):
    f = factorint(n)
    return (sum([f[i] for i in f]) == 2) and (max(f) % min(f) == 4)
def first_A064902(n):
    x = 1
    an = []
    while len(an) < n:
        if is_A064902(x):an.append(x)
        n += 2
    return an # _John Cerkan_, Apr 14 2018
```
