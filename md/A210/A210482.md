```Python
from functools import reduce
import numpy as np
def factors(n):
    return reduce(list.__add__, ([i, n//i] for i in range(1, int(n**0.5) + 1) if n % i == 0))
mul=1
for i in range(1,20):
    if len(factors(i))<3:
        mul*= i*i
        if len(factors(mul+1))< 3:
            print(mul+1)
```
