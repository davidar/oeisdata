## Python
```Python
from __future__ import division
from sympy import isprime
A210546_list = []
for l in range(1,20):
    q = (10**l-1)//9
    for i in range(l):
        for p in [2,3,5,7]:
            r = q+(p-1)*10**i
            s, t = str(r), str(r)[::-1]
            if s != t and isprime(r) and isprime(int(t)):
                A210546_list.append(r) # _Chai Wah Wu_, Aug 15 2017
```
