## R
```R
qnary = function(n, e, q){
  e = floor(n/4)
  q = n%%4
  if(n == 0 ){return(0)}
  if(e == 0){return(q)}
  else{return(c(qnary(e), (q)))}
}
m = 400
s = seq(2, m)
v = c(0)
for(i in s){
  x = qnary(i-1)
  x[which(x%%2!=0)] = 1
  x[which(x%%2==0)] = -1
  v[i] = sum(x)
}
```

## Python
```Python
import numpy as np
def qnary(n):
    e = n//4
    q = n%4
    if n == 0 : return 0
    if e == 0 : return q
    if e != 0 : return np.append(qnary(e), q)
m = 400
v = [0]
for i in range(1, m+1) :
    t = np.array(qnary(i))
    t[t%2 != 0] = 1
    t[t%2 == 0] = -1
    v = np.append(v, np.sum(t))
```
