## Python
```Python
from math import comb
from sympy import factorint
def A334082(n):
    m = 0
    for k in range(1,n):
        c = comb(n,k)
        l = (~c & c-1).bit_length()
        if l>0:
            P = (1<<l+1)-1
            for p, e in factorint(c>>l).items():
                if p > 1+P:
                    break
                P *= (p**(e+1)-1)//(p-1)
            else:
                continue
            m += 1
        else:
            m += 1
    return m # _Chai Wah Wu_, Jul 05 2023
```
