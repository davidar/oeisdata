## Maple
```Maple
A334454 := proc(n)
    local dcom,i,j;
    dcom := {} ;
    for i from 1 to n do
        for j from 1 to i do
            if not isprime(i*j) and i*j> 1 then
                dcom := dcom union {i*j} ;
            end if;
        end do:
    end do:
    print(n,dcom) ;
    nops(dcom) ;
end proc:
seq(A334454(n),n=1..70) ; # _R. J. Mathar_, Oct 02 2020
```
