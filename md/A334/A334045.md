## Python
```Python
def norbitwise(n):
    a = str(bin(n))[2:]
    b = str(bin(n-1))[2:]
    if len(b) < len(a):
        b = '0' + b
    c = ''
    for i in range(len(a)):
        if a[i] == b[i] and a[i] == '0':
            c += '1'
        else:
            c += '0'
    return int(c,2)
```
