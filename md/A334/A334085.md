## Python
```Python
def is_prime(n):
    if n == 2 or n == 3:
        return True
    if n % 2 == 0 or n < 2:
        return False
    for i in range(3,int(n**0.5)+1,2):
        if n % i == 0:
            return False
    return True
def A(n):
    if n < 3:
        raise ValueError('n must be > 2')
    else:
        a = 1
        for i in range(n):
            if is_prime(i):
                a *= i
    return math.gcd(n,a)
```
