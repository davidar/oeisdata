## Python
```Python
from sympy import primepi, integer_nthroot, primerange
def A334393(n):
    def f(x): return int(n-1+x-primepi(x)-sum(primepi(integer_nthroot(x, p)[0]) for p in primerange(x.bit_length())))
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    return m # _Chai Wah Wu_, Aug 13 2024
```
