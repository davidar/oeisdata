## Python
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def A334535(n):
    if n <= 2:
        return n
    i, a, b = 2, A334535(n-1), A334535(n-2)
    q = b
    while q >= n:
        i += 1
        q = A334535(n-i)
    return 2*A334535(q)+a-b # _Chai Wah Wu_, Jun 30 2020
```
