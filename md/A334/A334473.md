def highest_power_of_3(n):
    option = 0
    while 3**option <= n:
        option +=1
    return 3 ** (option -1)
def answer(n):
    x = highest_power_of_3(n)
    if x == n:
        return x
    else:
        if n < 2 * x:
            return n%x
        else:
            return x + 2 * (n%x)
number_of_terms = 100
answers = []
for i in range(1, number_of_terms + 1):
    answers.append(answer(i))
print(answers)
