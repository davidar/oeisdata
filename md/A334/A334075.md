from fractions import Fraction
from sympy import binomial, isprime
def A334075(n):
    b = binomial(2*n,n)
    return sum(Fraction(1,p) for p in range(2,n+1) if b % p != 0 and isprime(p)).denominator # _Chai Wah Wu_, Apr 14 2020
