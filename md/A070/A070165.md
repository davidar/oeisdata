## Python

def a(n):
    if n==1: return [1]
    l=[n, ]
    while True:
        if n%2==0: n/=2
        else: n = 3*n + 1
        if n not in l:
            l+=[n, ]
            if n<2: break
        else: break
    return l
for n in range(1, 101): print(a(n)) # _Indranil Ghosh_, Apr 14 2017
