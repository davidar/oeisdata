## Python
```Python
from sympy import prime
def a006577(n):
    if n==1: return 0
    x=0
    while True:
        if n%2==0: n//=2
        else: n = 3*n + 1
        x+=1
        if n<2: break
    return x
def a(n): return a006577(prime(n))
print([a(n) for n in range(1, 101)]) # _Indranil Ghosh_, Jun 05 2017
```
