from itertools import accumulate, islice
def A070906_gen(): # generator of terms
    yield 1
    blist, b = (1,), 1
    while True:
        for _ in range(3):
            blist = list(accumulate(blist, initial=(b:=blist[-1])))
        yield b
A070906_list = list(islice(A070906_gen(),30)) # _Chai Wah Wu_, Jun 22 2022
