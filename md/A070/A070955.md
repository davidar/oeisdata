from sympy import isprime
def afind(limit, startat=1):
    for k in range(startat, limit+1):
        s = str(k)
        if isprime( int(s + "0"*(k-1) + s[::-1]) ): print(k, end=", ")
        k += 10**(len(s)-1) if s[0] in "24568" else 1
afind(1500) # _Michael S. Branicky_, Nov 30 2021
