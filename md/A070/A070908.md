from itertools import accumulate, islice
def A070908_gen(): # generator of terms
    yield 1
    blist, b = (1,), 1
    while True:
        for _ in range(5):
            blist = list(accumulate(blist, initial=(b:=blist[-1])))
        yield b
A070908_list = list(islice(A070908_gen(),20)) # _Chai Wah Wu_, Jun 22 2022
