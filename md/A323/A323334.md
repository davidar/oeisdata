## Python
```Python
def aupton(terms):
  alst, s, pset = [], 0, set()
  for n in range(1, terms):
    for i in range(1, n+1):
      for j in range(i, n+1):
        p = i*j*n # k = n
        if p not in pset:
          pset.add(p)
          s += p
    alst.append(s)
  return alst
print(aupton(36)) # _Michael S. Branicky_, Jan 15 2021
```
