## Python
```Python
from sympy import isprime, nextprime
from sympy.ntheory import digits
def A323745(n):
    p = 2
    while True:
        m = 1
        for j in digits(p,n)[:0:-1]:
            for k in range(1,n):
                if k!=j and isprime(p+(k-j)*m):
                    break
            else:
                m *= n
                continue
            break
        else:
            return p
        p = nextprime(p) # _Chai Wah Wu_, Mar 25 2024
```
