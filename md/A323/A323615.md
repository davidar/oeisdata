## Python
```Python
def a323615(n):
  seq = []
  for i in range(n + 1):
    if i == 0: x = 1
    else:
      x = seq[i - 1] // i
      if x in seq or x == 0: x = seq[i - 1] * i
    seq.append(x)
  return seq
print(a323615(100))
```
