## Python
```Python
def row(n):
    b = bin(n)[2:]
    s = set(b[:i+1] + "0" + b[i+1:] for i in range(len(b)) if b[i] == "1")
    return sorted(int(w, 2) for w in s)
print([c for n in range(1, 29) for c in row(n)]) # _Michael S. Branicky_, Jul 24 2022
```
