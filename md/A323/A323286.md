def cdb(n):
    s, out = str(n), set()
    for l in range(1, len(s)+1):
        for i in range(len(s)+1-l):
            if s[i] == '0': continue
            t = int(s[i:i+l])
            out.add(int(s[:i] + str(2*t) + s[i+l:]))
            if t&1 == 0: out.add(int(s[:i] + str(t//2) + s[i+l:]))
    return sorted(out)
print([c for n in range(1, 25) for c in cdb(n)]) # _Michael S. Branicky_, Jul 24 2022
