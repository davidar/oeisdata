## Python
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def t(n, k):
    if n < k: return 0
    if k == 0: return 1 if n == 0 else 0
    r = 0
    i = 1
    while True:
        if i > n: break
        r += t(n - i, k-1)
        i *= 2
    return r
def T(n, k): return t(2**n, k) # _James Rayman_, Mar 30 2021
```
