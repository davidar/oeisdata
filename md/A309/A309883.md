## Python

def A003132(n):
    s = 0
    while n > 0:
        s, n = s+(n%10)**2, n//10
    return s
n, a = 0, 0
while n < 50:
    if A003132(a) == A003132(a*a):
        n = n+1
        print(n,a)
    a = a+1 # _A.H.M. Smeets_, Aug 23 2019
