## Python

```Python
def distinct_digits(n):
    p = math.floor(math.log(10**10)/math.log(n))
    while p >= 1:
        d = n**p
        if len(set(str(d))) == len(str(d)):
            return(p)
        else:
            p = p - 1
    return(0)
```
