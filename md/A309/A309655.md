```Python
#Lists a(1)...a(100)
from sympy import prime
sumP=0
for i in range(1,101):
  sumP+=prime(i)
  j=i+1
  diff=sumP
  while diff-prime(j) >=0:
   diff-=prime(j)
   j+=1
  print diff,',',
```
