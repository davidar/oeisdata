## Python
```Python
def c(n): s = str(n); return sum(map(int, s[::2])) == sum(map(int, s[1::2]))
def afind(limit):
    k, factk = 1, 1
    while k <= limit:
        if c(factk): print(k, end=", ")
        k += 1; factk *= k
afind(500) # _Michael S. Branicky_, Nov 18 2021
```
