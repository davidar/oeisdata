sa, n = [0], 0
print(n,sa[n])
while n < 50:
    i, j = 0, 0
    while i < n:
        if sa[i] == sa[n]:
            j = j+1
        i = i+1
    if j == 0:
        a = sa[n] - sa[n-sa[n]]
    else:
        a = j
    n = n+1
