## Python
```Python
def A309709(n):
    s = ""
    while n > 0:
        s, n = str(n%2)+s,n//2
    s, s4, i, j = "00"+s, s+"00", 0, 0
    while i < len(s):
        if s[i] != s4[i]:
            j = j+1
        i = i+1
    return j # _A.H.M. Smeets_, Aug 23 2019
```
