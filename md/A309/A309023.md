## Python
```Python
def Prog(length):
    L = 2
    seq = [1,2]
    while L < length:
        x = len(seq)-1
        while x > 0:
            if seq[-1] == seq[x-1]:
                m_minus_n = len(seq)-x
                a_n = seq[L-1]
                seq.append((m_minus_n)**a_n)
                x = -1
            else:
                x -= 1
            if x == 0:
                seq.append(0)
        L += 1
    return seq
```
