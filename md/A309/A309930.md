## Python
```Python
import decimal
from decimal import Decimal as D
def constant_from_cofr(clist):
    hn0, kn0 = 0, 1
    hn1, kn1 = 1, 0
    for n in clist:
        hn2 = (n * hn1) + hn0
        kn2 = (n * kn1) + kn0
        hn0, kn0 = hn1, kn1
        hn1, kn1 = hn2, kn2
    return D(hn2)/D(kn2)
if __name__ == "__main__":
    prec = 200
    decimal.getcontext().prec = prec
    glist = [x**3 for x in range(500)]
    print(', '.join(str(x) for x in str(constant_from_cofr(glist))[2:]))
```
