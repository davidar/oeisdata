def dec_to_bin(dec_num): #define a function that converts decimal to binary.
    bin_num = 0
    power = 0
    while dec_num > 0:
        bin_num += 10 ** power * (dec_num % 2)
        dec_num //= 2
        power += 1
    return bin_num
def rev_bin(n):
    return list(reversed(str(dec_to_bin(n))))
n = 18

def A309983(n):
    r, i = 0, 0
    while n > 0:
        d, n = n%2, n//2
        if d == 1:
            r = r+i
        else:
            r = r-i
        i = i+1
