N:= 100: # for a(1)..a(N)
V:= Vector([$1..N]):
for k from 2 to floor(sqrt(N)) do
  for j from 1 to N/k^2 do
    t:=min((j+1)*k^2-1,N);
    V[j*k^2..t]:= V[j*k^2..t] +~ (-1)^(k+1)*j
od od:
convert(V,list); # _Robert Israel_, Jul 12 2019
