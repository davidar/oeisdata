## Python
```Python
import decimal
prec = int(input('Precision: '))
decimal.getcontext().prec = prec
D = decimal.Decimal
def agm(a, b):
    for x in range(prec):
        a, b = (a + b) / 2,(a * b).sqrt()
    return a
print(agm(1, D(3).sqrt()/2))
```
