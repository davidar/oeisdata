from sympy import integer_nthroot
A309388_list, y, w = [], 1, 0
while len(A309388_list) < 10000:
    w += y
    z = 0
    for x in range(1,y+1):
        z += x
        if integer_nthroot(8*(w+z)+1,2)[1]:
            break
