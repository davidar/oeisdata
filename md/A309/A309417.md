## Python
```Python
import math
def digitsum(n):
   ds = 0
   while n > 0:
      ds += n % 10
      n = n // 10
   return ds
def steps(n):
   count = 0
   while n > 0:
      n = n - digitsum(n)
      count += 1
   return count
n = 1
for i in range(1,10):
   n = 10 * n
   print(steps(n))
```
