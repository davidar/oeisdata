from functools import lru_cache
@lru_cache(maxsize=None)
def A309288(n):
    if n <= 1:
        return n
    c, j = 0, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += ((j2-j)%2)*(1-2*(j%2))*A309288(k1)
        j, k1 = j2, n//j2
    return c+((n+1-j)%2)*(1-2*(j%2)) # _Chai Wah Wu_, Mar 31 2021
