## Python
```Python
import numpy as np
import math as m
n = 1
i = 0
inp = np.zeros(1)
out = inp
while n < 10001:
    k=m.fabs( m.trunc( m.sin(i) * m.pow( 10,m.floor( m.log10(n)+1 ) ) ) )
    if k==n:
        inp = np.append(inp,int(n))
        out = np.append(out,int(i))
        print(n,i)
        n += 1
        i = 0
        continue
    else:
        i+=1
```
