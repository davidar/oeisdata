## Python
```Python
seq = []
b = 2
for n in range(1, 100):
  seq += [n] + [-1] * n
while -1 in seq:
  i = seq.index(-1)
  seq[i] = 1
  k = 2
  blanks = b
  for s in range(i + 1, len(seq)):
    if seq[s] == -1:
      blanks -= 1
      if blanks < 0:
        seq[s] = k
        blanks = k * b
        k += 1
  b += 1
print(seq)
```
