## Python
```Python
from sympy.ntheory.digits import digits
def fromdigits(d, b):
    n = 0
    for di in d: n *= b; n += di
    return n
def afull():
    alst, diglst = [1], [1]
    for n in range(2, 11):
        andigs = digits(fromdigits(diglst, n-1), n)[1:]
        alst.append(int("".join(map(str, andigs))))
        diglst.extend(andigs)
    return alst
print(afull()) # _Michael S. Branicky_, Dec 05 2021
```
