## Python
```Python
from math import isqrt
def A188915(n):
    def bisection(f,kmin=0,kmax=1):
        while f(kmax) > kmax: kmax <<= 1
        while kmax-kmin > 1:
            kmid = kmax+kmin>>1
            if f(kmid) <= kmid:
                kmax = kmid
            else:
                kmin = kmid
        return kmax
    def f(x): return n+x-isqrt(x)-((m:=x.bit_length()-1)>>1)-(m&1)
    return bisection(f,n-1,n**2) # _Chai Wah Wu_, Sep 19 2024
```
