from math import sqrt
def A267077(n):
    if n == 0:
        return 1
    u,v,t,w = max(8,2*n),max(4,n)**2-9,4*n*(n+1),n**2
    while True:
        m,r = divmod(v,t)
        if not r and int(sqrt(m*w+1))**2 == m*w+1:
            return m
        v += u+1
        u += 2 # _Chai Wah Wu_, Jan 15 2016

    nsq = nn * nn
    ps = labmath.pell(nn*(4*nn+4), nn*(5*nn-4))
