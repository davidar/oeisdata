from itertools import count, islice
def agen(): # generator of terms
    astr, k, mink = "01", 2, 1
    while True:
        for k in range(mink, len(astr)+1):
            if astr[1:].count(astr[:k]) == 0:
                break
        mink = max(mink, k)
        astr += astr[:k]
        yield k
print(list(islice(agen(), 69))) # _Michael S. Branicky_, Mar 23 2024
