f:= proc(n) local s;
  s:= isqrt(10^(2*n+1));
  if s^2 < 10^(2*n+1) then s:= s+1 fi;
  s^2 - 10^(2*n+1)
end proc:
seq(f(n),n=0..40); # _Robert Israel_, Jan 17 2016
