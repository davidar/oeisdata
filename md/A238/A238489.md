## Python
```Python
# use version >= 3.8
from math import isqrt
for k in range(7777):
    s = isqrt(k)
    if s*s==k:  s-=1
    kxy = k + 2*s*(s+1) + 1   # k + s^2 + (s+1)^2
    r = isqrt(kxy)
    if r*r==kxy:  print(str(k), end=',')
```
