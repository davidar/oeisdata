def primfacs(n):
    i = 2
    primfacs = []
    while i * i <= n:
        while n % i == 0:
            primfacs.append(i)
            n = n / i
        i = i + 1
    if n > 1:
        primfacs.append(n)
    return primfacs
def sopfr(n):
    plist = list(primfacs(n))
    l = len(plist)
    s = 0
    while l > 0:
        s += plist[l - 1]
        l -= 1
    return s
def sd(n):
    d = 1
    s = n % sopfr(n)
    if s > 1:
        d += sd(s)
    return d
n=2
max=1000
rec = 0
lst = []
while n <= max:
    r = sd(n)
    if r > rec:
        lst.append(n)
        rec = r
    n += 1
print(lst)
