## Python
```Python
def isqrt(a):
    sr = 1 << (int.bit_length(int(a)) >> 1)
    while a < sr*sr:  sr>>=1
    b = sr>>1
    while b:
        s = sr + b
        if a >= s*s:  sr = s
        b>>=1
    return sr
for n in range(77):
    nn = 4**n
    s = isqrt(2*nn)
    if s*(s+1)//2 > nn:  s-=1
    d1 = nn - s*(s+1)//2
    d2 = (s+1)*(s+2)//2 - nn
    if d2 < d1:  d1 = -d2
    print(str(d1), end=',')
```
