## Python
```Python
def isqrt(a):
    sr = 1 << (int.bit_length(int(a)) >> 1)
    while a < sr*sr:  sr>>=1
    b = sr>>1
    while b:
        s = sr + b
        if a >= s*s:  sr = s
        b>>=1
    return sr
t = i = 0
while 1:
    t += i
    i += 1
    s = isqrt(t)
    if s*s==t:  s-=1
    txy = t + 2*s*(s+1) + 1   # t + s^2 + (s+1)^2
    r = isqrt(txy)
    if r*r==txy:  print(str(t), end=',')
```
