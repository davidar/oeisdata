## Python

from itertools import islice
def agen(): # generator of terms
    an, ratios = 1, set()
    while True:
        yield an
        k = 1
        q, r = divmod(max(k, an), min(k, an))
        while r != 0 or q in ratios:
            k += 1
            q, r = divmod(max(k, an), min(k, an))
        an = k
        ratios.add(q)
print(list(islice(agen(), 66))) # _Michael S. Branicky_, Feb 13 2023
