from sympy import prime, nextprime
def A360311(n):
    p = prime(n)
    q = nextprime(p)
    s, k = p+q, 1
    while s%(q:=nextprime(q)):
        k += 1
        s += q
    return s # _Chai Wah Wu_, Feb 06 2023
