from itertools import product
from sympy import primerange, primepi, factorint
def A360659(n):
    a = dict(zip(primerange(n+1),range(c:=primepi(n))))
    return (min(sum(sum(e for p,e in factorint(m).items() if b[a[p]])&1^1 for m in range(1,n+1)) for b in product((0,1),repeat=c))<<1)-n # _Chai Wah Wu_, May 31 2024
