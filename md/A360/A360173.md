## Python

```Python
def A360173_rowlist(row_n):
    A = [[0]]
    for i in range(0,row_n):
        A.append([])
        for j in range(0,len(A[i])):
            x = A[i][j]
            if x - i -1 >= 0:
                A[i+1].append(x-i-1)
            if x + i + 1 >= 0:
                A[i+1].append(x+i+1)
    return(A)
```
