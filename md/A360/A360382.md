## Python
```Python
def s(n):
    p=[k**n for k in range(360)]
    for k in range(4,360):
        for d in range(k-1,3,-1):
            if 4*p[d]>p[k]:
                cc=p[k]-p[d]
                for c in range(d-1,2,-1):
                    if 3*p[c]>cc:
                        bb=cc-p[c]
                        for b in range(c-1,1,-1):
                           if 2*p[b]>bb:
                               aa=bb-p[b]
                               if aa>0 and aa in p:
                                   a=round(aa**(1/n))
                                   return(n,k,[a,b,c,d])
for n in range(1,6):
    print(s(n))
```
