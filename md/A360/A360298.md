## Python
```Python
from functools import cache
@cache
def row(n):
    if n == 1: return [1]
    out = []
    for r in row(n-1): out += ([r//n] if r%n == 0 else []) + [r*n]
    return out
print([an for r in range(1, 13) for an in row(r)]) # _Michael S. Branicky_, Feb 02 2023
```
