## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    s = ""
    for k in count(1):
        s = bin(k)[2:] + s
        if isprime(int(s)): yield k
print(list(islice(agen(), 5))) # _Michael S. Branicky_, Feb 19 2023
```
