## Python
```Python
from itertools import combinations as C
def a(n):
    v = list(map(int, str(n)))
    return len(set(sum(c) for r in range(1, len(v)+1) for c in C(v, r)))
print([a(n) for n in range(89)]) # _Michael S. Branicky_, Feb 19 2023
```
