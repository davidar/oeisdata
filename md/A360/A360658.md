from functools import lru_cache
@lru_cache(maxsize=None)
def A360658(n):
    if n <= 1:
        return 1
    c, j = 0, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c -= ((j2*(j2-1))**2-(j*(j-1))**2>>2)*A360658(k1)
