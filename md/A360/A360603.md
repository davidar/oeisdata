## Python
```Python
from math import comb as binomial
from functools import cache
@cache
def A360603Row(n: int) -> list[int]:
    if n == 0: return [1]
    s = [2 ** (((k - n + 1) * (k - n)) // 2) * binomial(n - 1, k - 1) * A360603Row(k)[k] for k in range(1, n)]
    b = 2 ** (((n - 1) * n) // 2) - sum(s)
    return [0] + s + [b]
```
