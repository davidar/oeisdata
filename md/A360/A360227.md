## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    aset, an, s = {"1", "11"}, 2, "21"
    yield from [1, 11]
    while True:
        i = next(i for i in range(1, len(s)+1) if s[:i] not in aset and (i == len(s) or s[i] != "0"))
        an = int(str(s[:i]))
        s = s[i:] + str(an*11)
        aset.add(str(an))
        yield an
print(list(islice(agen(), 66))) # _Michael S. Branicky_, Apr 28 2023
```
