## Python
```Python
from itertools import count, islice
from sympy import primepi, isprime
def A360448_gen(): # generator of terms
    for i in count(2):
        k = (1<<i)+1
        for j in range(1,i):
            if isprime(m := k+(1<<j)):
                yield primepi(m)
A360448_list = list(islice(A360448_gen(),20)) # _Chai Wah Wu_, Mar 21 2023
```
