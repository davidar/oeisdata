## Python
```Python
def a(n, starte=0):
    e, p2, t = starte, 2**starte, 10**n
    while True:
        s2, ss = str(p2), set()
        for i in range(len(s2)-n+1):
            ss.add(s2[i:i+n])
            if len(ss) == t:
                return e
        e += 1
        p2 *= 2
print([a(n) for n in range(1, 4)]) # _Michael S. Branicky_, Feb 22 2023
```
