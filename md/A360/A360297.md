## Python

from sympy import prime, nextprime
def A360297(n):
    p = prime(n)
    q = nextprime(p)
    s, k = p+q, 1
    while s%(q:=nextprime(q)):
        k += 1
        s += q
    return k # _Chai Wah Wu_, Feb 03 2023
