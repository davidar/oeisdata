## Python
```Python
from sympy import divisors, totient as phi
def a(n):
    k, pk = 7, phi(7)
    while sum(pow(d, n, pk) for d in divisors(k, generator=True))%pk != 0:
        k += 1
        pk = phi(k)
    return k
print([a(n) for n in range(1, 16)]) # _Michael S. Branicky_, Dec 22 2021
```
