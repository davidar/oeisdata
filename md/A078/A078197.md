## Python
```Python
from sympy import E
digits_of_e = str(E.n(10**5))[1:-1] # raise to 10**6 for b-file
def a(n):
    nloc = digits_of_e.find(str(n))
    assert nloc > 0, ("Increase precision", n)
    return nloc
print([a(n) for n in range(63)]) # _Michael S. Branicky_, Jul 10 2022
```
