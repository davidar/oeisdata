## Python
```Python
def pd(n): return set(str(n)) <= set("2357")  # has only prime digits
def aupton(terms, startat=5):
  alst = [startat]
  for n in range(2, terms+1):
    m = alst[-1]; km = 2*m
    while not pd(km): km += m
    alst.append(km)
  return alst
print(aupton(9)) # _Michael S. Branicky_, May 25 2021
```
