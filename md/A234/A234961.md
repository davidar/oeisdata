from itertools import islice
from sympy import prevprime, nextprime
def agen():
    yield 1
    reach, expand = {2}, [2]
    while True:
        newreach = set()
        while len(expand) > 0:
            p = expand.pop()
            for q in prevprime(2*p), nextprime(2*p):
                if q not in reach:
                    newreach.add(q)
        yield len(newreach)
        reach |= newreach
        expand = list(newreach)
print(list(islice(agen(), 20))) # _Michael S. Branicky_, Jul 24 2022
