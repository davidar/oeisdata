import math
for n in range(67):
  x = n*n
  y = x*n
  a = (x+y)*0.5
  g = math.sqrt(x*y)
  while abs(a-g)>0.00000000001:
    x = a
    y = g
    a = (x+y)*0.5
    g = math.sqrt(x*y)
  print str(int(a))+',',
