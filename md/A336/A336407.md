## Python
```Python
from sympy import primepi
def A336407(n):
    if n == 1: return 3
    m, k = n, (r:=primepi(n)) + n + (n>>1)
    while m != k:
        m, k = k, (r:=primepi(k)) + n + (k>>1)
    return m-r-2 # _Chai Wah Wu_, Jul 31 2024
```
