## Python
```Python
from math import floor, log
def a(n):
    if n == 1: return 0
    count = 0
    while True:
        if n % 2: n = 3**n + 1
        else: n = int(floor(log(n, 2)))
        count += 1
        if n == 1: break
    return count
print([a(n) for n in range(1, 101)])
```
