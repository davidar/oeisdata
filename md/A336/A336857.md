## Python
```Python
def seq(num):
    res = []
    for i in range(1, num + 1):
        res.append(calc(i))
    return res
def calc(num):
    sum = 0
    start = int((num * (num - 1)) // 2) + 1
    for i in range(start, start + num):
        sum += i
    recip = 0
    for i in str(sum):
        recip += int(i)
    return recip
print(seq(260))
```
