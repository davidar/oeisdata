from sympy import isprime
n = 0
m = 2
while m >= 2:
    d = 0
    while d < m/2:
        p = m - d
        q = m + d
        if isprime(p) == 1 and isprime(q) == 1:
            if p*q < (m - 1) * (m - 1):
                n += 1
                print (m)
            break
        d += 1
    m += 1
