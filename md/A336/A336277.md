from functools import lru_cache
@lru_cache(maxsize=None)
def A336277(n):
    if n <= 1:
        return 1
    c, j = 1, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c -= ((j2*(j2-1))**2-(j*(j-1))**2>>2)*A336277(k1)
        j, k1 = j2, n//j2
    return c-((n*(n+1))**2-((j-1)*j)**2>>2) # _Chai Wah Wu_, Apr 04 2023
