from math import prod
from sympy import factorint, prime
def A336158(n): return prod(prime(i+1)**e for i,e in enumerate(sorted(factorint(n>>(~n&n-1).bit_length()).values(),reverse=True))) # _Chai Wah Wu_, Sep 16 2022
