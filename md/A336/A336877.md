def a(d, m=6):
    chi = [0, 1]
    for p in range(m):
        chi += [(d+1)*chi[k]+2*chi[2**p]-chi[2**p-k] for k in range(1, 2**p+1)]
    return chi
chi = a(2)
print(chi)
