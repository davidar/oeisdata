## Python
```Python
f = open("mu(n,mu).txt","a")
N = 10000
mu = [0]
x = []
f.write("0 0\n")
for n in range(1,N):
    for i in range(2,N):
        iChoose2 = (i*(i-1))/2
        if iChoose2 <= n:
            x.append(mu[int(n-iChoose2)]+i)
    mu.append(min(x))
    f.write(str(n)+" "+str(min(x))+"\n")
    x.clear()
f.close()
```
