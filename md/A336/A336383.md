## Python
```Python
def f(x):
    prod = 1
    for digit in str(x):
        prod *= int(digit)
    return x - prod
def a(n):
    i = 0
    iteration = 0
    while iteration != n:
        i += 1
        j = i
        iteration = 0
        new_j = f(j)
        while j != new_j:
            iteration += 1
            j = new_j
            new_j = f(j)
    return i
```
