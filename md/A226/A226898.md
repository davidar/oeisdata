## Python
```Python
from sympy import divisors, exp
def a(n):
    d = divisors(n)
    m = 1
    for i in range(len(d) - 1):
        t = exp(1)*d[i]
        m = max(sum(1 for j in range(i, len(d)) if d[j]<t), m)
    return m
print([a(n) for n in range(1, 51)]) # _Indranil Ghosh_, Jul 19 2017
```
