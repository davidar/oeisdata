## Python
```Python
from fractions import Fraction
from itertools import count, islice
def agen():
    rats = [Fraction(0, 1)]
    seen = {Fraction(0, 1)}
    for n in count(1):
        yield from [r.denominator for r in rats]
        newrats = []
        for r in rats:
            f = 1+r
            if f not in seen:
                newrats.append(1+r)
                seen.add(f)
            if r != 0:
                g = -1/r
                if g not in seen:
                    newrats.append(-1/r)
                    seen.add(g)
        rats = newrats
print(list(islice(agen(), 84))) # _Michael S. Branicky_, Jan 17 2022
```
