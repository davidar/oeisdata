## Python

from sympy import isprime
def nextsod(n, base):
    c, b, w = 0, base, 0
    while True:
        d = n%b
        if d+1 < b and c:
            return (n+1)*b**w + ((c-1)%(b-1)+1)*b**((c-1)//(b-1))-1
        c += d; n //= b; w += 1
def a(n):
    if n%3 == 0: return 0
    c, t = 0, 10**(n-1) + ((n-1)%9+1)*10**((n-1)//9)-1
    while t < 10**n:
        if isprime(t): c += 1
        t = nextsod(t, 10)
    return c
print([a(n) for n in range(1, 13)]) # _Michael S. Branicky_, Aug 02 2022
