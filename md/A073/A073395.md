## Python

from sympy import primefactors, factorint
def a001414(n):
    f=factorint(n)
    return sum([i*f[i] for i in f])
def a(n): return 0 if n==1 else sum(primefactors(n))*a001414(n)
print([a(n) for n in range(1, 101)]) # _Indranil Ghosh_, Jun 23 2017
