## Python

def A073790(n):
    s, q = '', n
    while q >= 9 or q < 0:
        q, r = divmod(q, -9)
        if r < 0:
            q += 1
            r += 9
        s += str(r)
    return int(str(q)+s[::-1]) # _Chai Wah Wu_, Apr 09 2016
