## Python
```Python
from math import isqrt
from sympy import mobius, primefactors
def A073482(n):
    def f(x): return n+x-sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1))
    kmin, kmax = 0,1
    while f(kmax) > kmax:
        kmax <<= 1
    while kmax-kmin > 1:
        kmid = kmax+kmin>>1
        if f(kmid) <= kmid:
            kmax = kmid
        else:
            kmin = kmid
    return max(primefactors(kmax),default=1) # _Chai Wah Wu_, Aug 28 2024
```
