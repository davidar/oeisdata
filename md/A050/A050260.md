## Python
```Python
from math import isqrt
from sympy import primerange
def ok(n):
    if n < 2: return False
    for p in primerange(2, isqrt(n)+1):
        isqrtp = isqrt(p)
        if abs(2*(n%p)-p) > p + 1 - isqrt(p) - (isqrtp**2 < p): return False
    return True
print(list(filter(ok, range(2, 350)))) # _Michael S. Branicky_, Aug 11 2021
```
