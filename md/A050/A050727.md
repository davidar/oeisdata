## Python
```Python
try: from gmpy2 import mpz; x = mpz(1)
except: x = 1
print(0)
k = 1
while True:
    print('\b'*42 + str(k), end='')
    x *= 6  # x == 6**k
    y, flag = x, True
    y, a = divmod(y, 10)
    while y > 6:
        b = a
        y, a = divmod(y, 10)
        if a == b:
            flag = False
            break
    if flag: print()
    k += 1
# _Lucas A. Brown_, Mar 02 2024
```
