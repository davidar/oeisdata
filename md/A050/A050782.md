```Maple
digrev:= proc(n) local L,d,i;
  L:= convert(n,base,10);
  d:= nops(L);
  add(L[i]*10^(d-i),i=1..d);
end proc:
f:= proc(n)
local d,d2,x,t,y;
if n mod 10 = 0 then return 0 fi;
if n < 10 then return 1 fi;
for d from 2 do
  if d::even then
    d2:= d/2;
    for x from 10^(d2-1) to 10^d2-1 do
       t:= x*10^d2 + digrev(x);
       if t mod n = 0 then return(t/n) fi;
    od
  else
    d2:= (d-1)/2;
    for x from 10^(d2-1) to 10^d2-1 do
      for y from 0 to 9 do
        t:= x*10^(d2+1)+y*10^d2+digrev(x);
        if t mod n = 0 then return(t/n) fi;
      od
    od
  fi
od;
end proc:
```

## Python
```Python
from __future__ import division
def palgen(l,b=10): # generator of palindromes in base b of length <= 2*l
    if l > 0:
        yield 0
        for x in range(1,l+1):
            n = b**(x-1)
            n2 = n*b
            for y in range(n,n2):
                k, m = y//b, 0
                while k >= b:
                    k, r = divmod(k,b)
                    m = b*m + r
                yield y*n + b*m + k
            for y in range(n,n2):
                k, m = y, 0
                while k >= b:
                    k, r = divmod(k,b)
                    m = b*m + r
                yield y*n2 + b*m + k
def A050782(n, l=10):
    if n % 10:
        x = palgen(l)
        next(x)  # replace with x.next() in Python 2.x
        for i in x:
            q, r = divmod(i, n)
            if not r:
                return q
        else:
            return 'search limit reached.'
    else:
        return 0 # _Chai Wah Wu_, Dec 30 2014
```
