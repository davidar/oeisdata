grow := proc(n,M) # searches out to a limit of M, returns [n,n+k] or [n,-1] if no k was found
local R,i;
R:=n;
for i from n+1 to M do
R:=R*10^length(i)+i;
if (i mod 2) = 0 then
if (R mod (i+1)) = 0 then return([n, i]); fi;
fi;
od:
[n, -1];
end;

def A332584(n):
    r, m = n, n + 1
    while True:
        r = r*10**(len(str(m))) + m
        if m % 2 == 0 and r % (m+1) == 0:
            return m
        m += 1 # _Chai Wah Wu_, Jun 12 2020
