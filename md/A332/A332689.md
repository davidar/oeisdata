## Python
```Python
from math import sqrt
def A332689(n):
    L = []; k = 4*n*n
    for x in range(1, int(2*sqrt(3)*n) + 1):
        for y in range(max(int(k/x) + 1, x), int((k + 2*n*sqrt(k + x*x))/x) + 1):
            if k*(x+y)%(x*y-k) == 0:
                s = x + y + k*(x+y)//(x*y-k)
                if s not in L: L.append(s)
    return len(L)  # _Ya-Ping Lu_, Dec 28 2023
```
