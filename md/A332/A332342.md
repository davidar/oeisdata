## Python
```Python
def cofr(p, q):
    return [] if q == 0 else [p // q] + cofr(q, p % q)
def t(n, k):
    return sum(cofr(n, k)[1:])
tr = []
for nk in range(1, 20):
    for k in range(1, nk+1):
        tr.append(t(nk+1-k, k))
print(tr)
```
