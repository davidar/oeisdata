## Python
```Python
from sympy import primerange
p = list(primerange(1, 200))
def a(n):
    out = p[0] + p[1]
    for i in range(1, n):
        if i % 2:
            out *= p[i + 1]
        else:
            out += p[i + 1]
    return out
for n in range(1, 25):
    print(a(n), end=", ")
```
