## Python
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def A344527_T(n,k):
    if n == 0:
        return 0
    c, j, k1 = 1, 2, n//2
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*A344527_T(k1,k)
        j, k1 = j2, n//j2
    return n*(n**(k-1)-1)-c+j
def A332468(n): return A344527_T(n,n) # _Chai Wah Wu_, Nov 02 2023
```
