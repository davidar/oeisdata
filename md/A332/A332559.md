```Maple
f:= proc(n) local k,p;
  p:= n;
  for k from 1 do
    p:= p*(n+k);
    if (p/(n+k+1))::integer then return n+k+1 fi
  od
end proc:
```

## Python
```Python
def a(n):
    k, p = 1, n*(n+1)
    while p%(n+k+1): k += 1; p *= (n+k)
    return n + k + 1
print([a(n) for n in range(1, 67)]) # _Michael S. Branicky_, Jun 06 2021
```
