## Ruby

def A(n)
  s = n
  t = n + 1
  while s % t > 0
    s += t
    t += 1
  end
  t - n - 1
end

def a(n):
    k, s = 1, 2*n+1
    while s%(n+k+1) != 0: k += 1; s += n+k
    return k
