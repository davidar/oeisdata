```Maple
  v:= max(s0,s1) mod n;
  R:= R,v;
```

## Python
```Python
a = [0,1]
s0,s1 = 0,1
for n in range(2, 1000):
    v = max(s0,s1) % n
    a.append(v)
    if (v & 1): s1 += v
    else:       s0 += v
print(a)
```
