## Python
```Python
def a(n):
    b = bin(n)[2:]
    t = b.rstrip("0")
    if t == "1": return int("1" + "0"*n + "1"*(n-1), 2)
    return int("1" + "0"*(n-b.count("0")) + b + "1"*(n-1-b.count("1")), 2)
print([a(n) for n in range(1, 26)]) # _Michael S. Branicky_, Feb 18 2023
```
