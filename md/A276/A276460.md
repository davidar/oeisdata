## Python
```Python
from __future__ import division
from sympy import divisors
from gmpy2 import is_square
A276460_list = [0]
for m in range(10**3):
    k = m**2+1
    for d in divisors(k):
        if d > m:
            A276460_list.append(k)
            break
        if not is_square(k//d - d):
            break # _Chai Wah Wu_, Sep 04 2016
```
