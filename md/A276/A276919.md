## Python
```Python
def A276919(n):
    ndict = {}
    for i in range(n):
        i3 = pow(i,3,n)
        for j in range(i+1):
            j3 = pow(j,3,n)
            m = (i3+j3) % n
            if m in ndict:
                if i == j:
                    ndict[m] += 1
                else:
                    ndict[m] += 2
            else:
                if i == j:
                    ndict[m] = 1
                else:
                    ndict[m] = 2
    count = 0
    for i in ndict:
        j = (1-i) % n
        if j in ndict:
            count += ndict[i]*ndict[j]
    return count # _Chai Wah Wu_, Jun 06 2017
```
