A:= Vector(100): q:= 2:
for n from 1 to 100 do
  p:= q; q:= nextprime(q);
  t:= 0;
  for i from p+1 to q-1 do t:= t + padic:-ordp(i,2) od;
  A[n]:= t
od:
convert(A,list); # _Robert Israel_, Apr 11 2021
