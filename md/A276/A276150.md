## Python
```Python
from sympy import prime, primefactors
def Omega(n): return 0 if n==1 else Omega(n//primefactors(n)[0]) + 1
def a276086(n):
    i=0
    m=pr=1
    while n>0:
        i+=1
        N=prime(i)*pr
        if n%N!=0:
            m*=(prime(i)**((n%N)/pr))
            n-=n%N
        pr=N
    return m
def a(n): return Omega(a276086(n))
```
