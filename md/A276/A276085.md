## Python

from sympy import primorial, primepi, factorint
def a002110(n):
    return 1 if n<1 else primorial(n)
def a(n):
    f=factorint(n)
    return sum(f[i]*a002110(primepi(i) - 1) for i in f)
print([a(n) for n in range(1, 101)]) # _Indranil Ghosh_, Jun 22 2017
