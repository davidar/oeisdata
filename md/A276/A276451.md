## Python
```Python
import math
def nCr(n,r):
    f = math.factorial
    return f(n) / f(r) / f(n-r)
# main program
for j in range(101):
    i = j/2
    if j%2==0:
        b = nCr(2*i*i,i)
    else:
        b = nCr(2*i*(i+1),i)
    if j%4==0:
        c = nCr((j*j/4),(j/4))
    elif j%4==1:
        c = nCr(((j-1)/2)*((j-1)/2+1),((j-1)/4))
    else:
        c = 0
    print(str(j)+" "+str((b-c)/2))
```
