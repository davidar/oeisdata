## Python

from sympy import factorint, factorial as f, primepi
def a(n):
    F=factorint(n)
    return 0 if n==1 else sum(F[i]*f(primepi(i)) for i in F)
print([a(n) for n in range(1, 121)]) # _Indranil Ghosh_, Jun 21 2017
