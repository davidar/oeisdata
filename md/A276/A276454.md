from math import comb as binomial
for j in range(1, 20):
    t = binomial(j * j, j)
    i = j // 2
    if j % 2 == 0:
        d = binomial(2 * i * i, i)
    else:
        d = binomial(2 * i * (i + 1), i)
    a = (t - d) // 4
    if j % 4 == 0:
        c = binomial((j * j // 4), (j // 4))
    elif j % 4 == 1:
        c = binomial(((j - 1) // 2) * ((j - 1) // 2 + 1), ((j - 1) // 4))
    else:
        c = 0
    b = (d - c) // 2
    print(str(j) + " " + str(a + b + c))
