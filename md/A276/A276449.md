## Python
```Python
import math
def nCr(n,r):
    f = math.factorial
    return f(n) / f(r) / f(n-r)
# main program
for j in range(101):
    if j%4 == 0:
        a = nCr((j*j/4),(j/4))
    elif j%4 == 1:
        a = nCr(((j-1)/2)*((j-1)/2+1),((j-1)/4))
    else:
        a = 0
    print(str(j)+" "+str(a))
```
