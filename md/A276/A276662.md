from sympy import primepi, primefactors, factorint
def A276662(n):
    if n == 1: return 311
    m, k = n, primepi(n) + n + (n>>1)
    while m != k:
        m, k = k, primepi(k) + n + (k>>1)
    while sum((f:=factorint(m)).values()) > 1:
        m = int(str(p:=min(f))+str(m//p))
    return m # _Chai Wah Wu_, Aug 02 2024
