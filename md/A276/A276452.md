## Python
```Python
import math
def nCr(n,r):
    f = math.factorial
    return f(n) / f(r) / f(n-r)
# main program
for j in range(101):
    a = nCr(j*j,j)
    i = j/2
    if j%2==0:
        b = nCr(2*i*i,i)
    else:
        b = nCr(2*i*(i+1),i)
    print(str(j)+" "+str((a-b)/4))
```
