## Python
```Python
from sympy import primepi, isprime
def A196274(n):
    if n == 1: return 4
    m, k = n-1, primepi(n) + n - 1 + (n>>1)
    while m != k:
        m, k = k, primepi(k) + n - 1 + (k>>1)
    for d in range(1,4):
        if not isprime(m+(d<<1)):
            return d # _Chai Wah Wu_, Jul 31 2024
```
