## Maple
```Maple
Digits:=10^3:
f:=proc(z,c,k) option remember;
  f(z,c,k-1)^2+c;
end;
a:=proc(n)
local epsilon, c, k;
  epsilon:=10.^(-n):
  c:=0.25+epsilon:
  f(0,c,0):=0:
  for k do
    if abs(f(0,c,k))>2 then
      break;
    fi;
  od:
  return(k);
end;
seq(a(n),n=0..11);
```
