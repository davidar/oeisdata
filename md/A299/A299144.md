p0=0
p1=1
def GCD(x,y):
    tmp = y
    y = x % y
    if y==0: return tmp
    return GCD(tmp, y)
n=0
for i in range(1,1000000):
    p0,p1 = p1, p0+p1
    for x in range(1000000):
        if GCD(p0,i+x)==1: break
    for j in range(n, x):

    if x>n: n=x
