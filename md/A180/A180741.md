## Python
```Python
from sympy import isprime
def afind(limit, startk=1):
    pow5, pow3 = 5**startk, 3**startk
    for k in range(startk, limit+1):
        if isprime(pow5 + pow3 - 1): print(k, end=", ")
        pow5 *= 5; pow3 *= 3
afind(1000) # _Michael S. Branicky_, Aug 21 2021
```
