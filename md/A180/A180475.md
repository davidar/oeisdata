## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen(): # generator of terms
    p = 2
    while True:
        t = p**5 + p**3 + 1
        if isprime(t):
            yield t
        p = nextprime(p)
print(list(islice(agen(), 21))) # _Michael S. Branicky_, Mar 12 2022
```
