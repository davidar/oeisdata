## Python
```Python
from sympy import isprime, nextprime, prevprime
def ok(n):
  if n <= 5: return n == 5
  return not isprime(n//2) and n == prevprime(n//2) + nextprime(n//2)
def a(n):
  k, pow3 = 1, 3**n
  while not ok(k*pow3): k += 1
  return k
print([a(n) for n in range(68)]) # _Michael S. Branicky_, May 04 2021
```
