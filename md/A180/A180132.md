## Python
```Python
from sympy import nextprime, prevprime
def sum2succ(n): return n == prevprime(n//2) + nextprime(n//2)
def a(n):
  if n < 2: return [5, 1][n]
  k, pow5 = 1, 5**n
  while not sum2succ(k*pow5): k += 1
  return k
print([a(n) for n in range(65)]) # _Michael S. Branicky_, May 01 2021
```
