## Python
```Python
from gmpy2 import t_mod, mpz
from operator import mul
from functools import reduce
A180484 = [mpz(n) for n in (str(x) for x in range(1,10**9)) if not
..........(n.count('0') or t_mod(mpz(n)**2*len(n),
..........reduce(mul, (mpz(d) for d in n))**2))]
```
