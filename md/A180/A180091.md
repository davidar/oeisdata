## Python

# The following program gives T(m,n)=a(m+1,n+1) for any m >= 0 and n >= 0:
def T(m,n):
    if n == 0:
        res = 1
    elif n == 1:
        res = max(m,n)
    elif m < n:
        res = T(n,m)
    else:
        res = 0
        for i in range(1,m+1):
            res += T(m-i,n-1)
        for j in range(2,n+1):
            res += T(m-1,n-j)
    return res # _Julien Rouyer_, Jun 02 2023
