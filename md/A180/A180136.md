## Python
```Python
from sympy import nextprime, prevprime
def sum2succ(n): return n == prevprime(n//2) + nextprime(n//2)
def a(n):
  if n == 0: return 5
  k, pow12 = 1, 12**n
  while not sum2succ(k*pow12): k += 1
  return k
print([a(n) for n in range(67)]) # _Michael S. Branicky_, May 01 2021
```
