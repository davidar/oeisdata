## Python
```Python
from __future__ import division
from sympy.ntheory.primetest import is_square
from sympy import totient, nextprime
A291340_list, p = [2], 3
while p < 10**6:
    if is_square(8*(p-1)*totient((p-1)//2)+1):
        A291340_list.append(p)
    p = nextprime(p)
```
