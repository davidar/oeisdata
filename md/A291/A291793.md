def step(w):
    i = 0
    while w[0] != alfabet[i]:
        i = i+1
    w = w+suffix[i]
    return w[n:len(w)]
alfabet, suffix, n, ws, w0, m = "01", ["00","1101"], 3, "100", "", 0
while m < 83:
    w0, m = w0+ws, m+1
    w, ww, i, a = w0, w0, 0, 0
    while w != "" and a == 0:
        w, i = step(w), i+1
        if i%1000 == 0:
            ww = w
        else:
            if w == ww or w == "":
                if w != "":
                    a = i%1000
                print(m,a) # _A.H.M. Smeets_, Jul 16 2020
