def u(n):
  if n < 0:
    return 0
  if n <= 1:
    return 1
  sum = 0
  sum -= (4 * n - 1) * u(n - 1)
  sum += n * (2 * n - 3) * (10 * n - 9) * u(n - 2)
  sum += 5 * (2 * n - 3) * (2 * n - 4) * (2 * n - 5) * u(n - 3)
  sum -= 2 * (2 * n - 3) * (2 * n - 4) * (2 * n - 5) * (2 * n - 6) * (2 * n - 7) * u(n - 4)
  return sum / (n + 1)
for i in range(1, 13):
