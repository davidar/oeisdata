#program to get all terms less than one million
def pattern(p):
    l=len(p)
    s=""
    for k in range(l-1):
        if p[k+1]>p[k]: s=s+"+"
        elif p[k+1]<p[k]: s=s+"-"
        else: s=s+"="
    return(s)
def is_prime(num):
    for k in range(2,num):
       if (num % k) == 0:
           return(0)
    return(1)
for i in range(1000,999999):
    if len(str(i)) % 2 == 0:
        p1=str(i)[0:int(len(str(i))/2)]
        p2=str(i)[int(len(str(i))/2):len(str(i))]
        if pattern(p1)==pattern(p2) and is_prime(i):  print(i)
