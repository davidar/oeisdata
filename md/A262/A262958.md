## Python

from gmpy2 import digits
def f1(n):
    s = digits(n,3)
    m = len(s)
    for i in range(m):
        if s[i] == '0':
            return(int(s[:i]+'1'*(m-i),3))
    return n
def f2(n):
    s = digits(n,4)
    m = len(s)
    for i in range(m):
        if s[i] == '0':
            return(int(s[:i]+'1'*(m-i),4))
        if s[i] == '2':
            return(int(s[:i]+'3'+'1'*(m-i-1),4))
    return n

for i in range(10**4):
