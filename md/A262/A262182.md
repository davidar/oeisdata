def prime(n):
    if n == 1: return True
    d = n + 1
    c = n - 1
    while c > 0 and d % c:
        d += n
        c -= 1
    return bool(c == 1)
n = 1
i = 1
while i <= 500:
    target = (i * (i + 1)) // 2
    if prime(target):
        print(n, target*target+1)
        n += 1
    i += 1
