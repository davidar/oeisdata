def A262038(n):
    sl = len(str(n))
    l = sl>>1
    if sl&1:
        w = 10**l
        n2 = w*10
        for y in range(n//(10**l),n2):
            k, m = y//10, 0
            while k >= 10:
                k, r = divmod(k,10)
                m = 10*m + r
            z = y*w + 10*m + k
            if z >= n:
                return z
    else:
        w = 10**(l-1)
        n2 = w*10
        for y in range(n//(10**l),n2):
            k, m = y, 0
