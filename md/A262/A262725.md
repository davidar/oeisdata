from math import factorial
def aupto(n):
    f = [1, 1]
    for D in range(1, 5):
        for j in range(1, D + 1):
            sign = (-1) ** j
            f.extend(sign * f[k] for k in range(1, factorial(D) + 1))
    return f[1:n] # _Paul Muljadi_, Sep 26 2024
