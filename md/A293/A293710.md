from sympy import expand
# also generates the terms a(n), where n < 0. For example a(-1) = 1, a(-2)= -4, ...
def a(n):
    if n == 0:
       return 0
    elif n == 1:
       return 0
    elif n== 2:
       return 1
    elif n < 0:
       return expand(a(n+3)- 4*a(n+2) - 4*a(n+1))
    else:
       return expand(4*a(n-1) + 4*a(n-2) + a(n-3))
m1 = 0
m2 = 25
for i in range (m1, (m2)+1):
    print(a(i), end=', ')
