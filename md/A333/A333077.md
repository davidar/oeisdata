## Python
```Python
import networkx as nx
def deBruijn(n):
    return nx.MultiDiGraph(((0, 0), (0, 0))) if n==0 else nx.line_graph(deBruijn(n-1))
def A333077(n):
    return sum(1 for _ in nx.find_cliques(nx.complement(nx.Graph(deBruijn(n)))))
```
