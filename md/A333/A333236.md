from sympy import n_order, multiplicity
def A333236(n):
    m2, m5 = multiplicity(2,n), multiplicity(5,n)
    return int(max(str(10**(max(m2,m5)+n_order(10,n//2**m2//5**m5))//n))) # _Chai Wah Wu_, Feb 07 2022
