def IsPrime(n):
    if n < 2:
        return 0
    elif n == 2 or n == 3:
        return 1
    elif n%2 == 0 or n%3 == 0:
        return 0
    else:
        d, dd = 5, 2
        while d*d <= n and n%d != 0:
            d, dd = d+dd, 6-dd
        if d*d <= n:
            return 0
        else:
            return 1
