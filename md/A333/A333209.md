from math import gcd
f0, f1, g0, g1, snum, sden, n = 1, 1, 1, 2, 0, 1, 0
while n < 12:
    n = n+1
    snum, sden = g0*g1*snum+sden, sden*g0*g1
    d = gcd(snum,sden*f0)
    print(n,sden*f0//d)
