def moda(n,a)
    kk,j = 0,1
    while n > 0:
        kk= kk-j*(n%10)**a
        n,j =int(n//10),-j
    return abs(kk)
for i in range (0,10**12):
    for t in range(2,21):
        if i==moda(i,t):
            print (i,t,moda(i,t))
