from math import gcd
f0, f1, snum, sden, n = 1, 1, 0, 1, 0
while n < 13:
    snum, sden, n = f0*f1*snum+sden, sden*f0*f1, n+1
    d = gcd(snum, sden*f0)
    print(n, sden*f0//d)
    f0, f1 = 2*f0+f1, f0+f1 # _A.H.M. Smeets_, May 16 2020
