from sympy import n_order, multiplicity
def A333442(n):
    if n == 1: return 0
    m2, m5 = multiplicity(2,n), multiplicity(5,n)
    r = max(m2,m5)+n_order(10,n//2**m2//5**m5)
    s = str(10**r//n).zfill(r)
    return s.index(max(s))+1 # _Chai Wah Wu_, Feb 07 2022
