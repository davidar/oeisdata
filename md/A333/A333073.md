## Python
```Python
from sympy import primorial, lcm
def A333073(n):
    f = 1
    for i in range(1,n+1):
        f = lcm(f,i)
    f = int(f)
    glist = []
    for i in range(1,n+1):
        glist.append(f//i)
    m = 1 if n < 2 else primorial(n,nth=False)//primorial(n//2,nth=False)
    k = m
    while True:
        p,ki = 0, -k
        for i in range(1,n+1):
            p = (p+ki*glist[i-1]) % f
            ki = (-k*ki) % f
        if p == 0:
            return k
        k += m # _Chai Wah Wu_, Apr 02 2020
```
