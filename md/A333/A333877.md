## Python
```Python
from sympy import isprime
from sympy.utilities.iterables import multiset_permutations
def A333877(n):
    for i in range(n-1,-1,-1):
        q = 2**n-1
        for d in multiset_permutations('0'*i+'1'*(n-1-i)):
            p = q-int(''.join(d),2)
            if isprime(p):
                return p # _Chai Wah Wu_, Apr 08 2020
```
