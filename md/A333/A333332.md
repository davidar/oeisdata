def closest_powers_of_2_to_10(n):
  smallest_error = 1
  a = []
  r = 0.2 # ratio test starts at 2/10
  k = 1
  while len(a) < n:
    error = abs(1-r)
    if error < smallest_error:
      smallest_error = error
      a.append(k)
      print(a)
    if r<1.0:
      r *= 2
    else:
      r /= 10
      k -= 1 # need to check the other power of 10
    k += 1
  return a
print(closest_powers_of_2_to_10(20))
