## Python
```Python
from functools import reduce
from operator import mul
from sympy import primefactors, factorial
def A333074(n):
    f, g = int(factorial(n)), []
    for i in range(n+1):
        g.append(int(f//factorial(i)))
    m = 1 if n < 2 else reduce(mul, primefactors(n))
    k = m
    while True:
        p, ki = 0, 1
        for i in range(n+1):
            p = (p+ki*g[i]) % f
            ki = (-k*ki) % f
        if p == 0:
            return k
        k += m # _Chai Wah Wu_, Apr 01 2020
```
