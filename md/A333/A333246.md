## Python
```Python
# Using graphillion
from graphillion import GraphSet
import graphillion.tutorial as tl
def A333246(n):
    universe = tl.grid(n - 1, n - 1)
    GraphSet.set_universe(universe)
    cycles = GraphSet.cycles().including(1)
    return cycles.len()
print([A333246(n) for n in range(2, 10)])
```
