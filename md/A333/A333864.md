## Python
```Python
# Using graphillion
from graphillion import GraphSet
import graphillion.tutorial as tl
def A333864(n):
    universe = tl.grid(n - 1, 2 * n - 1)
    GraphSet.set_universe(universe)
    cycles = GraphSet.cycles(is_hamilton=True)
    return cycles.len()
print([A333864(n) for n in range(2, 8)])
```
