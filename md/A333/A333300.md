## Python
```Python
def ok(n):
    base = 2
    while True:
        base3, base2 = base**3, base**2
        if base3 + base > n: return False
        for a in range(1, base):
            for b in range(base):
                if a == b: continue
                t = a*(base3 + base) + b*(base2 + 1)
                if t == n: return True
                elif t > n: break
        base += 1
print([k for k in range(751) if ok(k)]) # _Michael S. Branicky_, Oct 30 2021
```
