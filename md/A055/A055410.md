## C
```C
int A055410(int i)
{
    const int ring = i*i;
    int result = 0;
    for(int a = -i; a <= i; a++)
        for(int b = -i; b <= i; b++)
            for(int c = -i; c <= i; c++)
                for(int d = -i; d <= i; d++)
                    if ( ring >= a*a + b*b + c*c + d*d ) result++;
    return result;
} /* _Oskar Wieland_, Apr 08 2013 */
```
