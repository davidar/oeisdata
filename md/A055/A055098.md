## Python

from math import factorial, prod
def a(n):
    s = str(n); d, c = len(s), [s.count(str(i)) for i in range(10)]
    return (d-c[0])*factorial(d-1)//prod(map(factorial, c))
print([a(n) for n in range(1, 50)]) # _Michael S. Branicky_, Aug 24 2022
