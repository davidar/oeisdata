A055231 := proc(n)
    a := 1 ;
    if n > 1 then
        for f in ifactors(n)[2] do
            if op(2, f) = 1 then
                a := a*op(1, f) ;
            end if;
        end do:
    end if;
    a ;
end proc: # _R. J. Mathar_, Dec 23 2011
