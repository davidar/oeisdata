from math import gcd
def A055935_list(nmax):
    A = [1,1]
    for n in range(2,nmax+1):
        if n%2 > 1:
            A.append(2)
        else:
            A.append(sum(i for i in A if gcd(n,i) ==1))
    return A # _John Tyler Rascoe_, Jun 12 2024
