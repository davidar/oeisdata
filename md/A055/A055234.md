## Python
```Python
from math import prod
from itertools import count
from sympy import factorint
def A055234(n):
    for m in count(1):
        f = factorint(m)
        if n*m*prod((p-1)**2 for p in f)==prod(p**(e+2)-p for p,e in f.items()):
            return m # _Chai Wah Wu_, Aug 12 2024
```
