## Python
```Python
from sympy import isprime, prime
def a(n, wlimit=1000):
    if n == 2: return 0
    pn, w = str(prime(n)), 1
    while not isprime(int(pn+"9"*w)):
        w += 1
        if w > wlimit: return -1
    return int(pn+"9"*w)
print([a(n) for n in range(1, 26)]) # _Michael S. Branicky_, May 07 2022
```
