using Nemo, Primes
function A195441(n::Int)
    n < 4 && return ZZ([1,1,2,1][n+1])
    P = primes(2, div(n+2, 2+n%2))
    prod([ZZ(p) for p in P if p <= sum(digits(n+1, base=p))])
end
