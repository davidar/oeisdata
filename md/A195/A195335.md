## Maple
```Maple
read("transforms") ;
A195335 := proc(n)
        option remember;
        local prev,nxt,a ;
        if n =1 then
                2;
        else
                prev := procname(n-1) ;
                for nxt from 10^(n-1) to 10^n-1 do
                        if isprime(nxt) then
                                a := digcat2(prev,nxt) ;
                                if isprime(a) then
                                        return a ;
                                end if;
                        end if;
                end do:
                return -1 ;
        end if;
end proc: # _R. J. Mathar_, Sep 20 2011
```

## Python
```Python
from sympy import isprime, nextprime
def alst(nn):
  alst, astr = [2], "2"
  for n in range(2, nn+1):
    p = nextprime(10**(n-1))
    while not isprime(int(astr + str(p))): p = nextprime(p)
    alst.append(int(astr + str(p))); astr += str(p)
  return alst
print(alst(11)) # _Michael S. Branicky_, Dec 26 2020
```
