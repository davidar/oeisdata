## Python
```Python
from sympy import prime, isprime
def ok(n):
    i=1
    x=prime(i)
    while x<n:
        if isprime(n - x): return False
        i+=1
        x=prime(i)
    return True
print([n for n in range(1, 301) if ok(n)]) # _Indranil Ghosh_, Apr 29 2017
```
