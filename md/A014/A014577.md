## C++

/* code from the fxt library, about 5 CPU cycles per computation */
bool bit_paper_fold(ulong k)
{
  ulong h = k & -k; /* == lowest_one(k) */
  k &= (h<<1);
  return ( k==0 );
} /* _Joerg Arndt_, Apr 15 2010 */
