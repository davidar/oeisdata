## C
```C
unsigned hakmem175(unsigned x) {
    unsigned s, o, r;
    s = x & -x;  r = x + s;
    o = r ^ x;  o = (o >> 2) / s;
    return r | o;
}
unsigned A014311(int n) {
    if (n == 1) return 7;
```
