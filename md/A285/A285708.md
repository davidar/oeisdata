from sympy import divisors, gcd

def a007947(n): return max(i for i in divisors(n) if core(i) == i)
def a079277(n):
    k=n - 1
    while True:
        if a007947(k*n) == a007947(n): return k
        else: k-=1
def a285707(n): return 1 if n==1 else gcd(n, a079277(n))
