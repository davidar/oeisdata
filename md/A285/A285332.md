from operator import mul
from sympy import prime, primefactors
def a007947(n): return 1 if n<2 else reduce(mul, primefactors(n))
def a019565(n): return reduce(mul, (prime(i+1) for i, v in enumerate(bin(n)[:1:-1]) if v == '1')) if n > 0 else 1 # This function from _Chai Wah Wu_
def a065642(n):
    if n==1: return 1
    r=a007947(n)
    n = n + r
    while a007947(n)!=r:
        n+=r
    return n
def a(n):
    if n<2: return n + 1
    if n%2==0: return a019565(a(n//2))
    else: return a065642(a((n - 1)//2))
print([a(n) for n in range(51)]) # _Indranil Ghosh_, Apr 18 2017
