## Python
```Python
from sympy import fibonacci, gcd, lcm
def z(n):
    k=1
    while fibonacci(k)%n: k+=1
    return k
def l(n): return lcm(n, z(n))
def g(n): return gcd(n, fibonacci(n))
print([n for n in range(1, 201) if n==g(l(n))]) # _Indranil Ghosh_, Apr 09 2017
```
