from sympy import factorint, prime, primepi
from operator import mul
from functools import reduce
def a003961(n):
    f=factorint(n)
    return 1 if n==1 else reduce(mul, [prime(primepi(i) + 1)**f[i] for i in f])
def a242378(k, n):
    while k>0:
        n=a003961(n)
        k-=1
    return n
l=[2]
for n in range(1, 7):
    x=l[n - 1]
    l.append(x*a242378(n, x))
