## Python
```Python
from sympy import isprime
def heptagonal(n): return n*(5*n-3)//2
def aupto(limit):
  alst, n, hn = [], 1, heptagonal(1)
  while hn < limit:
    if isprime(hn+1): alst.append(hn+1)
    n, hn = n+1, heptagonal(n+1)
  return alst
print(aupto(94771)) # _Michael S. Branicky_, Feb 19 2021
```
