## Python
```Python
from sympy.ntheory import factorint,isprime
def all_prime(l):
    for i in l:
        if not(isprime(i)): return(False)
    return(True)
def all_equal(l):
    ll=len(l)
    set_l=set(l)
    lsl=list(set_l)
    llsl=len(lsl)
    return(llsl==1)
def pal(l):
    return(l == l[::-1])
n=350000
r=""
lp=[]
lexp=[]
def calc(n):
    global lp,lexp
    a=factorint(n)
    lp=[]
    for p in a.keys():
        lp.append(p)
    lexp=[]
    for exp in a.values():
        lexp.append(exp)
    return
for i in range(4,n):
    calc(i)
    if len(lexp)>1:
        if all_prime(lexp):
            if not(all_equal(lexp)):
                if pal(lexp):
                    r += ","+str(i)
print(r[1:])
```
