def a322674(k):
  seq = []
  i = 0
  while len(seq) <= k:
    j = 0
    while len(seq) <= k and j < i + 1:
      n = i - j
      m = j
      decn = int(bin(n).replace('0b', ''))
      decm = int(bin(m).replace('0b', ''))
      binProd = bin(n * m).replace('0b', '')
      decProd = str(decn * decm)
      seq.append(int(binProd == decProd))
      j += 1
    i += 1
  print(seq)
