## Python
```Python
from sympy import isprime
def A322743(n):
    i = 4 if n <= 1 else 2**n+1
    j = 1 if n <= 2 else 2
    while True:
        if not isprime(i):
            c = 0
            for m in range(len(bin(i))-2):
                if isprime(i^(2**m)):
                    c += 1
                if c > n:
                    break
            if c == n:
                return i
        i += j # _Chai Wah Wu_, Jan 03 2019
```
