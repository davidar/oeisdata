## Python
```Python
from sympy import isprime
def ok(n):
    if not isprime(n): return False
    if n < 8: return True
    o = oct(n)[2:]
    oi = (o[:i]+o[i+1:] for i in range(len(o)))
    return any(t[0] != '0' and ok(int(t, 8)) for t in oi)
print([k for k in range(492) if ok(k)]) # _Michael S. Branicky_, Jan 13 2022
```
