## Maple
```Maple
isA005381 := proc(n)
    not isprime(n) and not isprime(n-1) ;
end proc:
A005381 := proc(n)
    local a;
    option remember;
    if n = 1 then
        9;
    else
        for a from procname(n-1)+1 do
            if isA005381(a) then
                return a;
            end if;
        end do:
    end if;
end proc: # _R. J. Mathar_, Jul 14 2015
```
