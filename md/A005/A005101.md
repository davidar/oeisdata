A005101 := proc(n)
    option remember ;
    local a ;
    if n =1 then
        12 ;
    else
        a := procname(n-1)+1 ;
        while numtheory[sigma](a) <= 2*a do
            a := a+1 ;
        end do ;
        a ;
    end if ;
