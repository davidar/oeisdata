## Python
```Python
def a(n):
    k, prevk, steps, seen = n, n, 0, set()
    while k != 1:
        q, r = divmod(k, 9)
        if r == 0: k = q
        elif  k not in seen: k = int(str(k) + str(9-r))//9
        else: k = int(str(9-r) + str(k))//9
        steps, seen, prevk = steps+1, seen|{prevk}, k
    return steps
print([a(n) for n in range(1, 74)]) # _Michael S. Branicky_, Jun 21 2021
```
