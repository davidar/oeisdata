from itertools import combinations
from math import prod
from sympy import factorint, divisors
from sympy.ntheory.modular import crt
def A344769(n):
    c = 2*n-bin(n).count('1')
    plist = [p**q for p, q in factorint(2*n).items()]
    if len(plist) == 1:
        return int(c+1+(plist[0] % 2 - 2)*n)
    return int(c-min(min(crt([m,2*n//m],[0,-1])[0],crt([2*n//m,m],[0,-1])[0]) for m in (prod(d) for l in range(1,len(plist)//2+1) for d in combinations(plist,l)))) # _Chai Wah Wu_, Jun 03 2021
