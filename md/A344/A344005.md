# To compute the first M terms, from _N. J. A. Sloane_, Jun 18 2021
with(NumberTheory);
M:=100000;
W:=Array(1..M,0);
W[1]:=1; W[2]:=1; Lo:=[1,2]; # divisors of m
for m from 2 to M do
Ln:=Divisors(m+1); # divisors of m+1
  for d1 in Lo do
    for d2 in Ln do
      d:=d1*d2;
      if d<=M and W[d]=0 then W[d]:=m; fi;
    od: # d2
  od: # d1
  Lo:=Ln;
od: # od m
WW:=[seq(W[i],i=1..100)];

## Python
```Python
from itertools import combinations
from math import prod
from sympy import factorint
from sympy.ntheory.modular import crt
def A344005(n):
    if n == 1:
        return 1
    plist = [p**q for p, q in factorint(n).items()]
    return n-1 if len(plist) == 1 else int(min(min(crt([m,n//m],[0,-1])[0],crt([n//m,m],[0,-1])[0]) for m in (prod(d) for l in range(1,len(plist)//2+1) for d in combinations(plist,l)))) # _Chai Wah Wu_, Jun 04 2021
```
