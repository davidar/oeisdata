from sympy.ntheory.modular import crt
from sympy import factorint
from math import prod
from itertools import combinations
def A344763(n):
    plist = tuple(p**q for p, q in factorint(2*n).items())
    return 1-n if len(plist) == 1 else n-int(min(min(crt((m,2*n//m),(0,-1))[0],crt((2*n//m,m),(0,-1))[0]) for m in (prod(d) for l in range(1,len(plist)//2+1) for d in combinations(plist,l)))) # _Chai Wah Wu_, Jun 15 2022
