def divisibility_checking_number(a):
    if a%5 == 0 or a % 2 == 0:
        return(0)
    x = 1
    while (x*10 + 1)%a != 0 :
        x += 1
    return(x)
h = []
for i in range(1,250):
    if divisibility_checking_number(i) != 0:
        h.insert(len(h),divisibility_checking_number(i))
print(h)
