## Python
```Python
def a(n):
    d = {}
    for i in range(2, n+1):
        tmp = i
        j = 2
        while(tmp != 1):
            if(tmp % j == 0):
                d.setdefault(j, 0)
                tmp //= j
                d[j] += 1
            else:
                j += 1
    res = 1
    for i in d.values():
        res *= (i+1)
    return res // 2
```
