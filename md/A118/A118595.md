## Python

```Python
from gmpy2 import digits
def A118595(n):
    if n == 1: return 0
    y = (x:=1<<(n.bit_length()-2&-2))<<2
    return int((s:=digits(n-x,4))+s[-2::-1] if n<x+y else (s:=digits(n-y,4))+s[::-1]) # _Chai Wah Wu_, Jun 14 2024
```
