## Python
```Python
from sympy import integer_log
from gmpy2 import digits
def A118598(n):
    if n == 1: return 0
    y = 7*(x:=7**integer_log(n>>1,7)[0])
    return int((s:=digits(n-x,7))+s[-2::-1] if n<x+y else (s:=digits(n-y,7))+s[::-1]) # _Chai Wah Wu_, Jun 14 2024
```
