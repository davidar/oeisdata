from math import isqrt
from sympy import integer_nthroot, factorint
def A118896(n):
    m = 10**n
    return sum(isqrt(m//x**3) for x in range(1,integer_nthroot(m,3)[0]+1) if max(factorint(x).values(),default=0)<=1) # _Chai Wah Wu_, May 13 2023

# faster program
def A118896(n):
    def squarefreepi(n): return int(sum(mobius(k)*(n//k**2) for k in range(1, isqrt(n)+1)))
    m, c, l = 10**n, 0, 0
    j = isqrt(m)
    while j>1:
        k2 = integer_nthroot(m//j**2,3)[0]+1
        w = squarefreepi(k2-1)
        c += j*(w-l)
        l, j = w, isqrt(m//k2**3)
    c += squarefreepi(integer_nthroot(m,3)[0])-l
    return c # _Chai Wah Wu_, Sep 09 2024
