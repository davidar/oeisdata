def A118599(n):
    if n == 1: return 0
    y = (x:=1<<(m:=n.bit_length()-2)-m%3)<<3
    return int((s:=oct(n-x)[2:])+s[-2::-1] if n<x+y else (s:=oct(n-y)[2:])+s[::-1]) # _Chai Wah Wu_, Jun 14 2024
