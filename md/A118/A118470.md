## Python

def sd(n): return sum(map(int, str(n)))
def ok(n): return sd(n) + sd(n**2) + sd(n**3) == sd(n**4)
print([k for k in range(20000) if ok(k)]) # _Michael S. Branicky_, Dec 25 2021
