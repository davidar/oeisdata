from itertools import count, islice
def A081706_gen(): # generator of terms
    for n in count(0):
        if (n&-n).bit_length()&1:
            m = n<<2
            yield m-2
            yield m-1
A081706_list = list(islice(A081706_gen(),30)) # _Chai Wah Wu_, Jan 09 2023
