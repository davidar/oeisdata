## Python
```Python
from math import ceil
from fractions import Fraction
def a(n):
  b0 = b = Fraction((2*n+1), 2)
  while b.denominator != 1: b = b0*ceil(b)
  return b.numerator
print([a(n) for n in range(1, 43)]) # _Michael S. Branicky_, Mar 20 2021
```
