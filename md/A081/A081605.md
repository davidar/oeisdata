from itertools import count, islice
def A081605_gen(): # generator of terms
    a = -1
    for n in count(2):
        b = int(bin(n)[3:],3) + (3**(n.bit_length()-1)-1>>1)
        yield from range(a+1,b)
        a = b
A081605_list = list(islice(A081605_gen(),30)) # _Chai Wah Wu_, Oct 13 2023
