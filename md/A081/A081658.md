## Python
```Python
from functools import cache
@cache
def T(n: int, k: int) -> int:
    if k == 0: return 1
    if k % 2 == 1:  return 0
    if k == n: return -sum(T(n, j) for j in range(0, n - 1, 2))
    return (T(n - 1, k) * n) // (n - k)
for n in range(10):
    print([T(n, k) for k in range(n + 1)])  # _Peter Luschny_, Jun 05 2023
```
