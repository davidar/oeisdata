## Python
```Python
def A081134(n):
    kmin, kmax = 0,1
    while 3**kmax <= n:
        kmax *= 2
    while True:
        kmid = (kmax+kmin)//2
        if 3**kmid > n:
            kmax = kmid
        else:
            kmin = kmid
        if kmax-kmin <= 1:
            break
    return min(n-3**kmin, 3*3**kmin-n) # _Chai Wah Wu_, Mar 31 2021
```
