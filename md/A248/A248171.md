## Python
```Python
from itertools import count
def evil(n): return bin(n)[2:].count('1') % 2 == 0
def aupton(terms):
    alst, t = [], 0
    for k in count(1):
        t = int(str(t) + str(k))
        if evil(t):
            alst.append(t)
            t = 0
            if len(alst) >= terms: return alst
print(aupton(45)) # _Michael S. Branicky_, Dec 03 2021
```
