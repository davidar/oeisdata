## Python
```Python
from math import prod
def ok(n):
    s = str(n); d = list(map(int, s))
    if '0' in s: return False
    return set(s) == set(str(n-sum(d))) and set(s) == set(str(n-prod(d)))
print(list(filter(ok, range(127000)))) # _Michael S. Branicky_, Jul 18 2021
```
