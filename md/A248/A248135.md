## Python
```Python
from sympy import isprime, factorint
A248135_list = []
for n in range(1, 10**6):
    if not isprime(n):
        a = sum([int(n*e/p) for p, e in factorint(n).items()]) if n > 1 else 0
        if not sum(a % i for i in range(1, a)) % n:
            A248135_list.append(n)
```
