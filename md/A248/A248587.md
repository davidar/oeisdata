## Python
```Python
from sympy import nextprime, prevprime
A248587_list = []
for i in range(3,10**6):
    n = i**3
    p3 = prevprime(n//4)
    p2, p4 = prevprime(p3), nextprime(p3)
    p1 = prevprime(p2)
    q = p1+p2+p3+p4
    while q <= n:
        if q == n:
            A248587_list.append(p1)
        p1, p2, p3, p4 = p2, p3, p4, nextprime(p4)
        q = p1+p2+p3+p4 # _Chai Wah Wu_, Dec 31 2015
```
