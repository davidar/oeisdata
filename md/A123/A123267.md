```Python
from num2words import num2words
from itertools import islice
def n2w(n): return num2words(n, ordinal=True).replace(" and", "")
def agen(): # generator of terms
    s, idx = "t is the ", 0
    while True:
        idx = 1 + s.index("t", idx)
        yield idx
        s += n2w(idx) + ", "
print(list(islice(agen(), 56))) # _Michael S. Branicky_, Mar 18 2022
```
