## Python
```Python
from sympy.ntheory.digits import digits
def distinct(n, b): d = digits(n, b); return len(d) == len(set(d))
def a(n):
    if n == 1: return 1
    b = 2
    while not distinct(n, b): b += 1
    return b
print([a(n) for n in range(1, 106)]) # _Michael S. Branicky_, Jun 15 2022
```
