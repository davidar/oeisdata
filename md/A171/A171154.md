## Python
```Python
from sympy import isprime, primerange, prime
def a(n):
    if n == 1: return 2
    c = int("".join(map(str, [p for p in primerange(2, prime(n)+1)][::-1])))
    pow10 = 10
    while True:
        c *= 10
        for b in range(1, pow10, 2):
            if b%5 == 0: continue
            if isprime(c+b):
                return c+b
        pow10 *= 10
print([a(n) for n in range(1, 17)]) # _Michael S. Branicky_, Mar 12 2022
```
