## Python
```Python
from itertools import count, islice
def A171912gen(): # generator of terms
    b, bdict = 2, {2:(1,)}
    for n in count(2):
        yield b
        if len(l := bdict[b]) > 1:
            b = n-1-l[-2]
        else:
            b = 0
        if b in bdict:
            bdict[b] = (bdict[b][-1],n)
        else:
            bdict[b] = (n,)
A171912_list = list(islice(A171912gen(),20)) # _Chai Wah Wu_, Dec 21 2021
```
