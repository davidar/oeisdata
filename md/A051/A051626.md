A051626 := proc(n) local lpow,mpow ;
    if isA003592(n) then
       RETURN(0) ;
    else
       lpow:=1 ;
       while true do
          for mpow from lpow-1 to 0 by -1 do
              if (10^lpow-10^mpow) mod n =0 then
                 RETURN(lpow-mpow) ;
              fi ;
          od ;
          lpow := lpow+1 ;
       od ;
    fi ;
end: # _R. J. Mathar_, Oct 19 2006

def A051626(n):
    if isA003592(n):
        return 0
    else:
        lpow=1
        while True:
            for mpow in range(lpow-1,-1,-1):
                if (10**lpow-10**mpow) % n == 0:
                    return lpow-mpow
            lpow += 1 # _Kenneth Myers_, May 06 2016
