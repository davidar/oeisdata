## Python
```Python
from sympy import isprime
from itertools import islice
def agen(): # generator of terms
    yield from [2, 3]
    s, an = 5, 4
    while True:
        while not isprime(s+an): an += 2
        yield an
        an, s = an+2, s+an
print(list(islice(agen(), 60))) # _Michael S. Branicky_, Oct 30 2022
```
