from sympy import nextprime; p, q, r, s, t = 2, 3, 5, 7, 11
while t < 580000:
    if p + t == q + s == 2*r: print(r, end = ', ')
    p, q, r, s, t = q, r, s, t, nextprime(t) # _Ya-Ping Lu_, May 11 2024
