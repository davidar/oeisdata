## Python
```Python
from sympy import nextprime
def aupton(terms):
  prv, cur, nxt, alst = 0, 2, 3, []
  while len(alst) < terms:
    alst.append(prv if 2*cur - prv <= nxt else nxt)
    prv, cur, nxt = cur, nxt, nextprime(nxt)
  return alst
print(aupton(58)) # _Michael S. Branicky_, Jun 04 2021
```
