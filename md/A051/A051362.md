## Sage
```Sage
def is_A051362(n):
    prime = is_prime(n)
    if prime:
        L = ZZ(n).digits(10)
        for k in range(len(L)):
            K = L[:]; del K[k]
            prime = is_prime(ZZ(K, base=10))
            if not prime: break
    return prime
A051362_list = lambda n: filter(is_A051362, range(n))
```

## Python
```Python
from sympy import isprime
def ok(n):
    if n < 10 or not isprime(n): return False
    s = str(n)
    return all(isprime(int(s[:i]+s[i+1:])) for i in range(len(s)))
print([k for k in range(10**5) if ok(k)]) # _Michael S. Branicky_, Nov 02 2023
```
