A051270 := proc(n)
    option remember;
    local a;
    if n = 1 then
        2*3*5*7*11 ;
    else
        for a from procname(n-1)+1 do
            if A001221(a)= 5 then
                return a;
            end if;
        end do:
    end if;
end proc: # _R. J. Mathar_, Oct 13 2019
