## Python
```Python
from sympy import factorint
def ok(n):
    if n == 1: return True
    f = factorint(n)
    p, e = f.keys(), f.values()
    if max(e) >= 3: return False
    return all((pi**k)%pj!=1 for pi in p for pj in p if pj!=pi for k in range(1, f[pi]+1))
print([k for k in range(1, 162) if ok(k)]) # _Michael S. Branicky_, Feb 20 2023
```
