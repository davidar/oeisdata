## Python

from math import prod
from sympy import factorint
def A350743(n):
    f = list(factorint(n).items())
    return sum(1 for k in range(1,n+1) if prod(p**((q+1)*k)-1 for p,q in f)//prod(p**k-1 for p,q in f) % k == 0) # _Chai Wah Wu_, Jan 17 2022
