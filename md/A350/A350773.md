from sympy import divisor_count as tau
taulist = [1, 2, 2, 3, 2, 4, 2]
for k in range(2, 10000000):
    taulist.append(tau(k+6))
    del taulist[0]
    if sum(taulist) == 32: print(k, end=", ") # _Karl-Heinz Hofmann_, Jan 15 2022
