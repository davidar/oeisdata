## Python
```Python
from math import prod
def hd(n):
    while (n&1) == 0:  n >>= 1
    while (n%3) == 0:  n /= 3
    while (n%5) == 0:  n /= 5
    while (n%7) == 0:  n /= 7
    return(n)
def pd(n): return prod(map(int, str(n)))
def ok(n):
    if hd(n) > 9: return False
    return (p := pd(n)) > 9 and (q := pd(p)) > 9 and (r := pd(q)) > 9 and (s := pd(r)) > 9 and pd(s) < 10
print([k for k in range(10,476200000) if ok(k)])
```
