from itertools import count, islice
from collections import Counter
def A350578_gen(): # generator of terms
    yield 0
    b, bcounter = 0, Counter({0})
    for n in count(1):
        b += -n if b-n >= 0 and bcounter[b-n] <= bcounter[b+n] else n
        bcounter[b] += 1
        yield b
