from functools import cache
from math import comb as binomial
@cache
def ptan(n):
    return (0 if n % 2 == 0 else
    -sum(binomial(n,k)*ptan(n-k) if k > 0 else 1 for k in range(0,n+1,2)))
def A350972(n):
    t = ptan(n)
    return -t if t < 0 else t
