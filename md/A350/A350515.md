## Python

def a(n):
    r = n%6
    if r == 1 or r == 4: return (n-1)//3
    if r == 0 or r == 2: return n//2
    if r == 3 or r == 5: return (3*n+1)//2
print([a(n) for n in range(70)]) # _Michael S. Branicky_, Jan 02 2022
