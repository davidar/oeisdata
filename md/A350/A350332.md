## Python

```Python
from sympy import integer_nthroot, primerange
def aupto(limit):
    aset, maxp = set(), integer_nthroot(limit, 3)[0]
    for p in primerange(3, maxp+1):
        pp = p*p
        for q in primerange(p+1, limit//pp+1):
            if (q-1)%p != 0:
                aset.add(pp*q)
    return sorted(aset)
print(aupto(2060)) # _Michael S. Branicky_, Dec 25 2021
```
