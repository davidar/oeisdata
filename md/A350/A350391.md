## Python
```Python
from fractions import Fraction
from itertools import count, islice
def agen():
    rats = {Fraction(0, 1)}
    for n in count(1):
        yield max(r.denominator for r in rats)
        newrats = set()
        for r in rats:
            newrats.add(1+r)
            if r != 0:
                newrats.add(-1/r)
        rats = newrats
print(list(islice(agen(), 25))) # _Michael S. Branicky_, Jan 17 2022
```
