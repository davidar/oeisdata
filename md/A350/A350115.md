## Python
```Python
from sympy import integer_nthroot, isprime, primerange
def aupto(limit):
    aset, maxp = set(), integer_nthroot(limit, 4)[0]
    for p in primerange(1, maxp+1):
        m = p**2
        for t in range(m, limit//m, m):
            if isprime(t+1):
                aset.add(p**2*(t+1))
    return sorted(aset)
print(aupto(1844)) # _Michael S. Branicky_, Dec 14 2021
```
