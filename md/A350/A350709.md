## Python
```Python
def a(n):
    d, m = list(map(int, str(n))), [0, 0, 0]
    for di in d: m[di%3] += 1
    return int(str(len(d)) + "".join(map(str, m)))
print([a(n) for n in range(37)]) # _Michael S. Branicky_, Mar 28 2022
```
