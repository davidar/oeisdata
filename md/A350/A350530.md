def A350530_col(k,nmax):
    d = []
    c = [0]*nmax
    while 1:
        if not d or all(d[-1][:-1]):
            if d and d[-1][-1] == 0:
                c[len(d)-1] += 1 + (0 != 2*d[0][0] != k+1)
            elif len(d) < nmax:
                d.append([-1])
                for i in range(len(d)-1):
                    d[-1].append(d[-1][-1]-d[-2][i])
        while d and d[-1][0] == k:
            d.pop()
        if not d or len(d) == 1 and 2*d[0][0] >= k: return c
        for i in range(len(d)):
            d[-1][i] += 1
