## Python
```Python
from itertools import islice
def c(s, t): return sum(t.count(si) for si in s)
def agen(): # generator of terms
    an, target = 1, "1"
    while True:
        yield an
        k = an + 1
        while c(str(k), target) != 1: k += 1
        an, target = k, str(k)
print(list(islice(agen(), 75))) # _Michael S. Branicky_, Dec 31 2021
```
