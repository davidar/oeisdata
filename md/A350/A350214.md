## Python
```Python
from sympy import isprime
def a(n):
    an = 0
    while True:
        an, k = an+1, 1
        while isprime(int("3"*k+str(an))): k += 1
        if k-1 == n: return an
print([a(n) for n in range(1, 8)]) # _Michael S. Branicky_, Dec 20 2021
```
