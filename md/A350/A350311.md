## Python
```Python
def Interpretation(n):
    f0, f1, f2, r = 1, 1, 1, 0
    while n > 0:
        if n%2 == 1:
            r = r+f0
        n, f0, f1, f2 = n//2, f0+f2, f0, f1
    return r
n = 0
while n <= 69:
    print(Interpretation(n), end = ", ")
    n += 1
```
