## Python
```Python
from sympy import isprime; R = []
for i in range(1, 100):
    m = 1; L = [m]; m = (m*m+1)%10**i
    while m not in L: L.append(m); m = (m*m+1)%10**i
    del L[:L.index(m)]; {R.append(j) for j in L if isprime(j) and j not in R}
R.sort(); print(*R, sep = ", ")
```
