## Python
```Python
from sympy import isprime
def ok(n):
    if n == 0: return True
    count = {"0": 0, "1": 0}
    for bit in bin(n)[:1:-1]:
        count[bit] += 1
        if count["0"] > count["1"]: return False
    return isprime(n)
print([k for k in range(3, 500, 2) if ok(k)]) # _Michael S. Branicky_, Jan 07 2022
```
