## Python
```Python
from sympy import nextprime
from itertools import islice, repeat
def primes(p=2):
    while True: yield p; p = nextprime(p)
def agen():
    primesum = 0
    for p in primes():
        primesum += p
        yield from repeat(primesum, p)
print(list(islice(agen(), 66))) # _Michael S. Branicky_, Dec 18 2021
```
