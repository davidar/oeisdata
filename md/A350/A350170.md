## Python
```Python
from sympy import nextprime
from itertools import islice
def primes(p=2):
    while True: yield p; p = nextprime(p)
def agen():
    seq1, seq2 = primes(), primes()
    while True:
        p, q = next(seq1), next(seq2)
        yield p
        for i in range(q):
            p = next(seq1)
print(list(islice(agen(), 43))) # _Michael S. Branicky_, Dec 18 2021
```
