## Python
```Python
from sympy import isprime
def ok(n):
    if not isprime(n): return False
    if n < 10: return True
    s, c, d = str(n), 0, None
    for i in range(len(s)):
        di = int(s[:i]+s[i+1:])
        if isprime(di):
            c += 1
            if c > 1:
                return False
            d = di
    return ok(d) and len(str(d)) == len(s) - 1
print([k for k in range(3260) if ok(k)]) # _Michael S. Branicky_, Dec 31 2021
```
