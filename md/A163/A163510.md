from itertools import count, islice
def A163510_gen(): # generator of terms
    for n in count(1):
        k = n
        while k:
            yield (s:=(~k&k-1).bit_length())
            k >>= s+1
A163510_list = list(islice(A163510_gen(),30)) # _Chai Wah Wu_, Jul 17 2023
