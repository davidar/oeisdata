## Python
```Python
from sympy import primepi
def A163636(n):
    if n == 1: return 1
    m, k, n2 = n-1, primepi(n) + n - 1 + (n>>1), (n<<1)-1
    while m != k:
        m, k = k, primepi(k) + n - 1 + (k>>1)
    return (lambda x: (x+n2)*(x-n2+2)>>2)(m) # _Chai Wah Wu_, Jul 31 2024
```
