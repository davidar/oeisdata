## Python
```Python
from sympy import integer_nthroot
def a(n):
  target = 2 * int(str(n) + '0'*(n-len(str(n))))
  r, exact = integer_nthroot(target, 2)
  while r*(r+1)//2 >= target//2: r -= 1
  return (r+1)*(r+2)//2
print([a(n) for n in range(1, 21)]) # _Michael S. Branicky_, Jan 28 2021
```
