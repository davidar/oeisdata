from itertools import count, islice
def A077384_gen(): # generator of terms
    yield (a:=1)
    for n in count(2,4):
        yield (a:=a+n)
        yield (a:=a*(n+1))
        yield (a:=a-n-2)
        yield (a:=a*(n+3))
A077384_list = list(islice(A077384_gen(),20)) # _Chai Wah Wu_, Apr 19 2023
