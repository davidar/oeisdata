## Python
```Python
from sympy import isprime
def aupton(terms):
  k, alst = 0, []
  while len(alst) < terms:
    k += 1
    t = sum(5**i*int(di) for i, di in enumerate((bin(k)[2:])[::-1]))
    if isprime(t): alst.append(t)
  return alst
print(aupton(37)) # _Michael S. Branicky_, May 31 2021
```
