def b(n): return int("".join(map(str, range((n-1)*n//2+1, n*(n+1)//2+1))))
def a(n): return b(n+1)//b(n)
print([a(n) for n in range(1, 44)]) # _Michael S. Branicky_, Jul 31 2022
