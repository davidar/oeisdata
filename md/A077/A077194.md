def A077194(n):
    def traverse(remainder, d):
        add1, number = d
        return [((remainder*10)%n, (add1, number*10)),
                ((remainder+1)%n, (add1+1, number+1))]
    return dijkstra([(n+1, 0)]*n, 1%n, (1, 1), traverse)[0][1]  # _Jason Yuen_, Feb 28 2024
