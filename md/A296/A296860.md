## Python
```Python
def cwo(subs, s): # count with overlaps allowed
  c = i = 0
  while i != -1:
    i = s.find(subs, i)
    if i != -1: c += 1; i += 1
  return c
def ok(n): b = bin(n)[2:]; return cwo('101', b) < cwo('010', b)
print(list(filter(ok, range(1, 298)))) # _Michael S. Branicky_, May 11 2021
```
