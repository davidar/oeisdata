## Python
```Python
from itertools import filterfalse, count, combinations_with_replacement as cwr
n = 6
x = set([0,1])
for i in range(n):
    x = set([a[0] + a[1] for a in cwr(x,2)]).union(set([a[0]*a[1] for a in cwr(x,2)]))
    print(next(filterfalse(lambda b: b in x, count())))
```
