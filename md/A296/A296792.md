from itertools import accumulate, count, islice
def A296792_gen(): # generator of terms
    blist, m = tuple(), 1
    for i in count(1,2):
        yield (blist := tuple(accumulate(reversed(blist),initial=m)))[-1]
        m *= i
A296792_list = list(islice(A296792_gen(),30)) # _Chai Wah Wu_, Jun 11 2022
