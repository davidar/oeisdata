## Python
```Python
from gmpy2 import is_prime, gcd, mpz
A287198_list, n  = [], 2
while n <= 10**6:
    s = str(n)
    if not is_prime(n) and '0' not in s:
        k = n
        for i in range(len(s)-1):
            s = s[1:]+s[0]
            m = mpz(s)
            if is_prime(m) or gcd(k,m) > 1:
                break
            k *= m
        else:
            A287198_list.append(n)
    n += 1 # _Chai Wah Wu_, May 27 2017
```
