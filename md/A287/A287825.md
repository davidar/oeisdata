def a(n):
    if n in [0, 1, 2, 3, 4, 5]:
        return [1, 10, 82, 674, 5540, 45538][n]
    return 9*a(n-1) - 4*a(n-2) - 21*a(n-3) + 9*a(n-4) + 5*a(n-5)
