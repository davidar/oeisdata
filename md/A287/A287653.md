```Maple
N:= 100: # to get a(1) - a(N)
p:= 2: q:= 3: r:= 5: s:= 7:
count:= 0:
while count < N do
  p:= q; q:= r; r:= s; s:= nextprime(s);
  n:= p*q+q*r+r*s;
  if isprime(n) then count:= count+1; A[count]:= n fi
od:
seq(A[i],i=1..N); # _Robert Israel_, May 29 2017
```
