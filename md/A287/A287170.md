## Python
```Python
from sympy import factorint, primepi
def a087207(n):
    f=factorint(n)
    return sum([2**primepi(i - 1) for i in f])
def a069010(n): return sum(1 for d in bin(n)[2:].split('0') if len(d)) # this function from _Chai Wah Wu_
def a(n): return a069010(a087207(n)) # _Indranil Ghosh_, Jun 06 2017
```
