## Python
```Python
from gmpy2 import isqrt, mpz, digits
def A287298(n): # assumes n <= 62
    m = isqrt(mpz(''.join(digits(i,n) for i in range(n-1,-1,-1)),n))
    m2 = m**2
    d = digits(m2,n)
    while len(set(d)) < len(d):
        m -= 1
        m2 -= 2*m+1
        d = digits(m2,n)
    return int(m2) # _Chai Wah Wu_, May 24 2017
```
