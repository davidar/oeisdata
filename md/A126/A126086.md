## Python
```Python
# Naive version - see link for better version.
def f(a, b):
    if a == 0 or b == 0:
        return 1
    return f(a, b - 1) + f(a - 1, b) + f(a - 1, b - 1)
def g(a, b, c):
    if a == 0:
        return f(b, c)
    if b == 0:
        return f(c, a)
    if c == 0:
        return f(a, b)
    return (
        g(a, b, c - 1)
        + g(a, b - 1, c)
        + g(a - 1, b, c)
        + g(a, b - 1, c - 1)
        + g(a - 1, b, c - 1)
        + g(a - 1, b - 1, c)
        + g(a - 1, b - 1, c - 1)
    )
for n in range(6):
    print(g(n, n, n), end=", ")
```
