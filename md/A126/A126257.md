def A126257(n):
    if n:
        s, c = (1,), {1}
        for i in range(n-1):
            c.update(set(s:=(1,)+tuple(s[j]+s[j+1] for j in range(len(s)-1))+(1,)))
        return len(set((1,)+tuple(s[j]+s[j+1] for j in range(len(s)-1))+(1,))-c)
