def A126120_list(n) :
    D = [0]*(n+2); D[1] = 1
    b = True; h = 2; R = []
    for i in range(2*n-1) :
        if b :
            for k in range(h,0,-1) : D[k] -= D[k-1]
            h += 1; R.append(abs(D[1]))
        else :
            for k in range(1,h, 1) : D[k] += D[k+1]
        b = not b
    return R
