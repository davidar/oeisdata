## Python
```Python
from math import sqrt
def is_square(n):
  root = int(sqrt(n))
  return root*root == n
def find_sols(length):
    count = 0
    k=10**length
    for n in range(k,4*k-2):
        discr= (2*k-1)*(2*k-1) - 4*(k*(k-1)-(k-1)*n)
        if is_square(discr):
            count+=1
            b=(-(2*k-1)+sqrt(discr))/2
            print(n, k+b, n/(k+b))
    return count
for i in range(8):
    print(find_sols(i))
```
