## Python
```Python
def stepCount(x):
    x = int(x)
    steps = 0
    while True:
        if x == 1:
            break
        elif x % 2 == 0:
            x = x/2
            steps += 1
        else:
            x = x*3 + 1
            steps += 1
    return steps
n = 1
while True:
    print(stepCount(n) % n)
    n += 1
```
