## Python

```Python
from sympy import factorint, nextprime
def A288507(n):
    k, p, q = 1, 2, 3
    while True:
        if sum(factorint(q-p).values()) == n and sum(factorint(q+p).values()) == n:
            return k
        k += 1
        p, q = q, nextprime(q) # _Chai Wah Wu_, Jul 23 2017
```
