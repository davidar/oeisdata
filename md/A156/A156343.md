## Python
```Python
from sympy import isprime
def ok(n):
    if not isprime(n): return False
    s, counts = str(n), [0, 0]
    for c in s: counts[int(c in "2357")] += 1
    return counts[0] == counts[1]
print([k for k in range(10**4) if ok(k)]) # _Michael S. Branicky_, Apr 23 2024
```
