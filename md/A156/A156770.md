## Python
```Python
from sympy import isprime
from itertools import islice
def agen():
    an = 1
    while True:
        yield an
        s, an = str(an), an+1
        while not isprime(int(s+str(an))): an += 1
print(list(islice(agen(), 56))) # _Michael S. Branicky_, Oct 17 2022
```
