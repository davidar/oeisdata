from sympy import divisors
def ok(n): return any(n==d*int(s[::-1]) for d in divisors(n)[1:-1] if (s:=str(d))!=s[::-1] and s[-1]!="0")
print([k for k in range(36000) if ok(k)]) # _Michael S. Branicky_, Sep 07 2024
