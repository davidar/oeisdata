## Python
```Python
from sympy import prime, primepi
def A129145(n):
    if n&1: return prime((n>>1)+2)
    if n==2: return 9
    r = n>>1
    m, k = r, primepi(r) + r + (r>>1)
    while m != k:
        m, k = k, primepi(k) + r + (k>>1)
    return m # _Chai Wah Wu_, Aug 01 2024
```
