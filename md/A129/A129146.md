from sympy import primepi, prime
def A129146(n):
    if n == 1: return -6
    m, k = n, primepi(n) + n + (n>>1)
    while m != k:
        m, k = k, primepi(k) + n + (k>>1)
    return prime(n+1)-m # _Chai Wah Wu_, Aug 01 2024
