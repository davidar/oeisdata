N = 10^6;
Avail = ones(1,N);
A = zeros(1,N);
A(1) = 1; A(2) = 2;
Avail([1,2]) = 0;
for n=3:N
  q = round(A(n-1)/gcd(A(n-1),A(n-2)));
  b = find(Avail(q*[1:floor(N/q)]),1,'first');
  if numel(b) == 0
     break
  end
  A(n) = q*b;
  Avail(A(n)) = 0;
end
