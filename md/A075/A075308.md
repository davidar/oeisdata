from sympy import mobius, integer_nthroot
def A075308(n): return int(sum(mobius(x)*(integer_nthroot(10**(n-1),x)[0]-integer_nthroot(10**n,x)[0]) for x in range(2,((10**(n-1)).bit_length())))-sum(mobius(x)*(integer_nthroot(10**n,x)[0]-1) for x in range((10**(n-1)).bit_length(),(10**n).bit_length()))) if n>2 else n<<2 # _Chai Wah Wu_, Aug 13 2024
