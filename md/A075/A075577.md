## Python
```Python
from sympy import isprime
def aupto(limit):
  alst, is2 = [], False
  for k in range(1, int(limit**.5) + 2):
    is1, is2 = is2, isprime(k**2 + (k+1)**2)
    if is1 and is2: alst.append(k**2)
  return alst
print(aupto(1500000)) # _Michael S. Branicky_, Apr 25 2021
```
