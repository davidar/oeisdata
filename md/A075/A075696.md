## Python
```Python
from sympy import prime, primerange
def auptopn(lim):
    alst = []
    for k, pk in enumerate(primerange(2, prime(lim)+1), start=1):
        if str(sum(map(int, str(k)))) in str(pk): alst.append(k)
    return alst
print(auptopn(405)) # _Michael S. Branicky_, Jul 03 2021
```
