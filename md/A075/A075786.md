## Python
```Python
from math import isqrt
def ispal(n): s = str(n); return s == s[::-1]
def athrough(digits):
  found, limit = {1}, 10**digits
  for k in range(2, isqrt(limit) + 1):
    kpow = k*k
    while kpow < limit:
      if ispal(kpow): found.add(kpow)
      kpow *= k
  return sorted(found)
print(athrough(9)) # _Michael S. Branicky_, Mar 23 2021
```
