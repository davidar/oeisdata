from sympy import prime, primerange
def auptopn(lim):
    alst = []
    for k, pk in enumerate(primerange(2, prime(lim)+1), start=1):
        sdk =  sum(map(int, str(k)))
        if str(sdk) in str(pk): alst.append(sdk)
    return alst
print(auptopn(1010)) # _Michael S. Branicky_, Jul 03 2021
