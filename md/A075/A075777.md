```Maple
A:= Vector(N,1)*6*N;

  for p2 from p1 to floor(N/p1) do
     for p3 from p2 to floor(N/p1/p2) do
        n:= p1*p2*p3;
```

## Python
```Python
import math
def cubestepdown(n):
    s1_0 = int(math.ceil(n ** (1 / 3.0)))
    minSA = -1
    s1 = s1_0
    while s1>=1:
        while n % s1 > 0:
            s1 = s1 - 1
        s1quot = int(n/s1)
        s2_0 = int(math.ceil(math.sqrt(n/s1)))
        s2 = s2_0
        while s2>=1:
            while s1quot % s2 > 0:
                s2 = s2 - 1
            s3 = int(n / (s1 * s2))
            SA = 2*(s1*s2 + s1*s3 + s2*s3)
            if minSA==-1:
                minSA = SA
            else:
                if SA<minSA:
                    minSA = SA
            s2 = s2 - 1
        s1 = s1 - 1
    return minSA
# _Scott B. Farrar_, Sep 29 2015
```
