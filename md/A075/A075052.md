## Python
```Python
from math import isqrt
from sympy import integer_nthroot
def A075052(n):
    def f(x): return n-1+x+integer_nthroot(x,6)[0]-(b:=integer_nthroot(x,3)[0])-(a:=isqrt(x)), a, b
    m = n-1
    k, a, b = f(n-1)
    while m != k:
        m = k
        k, a, b = f(k)
    return min((a+1)**2,(b+1)**3)-m # _Chai Wah Wu_, Aug 09 2024
```
