from sympy import primepi
def depth(k):
    d = 0
    while k > 1:
        k -= primepi(k)
        k += primepi(k)
        d += 1
    return d
m = 1
for n in range (0, 101):
    a = 0
    while depth(m + a) == n:
        a += 1
    print(a)
    m += a
