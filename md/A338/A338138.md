myList = [1]
roots = [1]
s = 1
t = 0
for n in range(9999):
    b = 2
    while t == 0:
        if(b**3-s > 0 and not b**3-s in myList):
            myList.append(b**3-s)
            s = b
            roots.append(s)
            t = 1
        else:
            b += 1
    t=0
print("roots: ",roots)
