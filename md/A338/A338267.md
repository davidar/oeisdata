from sympy import prime, integer_nthroot
def A338267(n):
    p, q, r = prime(n)**2, prime(n+1)**2, prime(n+2)**2
    return (integer_nthroot(4*p*q-(p+q-r)**2,2)[0]+2)//4 # _Chai Wah Wu_, Oct 19 2020
