## Python
```Python
def A338420(N):
    return list(filter(isA338420,range(1,N+1)))
def isA338420(n):
    counter=0
    if n==2 or n==4:
        return True
    if n%2==0:
        counter=1
    for b in range(3,(n//2) +1):
        if (n+1)%b!=0:
            counter=main_base_check(int(n/b),b)+counter
    return counter==1
def main_base_check(m,b):
    while m!=0:
        if m%b == b-1:
            return 1
        m = m//b
    return 0
```
