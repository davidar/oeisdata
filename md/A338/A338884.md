## Python
```Python
from sympy import isprime
for n in range(1,101):
    a = 0
    k = i = 1
    while isprime(i) == 0:
        a += 1
        k = 2*k
        for i in range(k*n + 1, k*n + k):
            if isprime(i) == 1: break
    print(a)
```
