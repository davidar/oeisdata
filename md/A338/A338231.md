## Python
```Python
from sympy import divisor_sigma, integer_nthroot
from sympy.ntheory.factor_ import core
def A338231(n):
    return n*(n+1)//2-divisor_sigma(integer_nthroot(n//core(n,2),2)[0]) # _Chai Wah Wu_, Jan 31 2021
```
