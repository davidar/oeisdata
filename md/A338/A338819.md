## Python
```Python
import numpy as np
def n_id_sequence(iterations):
    sequence = []
    for i in range(1,iterations+1):
        matrix = i*(np.identity(i))
        for row in matrix:
            for entry in row:
                sequence.append(int(entry))
    return sequence
```
