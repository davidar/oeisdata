from collections import Counter
from sympy import factorint
def A338713(n):
    f = Counter(factorint(n))
    return sum(1 for i in range(1,n+1) if (l:=tuple((f+Counter(factorint(i))).values()))==(1,1) or l==(3,)) # _Chai Wah Wu_, Oct 14 2023
