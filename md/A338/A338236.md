## Python

from sympy import divisor_count, integer_nthroot
from sympy.ntheory.factor_ import core
def A338236(n):
    return integer_nthroot(n,2)[0]-divisor_count(integer_nthroot(n//core(n,2),2)[0]) # _Chai Wah Wu_, Jan 31 2021
