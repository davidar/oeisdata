## Python
```Python
from sympy import prevprime
def ok(n):
    if n < 2: return False
    p, q, s = [prevprime(n+1)] * 3
    while True:
        if p == 2: return False
        if s == n: return True
        elif s < n: p = prevprime(p); s += p
        else: s -= q; q = prevprime(q)
print([k for k in range(150) if ok(k)]) # _Michael S. Branicky_, Feb 21 2022 after _Alois P. Heinz_
```
