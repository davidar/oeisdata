from collections import Counter
from sympy import factorint
def A338714(n):
    f = Counter(factorint(n))
    return sum(1 for i in range(1,n+1) if (l:=tuple(sorted((f+Counter(factorint(i))).values())))==(1,2) or l==(5,)) # _Chai Wah Wu_, Oct 14 2023
