## Python
```Python
from sympy import divisor_count
def A284597(n):
    count, starti, s, i = 0,1,0,1
    while True:
        d = divisor_count(i)
        if d < s:
            if count == n:
                return starti
            starti = i
            count = 0
        s = d
        i += 1
        count += 1 # _Chai Wah Wu_, May 04 2017
```
