from functools import lru_cache
@lru_cache(maxsize=None)
def a(n):
    if n <= 0: return 0
    if n < 17:
        return [7, 26, 8, 8, 8, 8, 8, 26, 7, 8, 16, 16, 16, 16, 16, 26][n-1]
