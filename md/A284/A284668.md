def steps(n):
if n==1:
  return 0
else:
  if (n%2)==0:
  return 1+steps(int(n/2))
  else:
  return 1+steps(3*n+1)
def max_steps(i):
    a=max([[i,steps(i)] for i in range(1,10**(i))],key=lambda x:x[1])
    return a[0]
