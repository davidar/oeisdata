## Python
```Python
from sympy import lcm
def a003418(n): return 1 if n<1 else lcm(range(1, n + 1))
def a(n):
    m=1
    while True:
        if a003418(m)%n==0: return m
        else: m+=1
print([n//a(n) for n in range(1, 101)]) # _Indranil Ghosh_, Apr 04 2017
```
