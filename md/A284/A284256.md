## Python
```Python
from sympy import primefactors
def A(n):
    pf = primefactors(n)
    if pf: min_pf2 = min(pf)**2
    for i in pf:
        if i > min_pf2: return i
    return 1
def a(n): return 0 if A(n)==1 else 1 + a(n//A(n))
print([a(n) for n in range(1, 151)]) # _Indranil Ghosh_, Mar 24 2017
```
