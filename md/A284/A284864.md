from sympy import binomial, factorial
def T(n, k): return sum([binomial(k, j)*(-1)**(j - k)*(2 + 3*j)**n/factorial(k) for j in range(k + 1)])
def a(n): return sum([T(n, k) for k in range(n + 1)]) # _Indranil Ghosh_, Apr 10 2017
