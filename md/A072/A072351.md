      f:= `if`(n=1, [1$2], F(n-1));
      do f:= [f[2], f[1]+f[2]];
         if length(f[1])<length(f[2]) then break fi
      od; f
    end:

def A072351_list(n):
    list = [1]
    x, y = 1, 1
    while len(list) < n:
        if len(str(x)) < len(str(y)):
            list.append(y)
        x, y = y, x + y
    return list
