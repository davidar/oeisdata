## Python
```Python
from math import isqrt
from sympy import mobius, integer_nthroot, perfect_power
def A072776(n):
    def g(x): return int(sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1)))-1
    def f(x): return n-2+x-sum(g(integer_nthroot(x,k)[0]) for k in range(1,x.bit_length()))
    kmin, kmax = 1,2
    while f(kmax) >= kmax:
        kmax <<= 1
    while True:
        kmid = kmax+kmin>>1
        if f(kmid) < kmid:
            kmax = kmid
        else:
            kmin = kmid
        if kmax-kmin <= 1:
            break
    return 1 if not (p:=perfect_power(kmax)) else p[1] # _Chai Wah Wu_, Aug 19 2024
```
