## Python
```Python
def A072353_list(n):
    list = []
    x, y, index = 1, 1, 1
    while len(list) < n:
        if len(str(x)) < len(str(y)):
            list.append(index)
        x, y = y, x + y
        index += 1
    return list
print(A072353_list(57)) # _M. Eren Kesim_, Jul 19 2021
```
