## Python
```Python
from sympy import isprime
B=[3, 1, 2, 2, 3, 2, 3, 4, 3, 2]
def a(n):
    s=0
    for i in str(n):
        s+=B[int(i)]
    return s+4
i=0
j=1
while j<=10000:
    if isprime(a(i))==True :
        print str(j)+" "+str(i)
        j+=1
    i+=1
```
