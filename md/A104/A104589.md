from sympy import isprime
from itertools import islice
def A104589_gen(): # generator of terms
    a, b = 1, 1
    while True:
        yield a
        a += b
        b += a if isprime(a) else 0
A104589_list = list(islice(A104589_gen(),50)) # _Chai Wah Wu_, Jun 03 2024
