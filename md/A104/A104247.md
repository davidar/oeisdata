## Python
```Python
from sympy import isprime, nextprime
def sd(n): return sum(map(int, str(n)))
def aupto(limit):
    alst, k, p, s = [], 1, 2, 2
    while s <= limit:
        if isprime(s): alst.append(s)
        k += 1; p = nextprime(p); s += sd(p)
    return alst
print(aupto(2579)) # _Michael S. Branicky_, Jul 18 2021
```
