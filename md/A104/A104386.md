## Python
```Python
from itertools import count, islice
from sympy import isprime, prevprime, primepi
def agen():
    for d in count(1):
        ru = int("1"*d)
        for r in range(ru, 10*ru, ru):
            if r > 2:
                p = prevprime(r)
                if isprime(r + (r-p)) and prevprime(r+(r-p)) == p:
                    yield primepi(p)
print(list(islice(agen(), 7))) # _Michael S. Branicky_, Jun 30 2022
```
