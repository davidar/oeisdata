## Python
```Python
from sympy import binomial
def A001263(n,m):
    return binomial(n-1,m-1)*binomial(n,m-1)//m
def A104711(n,m):
    a = 0
    for k in range(m,n+1):
        a += A001263(k,m)
    return a
print([A104711(n,m) for n in range(20) for m in range(1,n+1)]) # _R. J. Mathar_, Oct 11 2009
```
