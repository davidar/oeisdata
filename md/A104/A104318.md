## Python
```Python
from sympy import isprime
def aupton(terms):
    alst, i, s2 = [], 1, "1"
    while len(alst) < terms:
        i += 1
        s1, s2 = s2, str(i**3)
        t = int(s2+s1)
        if isprime(t):
            alst.append(t)
    return alst
print(aupton(16)) # _Michael S. Branicky_, Jan 08 2022
```
