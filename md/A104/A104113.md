## Python
```Python
def expr(t, d): # can you express target t with digits d, only adding +'s
    if t < 0: return False
    if t == int(d): return True
    return any(expr(t-int(d[:i]), d[i:]) for i in range(1, len(d)))
def aupto(limit):
    alst, k, k2 = [], 0, 0
    while k2 <= limit:
        if expr(k, str(k2)):
            alst.append(k2)
        k, k2 = k+1, k2 + 2*k + 1
    return alst
print(aupto(7500000)) # _Michael S. Branicky_, Sep 27 2021
```
