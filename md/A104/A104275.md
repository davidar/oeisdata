from sympy import primepi
def A104275(n):
    if n <= 2: return ((n-1)<<2)+1
    m, k = n-1, (r:=primepi(n-1)) + n - 1 + (n-1>>1)
    while m != k:
        m, k = k, (r:=primepi(k)) + n - 1 + (k>>1)
    return r+n-1 # _Chai Wah Wu_, Aug 02 2024
