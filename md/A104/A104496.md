## Python
```Python
from itertools import accumulate
def A104496_list(size):
    if size < 1: return []
    L, accu = [1], [1]
    for n in range(size-1):
        accu = list(accumulate(accu + [-accu[0]]))
        L.append(-(-1)**n*accu[-1])
    return L
print(A104496_list(29)) # _Peter Luschny_, Apr 25 2016
```
