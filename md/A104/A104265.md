from sympy import integer_nthroot
def A104265(n):
    m, a = integer_nthroot((10**n-1)//9,2)
    if not a:
        m += 1
    k = m**2
    while '0' in str(k):
        m += 1
        k += 2*m-1
    return k # _Chai Wah Wu_, Mar 24 2020
