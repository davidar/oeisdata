def aupton(terms):
  c, k, kk = [0 for i in range(terms)], 1, 1
  while kk < 10**terms:
    s = str(kk)
    c[len(s)-1], k, kk = c[len(s)-1] + (s.count('0')==0), k+1, kk + 2*k + 1
  return c
print(aupton(14)) # _Michael S. Branicky_, Mar 06 2021
