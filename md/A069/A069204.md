## Python
```Python
from sympy import isprime
def ok(n): return isprime(n) and isprime(r:=int(str(n)[::-1])) and r!=n and n%4==r%4
print([k for k in range(3100) if ok(k)]) # _Michael S. Branicky_, Aug 12 2023
```
