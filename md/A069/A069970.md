## Python
```Python
from itertools import count, islice
from sympy import divisor_sigma as sigma
def agen(): # generator of terms
    dr2, dr1 = 1, 3
    for k in count(3):
        dr = sigma(int(str(k)[::-1]))
        if dr == dr1 + dr2:
            yield k
        dr2, dr1 = dr1, dr
print(list(islice(agen(), 25))) # _Michael S. Branicky_, May 29 2024
```
