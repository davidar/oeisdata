## Python
```Python
from sympy import isprime
def aupton(terms):
    alst, astr = [8], '8'
    for n in range(2, terms+1):
        an = 1
        while not isprime(int(astr + str(an))): an += 2
        alst, astr = alst + [an], astr + str(an)
    return alst
print(aupton(62)) # _Michael S. Branicky_, Aug 03 2021
```
