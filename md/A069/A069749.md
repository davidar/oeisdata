## Python
```Python
from sympy import isprime
from itertools import count, islice, product
def agen(): # generator of terms
    c = 2
    for d in count(2):
        yield c
        for first in product("23", repeat=d-1):
            t = int("".join(first) + "3")
            if isprime(t): c += 1
print(list(islice(agen(), 20))) # _Michael S. Branicky_, May 23 2024
```
