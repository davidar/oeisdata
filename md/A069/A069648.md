import sympy
from sympy import factorint
def DigitSum(x):
    return sum(int(i) for i in str(x))
def PowExp(p):
    n = 2
    while n < 10000*(10**(int(2**p/9)/p)):
        if DigitSum(n**p) != 1:
            count = 0
            for i in list(factorint(DigitSum(n**p)).values()):
                if (int(i)/p) % 1 == 0:
                    count += 1
            if count == len(list(factorint(DigitSum(n**p)).values())):
                return n
            else:
                n += 1
        else:
            n += 1
print(1)
x = 2
while x < 20:
    print(PowExp(x))
    x += 1
