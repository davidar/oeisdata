## Python
```Python
from sympy import divisor_count
from itertools import count, islice
def agen(an=1, startn=1):
    for n in count(startn):
        while divisor_count(an) != n: an += 1
        yield an
print(list(islice(agen(), 22))) # _Michael S. Branicky_, Jan 16 2022
```
