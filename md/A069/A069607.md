## Python
```Python
from sympy import isprime
def aupton(terms):
  astr, alst = '5', [5]
  for n in range(2, terms+1):
    an = 1
    while not isprime(int(astr + str(an))): an += 1
    astr, alst = astr + str(an), alst + [an]
  return alst
print(aupton(63)) # _Michael S. Branicky_, May 03 2021
```
