## Python
```Python
from sympy import nextprime
from itertools import islice
def agen(): # generator of terms
    p, hp, q, hq = 2, "2", 3, "3"
    while True:
        if hp == hq: yield p
        p, q = q, nextprime(q)
        hp, hq = hq, "".join(sorted(str(q)))
print(list(islice(agen(), 38))) # _Michael S. Branicky_, Feb 19 2024
```
