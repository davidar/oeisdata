## Python

from sympy import sqrt
import math
def T(n, k): return int(math.ceil((n + 1)/k - (k + 1)/2))
for n in range(1, 21): print([T(n, k)**2 for k in range(1, int(math.floor((sqrt(8*n + 1) - 1)/2)) + 1)]) # _Indranil Ghosh_, Apr 25 2017
