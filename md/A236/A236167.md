## Python
```Python
from sympy import isprime
def afind(startat=0, limit=10**9):
  pow47 = 47**startat
  for k in range(startat, limit+1):
    q, r = divmod(pow47+1, 48)
    if r == 0 and isprime(q): print(k, end=", ")
    pow47 *= 47
afind(limit=300) # _Michael S. Branicky_, May 19 2021
```
