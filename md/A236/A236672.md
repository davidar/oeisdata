## Maple
```Maple
R:= 9: x:= 9:
for i from 2 to 20 do
  for y from 1 by 2 do
    z:= x*10^(1+ilog10(y)) + y;
    if isprime(z) then
      R:= R,z; x:= z; break
    fi
od od:
R; # _Robert Israel_, Nov 22 2023
```

## Python
```Python
import sympy
from sympy import isprime
def b(x):
  num = str(x)
  n = 1
  while n < 10**3:
    new_num = str(x) + str(n)
    if isprime(int(new_num)):
      print(int(new_num))
      x = new_num
      n = 1
    else:
      n += 1
b(9)
```
