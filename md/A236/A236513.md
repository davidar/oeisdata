## Python
```Python
from itertools import combinations
from sympy import isprime
def A236513(n):
    l, k, c = n-1, 2**n, 0
    while True:
        for d in combinations(range(l-1,-1,-1),l-n+1):
            m = k-1 - sum(2**(e) for e in d)
            if isprime(m):
                c += 1
                if c == n:
                    return m
        l += 1
        k *= 2 # _Chai Wah Wu_, Sep 02 2021
```
