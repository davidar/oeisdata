## Maple
```Maple
A100612 := proc(n)
    local p,lf,kf,k ;
    p := ithprime(n) ;
    lf := 1 ;
    kf := 1 ;
    for k from 1 to p-1 do
        kf := modp(kf*k,p) ;
        lf := lf+modp(kf,p) ;
    end do:
    lf mod p ;
end proc:
for n from 1 do
    p := ithprime(n) ;
    rp := A100612(n) ;
    prp := p-rp ;
    if min(rp,prp) <= 2 then
        print(p) ;
    end if;
end do: # _R. J. Mathar_, Feb 17 2014
```

## Python
```Python
from sympy import isprime
def afind(limit):
    f = 1 # (p-1)!
    s = 2 # sum(0! + 1! + ... + (p-1)!)
    for p in range(2, limit+1):
        if isprime(p):
            r_p = s%p
            if min(r_p, p-r_p) <= 2:
                print(p, end=", ")
        s += f*p
        f *= p
afind(11000) # _Michael S. Branicky_, Jan 03 2022
```
