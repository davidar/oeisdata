A065608:= Vector(N):
for a from 1 to floor(sqrt(N)) do for b from a to N/a do
   if b = a then
     A065608[a*b] := A065608[a*b] + a - 1
   else
     A065608[a*b] := A065608[a*b] + a + b - 2;
   fi
od od:
