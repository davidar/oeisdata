```Python
#Generates and prints a list containing the terms, up to the term with the index of seq_limit
seq_limit=9
seq_list=[1,1]
for seq_no in range(3,seq_limit):
    seq_list.append((seq_list[-1]+seq_list[-2])**(seq_no-2))
print(seq_list)
```
