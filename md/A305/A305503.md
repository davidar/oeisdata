## Python
```Python
import numpy as np
import itertools
def findsubsets(S, m):
    return itertools.combinations(S, m)
def mstd(a):
    a1 = set()
    a2 = set()
    for i in a:
        for j in a:
            a1.add(i + j)
            a2.add(i - j)
    return len(a1) > len(a2)
def a(n):
    ans = 0
    Nn = list(range(n))
    for k in range(1, n):
        if any(mstd(i) for i in findsubsets(Nn, k)):
            ans = k
    return ans
```
