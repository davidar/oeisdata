## Python
```Python
from sympy import isprime
def ok(n):
    if not isprime(n): return False
    if n < 10: return True
    s = str(n)
    si = (s[:i]+s[i+1:] for i in range(len(s)))
    return any(t[0] != '0' and ok(int(t)) for t in si)
print([k for k in range(440) if ok(k)]) # _Michael S. Branicky_, Jan 28 2023
```
