from sympy import binomial
def T(n, k): return binomial(n + 1, k)*binomial(2*(n + 1), n - k)//(n + 1)
for n in range(21): print([T(n, k) for k in range(n + 1)]) # _Indranil Ghosh_, Nov 07 2017
