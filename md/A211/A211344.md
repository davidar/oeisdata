Seq = sym(zeros(55,1)) ;
Filledlines = 0 ;
for m=1:10
    for n=1:m
        Sum = sym(0) ;
        for k=0:2^m-1
            if mod(  floor( k/2^(m-n) )  ,2) == 0
               Sum = Sum + 2^sym(k) ;
            end
        end
        Seq( Filledlines + n ) = Sum ;
    end
    Filledlines = Filledlines + m ;
end
