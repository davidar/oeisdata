## Python
```Python
from sympy import isprime
def A211000(nmax):
    walk, turn = [(0,0),(1,1)], 1
    for n in range(1,nmax):
        p1, p2 = walk[-2], walk[-1]
        if isprime(n): # Go straight
            walk.append((2*p2[0]-p1[0],2*p2[1]-p1[1]))
        else:          # Turn
            if n>5 and isprime(n-1): turn *= -1
            walk.append((p2[0]-turn*(p1[1]-p2[1]),p2[1]+turn*(p1[0]-p2[0])))
    return walk[:nmax+1]
```
