from __future__ import division
def A211034(n):
    x,y,z = n//3 + 1, (n-1)//3 + 1, (n-2)//3 + 1
    return x**2*y**2 + 2*x**2*y*z + x**2*z**2 + 2*x*y**3 + 6*x*y**2*z + 6*x*y*z**2 + 2*x*z**3 + 2*y**3*z + 2*y*z**3 # _Chai Wah Wu_, Nov 28 2016
