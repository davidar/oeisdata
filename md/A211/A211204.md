def isPrime(n):
    if (n%2==0): return n==2
    for i in range(3,int(n**0.5+1),2):
        if (n%i==0): return False
    return n>1
def nextPrime(n):
    n=n+1
    while not isPrime(n): n=n+1
    return n
def a(n):
    p,L,S=2,[],[]
    while len(L)<n-1:
        p,S1,i=nextPrime(p),S,0
        while (i<len(L)) and ((len(S[i])+2<L[i]) or (p%L[i] in S[i])):
            S1[i].add(p%L[i])
            i=i+1
        if i==len(L):
            S1.append(set(L))
            S=S1
            L.append(p)
    return p
