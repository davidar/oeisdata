from __future__ import division
def A211071(n):
    if n % 3 == 0:
        return 8*n**4//27
    elif n % 3 == 1:
        return (8*n**4 + 4*n**3 - 3*n**2 - 2*n - 7)//27
    else:
        return (8*n**4 + 8*n**3 - 12*n**2 - 16*n - 4)//27 # _Chai Wah Wu_, Nov 30 2016
