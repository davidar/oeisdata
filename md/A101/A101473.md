from itertools import accumulate, islice
def A101473_gen(): # generator of terms
    blist, a, b = tuple(), 0, 1
    while True:
        yield (blist := tuple(accumulate(reversed(blist),initial=a)))[-1]
        a, b = b, 2*a+b
A101473_list = list(islice(A101473_gen(),30)) # _Chai Wah Wu_, Jun 11 2022
