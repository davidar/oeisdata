## Python
```Python
from sympy import isprime, prime, primorial
def a(n):
    k, t = 1, primorial(n)**prime(n)
    while True:
        if isprime(k*t - 1): return k
        k += 1
print([a(n) for n in range(1, 15)]) # _Michael S. Branicky_, Jan 29 2022
```
