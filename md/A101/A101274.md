from itertools import count, islice
def A101274_gen(): # generator of terms
    aset1, aset2, alist, n = {1}, set(), [1], 1
    for k in count(2):
        bset2 = {k<<1}
        if (k<<1) not in aset2:
            for d in aset1:
                if (m:=d+k) in aset2:
                    break
                bset2.add(m)
            else:
                yield k-n
                n = k
                alist.append(k)
                aset1.add(k)
                aset2.update(bset2)
A101274_list = list(islice(A101274_gen(),30)) # _Chai Wah Wu_, Sep 05 2023
