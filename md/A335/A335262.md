## Python
```Python
def T(num_rows):
    t, s = 1, 1
    L, R = [0], [0]
    for n in range(1, num_rows):
        R.reverse()
        R.insert(0, t)
        L.extend(R)
        t, s = t+s+1, s+1
    return L
print(T(12))
```
