## Python
```Python
from sympy.ntheory import isprime as isp
i = []
for a in range(10,1000000):
    if isp(a):
        b = str(a)
        d=[]
        for c in range(0,len(b)-1):
            ee = abs(int(b[c])-int(b[c+1]))
            d.append(str(ee))
        f = ''.join(d)
        g = b[::-1]
        if isp(int(f)) and isp(int(g)):
            if len(b)<3:
                i.append(b)
            else:
                if f in i:
                    i.append(b)
print(','.join(i))
```
