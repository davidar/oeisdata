## Python

from sympy.ntheory.factor_ import digits
from functools import reduce
def drop(x,n,k):
  # Drop all digits k from x in base n.
  return reduce(lambda x,j:n*x+j if j!=k else x,digits(x, n)[1:],0)
def A335506(n):
  if n==0:
    return 1
  else:
    return drop(5*A335506(n-1),10,7)
