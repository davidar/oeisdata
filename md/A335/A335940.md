import numpy as np
def primeFactors(n):
    x=[]
    while n % 2 == 0:
        x.append(2),
        n = n / 2
    for i in range(3,int(np.sqrt(n))+1,2):
        while n % i== 0:
            x.append(i),
            n = n / i
    if n > 2:
        x.append(n)
    if len(x)==0:
        x.append(1)
    if len(x)!=1:
        y=x[-1]-x[0]
    else:
        y=x[0]
    return y

nums = list(range(1,101))
final=[]
for i in nums:
    final.append(primeFactors(i))
final = [int(i) for i in final]

from sympy import primefactors, isprime
def A335940(n):
    if isprime(n):
        return n
    else:
        pf = primefactors(n)
        return max(pf)-min(pf) # _Chai Wah Wu_, Jul 01 2020
