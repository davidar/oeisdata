## Python
```Python
import math
n_max = 1000000
a_last = 0
list1 = [a_last]
print(0)
for n in range(1, n_max+1):
    m = a_last - n
    if m >= 0 and m not in list1:
        a = m
    else:
        a = a_last + math.ceil(n/2)
    list1.append(a)
    print(a)
    a_last = a
```
