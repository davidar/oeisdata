## Python
```Python
from sympy import factorint
def A335645(n):
  d = 1
  while True:
    half = (d+1)//2
    for left in range(10**(half-1), 10**half):
      strleft = str(left)
      if d%2 == 0:
        m = int(strleft + strleft[::-1])
      else:
        m = int(strleft + (strleft[:-1])[::-1])
      if len(factorint(m)) == n:
        return m
    d += 1
print([A335645(n) for n in range(8)]) # _Michael S. Branicky_, Oct 02 2020
```
