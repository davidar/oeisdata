## Python
```Python
from math import prod, isqrt
from sympy import isprime, primepi, primerange, integer_nthroot, primeomega
def A335097(n):
    if n==1: return 0
    if isprime(n): return primepi(n)-1
    def g(x,a,b,c,m): yield from (((d,) for d in enumerate(primerange(b,isqrt(x//c)+1),a)) if m==2 else (((a2,b2),)+d for a2,b2 in enumerate(primerange(b,integer_nthroot(x//c,m)[0]+1),a) for d in g(x,a2,b2,c*b2,m-1)))
    return int(sum(primepi(n//prod(c[1] for c in a))-a[-1][0] for a in g(n,0,1,1,primeomega(n)))-1) # _Chai Wah Wu_, Aug 28 2024
```
