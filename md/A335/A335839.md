## Python
```Python
def digsum(n,b):
    s = 0
    while n > 0:
        n, d = n//b, n%b
        s = s+d
    return s
p = [2,3,5,7,11,13]
n, a = 0, 0
while n <= 20:
    s2, i = digsum(a,2), 1
    while i < len(p) and digsum(a,p[i]) == s2:
        i = i+1
    if i == len(p):
        print(a, end = ", ")
        n = n+1
    a = a+1 # _A.H.M. Smeets_, May 16 2021
```
