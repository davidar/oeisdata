## Python
```Python
from itertools import count, islice
def A176997_gen(startvalue=1): # generator of terms >= startvalue
    if startvalue <= 1:
        yield 1
    k = 1<<(s:=max(startvalue,1))-1
    for n in count(s):
        if k % n == 1:
            yield n
        k <<= 1
A176997_list = list(islice(A176997_gen(),30)) # _Chai Wah Wu_, Jun 30 2022
```
