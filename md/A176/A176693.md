from math import isqrt
def A176693(n):
    def f(x): return n-1+(x>>1)+(x&1)-(isqrt(x)+1>>1)
    m, k = n-1, f(n-1)
    while m != k: m, k = k, f(k)
    return m # _Chai Wah Wu_, Sep 18 2024
