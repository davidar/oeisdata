## Sage

```Sage
def b(n):
    if (n==0): return 0
    elif (n==1): return 1
    elif (n==2): return 4
    elif (n==3): return 13
    else: return 4*b(n-1) -3*b(n-2) +2*b(n-3) -b(n-4)
def T(n, k): return b(n) - b(k) - b(n-k) + 1
[[T(n, k) for k in (0..n)] for n in (0..12)] # _G. C. Greubel_, May 06 2019
```
