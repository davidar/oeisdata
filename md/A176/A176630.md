## Python
```Python
def agen():
  k = 0
  while True:
    strk = str(k)
    revstrk = strk[::-1]
    if revstrk != strk:
      if int(revstrk) == int((bin(k)[2:])[::-1], 2):
        yield k
    k += 1
g = agen()
print([next(g) for i in range(11)]) # _Michael S. Branicky_, Dec 29 2020
```
