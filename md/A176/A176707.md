## Python
```Python
from sympy import factorint
def aupton(terms):
  alst, m = [], 4
  while len(alst) < terms:
    f = factorint(m)
    if sum(f.values()) == 2: # semiprime
      alst.append(sum(sum(map(int, str(p))) for p in f.keys()))
    m += 1
  return alst
print(aupton(81)) # _Michael S. Branicky_, Feb 05 2021
```
