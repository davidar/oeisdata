## Python
```Python
def A308935(n):
    n2, m, m2 = n**2*(n**2+1), n+1, ((n+1)**2*((n+1)**2+1)) % (n**2*(n**2+1))
    while m2:
        m2, m = (m2 + 2*(2*m+1)*(m**2+m+1)) % n2, (m+1) % n2
    return m # _Chai Wah Wu_, Jul 01 2019
```
