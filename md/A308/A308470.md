## Python
```Python
from fractions import gcd
def A000010(n):
    if n == 1:
        return 1
    d, m = 1, 0
    while d < n:
        if gcd(d,n) == 1:
            m = m+1
        d = d+1
    return m
n = 0
while n < 30:
    n = n+1
    print(n,(gcd(A000010(n),4*n**2-1)-1)//2) # _A.H.M. Smeets_, Aug 18 2019
```
