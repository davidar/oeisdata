def Prog(length):
    L = 2
    seq = [0,0]
    while L < length:
        x = len(seq)-1
        while x > 0:
            if seq[-1] == seq[x-1]:
                m = len(seq)-x
                a_n = seq[L-1]
                seq.append(a_n+m)
                x = -1
            else:
                x -= 1
            if x == 0:
                seq.append(1)
        L += 1
    return seq
