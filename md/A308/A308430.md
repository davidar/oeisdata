## Python
```Python
import gmpy2
def dec2bin(x):
    return str(bin(x))[2:]
def digitBalance(string):
    s = 0
    for char in string:
        if int(char) > 0:
            s -= 1
        else:
            s += 1
    return s
N = 100 # number of terms
seq = [0]
prime = 2
for i in range(N-1):
    prime = gmpy2.next_prime(prime)
    binary = dec2bin(prime)
    truncated = binary[1:-1]
    term = seq[-1] + digitBalance(truncated)
    seq.append(term)
print(seq) # _Jonas K. Sønsteby_, May 27 2019
```
