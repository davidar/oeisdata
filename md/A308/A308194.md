from sympy import divisors
def A308194(n):
    c, x = 0, n
    while x != 5:
        d = divisors(x)
        l = len(d)
        x = d[(l-1)//2] + d[l//2]
        c += 1
    return c # _Chai Wah Wu_, Jun 14 2019
