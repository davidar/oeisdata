from itertools import islice
from sympy import divisors
def A308751_gen(): # generator of terms
    bset, s = {1}, 3
    yield 2
    while True:
        for d in divisors(s):
            if d not in bset:
                yield s//d
                bset.add(d)
                s += d
                break
A308751_list = list(islice(A308751_gen(),30)) # _Chai Wah Wu_, Jan 25 2022
