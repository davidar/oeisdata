## Python
```Python
from functools import reduce
from operator import mul
from sympy import prime, primefactors
def A308439(n):
    return min(primefactors(1 + reduce(mul,(prime(i+1) for i,j in enumerate(bin(n)[:1:-1]) if j == '1')))) # _Chai Wah Wu_, Jun 03 2019
```
