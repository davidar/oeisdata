def seqr(n):
    sequence = [n]
    i = 1
    while True:
        if n - i > 0 and n - i not in sequence:
            n -= i
            sequence.append(n)
        elif n + i not in sequence:
            n += i
            sequence.append(n)
        else:
            break
        i += 1
    return len(sequence)
print([seqr(n) for n in range(1000)])
