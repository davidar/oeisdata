## Python

from sympy import primepi
def A308575(n):
    n2, t1 = 2**(n-1), 0
    k = n2 - 1
    kp = primepi(k)
    kp2 = primepi(k+n2)-kp
    while kp2 < kp or t1 >= kp:
        k += n2
        t1, t2 = kp, kp2
        kp2 = primepi(k+n2) - kp2
        kp = t2
    return 2*kp # _Chai Wah Wu_, Jun 13 2019
