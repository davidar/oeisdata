## Maple
```Maple
f:= proc(n) local d,p,q;
  for d from 0 do
    p:= nextprime(n*10^d-1);
    do
      q:= nextprime(p);
      if q - p = 2*n then return p fi;
      if q >= (n+1)*10^d then break fi;
      p:= q;
    od;
  od;
end proc:
map(f, [$1..50]);
```
## Python
```Python
from sympy import nextprime
def A308438(n):
    l, p = 1, nextprime(n)
    while True:
        q = nextprime(p)
        if q-p == 2*n:
            return p
        p = q
        if p >= (n+1)*l:
            l *= 10
            p = nextprime(n*l) # _Chai Wah Wu_, May 31 2019
```
