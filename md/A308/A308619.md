import numpy as np
def A308619(n):
    NegVanEck = np.array([0])
    #first value is 0
    i=NegVanEck[0]
    while i < n:
        last = NegVanEck[-1]
        pos = np.where(NegVanEck == last)
        if pos[0].size == 1:
            NegVanEck = np.append(NegVanEck, -last)
        else:
            NegVanEck = np.append(NegVanEck, pos[0][-1] - pos[0][-2])
        i += 1
    return NegVanEck
print(A308619(1000)) #prints first 1000 integers of the sequence
