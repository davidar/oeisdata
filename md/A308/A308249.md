## Python
```Python
dig = "0123456789AB"
def To12(n):
    s = ""
    while n > 0:
        s, n = dig[n%12]+s, n//12
    return s
n, m = 1, 0
print(n,m*m)
while n < 100:
    m = m+1
    m2, m1 = To12(m*m), To12(m)
    i, i2, i1 = 0, len(m2), len(m1)
    while i < i1 and (m2[i2-i-1] == m1[i1-i-1]):
        i = i+1
    if i == i1:
        print(n,m*m)
n = n+1 # _A.H.M. Smeets_, Aug 09 2019
```
