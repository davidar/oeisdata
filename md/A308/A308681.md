from itertools import count, islice, accumulate
from operator import sub
def A308681_gen(): # generator of terms
    blist, m = tuple(), 1
    for i in count(1):
        yield (blist := tuple(accumulate(reversed(blist),func=sub,initial=m)))[-1]
        m *= (2*i-1)
