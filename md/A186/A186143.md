## Python
```Python
from sympy import isprime
def a(n):
    an = 0
    while True:
        an = an+1
        while not all(isprime(int("3"*k+str(an))) for k in range(1, n+1)):
            an += 1
        return an
print([a(n) for n in range(1, 9)]) # _Michael S. Branicky_, Mar 10 2022
```
