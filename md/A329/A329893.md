from math import prod
def A329893(n):
    c, s = [0]*(m:=n.bit_length()), bin(n)[2:]
    for i in range(m):
        if s[i]=='1':
            for j in range(m-i):
                c[j] = c[j]+1
        else:
            for j in range(m-i):
                c[j] = -c[j]
