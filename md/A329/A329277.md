## Python
```Python
from gmpy2 import is_prime
from sympy import factorint
def gradus(n):
    sum  = 0
    factors = factorint(n)
    for p,a in factors.items():
        sum += (p - 1)*a
    return sum + 1
if __name__ == "__main__":
    glist = []
    for x in range(1, 80):
        glist.append(gradus(x))
    while True:
        for p in glist:
            a = 0
            if not is_prime(p):
                glist = [gradus(x) for x in glist]
                a = 1
        if a == 0:
            break
    print(', '.join([str(x) for x in glist]))
```
