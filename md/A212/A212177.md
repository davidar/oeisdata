## Python
```Python
from math import isqrt
from sympy import mobius, factorint
def A212177(n):
    def f(x): return n+sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1))
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    return sum(1 for e in factorint(m).values() if e>1) # _Chai Wah Wu_, Jul 22 2024
```
