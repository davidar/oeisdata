## Python
```Python
from itertools import count, islice
from sympy import isprime
def A181741_gen(): # generator of terms
    m = 2
    for t in count(1):
        r=1<<t-1
        for k in range(t-1,0,-1):
            if isprime(s:=m-r-1):
                yield s
            r>>=1
        m<<=1
A181741_list = list(islice(A181741_gen(),30)) # _Chai Wah Wu_, Jul 08 2022
```
