## Python
```Python
from sympy import isprime
def afind(limit, startk=1):
    k = startk
    pow2 = 2**k
    pow3 = 3**k
    for k in range(startk, limit+1):
        q, r = divmod(pow2+pow3, 13)
        if r == 0 and isprime(q):
            print(k, end=", ")
        pow2 *= 2
        pow3 *= 3
afind(1000) # _Michael S. Branicky_, Dec 28 2021
```
