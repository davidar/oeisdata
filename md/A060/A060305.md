## Python
```Python
from itertools import count
from sympy import prime
def A060305(n):
    x, p = (1,1), prime(n)
    for k in count(1):
        if x == (0,1):
            return k
        x = (x[1], (x[0]+x[1]) % p) # _Chai Wah Wu_, May 31 2022
```
