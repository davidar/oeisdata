```Python
def Trow(n: int) -> list[int]:
    row: list[int] = [0] * (n + 1); row[n] = 2**n
    for k in range(n - 2, -1, -2):
        row[k] = -(row[k + 2] * (k + 2) * (k + 1)) // (2 * (n - k))
    return row  # _Peter Luschny_, Jan 08 2023
```
