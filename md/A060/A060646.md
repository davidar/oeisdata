## Python
```Python
from sympy import nextprime
from itertools import count, islice
def agen(): # generator of terms
    n, pj = 1, 2
    for j in count(1):
        while n - j + 1 < pj: yield j; n += 1
        pj = nextprime(pj)
print(list(islice(agen(), 76))) # _Michael S. Branicky_, Aug 09 2022
```
