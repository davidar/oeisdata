A060973 := proc(n)
    option remember;
    if n <= 2 then
        return n-1;
    fi;
    if n mod 2 = 0 then
        2*procname(n/2)
    else
        procname((n-1)/2)+procname((n+1)/2);
    fi;
end proc:  # _R. J. Mathar_ Nov 30 2014
