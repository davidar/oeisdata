## Python
```Python
from functools import cache;
from sympy import primefactors, prod
@cache
def a(n): return 1 if n == 0 else a(n-1) + prod(primefactors(a(n-1)))
print([a(n) for n in range(42)]) # _Trey Deitch_, Jun 08 2024
```
