## Python
```Python
from sympy import prime
def a(n, k):
    p=prime(n)
    i=z=0
    while p**i<=k:
        if k%(p**i)==0: z=i
        i+=1
    return z
for n in range(1, 10): print([a(n - k + 1, k) for k in range(1, n + 1)]) # _Indranil Ghosh_, Jun 24 2017
```
