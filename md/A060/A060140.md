## Python
```Python
import heapq
from itertools import islice
def agen(): # generator of terms, using recursion in Comments
    yield 0; x, h = None, [1]
    while True:
        x = heapq.heappop(h)
        yield x
        for t in [3*x, 9*x+1]:  heapq.heappush(h, t)
print(list(islice(agen(), 51))) # _Michael S. Branicky_, Aug 30 2022
```
