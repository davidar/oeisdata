## Python
```Python
def summarize_lst(lst):
  ans = []
  for d in sorted(set(lst)): ans += [lst.count(d), d]
  return ans
def aupton(nn):
  alst, arunninglst = [1], [1]
  for n in range(nn):
    nxt_lst = summarize_lst(arunninglst)
    arunninglst += nxt_lst
    alst.append(int("".join(map(str, nxt_lst))))
  return alst
print(aupton(10)) # _Michael S. Branicky_, Jan 11 2021
```
