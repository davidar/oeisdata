## Maple
```Maple
A060839 := proc(n)
    local a,pf,p,r;
    a := 1 ;
    for pf in ifactors(n)[2] do
        p := op(1,pf);
        r := op(2,pf);
        if p = 2 then
            ;
        elif p =3 then
            if r >= 2 then
                a := a*3 ;
            end if;
        else
            if modp(p,3) = 2 then
                ;
            else
                a := 3*a ;
            end if;
        end if;
    end do:
    a ;
end proc:
```
