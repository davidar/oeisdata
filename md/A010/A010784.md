## Scala
```Scala
def hasDistinctDigits(n: Int): Boolean = {
  val numerStr = n.toString
  val digitSet = numerStr.split("").toSet
  numerStr.length == digitSet.size
}
(0 to 99).filter(hasDistinctDigits) // _Alonso del Arte_, Jan 09 2020
```
