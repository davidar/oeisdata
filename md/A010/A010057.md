A010057 := proc(n)
    if n = 0 then
        1;
    else
        for pe in ifactors(n)[2] do
            if modp(op(2,pe),3) <> 0 then
                return 0 ;
            end if;
        end do:
    end if;
    1 ;
end proc: # _R. J. Mathar_, Feb 07 2023
