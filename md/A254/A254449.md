def A254449(n):
    if n == 0:
        return 0
    i, m, s = 1, 1, '4'*n
    s2 = s+'4'
    while True:
        m *= i
        sn = str(m)
        if s in sn and s2 not in sn:
            return i
        i += 1 # _Chai Wah Wu_, Dec 29 2015
