## Python

from mpmath import *
mp.dps=106
x=3 - 2*sqrt(2)
Ti2x=(j/2)*(polylog(2, -j*x) - polylog(2, j*x))
C = 2/5 - catalan/10 + (3/10)*Ti2x + log(3) - (7*sqrt(2)/10)*atan(1/sqrt(8))
print([int(n) for n in str(C.real).replace('.', '')]) # _Indranil Ghosh_, Jul 04 2017
