## Python
```Python
def A254073(n):
    ndict = {}
    for i in range(n):
        m = pow(i,3,n)
        if m in ndict:
            ndict[m] += 1
        else:
            ndict[m] = 1
    count = 0
    for i in ndict:
        ni = ndict[i]
        for j in ndict:
            k = (1-i-j) % n
            if k in ndict:
                count += ni*ndict[j]*ndict[k]
    return count # _Chai Wah Wu_, Jun 06 2017
```
