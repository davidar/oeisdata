from sympy import factorint, prevprime, floor
from operator import mul
def a064216(n):
    f=factorint(2*n - 1)
    return 1 if n==1 else reduce(mul, [prevprime(i)**f[i] for i in f])
def a254103(n):
    if n==0: return 0
    if n%2==0: return 3*a254103(n/2) - 1
    else: return floor((3*(1 + a254103((n - 1)/2)))/2)
