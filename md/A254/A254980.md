## Python
```Python
from mpmath import *
mp.dps=105
x=3 - 2*sqrt(2)
Ti2x=(j/2)*(polylog(2, -j*x) - polylog(2, j*x))
C = 2*log(3) - (2/3)*catalan + 2*Ti2x - sqrt(8) * atan(1/sqrt(8))
print([int(n) for n in list(str(C.real)[2:-1])]) # _Indranil Ghosh_, Jul 03 2017
```
