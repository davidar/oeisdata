## Python
```Python
def comma_parent(n): # A367616(n)
    y = int(str(n)[0])
    x = (n-y)%10
    k = n - y - 10*x
    return k if k > 0 else -1
def a(n):
    an = n
    while (cp:=comma_parent(an)) > 0: an = cp
    return an
print([a(n) for n in range(1, 76)]) # _Michael S. Branicky_, Dec 18 2023
```
