# Alternative:  (For Python include 'import math' for math.gcd.)
def a(n):
    cop = [int(gcd(k, n) == 1) for k in (0..n)]
    return sum(p * n^k for k, p in enumerate(cop))
print([a(n) for n in range(21)])
