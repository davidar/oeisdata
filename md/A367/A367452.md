## Python
```Python
from math import isqrt
from sympy import mobius, primenu
def A367452(n):
    def f(x): return n+x-sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1))
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    return (c:=primenu(m))*(c-1)>>1 # _Chai Wah Wu_, Aug 12 2024
```
