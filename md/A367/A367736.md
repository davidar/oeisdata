from itertools import islice
from collections import Counter
def A367736_gen(): # generator of terms
    c, b = {0:1}, 1
    while True:
        yield b
        d = Counter(c)
        for k in c:
            e = c[k]
            d[k+b] += e
            d[k-b] += e
        c = d
        b = c[b]
A367736_list = list(islice(A367736_gen(),20)) # _Chai Wah Wu_, Feb 05 2024
