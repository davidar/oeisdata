## Python

from itertools import count, islice, pairwise
def S(): yield from (str(2**i) for i in count(0))
def agen(): yield from (int(t[-1]+u[0]) for t, u in pairwise(S()))
print(list(islice(agen(), 80))) # _Michael S. Branicky_, Nov 22 2023
