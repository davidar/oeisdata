## Python
```Python
def a(n: int) -> int:
    s = str(n)
    if n <= 1 or (s[0] == '1' and set(s[1:]) == {'0'}):
        return -1
    pwr, e = 1, 0
    while str(pwr)[0] != '9':
        pwr *= n
        e += 1
    return e
```
