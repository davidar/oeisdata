## Python
```Python
from itertools import count
from sympy import isprime
def A367081(n):
    for k in count(1):
        a, c = (1<<k)-1, 0
        for i in range(k-2,0,-1):
            if isprime(a^(1<<i)):
                c += 1
            if c >= n+1:
                break
        if c == n:
            return k # _Chai Wah Wu_, Nov 11 2023
```
