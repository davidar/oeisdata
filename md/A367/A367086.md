from itertools import islice
def A367086_gen(): # generator of terms
    a, b, c, i = 1, 4, -1, 1
    while True:
        while (a:=a*3)<b:
            c += 1
            yield i
        b <<= 2
        i += 1
        c += 2
A367086_list = list(islice(A367086_gen(),30)) # _Chai Wah Wu_, Nov 18 2023
