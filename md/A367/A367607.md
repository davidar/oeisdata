## Python
```Python
from sympy.ntheory.factor_ import digits
def a(n):
    b = n + 3*(n%3)
    return next((int("".join(map(str, d3))) for y in [1, 2] if (d3:=digits(b+y, 3)[1:])[0] == y), -1)
print([a(n) for n in range(1, 61)]) # _Michael S. Branicky_, Dec 11 2023
```
