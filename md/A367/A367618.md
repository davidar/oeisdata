## Python
```Python
from functools import cache
from sympy.ntheory.factor_ import digits
def a(n, base=3):
    y = digits(n, base)[1]
    x = (n-y)%base
    k = n - y - base*x
    return k if k > 0 else -1
print([a(n) for n in range(1, 88)])
```
