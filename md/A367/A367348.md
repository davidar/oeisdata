from itertools import islice
def agen(): # generator of terms
    yield 1
    an = 11
    while True:
        yield an
        s = str(an)
        an += int(s[0]) + 10*int(s[-1])
print(list(islice(agen(), 55))) # _Michael S. Branicky_, Nov 17 2023
