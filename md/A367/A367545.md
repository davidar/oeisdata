## Sage

. # For Python include 'import math' for math.gcd.
def a(n):
    cop = [int(gcd(i, n) == 1) for i in range(n + 1)]
    return sum(p * (-2)^k for k, p in enumerate(cop))
print([a(n) for n in range(34)])
