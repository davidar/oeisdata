def a(n):
    y = int(str(n)[0])
    x = (n-y)%10
    k = n - y - 10*x
    kk = k + 10*x + y-1
    return k if k > 0 and int(str(kk)[0]) != y-1 else -1
print([a(n) for n in range(1, 86)]) # _Michael S. Branicky_, Dec 16 2023
