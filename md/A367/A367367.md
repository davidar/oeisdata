```Python
def A367367(n):
    kmin, kmax = 1, 1
    while kmax*(n-1)**kmax >= n**kmax-1:
        kmax <<= 1
    while True:
        kmid = kmax+kmin>>1
        if kmid*(n-1)**kmid < n**kmid-1:
            kmax = kmid
        else:
            kmin = kmid
        if kmax-kmin <= 1:
            break
    return kmin # _Chai Wah Wu_, Dec 21 2023
```
