## Python

from itertools import combinations
def A367396(n): return sum(1 for k in range(3,n+1) for w in (set(d) for d in combinations(range(1,n+1),k)) if any({a,k-a}<=w for a in range(1,k+1>>1))) # _Chai Wah Wu_, Nov 21 2023
