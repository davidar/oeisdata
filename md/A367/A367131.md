from sympy import divisor_sigma
memo = {0: 2}
def a000058(n):
    if n not in memo:
        memo[n] = a000058(n - 1)**2 - a000058(n - 1) + 1
    return memo[n]
a = lambda n: divisor_sigma(a000058(n))
print([a(n) for n in range(10)])
