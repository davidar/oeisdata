## Python
```Python
cache = {}
def F(a,b,n):
    if (a,b,n) in cache: return cache[(a,b,n)]
    if a==0 and b==0: return 1
    if a > n or b > n: return 0
    if a < 0 or b < 0: return 0
    cache[(a,b,n)] = F(a-1,b-2,n) + F(a-2,b-1,n) + F(a+1,b-2,n) + F(a-2,b+1,n)
    return cache[(a,b,n)]
for i in range (30):
    print(F(i,i,i), end=', ')
```
