z_list = [-1, 2, 4]
m_list = [-1, 0, 1]
n = 2
for n in range(2, 100):
    if m_list[n] in z_list:
        m_list.append(m_list[n] + 1)
        z_list.append(m_list[n+1] + n+2)
    else:
        m_list.append(m_list[n])
        z_list.append(m_list[n+1])
print(z_list[1:])
