from itertools import islice
def A367083_gen(): # generator of terms
    yield 1
    a, b = 1, 4
    while True:
        while (a:=a*3)<b:
            yield a
        yield from (b,a)
        b <<= 2
A367083_list = list(islice(A367083_gen(),30)) # _Chai Wah Wu_, Nov 08 2023
