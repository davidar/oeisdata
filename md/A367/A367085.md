from itertools import islice
def A367085_gen(): # generator of terms
    a, b, c = 1, 4, 0
    while True:
        while (a:=a*3)<b:
            yield (c:=c+1)
        b <<= 2
        c += 1
A367085_list = list(islice(A367085_gen(),30)) # _Chai Wah Wu_, Nov 09 2023
