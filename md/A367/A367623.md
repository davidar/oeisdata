## Python
```Python
from sympy.ntheory.factor_ import digits
def a(k, base=3):
    m = k + base*(k%base)
    return len([m+y for y in range(1, base) if digits(m+y, base)[1] == y])
print([a(n) for n in range(1, 96)]) # _Michael S. Branicky_, Dec 23 2023
```
