from itertools import islice
def A367296_gen(): # generator of terms
    a, b, c, l = 8, 9, 1, 0
    while True:
        if a<=c:
            if c<b:
                yield l
            a *= 10
            b *= 10
        c <<= 1
        l += 1
A367296_list = list(islice(A367296_gen(),30)) # _Chai Wah Wu_, Nov 13 2023
