from math import prod
from sympy import primerange
def A367566(n): return prod(p for p in primerange(n+2) if all((m:=n*pow(k,n,p)%p)==1 or m==p-1 for k in range(1,p))) # _Chai Wah Wu_, Nov 24 2023
