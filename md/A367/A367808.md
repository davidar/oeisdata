## Python
```Python
from functools import cache
@cache
def b(n: int) -> list[int]:
    if n == 0: return [1]
    row = [b(n - 1)[n - 1]] + b(n - 1)
    for k in range(1, n + 1): row[k] += row[k - 1]
    return row
def a(n): return sum(b(n)[k] * 2 ** (n - k) for k in range(n + 1))
print([a(n) for n in range(22)])
```
