def is_A367420(k):
    a, b = 1, 1
    while((a,b)!=(0,1)):
        if (a==k): return False
        a, b = b, (a+b)%(2*k)
    return True
print([k for k in range(1, 1000) if is_A367420(k)])  # _Robin Visser_, Jan 08 2024
