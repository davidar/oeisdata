## Python
```Python
from sympy import prime
from itertools import count, islice
def a(n):
    return int("".join(map(str, range(1 if n<2 else prime(n-1)+1, prime(n)+1))))
print([a(n) for n in range(1, 31)]) # _Michael S. Branicky_, Nov 18 2023
```
```Python
# faster for generating initial segment of sequence
from sympy import isprime
from itertools import count, islice
def agen():
    cat = ""
    for i in count(1):
        cat += str(i)
        if isprime(i): yield int(cat); cat = ""
print(list(islice(agen(), 30))) # _Michael S. Branicky_, Nov 18 2023
```
