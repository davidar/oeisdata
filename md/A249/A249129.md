import Data.List (delete)
a249129 n = a249129_list !! n
a249129_list = 1 : 0 : 2 : f 3 2 [3..] where
   f :: Int -> Int -> [Int] -> [Int]
   f k x zs'@(z:zs)
     | r == 0 = y : f (k+1) y (delete (x + y) $ delete y zs')
     | r == 1 = z : f (k+1) z (delete (x + z) zs)
