```Python
def isqrt(a):
    sr = 1 << (a.bit_length() >> 1)
    while a < sr * sr:
        sr >>= 1
    b = sr >> 1
    while b:
        s = sr + b
        if a >= s * s:
            sr = s
        b >>= 1
    return sr
for j in range(99):
    i = 2**j
    r = isqrt(i)
    if r * r == i:
        continue
    if r & 1:
        a = ((r + 1) * (r + 1) + i) // 2
    else:
        a = (i + r * r) // 2
    print(a, end=', ')
```
