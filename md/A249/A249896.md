## Maple
```Maple
R:= 6: x:= 6: s:= 2:
for iter from 1 while length(x) < 1000 do
  for d from 1 do
   if d::even then sp:= 1+ 10^(d/2)*s
   else
     sp:= ceil(sqrt(10^d*x));
     if sp mod 10 = 0 then sp:=sp+1; fi
   fi;
   if sp^2 < (x+1)*10^d then
     x:= sp^2; s:= sp; R:= R, x; break
   fi
  od;
od:
R; # _Robert Israel_, Nov 25 2020
```
