L:= [seq(i, i=1..10^3)]:
for n from 2 while n < nops(L) do
  r:= L[n];
  L:= subsop(seq(r*i=NULL, i=1..nops(L)/r), L);
od:

def listed(n):
   L=list(range(1,n+1));j=1
   while L[j] <= len(L):
      L=[L[i] for i in range(len(L)) if (i+1)%L[j]!=0]
