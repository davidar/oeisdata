## C

// Input: n (no negative offset/term number), Output: a(n)
int A249919 (int n) {
   int m=0, r=0;
   if (n) {
      while (n!=1) {
         m=n&1; //equivalent to m=n%2;
         n=n>>1; //equivalent to n/=2;
         if (m) {
            r+=2;
         } else {
            r+=6;
         }
      }
      r+=2;
   } else {
      r+=6;
   }
   return r;
}
