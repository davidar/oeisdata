## Python
```Python
from sympy import prime
def a(n):
    b, p = 2, prime(n)
    p3 = p**3
    while pow(b, p-1, p3) != 1: b += 1
    return b
print([a(n) for n in range(1, 46)]) # _Michael S. Branicky_, Sep 26 2021
```
