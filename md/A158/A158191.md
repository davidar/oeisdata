## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen(): # generator of terms
    p, s = 2, "2"
    while True:
        yield p
        q = 2
        while not isprime(p:=int(s+str(q))):
            q = nextprime(q)
        s += str(q)
print(list(islice(agen(), 15))) # _Michael S. Branicky_, May 26 2023
```
