## Python
```Python
from sympy import isprime
def A158962(n):
    m = 1
    while True:
        for i in range(n):
            if not isprime(int(str(m)*(i+1))-1):
                break
        else:
            return m
        m += 1 # _Chai Wah Wu_, Apr 14 2021
```
