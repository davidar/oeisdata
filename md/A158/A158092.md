from itertools import count, islice
from collections import Counter
def A158092_gen(): # generator of terms
    ccount = Counter({0:1})
    for i in count(1):
        bcount = Counter()
        for a in ccount:
            bcount[a+(j:=i**2)] += ccount[a]
            bcount[a-j] += ccount[a]
        ccount = bcount
        yield(ccount[0])
A158092_list = list(islice(A158092_gen(),20)) # _Chai Wah Wu_, Jan 29 2024
