## Python
```Python
from sympy import isprime
def ok(n): s = str(n); return (s.count("0") > 1 or s.count("1") > 1 or ("2" in s and "4" in s) or ("3" in s and "9" in s)) and isprime(n)
print([k for k in range(1172) if ok(k)]) # _Michael S. Branicky_, Nov 29 2022
```
