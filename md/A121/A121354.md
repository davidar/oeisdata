## Python
```Python
from sympy import cacheit
@cacheit
def A121354(n):
    if n <= 1:
        return n
    else:
        return (3*n-1)*A121354(n-1)-A121354(n-2)
print([A121354(n) for n in range(20)]) # Oct 14 2009
```
