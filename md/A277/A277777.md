```Python
from gmpy2 import *
def f(n):
  for k in range(n - 2, 0, -1):
    if pow(k, 2, n) == 1:
      return k
def A277777(L):
  return [j for j in [f(k) for k in range(3, L + 1)] if j > 1] # _Dar√≠o Clavijo_, Oct 15 2022
```
