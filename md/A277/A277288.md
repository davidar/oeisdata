## Sage

def A277288_list(search_limit):
    n, t, r = 1, Integer(3), [1]
    while n < search_limit:
        n += 1
        t *= 3
        if n.divides(t+5): r.append(n)
    return r # _Peter Luschny_, Oct 10 2016
