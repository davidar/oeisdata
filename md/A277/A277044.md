def t(n):
    s=0
    for a in range(0,n+1):
        for b in range(0,n+1):
            for c in range(0,n+1):
                for d in range(0,n+1):
                   if (a!=b  and a!=d and b!=d and c!=a and c!=b and c!=d):
                        if (a*d-b*c)%2==0:
                            s+=1
    return s
for i in range(0,201):
    print str(i)+" "+str(t(i))
