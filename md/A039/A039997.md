## Python
```Python
from sympy import isprime
def a(n):
    s = str(n)
    ss = (int(s[i:j]) for i in range(len(s)) for j in range(i+1, len(s)+1))
    return len(set(k for k in ss if isprime(k)))
print([a(n) for n in range(1, 106)]) # _Michael S. Branicky_, Aug 07 2022
```
