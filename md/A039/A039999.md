## Python
```Python
from sympy import isprime
from itertools import permutations
def a(n): return len(set(t for p in permutations(str(n)) if isprime(t:=int("".join(p)))))
print([a(n) for n in range(1, 106)]) # _Michael S. Branicky_, Feb 17 2024
```
