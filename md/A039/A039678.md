## Python

from sympy import prime
from sympy.ntheory.residue_ntheory import nthroot_mod
def A039678(n): return 2**2+1 if n == 1 else int(nthroot_mod(1,(p:= prime(n))-1,p**2,True)[1]) # _Chai Wah Wu_, May 18 2022
