## Python

```Python
def aupton(terms):
  k, n, alst = 1, 1, []
  while len(alst) < terms:
    while not all(str(i*k).count('4') > 0 for i in range(1, n+1)): k += 1
    while str(n*k).count('4') > 0: alst.append(k); n += 1
    k += 1
  return alst[:terms]
print(aupton(39)) # _Michael S. Branicky_, Apr 24 2021
```
