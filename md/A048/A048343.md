## Python
```Python
def A048343(n):
....y, plist = 0, []
....for i in range(10**(n-1),10**n):
........s1 = str(i)
........s2 = s1[::-1]
........if s1 != s2:
............p = i*int(s2)
............if not p in plist:
................sp = str(p)
................if sp == sp[::-1]:
....................plist.append(p)
....................y += 1
....return y # _Chai Wah Wu_, Sep 05 2014
```
