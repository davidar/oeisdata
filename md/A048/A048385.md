## Python
```Python
def digits(n):
    d=[]
    while n>0:
        d.append(n%10)
        n=n//10
    return d
def sqdig(n):
    new=0
    num=digits(n)
    spacing=0
    while num:
        k=num.pop(0)
        new+=(10**(spacing))*(k**2)
        if k>3:
            spacing+=1
        spacing+=1
    return new
# _David Nacin_, Aug 19 2012
```
