## Python
```Python
from functools import cache
@cache
def f(m):
    if m == 1: return {1}
    out = set()
    for j in range(1, m//2+1):
        for x in f(j):
            for y in f(m-j):
                out.update([x + y, x * y])
    return out
def a(n): return len(f(n))
print([a(n) for n in range(1, 40)]) # _Michael S. Branicky_, Aug 03 2022
```
