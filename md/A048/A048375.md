## Python
```Python
from math import isqrt
def issquare(n): return isqrt(n)**2 == n
def ok(n):
  d = str(n)
  for i in range(1, len(d)):
    if d[i] != '0' and issquare(int(d[:i])) and issquare(int(d[i:])):
      return True
  return False
print([r for r in range(2851) if ok(r*r)]) # _Michael S. Branicky_, Jul 13 2021
```
