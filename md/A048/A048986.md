## Python
```Python
from sympy import factorint, isprime
def f(n):
    if n == 1: return 1
    return int("".join(bin(p)[2:]*e for p, e in factorint(n).items()), 2)
def a(n):
    if n == 1: return 1
    while not isprime(n): n = f(n)
    return n
print([a(n) for n in range(1, 56)]) # _Michael S. Branicky_, Oct 07 2022
```
