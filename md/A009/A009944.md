## Python
```Python
from sympy import divisors
def ok(n):
    if n%10==0: return False
    t = str(n)[::-1]
    return any(t==str(d)+str(n//d) for d in divisors(n, generator=True))
print([k for k in range(10**6) if ok(k)]) # _Michael S. Branicky_, Apr 13 2024
```
