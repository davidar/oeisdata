from itertools import groupby
def A361016(n):
    c = 0
    for k, g in groupby(bin(n)[2:]):
        c = c+len(list(g)) if k == '1' else (-c if len(list(g))&1 else c)
    return int(not c) # _Chai Wah Wu_, Mar 02 2023
