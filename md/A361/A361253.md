from sympy import integer_nthroot
def A361253(n):
    if n <= 1:
        return n
    a, b = integer_nthroot(c:=n,2)
    while b:
        a, b = integer_nthroot(c:=a,2)
    return c # _Chai Wah Wu_, Mar 17 2023
