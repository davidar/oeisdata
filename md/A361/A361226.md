# Seen as a triangle:
from functools import cache
@cache
def Trow(n: int) -> list[int]:
    if n == 0: return [0]
    r = Trow(n - 1)
    return [r[k] + k * k if k < n else r[n - 1] + n - 1 for k in range(n + 1)]
for n in range(7): print(Trow(n)) # _Peter Luschny_, Mar 05 2023
