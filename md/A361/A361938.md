def seqx_it(n):
  a0 = 1
  a1 = 0
  sequence_store = [a0,a1]
  for i in range (2,n):
    a2 = (i//2) * (a1 + a0)
    sequence_store.append(a2)
    a0 = a1
    a1 = a2
