## Python
```Python
from sympy import divisors, isprime
def a(n):
    if n == 1 or isprime(n): return n
    return int("".join(str(d) for d in divisors(n)[-2:0:-1]))
print([a(n) for n in range(1, 51)]) # _Michael S. Branicky_, Mar 21 2023
```
