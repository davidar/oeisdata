```Maple
gsigma := proc(n,z,k)
    local a,d ;
    a := 0 ;
    for d in numtheory[divisors](n) do
        if modp(n,d^k) = 0 then
            a := a+d^z ;
        end if ;
    end do:
    a ;
end proc:
seq( gsigma(n,2,3),n=1..80) ;
```
