from functools import lru_cache
@lru_cache(maxsize=None)
def f(n):
    if n < 10: return {n}
    s = str(n)
    return {e for i in range(1, len(s)) if s[i]!="0" or i==len(s)-1 for e in f(int(s[:i])*int(s[i:]))}
def A361338(n):
    return len(f(n))
