## Python
```Python
from sympy import divisors, isprime
def ok(n): return n&1 and all(d&1 or isprime(d**2+1) for d in divisors(n**2+1))
print([k for k in range(7000) if ok(k)]) # _Michael S. Branicky_, Apr 17 2023
```
