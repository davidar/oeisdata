## Python
```Python
from math import gcd
from itertools import count, islice
from sympy import nextprime
def A361627_gen(): # generator of terms
    p, s = 2, 2
    for n in count(1):
        if gcd(n,s) > 1:
            yield n
        s += (p:=nextprime(p))
A361627_list = list(islice(A361627_gen(),20)) # _Chai Wah Wu_, Mar 22 2023
```
