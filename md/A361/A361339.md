from itertools import count
def agen():
    n, adict = 1, dict()
    for k in count(1):
        v = A361338(k) # uses A361338() and reach1() in A361338
        if v not in adict: adict[v] = k
        while n in adict: yield adict[n]; n += 1
        if n == 11: return
