def Ofmp(f):
    c = 0
    while len(f) > 1:
        f.sort()
        m = f[0] + f[1]
        c += m
        f[0] = m
        f.pop(1)
    return c
a = lambda n: Ofmp(list(range(1, n+1)))
print([a(n) for n in range(1, 49)]) # _Dar√≠o Clavijo_, Aug 21 2024
