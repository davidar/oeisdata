## Python

def f(n):
    return sum((6, 2, 5, 5, 4, 5, 6, 3, 7, 6)[int(d)] for d in str(n))
def A328330(n):
    c, m = 1, f(n)
    while m != n:
        c += 1
        n, m = m, f(m)
    return c # _Chai Wah Wu_, Oct 27 2020
