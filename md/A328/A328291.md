from __future__ import division
def A328291(n):
    if n > 9876543210 or n % 100 == 0:
        return 0
    k = 9876543210//n
    m = k*n
    s = str(m)
    while len(set(s)) != len(s):
        k -= 1
        m -= n
        s = str(m)
    return k # _Chai Wah Wu_, Oct 13 2019
