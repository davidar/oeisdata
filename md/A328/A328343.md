## Python
```Python
import math
for n in range(1, 100):
   for b in range(1, 10000000):
      d = (6*b*b*(n+1)+6*b*n*(n+1)+2*n*n*n+3*n*n+n)
      w = int((math.sqrt(d/6)))
      a = w
      if 6*a*a-6*b*b*(n+1)-6*b*n*(n+1)-2*n*n*n-3*n*n-n == 0:
         print(a,b,n+1)
      a = w+1
      if 6*a*a-6*b*b*(n+1)-6*b*n*(n+1)-2*n*n*n-3*n*n-n == 0:
         print(a,b,n+1)
      a = w-1
      if 6*a*a-6*b*b*(n+1)-6*b*n*(n+1)-2*n*n*n-3*n*n-n == 0:
         print(a,b,n+1)
```
