from functools import lru_cache
@lru_cache(maxsize=None)
def A328967(n):
    if n == 0:
        return 1
    c, j = n-1, 1
    k1 = (n-1)//j
    while k1 > 1:
        j2 = (n-1)//k1 + 1
        c += (j2-j)*A328967(k1)
        j, k1 = j2, (n-1)//j2
    return j-c # _Chai Wah Wu_, Mar 31 2021
