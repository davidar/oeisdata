## Python
```Python
def f(p0, n, m, cm):
    full, t, p = True, 0, p0
    while p<n:
        sm = m<<p
        if (m & sm) == 0:
            t += f(p+1, n, m|sm, cm|(1<<p))
            full=False
        p+=1
    if full:
        for k in range(1, p0):
            if ((cm>>k)&1)==0 and ((m<<k)&m)==0:
                full=False
                break
    return 1 if full else t
def a325867(n):
    return f(1, n, (1<<n)+1, 0)
```
