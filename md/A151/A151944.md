# alst(), moves(), swap() in A089473
def T(m, n):  # chr(45) is '-'
    start, shape = "".join(chr(45+i) for i in range(m*n)), (m, n)
    return len(alst(start, shape))-1
def auptodiag(maxd):
    for d in range(1, maxd+1):
        for m in range(1, d+1):
            n = d-m+1
            print(T(m, d-m+1), end=", ")
auptodiag(5) # _Michael S. Branicky_, Aug 02 2021
