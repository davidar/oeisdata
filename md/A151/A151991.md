## Python
```Python
def exis(n):
    for x in range(1,n):
        for y in range(x+1):
            for z in range(y+1):
                if x*y*z==(k-x)*(k-y)*(k-z):
                    return True
    return False
for k in range(1, 200, 2):
    if not exis(k):
        print(str(k), end=',')
```
