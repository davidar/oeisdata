from math import factorial, gcd
from sympy.functions.combinatorial.numbers import stirling
def A370550(n): return (a:=factorial(n))//gcd(a,sum(stirling(n+1,n-(k<<1),kind=1)*(-1 if k&1 else 1) for k in range((n>>1)+1))) # _Chai Wah Wu_, Feb 22 2024
