from itertools import count, islice
def A370956_gen(): # generator of terms
    a, aset, b, c = 0, set(), 0, -1
    for n in count(1):
        aset.add(a)
        if a==b:
            if n-1>c:
                c = n-1
                yield c
            while b in aset:
                b += 1
        a = next(a for a in count(a%n,n) if a not in aset)
