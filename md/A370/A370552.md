def A370552(n): return (a:=factorial(n))//gcd(a,sum(stirling(n+1,(k<<1)+1,kind=1)*(-1 if k&1 else 1) for k in range((n>>1)+1))) # _Chai Wah Wu_, Feb 22 2024
