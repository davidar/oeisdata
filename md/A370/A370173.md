## Python

from functools import cache
@cache
def T(n, k):
    if k > n: return 0
    if n == 0: return 1
    if k == 0: return -1 if n == 1 or n == 2 else 0
    return T(n-1, k-1) + T(n-2, k-1)
for n in range(9):
    print([T(n, k) for k in range(n+1)])  # _Peter Luschny_, Feb 28 2024
