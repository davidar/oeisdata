## Python
```Python
from sympy import primefactors
from sympy.utilities.iterables import multiset_permutations
def A370361(n): return min((max(primefactors(a:=int(''.join(s)))),a) for i in range(10) for j in range(i+1,10) for k in range(1,n) for s in multiset_permutations(str(i)*k+str(j)*(n-k)) if s[0] != '0' and s[-1] != '0')[0] # _Chai Wah Wu_, Mar 09 2024
```
