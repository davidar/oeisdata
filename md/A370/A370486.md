from itertools import count, islice
from sympy import sqrt_mod
def A370486_gen(): # generator of terms
    for j in count(0):
        b = 10**j
        a = b*10+1
        for k in sorted(sqrt_mod(0,a,all_roots=True)):
            if a*(b+4) <= k**2 < a*(a+3):
                yield k-2
A370486_list = list(islice(A370486_gen(),20))
