## Python

def map_5x(n): return 5*n+1 if n%2 == 1 else n//3
def A370394(n):
    m = n; ct = 0
    while m != 0: m = map_5x(m); ct += 1
    return ct
for n in range(77): print(A370394(n), end = ', ')
