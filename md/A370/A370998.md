from itertools import count
from sympy import prime, isprime
def A370998(n):
    ptuple = tuple(prime(k) for k in range(1,n+1))
    return next(filter(lambda m:not any(isprime(p+m) for p in ptuple),count(2,2)))>>1 # _Chai Wah Wu_, Mar 21 2024
