from itertools import count, islice
def A370952_gen(): # generator of terms
    a, aset = 1, {0,1}
    for p in count(3,2):
        yield a
        for b in count(a%p,p):
            if b not in aset:
                aset.add(b)
                a = b
                break
A370952_list = list(islice(A370952_gen(),20)) # _Chai Wah Wu_, Apr 17 2024
