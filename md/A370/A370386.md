## Python
```Python
def A370386_rowlist(maxrow):
    A,C = [[(1,)]],[[1]]
    for i in range(maxrow):
        A.append([])
        C.append([])
        for j in A[i]:
            for k in j:
                x = j + (j[-1] + k,)
                A[i+1].append(x)
                C[i+1].append(x[-1])
    return(C)
```
