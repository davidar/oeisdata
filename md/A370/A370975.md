from itertools import count, islice
def A370975_gen(): # generator of terms
    a, aset = 1, {0,1}
    for p in count(3,2):
        for b in count(a%p,p):
            if b not in aset:
                aset.add(b)
                yield (b-a)//p
                a = b
                break
