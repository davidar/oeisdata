## Python

from itertools import count, islice
from sympy import isprime
def A370606_gen(): # generator of terms
    a = 1
    while True:
        for q in count(a,a):
            if isprime(q+1):
                yield q+1
                a = (q&-q)<<1
                break
A370606_list = list(islice(A370606_gen(),30)) # _Chai Wah Wu_, Feb 23 2024
