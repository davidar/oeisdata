from math import factorial, gcd
from sympy.functions.combinatorial.numbers import stirling
def A370553(n): return (a:=sum(stirling(n+1,k<<1,kind=1)*(1 if k&1 else -1) for k in range((n+1>>1)+1)))//gcd(a,factorial(n)) # _Chai Wah Wu_, Feb 22 2024
