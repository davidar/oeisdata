## Python
```Python
def a(n):
  t = 1
  while True:
    a_n = (t << n) + 1
    for r in range(2, t+2):
      if pow(r, 1 << (n-1), a_n) + 1 == a_n:
        return a_n
    t += 1
print([a(n) for n in range(1,31)])
```
