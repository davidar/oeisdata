def A370680(n):
    a = 10**(n-1)
    b, c = 10*a, 0
    for x in range(a,b):
        s = set(str(x))
        for y in range(x,b):
            if len(s|set(str(y))) < len(set(str(x*y))):
                c += 1
    return c # _Chai Wah Wu_, Feb 28 2024
