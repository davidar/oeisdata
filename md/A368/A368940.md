from itertools import groupby
def a(n):
    seen, k, c = set(), 1, 0
    while k not in seen:
        seen.add(k)
        c += 1
        s = str(k)
        t = "".join(k for k, g in groupby(s))
        k = k*n if s == t else int(t)
    return c
print([a(n) for n in range(1, 45)]) # _Michael S. Branicky_, Jan 11 2024
