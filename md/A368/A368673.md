## Python

from math import isqrt
from sympy import factorint, mobius
def A368673(n): return sum(mobius(k)*((n-1)//k**2) for k in range(1,isqrt(n-1)+1))-(1<<len(f:=factorint(n)))+int(max(f.values(),default=1)==1) # _Chai Wah Wu_, Jan 03 2024
