## Python
```Python
from functools import cache
@cache
def F(k, n):
    return sum(F(k, n-j) for j in range(1, min(k, n))) if n > 1 else n
def Trow(n):
    return list(F(k+1, n+1-k) - F(k+1, n-k) for k in range(n+1))
print(flatten([Trow(n) for n in range(12)]))
```
