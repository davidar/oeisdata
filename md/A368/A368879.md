## Python
```Python
from sympy import isprime
def aupto(LIMIT):
    cands = set(k + int("".join(sorted(str(k)))) for k in range(LIMIT))
    return sorted(k for k in cands if k <= LIMIT and isprime(k))
print(aupto(1200)) # _Michael S. Branicky_, Jan 15 2024
```
