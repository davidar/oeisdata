## Python
```Python
from fractions import Fraction
from itertools import count, islice
def agen(): # generator of terms
    A = H = 0
    for n in count(1):
        A += n
        H += Fraction(1, n)
        yield ((A*Fraction(1, n) - n/H)).denominator
print(list(islice(agen(), 31))) # _Michael S. Branicky_, Jan 24 2024
```
