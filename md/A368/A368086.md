## Python
```Python
from itertools import count, islice
def chap():
    for k in count(1): yield from list(map(int, str(k)))
def agen(): # generator of terms
    an = 0
    for c in chap():
        yield an
        an = c - an
print(list(islice(agen(), 80))) # _Michael S. Branicky_, Dec 11 2023
```
