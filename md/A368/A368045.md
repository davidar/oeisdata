## Python
```Python
from functools import cache
@cache
def Trow(n: int) -> list[int]:
    if n == 0: return [0]
    row = Trow(n - 1) + [0]
    for k in range(n): row[k] += n * n
    row[n] = row[n - 1] + n * n
    return row
print([k for n in range(10) for k in Trow(n)])
```
