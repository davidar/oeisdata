def A368844(n):
    l = len(s:=bin(n)[3:])
    return sum(1 for i in range(l-2) for j in range(1,l-i+1>>1) if s[i:i+(j<<1)+1:j]=='000') # _Chai Wah Wu_, Jan 10 2024
