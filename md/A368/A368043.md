## Python
```Python
from functools import cache
@cache
def T_row(n: int) -> list[int]:
    if n == 0: return [1]
    row = T_row(n - 1) + [0]
    for k in range(n): row[k] *= 2
    row[n] = row[n - 1] * 2
    return row
for n in range(11): print(T_row(n))
```
