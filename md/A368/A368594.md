from itertools import count
def a(n) :
  """For integer n, the least number of signed Lucas terms required to sum to n."""
  f = [2, 1];    # Lucas numbers, starting with Lucas(0)
  while f[-1] <= (n or 1) :
    f.append(f[-2]+f[-1])
  a = [0 for _ in range(f[-1]+1)]
  for i in f :
    a[i] = 1
  for c in count(2) :
    if not all(a[4:]) :
      for i in range(4, f[-1]) :
        if not a[i] :
          for j in f :
            if j >= i :
              break
            if a[i-j] == c-1 :
              a[i] = c
              break
          if not a[i]:
            for j in f :
              if i+j >= len(a) :
                if j != 2:
                  break
              elif a[i+j] == c-1 :
                a[i] = c
                break;
    else :
      break
  return a[n]
