from itertools import count, islice
def A368382_gen(): # generator of terms
    a, aset, p = 1, {0,1}, 2
    while True:
        yield a
        for b in count(a%p,p):
            if b not in aset:
                aset.add(b)
                a, p = b, 3 if p == 2 else p+2
                break
A368382_list = list(islice(A368382_gen(),40)) # _Chai Wah Wu_, Mar 05 2024
