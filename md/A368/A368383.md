from itertools import count
def A368383(n):
    a, aset, p = 1, {0,1}, 2
    for c in count(1):
        if a==n: return c
        for b in count(a%p,p):
            if b not in aset:
                aset.add(b)
                a, p = b, 3 if p == 2 else p+2
                break # _Chai Wah Wu_, Mar 05 2024

def A368383_gen(): # uses A368382_gen/imports from _Chai Wah Wu_ in A368382
