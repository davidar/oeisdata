## Python
```Python
def c(n): return len(set(d for d in str(n) if d>'1' and n%int(d)==0))
def a(n):
  m = 2*10**(n-1)
  while c(m) < n: m += 1
  return m
print([a(n) for n in range(1, 7)]) # _Michael S. Branicky_, Feb 24 2021
```
