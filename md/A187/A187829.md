def hexmatch(inp):
    h=hex(inp)
    ho=''
    for i in h[2:]:
        if i=='a': ho+='10'
        elif i=='b' :ho+='11'
        elif i=='c' :ho+='12'
        elif i=='d' :ho+='13'
        elif i=='e' :ho+='14'
        elif i=='f' :ho+='15'
        else: ho +=i
    return ho
def loop(start, stop):
    rec=100000000
    for i in range(start, stop):
        val=int(hexmatch(i))
        diff=val-i
