## Python
```Python
from sympy import n_order, nextprime
from itertools import islice
def A187614_gen(): # generator of terms
    yield from (2,3,5)
    p = 7
    while True:
        if len(set('0'+str(10**(n_order(10, p))//p))) < 10:
            yield p
        p = nextprime(p)
A187614_list = list(islice(A187614_gen(),20)) # _Chai Wah Wu_, Mar 03 2022
```
