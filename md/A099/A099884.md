## Python
```Python
from sympy import binomial
def a(n):
    return sum((binomial(n, i)%2)*2**i for i in range(n + 1))
def T(n, k): return 2**(n - k)*a(k)
for n in range(21): print([T(n, k) for k in range(n + 1)]) # _Indranil Ghosh_, Apr 11 2017
```
