def A099174_triangle(dim):
    M = matrix(ZZ,dim,dim)
    for n in (0..dim-1): M[n,n] = 1
    for n in (1..dim-1):
        for k in (0..n-1):
            M[n,k] = M[n-1,k-1]+(k+1)*M[n-1,k+1]
    return M

def Trow(n: int) -> list[int]:
