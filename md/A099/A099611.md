from itertools import count
from sympy import factorint
def A099611(n):
    for i in count(n**2-(n%2)-1,-2):
        fs = factorint(i)
        if len(fs) == 2 == sum(fs.values()):
            return i # _Chai Wah Wu_, Dec 06 2021
