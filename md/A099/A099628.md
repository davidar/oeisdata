## Python

from itertools import count, islice
def A099628_gen(): # generator of terms
    m = 1
    for n in count(1):
        m *= 2
        r, k = m-1,1
        for _ in range(n):
            yield r+k
            k *= 2
A099628_list = list(islice(A099628_gen(),40)) # _Chai Wah Wu_, Nov 15 2022
