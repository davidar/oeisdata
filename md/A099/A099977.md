from itertools import accumulate, islice
def A099977_gen(): # generator of terms
    yield 1
    blist, b = (1,2), 1
    while True:
        for _ in range(2):
            blist = list(accumulate(blist, initial=(b:=blist[-1])))
        yield b
A099977_list = list(islice(A099977_gen(),30)) # _Chai Wah Wu_, Jun 22 2022
