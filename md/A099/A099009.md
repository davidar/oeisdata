def extend(base, start, n):
    if n == 0:
        yield base
        return
    for i in range(start, 10):
        for x in extend(base + str(i), i, n-1):
            yield x
def drive(n):
    result = []
    for lo in extend("", 0, n):
        ilo = int(lo)
        if ilo == 0 and n > 1:
            continue
        hi = lo[::-1]
        diff = str(int(hi) - ilo)
        diff = "0" * (n - len(diff)) + diff
        if sorted(diff) == list(lo):
            result.append(diff)
    return sorted(result)
for n in range(1, 17):
    # print("Length", n)
    # print('-' * 40)
    for r in drive(n):
        print(r, end=', ')
