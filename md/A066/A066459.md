A066459 := proc(n)
    local a,d;
    a := 1 ;
    for d in convert(n,base,10) do
        a := a*d! ;
    end do:
    a ;
end proc: # _R. J. Mathar_, Aug 07 2014

## Python

import math
def A066459(n):
    s=1
    for i in str(n):
        s*=math.factorial(int(i))
    return s # _Indranil Ghosh_, Jan 11 2017
