## Sage
```Sage
def A066325_row(n):
    T = [0]*(n+1)
    if n==1: return [1]
    for m in (1..n-1):
        a,b,c = 1,0,0
        for k in range(m,-1,-1):
            r = a - (k+1)*c
            if k < m : T[k+2] = u;
            a,b,c = T[k-1],a,b
            u = r
        T[1] = u;
    return T[1:]
for n in (1..11): A066325_row(n)  # _Peter Luschny_, Nov 01 2012
```

## Python
```Python
from sympy import Poly
from sympy.abc import x
def H(n, x): return 1 if n==0 else x if n==1 else x*H(n - 1, x) - (n - 1)*H(n - 2, x)
def a(n): return Poly(H(n, x), x).all_coeffs()[::-1]
for n in range(21): print(a(n)) # _Indranil Ghosh_, May 26 2017
```
