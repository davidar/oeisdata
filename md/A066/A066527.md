## Python
```Python
from sympy import integer_nthroot, isprime, nextprime
def istri(n): return integer_nthroot(8*n+1, 2)[1]
def afind(limit):
    s, p = 2, 2
    while s < limit:
        if istri(s): print(s, end=", ")
        p = nextprime(p)
        s += p
afind(10**11) # _Michael S. Branicky_, Oct 28 2021
```
