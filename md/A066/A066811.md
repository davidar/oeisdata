## Python
```Python
from sympy import isprime
def agen():
  k, str1tok = 1, '1'
  while True:
    if isprime(int(str1tok)): yield k
    k, str1tok = k + 2, str1tok + str(k + 2)
g = agen()
print([next(g) for i in range(5)]) # _Michael S. Branicky_, Mar 19 2021
```
