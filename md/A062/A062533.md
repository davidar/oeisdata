from sympy import primepi, totient
def A062533(n):
    if n == 1: return 1
    m, k = n-1, primepi(n) + n - 1 + (n>>1)
    while m != k:
        m, k = k, primepi(k) + n - 1 + (k>>1)
    return totient(m) # _Chai Wah Wu_, Jul 31 2024
