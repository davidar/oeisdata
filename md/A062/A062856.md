## Python
```Python
from itertools import count
from collections import Counter
def A062856(n):
    if n == 1: return 1
    c = Counter()
    for m in count(1):
        for i in range(1,m):
            ij = i*m
            c[ij] += 2
            if c[ij]>=n:
                return ij
        c[m*m] = 1 # _Chai Wah Wu_, Oct 16 2023
```
