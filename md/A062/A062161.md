# Generalized algorithm of L. Seidel (1877)
def A062161_list(n) :
    R = []; A = {-1:0, 0:0}
    k = 0; e = 1
    for i in range(n) :
        Am = 1 if e == -1 else 0
        A[k + e] = 0
        e = -e
        for j in (0..i) :
            Am += A[k]
            A[k] = Am
            k += e
        # print [A[z] for z in (-i//2..i//2)]
        R.append(A[e*i//2])
    return R

def A062161_gen(): # generator of terms

    while True:
        yield (blist := tuple(accumulate(reversed(blist),initial=(m := 1-m))))[-1]
