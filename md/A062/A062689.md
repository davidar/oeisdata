## Python
```Python
def centered(s, t):
    q, r = divmod(len(t)-len(s), 2)
    return r == 0 and t[q:].startswith(s)
def a(n):
    k, target = 0, str(n)
    while not centered(target, str(k**2)): k += 1
    return k**2
print([a(n) for n in range(43)]) # _Michael S. Branicky_, Feb 16 2022
```
