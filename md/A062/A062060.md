## Python
```Python
def a(n):
    l=[n]
    while True:
        if n%2==0: n//=2
        else: n = 3*n + 1
        if n not in l:
            l.append(n)
            if n<2: break
        else: break
    return len([1 for i in l if i%2])
print([n for n in range(40, 1501) if a(n)==10]) # _Indranil Ghosh_, Apr 14 2017
```
