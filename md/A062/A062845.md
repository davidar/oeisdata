## Python
```Python
def BaseUp(n,b):
    up, b1 = 0, 1
    while n > 0:
        up, b1, n = up+(n%b)*b1, b1*(b+1), n//b
    return up
n, k = 1, 0
print(1,0)
while n < 35:
    n, k = n+1, k+1
    while BaseUp(k,2)%k != 0:
        k = k+1
    print(n,k) # _A.H.M. Smeets_, Mar 31 2020
```
