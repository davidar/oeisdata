## Python
```Python
from math import isqrt
from sympy import mobius
def A062503(n):
    def f(x): return n-1+x-sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1))
    kmin, kmax = 1,2
    while f(kmax) >= kmax:
        kmax <<= 1
    while True:
        kmid = kmax+kmin>>1
        if f(kmid) < kmid:
            kmax = kmid
        else:
            kmin = kmid
        if kmax-kmin <= 1:
            break
    return kmax**2 # _Chai Wah Wu_, Aug 19 2024
```
