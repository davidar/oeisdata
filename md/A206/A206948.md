def a(n, adict={0:0, 1:0, 2:0, 3:2, 4:19, 5:131}):
  if n in adict:
    return adict[n]
  adict[n]=11*a(n-1)-40*a(n-2)+55*a(n-3)-30*a(n-4)+6*a(n-5)
  return adict[n]
