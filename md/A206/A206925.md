## Python

def A206925(n):
    s = bin(n)[2:]
    k = len(s)
    return sum(1 for i in range(k) for j in range(i+1,k+1) if s[i:j] == s[j-1:i-1-k:-1]) # _Chai Wah Wu_, Jan 31 2023
