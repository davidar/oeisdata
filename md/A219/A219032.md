from sympy import integer_nthroot
def A219032(n):
    s = str(n*n)
    m = len(s)
    return len(set(filter(lambda x: integer_nthroot(x,2)[1], (int(s[i:j]) for i in range(m) for j in range(i+1,m+1))))) # _Chai Wah Wu_, Oct 19 2021
