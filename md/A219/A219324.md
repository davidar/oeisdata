## Python
```Python
from sympy import Matrix
A219324_list = []
for n in range(1,10**4):
    s = [int(d) for d in str(n)]
    m = len(s)
    if n == Matrix(m, m, lambda i, j: s[(i-j) % m]).det():
        A219324_list.append(n) # _Chai Wah Wu_, Oct 18 2021
```
