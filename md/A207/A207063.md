## Python
```Python
def aupton(terms):
    alst = [0]
    for n in range(2, terms+1):
        an = alst[-1] + 1
        while bin(an^alst[-1]).count('1') != 2:  an += 1
        alst.append(an)
    return alst
print(aupton(54)) # _Michael S. Branicky_, Jul 07 2021
```
