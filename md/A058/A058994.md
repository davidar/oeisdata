## Python

```Python
from sympy import isprime
k, m, A058994_list = 1, 7,  []
while k <= 10**3:
    if isprime(int(str(m)[::-1])):
        A058994_list.append(k)
    k += 1
    m *= 7 # _Chai Wah Wu_, Mar 09 2021
```
