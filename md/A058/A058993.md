## Python

from sympy import isprime
k, m, A058993_list = 1, 5,  []
while k <= 10**3:
    if isprime(int(str(m)[::-1])):
        A058993_list.append(k)
    k += 1
    m *= 5 # _Chai Wah Wu_, Mar 09 2021
