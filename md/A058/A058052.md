## Python
```Python
from numba import njit
@njit
def d(x, y, n):
  for k in range(n):
    mask = (1 << (n-k)) - 1
    if x & mask == (y >> k): return k
  return n
@njit
def a(n):
  s = 0
  for x in range(2**(n-1)):
    for y in range(2**n):
      s += d(x, y, n)
  return 2*s
print([a(n) for n in range(1, 15)]) # _Michael S. Branicky_, Feb 18 2021
```
