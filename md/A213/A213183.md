a = [1]*992
R = 1
i = 2
while i<900:
        for t in range(R):
                a[i] = a[i-R]
                i += 1
        R += 1
        a[i] = a[i-1] + 1
        i += 1
        a[i] = a[i-1] + 1
        i += 1
for i in range(1,99):
        print(a[i], end=',')
