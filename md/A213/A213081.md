## Python

src = [0]*1024
dst = [0]*1024
for i in range(1,39):
    dst[0] = dst[i-1] = i
    for j in range(1,i-1):
        dst[j] = src[j-1]^src[j]
    for j in range(i):
        src[j] = dst[j]
        print(dst[j], end=',')
