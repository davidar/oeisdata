from fractions import Fraction
num_terms = 100
S = [1, 2]
for n in range(num_terms-2):
    s = Fraction((S[n]+S[n+1]+1),S[n+1]).numerator
    S.append(s)
print(S) # Should print the sequence to the length specified.
