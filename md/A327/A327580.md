```Python
import numpy as np
for n in range(1, 21):
    A = [ sum([ np.cos(np.pi * (j*k-1)/2) for k in range(1, n//j+1)])
            for j in range(1, n+1) ]
    print(np.array(np.rint(A), dtype=int))
```
