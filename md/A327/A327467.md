## Python
```Python
from sympy import sieve as prime
def A327467(n):
    array, np, k = [2], 1, 1
    while n not in array:
        temp = []; np += 1; k += 1
        for item in array:
            temp.append(item + prime[k])
            temp.append(abs(item - prime[k]))
        array = set(temp)
    return np
print([A327467(n) for n in range(0, 100)]) # _Karl-Heinz Hofmann_, May 30 2023
```
