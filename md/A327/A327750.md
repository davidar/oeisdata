## Python
```Python
def test(n):
    m, p = n, 1
    while m > 0:
        m, p = m//10, p*(m%10)
    if p == 0:
        return 0
    m, q = n+p, 1
    while m > 0:
        m, q = m//10, q*(m%10)
    return p == q
n, a = 0, 0
while n < 100:
    a = a+1
    if test(a):
        n = n+1
    print(n,a) # _A.H.M. Smeets_, Sep 25 2019
```
