## Python
```Python
def aupton(nn):
  alst = [0, 1, 1]
  for n in range(3, nn+1):
    b = list(map(int, bin(alst[n-1])[2:]))
    alst.append(alst[n-1] + sum(bi*alst[n-i-2] for i, bi in enumerate(b)))
  return alst[:nn+1]
print(aupton(37)) # _Michael S. Branicky_, Jan 19 2021
```
