## Python
```Python
def a(n):
    ternary = []
    i = math.floor(math.log(n,3))
    while i >= 0:
        for j in range (2, -1, -1):
            if j*(3**i) <= n:
                if (j==2): ternary.append(-1)
                else: ternary.append(j)
                n -= j*(3**i)
                break
        i -=1
    for k in range (0, len(ternary)):
        n += ternary[k]*(3**k)
    return n
```
