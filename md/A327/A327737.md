## Go
```Go
package main
import (
    "fmt"
    "strconv"
)
func main() {
    // Due to limitations in strconv, this will only work for the first 36 terms
    for i := 1; i <= 36; i++ {
        count := i
        for base := 2; base <= i; base++ {
            count += len(strconv.FormatInt(int64(i), base))
        }
        fmt.Printf("%d, ", count)
    }
}
```
