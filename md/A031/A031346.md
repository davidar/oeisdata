## Python

from operator import mul
from functools import reduce
def A031346(n):
    mp = 0
    while n > 9:
        n = reduce(mul, (int(d) for d in str(n)))
        mp += 1
    return mp
# _Chai Wah Wu_, Aug 23 2014
