## Maple
```Maple
A031923 := proc(n)
    local n0,i,r,s,m ;
    n0 := n-1 ;
    i := floor(n0/4) ;
    r := combinat[fibonacci](i+2) ;
    s := combinat[fibonacci](i+3) ;
    m := modp(n0,4) ;
    r^(4-m)*s^m ;
end proc:
seq(A031923(n),n=1..50) ; # _R. J. Mathar_, Jan 23 2022
```
