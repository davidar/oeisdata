## Perl
```Perl
for my $half ( 1 .. 4 ) {
  my $N = 2 * $half;  # only even widths apply
  my $vector = (1 << ($N-1)) | ((1 << ($N/2-1)) - 1);  # first key
  my $n = 1; $n *= $_ for 2 .. $N;    # N!
  my $d = 1; $d *= $_ for 2 .. $N/2;  # (N/2)!
  for (1 .. $n/($d*$d*2)) {
    print "$vector, ";
    my ($v, $d) = ($vector, 0);
    until ($v & 1 or !$v) { $d = ($d << 1)|1; $v >>= 1 }
    $vector += $d + 1 + (($v ^ ($v + 1)) >> 2);  # next key
  }
} # _Ruud H.G. van Tol_, Mar 30 2014
```
