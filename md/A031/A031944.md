## Python

from sympy.ntheory import count_digits
def ok(n): c = count_digits(n, 3); return all(c[d] > 0 for d in [0, 1, 2])
print([k for k in range(136) if ok(k)]) # _Michael S. Branicky_, Nov 15 2021
