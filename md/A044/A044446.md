## Python
```Python
from sympy.ntheory.factor_ import digits
def has10(k): s = "".join(map(str, digits(k, 4)[1:])); return "10" in s
def ok(n): return has10(n) and not has10(n+1)
print(list(filter(ok, range(965)))) # _Michael S. Branicky_, Apr 20 2021
```
