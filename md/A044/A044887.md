## Python
```Python
from itertools import groupby
def ok(n):
    rl_sums = [0, 0]
    for k, g in groupby(hex(n)[2:]):
        rl = len(list(g))
        rl_sums[rl%2] += rl
    return rl_sums[0] == rl_sums[1]
print(list(filter(ok, range(4359)))) # _Michael S. Branicky_, Sep 11 2021
```
