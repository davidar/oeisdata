## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def A237627_gen(): # generator of terms
    p = 1
    while (p:=nextprime(p)):
        if isprime((p-1)**2+1):
            yield p*((p-1)**2+1)
A237627_list = list(islice(A237627_gen(),20)) # _Chai Wah Wu_, Feb 27 2023
```
