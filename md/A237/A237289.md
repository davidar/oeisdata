```Maple
isSumDist := proc(n,k)
    local dvs ;
    dvs := numtheory[divisors](n) ;
    for s in combinat[powerset](dvs) do
        add(m,m=op(s)) ;
        if % = k then
            return true;
        end if;
    end do:
    false ;
end proc:
A237289 := proc(n)
    local a;
    a := 0 ;
    for k from 1 to numtheory[sigma](n) do
        if not isSumDist(n,k) then
            a := a+k;
        end if;
    end do:
    a ;
end proc:
```

## Python
```Python
from sympy import divisors
def A237289(n):
    ds = divisors(n)
    c, s = {0}, sum(ds)
    for d in ds:
        c |=  {a+d for a in c}
    return (s*(s+1)>>1)-sum(a for a in c if 1<=a<=s) # _Chai Wah Wu_, Jul 05 2023
```
