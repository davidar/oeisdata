with(numtheory);
A216384:= proc(i)
local a,b,n,p,pfs;
for n from 1 to i do
pfs:=ifactors(n)[2]; a:=n*add(op(2,p)/op(1,p),p=pfs);
if a<n then b:=a;
   while b<n do
     pfs:=ifactors(a)[2]; a:=a*add(op(2,p)/op(1,p),p=pfs);
     if a=0 then break; else b:=b+a; fi;
   od;

end:
