## Python
```Python
from sympy.ntheory.factor_ import core
from sympy import primefactors
def ok(n):
    if n<3: return True
    if core(n) == n:
        for p in primefactors(n):
            if (n - 2)%(p - 1): return False
        return True
    return False
print([n for n in range(1, 501) if ok(n)]) # _Indranil Ghosh_, Apr 23 2017
```
