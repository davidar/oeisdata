def a(n):
   if n < 3:
      return n
   l = [0,1,2]
   for i in range(n-2):
      l[i%3] = ( (16*i-64*i**3)*l[i%3]+(12+72*i+112*i**2+32*i**3)*l[(i+1)%3]+(-26-62*i-4*i**3-36*i**2)*l[(i+2)%3] ) / (-5-7*i-2*i**2)
   return l[i%3]
