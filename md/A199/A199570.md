def A199570_list(row):
    A = [1]
    for i in range(2,row+1):
        z = 2*(3**(i-2))
        for j in range(1,z+1):
            if j%2 != 0: A.append(A[int((j-1)/2)])
            else: A.append(i)
    return(A) # _John Tyler Rascoe_, Feb 19 2023
