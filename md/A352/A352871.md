## Python
```Python
def sumdigits(n: int) -> int:
    return sum(map(int, str(n)))
def a(n: int) -> int:
    i = 0
    while True:
        denom = sumdigits(n)
        if denom == 1:
            i = -1
            break
        elif n % denom == 0:
            i = i + 1
            n = n // denom
        else:
            break
    return i
```
