## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen(): # generator of terms
    p, q = 3, 5
    while True:
        if q == p+2:
            t, s = (p*q-1)//2, p+q
            if isprime(t) and isprime(t+s) and isprime(t-s):
                yield p
        p, q = q, nextprime(q)
print(list(islice(agen(), 36))) # _Michael S. Branicky_, Apr 10 2022
```
