## Python
```Python
from sympy import ceiling, log
def A352040(n):
    k = 10*n-1+int(ceiling((10*n-1)*log(5,2)))
    s = str(c := 2**k)
    while any(s.count(d) < n for d in '0123456789'):
        c *= 2
        k += 1
        s = str(c)
    return k   # _Chai Wah Wu_, Apr 16 2022
```
