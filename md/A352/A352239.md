## Python
```Python
from itertools import count, islice
def A352239_gen(): # generator of terms
    for l in count(0):
        k14 = 14**(14*10**l)
        for k in range(14*10**l,15*10**l):
            if str(k14)[:2] == '14':
                yield k
            k14 *= 14
A352239_list = list(islice(A352239_gen(),20))
```
