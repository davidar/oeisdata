## Python
```Python
from math import gcd
from itertools import islice
def agen(): # generator of terms
    aset, b, c = {1, 3, 5}, 3, 5
    yield from [1, b, c]
    while True:
        k = 1
        while k in aset or any(gcd(t, k) != 1 for t in [b, c]): k+= 1
        b, c = c, k
        aset.add(k)
        yield k
print(list(islice(agen(), 68))) # _Michael S. Branicky_, Apr 14 2022
```
