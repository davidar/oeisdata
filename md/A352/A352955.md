## Python
```Python
from sympy import isprime, factorint
from itertools import islice
def rem2(n):
    while n%2 == 0: n //= 2
    return n
def agen():
    b, c = 11, 19
    yield 11
    while True:
        yield c
        k = (c+2)//b + 1
        m = b*k
        while not isprime(m-c) or min(factorint(rem2(k)), default=b+1) < b:
            m += b
            k += 1
        b, c = c, m-c
print(list(islice(agen(), 17))) # _Michael S. Branicky_, Apr 12 2022
```
