## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    alst = [1]
    for n in count(2):
        yield alst[-1]
        b = alst[-1] + 1
        alst.append(sum(alst[-1-i]*b**i for i in range(len(alst))))
print(list(islice(agen(), 7))) # _Michael S. Branicky_, May 22 2022
```
