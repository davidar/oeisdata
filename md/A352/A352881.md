def a(n):
  D = {}
  for x in range(2, 10**n):
    for y in range(x, 10**n):
      z, r = divmod(x * y, x + y)
      if r == 0:
        if z in D:
          D[z] += 1
        else:
          D[z] = 1
  best_Z = 0
  for z in D:
    if D[z] > best_Z:
      best_Z = D[z]
      best_z = z
  return best_z
