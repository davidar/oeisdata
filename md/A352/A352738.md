from itertools import islice
def A352738_gen(): # generator of terms
    c, k, ks, m, ms = 0, 1, 2, 1, 1
    while True:
        for n in range(ks,ks+2*k):
            c += n
            if c == ms:
                yield c
            elif c > ms:
                ms += 2*m+1
                m += 1
        ks += 2*k+1
        k += 1
