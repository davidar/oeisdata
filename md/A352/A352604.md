## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen():
    p = 2
    while True:
        if isprime(p**2 + 3*p + 1) and isprime(p**2 + p - 1):
            yield p
        p = nextprime(p)
print(list(islice(agen(), 53))) # _Michael S. Branicky_, Mar 22 2022
```
