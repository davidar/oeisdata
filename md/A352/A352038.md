from math import prod
from sympy import factorint
def A352038(n): return prod((p**(10*(e+1))-1)//(p**10-1) for p, e in factorint(n).items() if p > 2) - (n**10 if n % 2 else 0) # _Chai Wah Wu_, Mar 01 2022
