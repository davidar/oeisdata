## Maple
```Maple
A352047 := proc(n)
    local a,d ;
    a := 0 ;
    for d in numtheory[divisors](n) do
        if type(d,'odd') and d < n then
            a := a+n/d ;
        end if;
    end do:
    a ;
end proc:
```
