from itertools import count, islice
def agen():
    b = "1"
    for k in count(1):
        bk = bin(k)[2:]
        idx = b.find(bk) + 1
        if idx == 0: yield k
        b += bin(idx)[2:]
print(list(islice(agen(), 73))) # _Michael S. Branicky_, Mar 18 2022
