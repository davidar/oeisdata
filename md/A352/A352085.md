## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    for m in count(1):
        q, r = divmod(bin(m**2).count('1'), bin(m).count('1'))
        if r == 0:
            yield q
print(list(islice(agen(), 100))) # _Michael S. Branicky_, Mar 05 2022
```
