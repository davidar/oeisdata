from itertools import islice
def A352716_gen(): # generator of terms
    yield 1
    l1, s, b = 1, 2, set()
    while True:
        i = s
        while True:
            if not (i in b or (i & l1).bit_count() % 2):
                yield i
                l1 = i
                b.add(i)
                while s in b:
                    b.remove(s)
                    s += 1
                break
            i += 1
A352716_list = list(islice(A352716_gen(),20)) # _Chai Wah Wu_, Apr 02 2022
