## Python
```Python
from sympy import nextprime
from itertools import islice
def agen():
    p, evod = 2, [0, 1]
    while True:
        yield evod[1] - evod[0]
        p = nextprime(p); evod[bin(p).count('1')%2] += 1
print(list(islice(agen(), 97))) # _Michael S. Branicky_, Dec 21 2021
```
