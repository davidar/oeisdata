from itertools import count, islice
def A130241_gen(): # generator of terms
    a, b = 1, 3
    for i in count(1):
        yield from (i,)*(b-a)
        a, b = b, a+b
A130241_list = list(islice(A130241_gen(),40)) # _Chai Wah Wu_, Jun 08 2022
