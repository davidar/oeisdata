from math import gcd
def A130335(n):
    k, Tn, Tm = 1, n*(n+1)//2, (n+1)*(n+2)//2
    while gcd(Tn,Tm) != 1:
        k += 1
        Tm += k+n
    return k # _Chai Wah Wu_, Sep 16 2021
