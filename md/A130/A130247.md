from itertools import islice, count
def A130247_gen(): # generator of terms
    yield from (1,0)
    a, b = 3, 4
    for i in count(2):
        yield from (i,)*(b-a)
        a, b = b, a+b
A130247_list = list(islice(A130247_gen(),40)) # _Chai Wah Wu_, Jun 08 2022
