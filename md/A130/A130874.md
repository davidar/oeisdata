## Python
```Python
from sympy.ntheory.factor_ import antidivisors
def A130874():
    sum = 1
    i = 2 #(offset-1)
    while True:
        i += 1
        for j in antidivisors(i):
            sum *= j
        yield sum
        if i == 50:#Generator stops after calculating a(50)
            break
for i in A130874():
    print(i) # _Hakan Icoz_, Dec 26 2021
```
