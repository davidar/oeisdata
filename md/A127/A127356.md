## Python
```Python
from sympy import isprime, nextprime, prime
def a(n):
    if n == 1: return 1
    k, pn = 2, prime(n)
    while not isprime(pn + k*k): k += 2
    return k
print([a(n) for n in range(1, 81)]) # _Michael S. Branicky_, Nov 11 2022
```
