## Python
```Python
def A127705_list(n):
    upto = n + 1
    p = [i for i in range(upto)]
    for i in range(2, upto):
        for j in range(i + i, upto , i):
            p[j] -= p[i]
    return p[1::]
print(A127705_list(1000))  # _Peter Luschny_, Sep 02 2023
```
