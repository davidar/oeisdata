## Python
```Python
import math
def binomial(n,m):
    a=1
    for k in range(n-m+1,n+1):
        a *= k
    return a//math.factorial(m)
def A127548(n):
    if n == 0:
        return 1
    a=0
    for s in range(1,n+1):
        a += (-1)**(n-s)*binomial(s+n-1,2*s-1)*math.factorial(s)
    return a
for n in range(30):
    print(A127548(n))
```
