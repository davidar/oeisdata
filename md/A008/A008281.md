from functools import cache
@cache
def seidel(n):
    if n == 0: return [1]
    rowA = seidel(n - 1)
    row = [0] + seidel(n - 1)
    row[1] = row[n]
    for k in range(2, n + 1): row[k] = row[k - 1] + rowA[n - k]
    return row
def A008281row(n): return seidel(n)
