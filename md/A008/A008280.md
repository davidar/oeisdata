## Sage

# Algorithm of L. Seidel (1877)
# Prints the first n rows of the triangle.
def A008280_triangle(n) :
    A = {-1:0, 0:1}
    k = 0; e = 1
    for i in range(n) :
        Am = 0
        A[k + e] = 0
        e = -e
        for j in (0..i) :
            Am += A[k]
            A[k] = Am
            k += e
        print([A[z] for z in (-i//2..i//2)])
A008280_triangle(10) # _Peter Luschny_, Jun 02 2012
