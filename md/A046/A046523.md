## Python
```Python
from sympy import factorint
def P(n):
    f = factorint(n)
    return sorted([f[i] for i in f])
def a(n):
    x=1
    while True:
        if P(n) == P(x): return x
        else: x+=1 # _Indranil Ghosh_, May 05 2017
```
