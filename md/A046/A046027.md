## Python
```Python
from math import isqrt
from sympy import mobius, factorint
def A046027(n):
    def f(x): return n+sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1))
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    s = factorint(m)
    return next(p for p in sorted(s) if s[p]>1) # _Chai Wah Wu_, Jul 22 2024
```
