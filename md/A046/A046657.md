## Python
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def A046657(n): # based on second formula in A018805
    if n == 0:
        return 0
    c, j = 0, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*(4*A046657(k1)-1)
        j, k1 = j2, n//j2
    return (n*(n-1)-c+j)//4 # _Chai Wah Wu_, Mar 25 2021
```
