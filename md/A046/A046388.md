from sympy import factorint
def ok(n):
    if n < 2 or n%2 == 0: return False
    f = factorint(n)
    return len(f) == 2 and sum(f.values()) == 2
print([k for k in range(304) if ok(k)]) # _Michael S. Branicky_, May 03 2022
