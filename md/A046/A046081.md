## Python
```Python
from sympy import factorint
def a(n):
    p1, p2 = 1, 1
    for i in factorint(n).items():
        if i[0] % 4 == 1:
            p2 *= i[1] * 2 + 1
        p1 *= i[1] * 2 + 1 - (2 if i[0] == 2 else 0)
    return (p1 + p2)//2 - 1
print([a(n) for n in range(1, 100)])  # _Oleg Sorokin_, Mar 02 2023
```
