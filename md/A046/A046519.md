from sympy import nextprime
def A046519(n):
    m = 10**(n-1)
    while (m:=nextprime(m)):
        k = m
        while k not in {1,37,58,89,145,42,20,4,16}:
            k = sum((0, 1, 4, 9, 16, 25, 36, 49, 64, 81)[ord(d)-48] for d in str(k))
        if k == 1:
            return m # _Chai Wah Wu_, Aug 02 2023
