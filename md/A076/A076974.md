## Python
```Python
def A076974():
    D = {}
    q = 2
    while True:
        if q not in D:
            yield q
            D.setdefault(q+2, []).append(q)
        else:
            for p in D[q]:
                D.setdefault(p+q, []).append(p)
            del D[q]
        q += 1
```
