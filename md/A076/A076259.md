## Python
```Python
from math import isqrt
from sympy import mobius
def A076259(n):
    def f(x): return n+x-sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1))
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    r, k = n+1, f(n+1)+1
    while r != k:
        r, k = k, f(k)+1
    return int(r-m) # _Chai Wah Wu_, Aug 15 2024
```
