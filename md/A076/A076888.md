from sympy import divisor_count
def A076888(n):
    y = 10*(x:=10**(len(str(n+1>>1))-1))
    return divisor_count(int((c:=n+1-x)*x+int(str(c)[-2::-1] or 0) if n+1<x+y else (c:=n+1-y)*y+int(str(c)[::-1] or 0))) # _Chai Wah Wu_, Jul 24 2024
