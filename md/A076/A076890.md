from sympy import primepi
def A076890(n):
    if n == 1: return 0
    y = 10*(x:=10**(len(str(n+1>>1))-1))
    return primepi((c:=n+1-x)*x+int(str(c)[-2::-1]or 0) if n<x+y else (c:=n+1-y)*y+int(str(c)[-1::-1]or 0)) # _Chai Wah Wu_, Jun 14 2024
