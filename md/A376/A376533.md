## Python

def A376533(n):
    m, r, c, a, b, d = (1<<(n<<1))-1, n**2, 1, 3, n**2, 1
    for k in range(1,r+1):
        c += b//d*a&m
        a = a*3&m
        b *= r-k
        d *= k+1
    return c>>(n<<1) # _Chai Wah Wu_, Oct 09 2024
