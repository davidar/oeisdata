## Python
```Python
def is_valid_k(k):
    k_squared = k ** 2
    for digit in str(k_squared):
        d = int(digit)
        if d == 0 or k % d != 0:
            return False
    return True
def find_valid_k(max_k):
    valid_k = []
    for k in range(1, max_k + 1):
        if is_valid_k(k):
            valid_k.append(k)
    return valid_k
max_k = 10000
result = find_valid_k(max_k)
print(result)
```
