from math import gcd
def A376203(n):
    c, m = 0, (n<<1)-1
    for x in range(1,m):
        if gcd(x,m) == 1:
            for y in range(x,m):
                if gcd(y,m)==gcd(z:=x+y,m)==1 and not (w:=z**2-x*y)//gcd(w,x*y*z)%m:
                    c += 1
    return c>>1 # _Chai Wah Wu_, Oct 06 2024
