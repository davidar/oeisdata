def A376007(n):
    if (m:=n&7)==0 or m==4:
        return n*(n**2-4)+8>>3
    elif m==1 or m==7:
        return n*(n**2-9)+24>>4
    elif m==2 or m==6:
        return n**3>>3
    else:
        return n*(n**2-1)+8>>4 # _Chai Wah Wu_, Sep 27 2024
