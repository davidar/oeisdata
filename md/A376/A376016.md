## Python
```Python
from math import comb
from itertools import takewhile
from sympy import divisors
def A376016(n): return sum(d**(m:=n//d)*comb(m,d) for d in takewhile(lambda d:d**2<=n,divisors(n))) # _Chai Wah Wu_, Sep 06 2024
```
