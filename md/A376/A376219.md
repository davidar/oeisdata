from itertools import count, islice
from math import gcd
from sympy import isprime
def A376219_gen(): # generator of terms
    c = 0
    for n in count(1):
        for l in count(1):
            if gcd(n,(m:=10**l)+1)==1:
                r = m//10
                a = m*(n+r)+r
                for k in range(r,m):
                    if isprime(a):
                        if k>c:
                            yield n
                            c = k
                        break
                    a += m+1
                else:
                    continue
                break
