## Python
```Python
from sympy import integer_nthroot
def A376279(n):
    def f(x): return n-1+x-x//3-integer_nthroot(x,3)[0]+integer_nthroot(x//27,3)[0]
    m, k = n-1, f(n-1)
    while m != k: m, k = k, f(k)
    return m
```
