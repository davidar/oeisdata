```Maple
a:=proc(n)
 local x,y,count;
  count:=0:
  for x from 0 to n-1 do
   for y from x to n-1 do
     if (x+y) mod n =(x^2+y^2) mod n  then count:=count+1; fi;
   od:
  od:
  count;
end:
```

## Python
```Python
def A376646(n):
    c = 0
    for x in range(n):
        m = x*(1-x)%n
        c += sum(1 for y in range(x,n) if y*(y-1)%n==m)
    return c # _Chai Wah Wu_, Oct 02 2024
```
