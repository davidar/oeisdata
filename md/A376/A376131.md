## Python
```Python
def f0(n):
    if n <= 2:
        return 0
    else:
        return (n+1) // 2 - 1 + f0((n+1)//2 - 1)
def a(n):
    numchip = 2*n
    total = 0
    firetime = f0(numchip)
    l = 0
    while firetime > 0:
        total += (2**l) * firetime
        numchip = (numchip+1)//2 - 1
        firetime = f0(numchip)
        l += 1
    return total
print([a(n) for n in range(1, 66)])
```
