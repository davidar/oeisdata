## Python
```Python
def A376427(n):
    s = set()
    for x in range(n):
        for y in range(x,n):
            xy, xyp = x*y%n, (x+y)%n
            for z in range(y,n):
                try:
                    s.add((xyp+z+pow(xy*z%n,-1,n))%n)
                except:
                    continue
    return len(s) # _Chai Wah Wu_, Sep 23 2024
```
