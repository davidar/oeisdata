## Python
```Python
def f(n): return (d:=list(map(int, str(n))))[0] * sum(di*di for di in d)
def ok(n):
    if n == 1: return True
    traj = {n}
    while (n:=f(n)) not in traj: traj.add(n)
    return 1 in traj
print([k for k in range(750) if ok(k)]) # _Michael S. Branicky_, Sep 18 2024
```
