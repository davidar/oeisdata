def expand(m):
  i = len(m)
  res = [[0 for _ in range(2*i)] for _ in range(2*i)]
  for x in range(i):
    for y in range(i):
      res[y][x] = m[y][x]
      res[y][x+i] = m[y][x] + x
      res[y+i][x] = m[y][x] - x
      res[y+i][x+i] = m[y][x] + y
  return res
a = []
m = [[1]]
for _ in range(11):
  m = expand(m)
for i in range(len(m)):
  for j in range(i+1):
    a.append(m[j][i-j])
