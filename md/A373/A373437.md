    local n, result:
    result := []:

        if sigma(sigma(2*n)) = 2*sigma(sigma(n)) then
            result := [op(result), n]:
        end if

## Python
```Python
from sympy import divisor_sigma as sigma
def P(q):
    result = []
    for n in range(1, q + 1):
        if sigma(sigma(2 * n)) == 2 * sigma(sigma(n)):
            result.append(n)
    print(result)
P(10**3)
```
