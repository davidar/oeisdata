def aupton(terms):
    a = [0, 0, 0]
    for n in range(3, terms):
        next_term = 2**(n-3) + a[n-1] + a[n-2] + a[n-3] - 1
        a.append(next_term)
    return a[:terms]
print(aupton(35))
