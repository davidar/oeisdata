from sympy import integer_nthroot
def A373727(n): return max(sum(int(d) for d in str(m**3)) for m in range(1+integer_nthroot(10**(n-1)-1,3)[0],1+integer_nthroot(10**n-1,3)[0])) # _Chai Wah Wu_, Jun 26 2024
