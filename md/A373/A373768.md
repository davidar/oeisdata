## Python
```Python
from itertools import permutations
def a(n):
    perms = permutations(range(n))
    not_term = set()
    i = 0
    for p in perms:
        if p not in not_term:
            P = tuple(p[i] for i in p)
            while P != p:
                not_term.add(P)
                P = tuple(p[i] for i in P)
            i += 1
    return i
```
