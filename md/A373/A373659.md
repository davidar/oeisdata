from functools import cache
@cache
def T(n, k):
    return T(n, k - 1) + sum(T(2 * j, k - 1) * T(n - 1 - 2 * j, k)
    for j in range(1 + (n - 1) // 2)) if k > 0 else 1
def a(n): return T(n, n)
print([a(n) for n in range(21)])
