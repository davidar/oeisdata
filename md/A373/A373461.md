## Python
```Python
from sympy.ntheory.primetest import is_square
from sympy.core.power import isqrt
A003059 = lambda n: isqrt((n)-1)+1
def a(n):
  i = 1
  while True:
    s = A003059(n*i)
    if is_square(m:=pow(s,2,n)):
      return s-isqrt(m)
    i+=1
print([a(n) for n in range(1, 69)])
```
