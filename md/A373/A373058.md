## Python
```Python
from math import prod, isqrt
from sympy import mobius, factorint
def A373058(n):
    def f(x): return n+sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1))
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    return prod((p**(e+1)-1)//(p-1)-1 for p, e in factorint(m).items())-m # _Chai Wah Wu_, Jul 22 2024
```
