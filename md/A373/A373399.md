## Python
```Python
def a(n):
    target = bin(n)[2:]
    for m in range(1, n):
        b = bin(m)[2:]
        mstar = b*(2*len(target)//len(b))
        if target in mstar:
            return m
    return n
print([a(n) for n in range(1, 74)]) # _Michael S. Branicky_, Jun 14 2024
```
