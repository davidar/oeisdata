from itertools import count, islice
from sympy import isprime, nextprime
def A373799_gen(): # generator of terms
    a, p = 1, 3
    for i in count(1):
        if isprime(a):
            yield i
            a = p-1
        else:
            a = (a<<1)+1
        p = nextprime(p)
