def fib(arb1, arb2, nth):
    if nth == 0:
        return arb1
    if nth == 1:
        return arb2
    x = [0]*nth
    x[0] = arb1
    x[1] = arb2
    for i in range(2, nth, 1):
        x[i] = x[i-1]+x[i-2]
    return x[-1]
def fib2d(n):
    return fib(1, fib(1, 1, n), n)
def fib3d(n):
    return fib(1, fib(1, fib(1, 1, n), n), n)
def slowfibnd(n): # This is an inelegant way to generate a(n)
    begin = "fib(1, 0+1, n)"
    for x in range(n-1):
        begin = begin.replace('0+1', 'fib(1, 0+1, n)')
    return eval(begin)
