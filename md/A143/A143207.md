## Python
```Python
from sympy import integer_log
def A143207(n):
    def bisection(f,kmin=0,kmax=1):
        while f(kmax) > kmax: kmax <<= 1
        while kmax-kmin > 1:
            kmid = kmax+kmin>>1
            if f(kmid) <= kmid:
                kmax = kmid
            else:
                kmin = kmid
        return kmax
    def f(x):
        c = n+x
        for i in range(integer_log(x,5)[0]+1):
            for j in range(integer_log(m:=x//5**i,3)[0]+1):
                c -= (m//3**j).bit_length()
        return c
    return bisection(f,n,n)*30 # _Chai Wah Wu_, Sep 16 2024
```
