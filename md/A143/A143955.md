## Python
```Python
from functools import cache
@cache
def B(n, k):
    if n <= 0 or k <= 0: return 0
    if n == k: return 1
    return B(n - 1, k) + B(n, k - 1)
def A143955(k):
    return B(k + 3, k - 2)
print([A143955(n) for n in range(26)]) # _Peter Luschny_, May 15 2022
```
