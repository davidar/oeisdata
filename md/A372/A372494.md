from sympy import sqrt_mod_iter
def A372494(n):
    k = (1<<n)-1
    k2 = k>>1
    return min((d>>1 if d&1 else (d>>1)+k2 for d in sqrt_mod_iter(-3,k)),default=-1) # _Chai Wah Wu_, May 03 2024
