import math
L=20 #tr are terms A372477 below L**2+1
numb=set()
Lmax=math.trunc((1+2*math.sqrt(3*L**2+1))/3)+1
tr=set()
tr.add(1)
for n in range(2,Lmax):
  for k in range(0,n):
    p1=n*n+k*k-k*n
    p2=p1+k-n
    if p1<=L**2:
      tr.add(p1)
    if p2<=L**2:
      tr.add(p2)
for k in  range (1, L**2):
  numb.add(k)
print(sorted(numb-tr))
