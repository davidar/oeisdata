## Python
```Python
from math import isqrt
def ok(m):
    tot = m**2*(m**2+1)//2
    for k in range(1, m):
        skm = tot - k**2*(k**2-1)//2
        if isqrt(skm)**2 == skm:
            return True
    return False
print([m for m in range(621) if ok(m)]) # _Michael S. Branicky_, May 10 2024
```
