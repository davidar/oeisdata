from itertools import count
def A372052(n):
    a, aset = 1, {0,1}
    for p in count(3,2):
        if a == n:
            return p>>1
        for b in count(a%p,p):
            if b not in aset:
                aset.add(b)
                a = b
                break # _Chai Wah Wu_, Apr 17 2024
