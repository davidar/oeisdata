## Python

from sympy import isprime, primefactors
for k in range(10**6):
    s0, s1 = sum(primefactors(k)), sum(primefactors(k + 1))
    if s0 == s1 and isprime(s0): print(k, end=', ') # _Jason Yuen_, Jun 05 2024
