## Python
```Python
from sympy import prod
def a(n):
  k, QS = 0,[]
  for i in range((n >> 1) + 1):
    if k > 0: QS.append(k)
    k += (i << 1) + 1
    k %= n
  return prod(set(QS))
print([a(n) for n in range(1, 34)])
```
