```Maple
R:= 0: RL:= [1]: nextp:= 2: m:= 1: count:= 0:
for i from 1 while count < 100 do
  L:= convert(i,base,2);
  if i = nextp then
    nextp:= 2*nextp;
    if R[1+nops(RL)] = m then RL:= [op(RL),m+1] fi;
    m:= m+1;
  fi;
  if convert(L[RL],`+`)::even
  then R:= R,i; count:= count+1
  fi
od:
R; # _Robert Israel_, May 28 2024
```

## Python
```Python
from itertools import count, islice
def agen():  # generator of terms
    aset = 0 # stored as a bitmask
    for k in count(0):
        if (k&aset).bit_count()%2 == 0:
            yield k
            aset += (1<<k)
print(list(islice(agen(), 63))) # _Michael S. Branicky_, Apr 28 2024
```
