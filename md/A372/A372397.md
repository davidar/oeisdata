## Python
```Python
def H(n,k):
    if n == 0:
        return 0
    else:
        i, x = 0, n-1
        while i < k:
            i, x = i+1, H(x,k)
        return n-x
n, nn = 0, 0
while n < 50:
    if nn == 0:
        Hno = H(nn,4)
    else:
        Hnn = H(nn,4)
        if Hnn == Hno:
            n += 1
            print(Hnn, end = ", ")
        Hno = Hnn
    nn += 1
```
