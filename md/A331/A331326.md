def A331326():
    sa, sb, ta, tb, n = 1, 2, 1, 0, 2
    yield 0
    yield ta
    while(True):
        s = 2*n*sb - ((n-1)**2)*sa
        t = 2*(n-1)*tb - ((n-1)**2)*ta
        sa, sb, ta, tb = sb, s, tb, t
        n += 1
        yield (s - t)//2
