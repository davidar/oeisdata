## Python
```Python
from itertools import count
def A331250(n):
    m, c = n**2<<4, 0
    for k in count(1):
        if (k**2<<2) - 1 > m:
            break
        for j in range((k>>1)+1,k+1):
            for i in range(k-j+1,j+1):
                if ((-i + j + k)*(i - j + k)*(i + j - k)*(i + j + k)) > m:
                    break
                c += 1
    return c # _Chai Wah Wu_, Aug 25 2023
```
