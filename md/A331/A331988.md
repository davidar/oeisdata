```Python
from itertools import permutations, combinations_with_replacement
def A331988(n,k): # compute T(n,k)
    if k == 1:
        count = 1
        for i in range(1,n):
            count *= i+1
        return count
    ntuple, count = tuple(range(1,n+1)), 0
    for s in combinations_with_replacement(permutations(ntuple,n),k-2):
        t = list(ntuple)
        for d in s:
            for i in range(n):
                t[i] += d[i]
        t.sort()
        w = 1
        for i in range(n):
            w *= (n-i)+t[i]
        if w > count:
            count = w
    return count
```
