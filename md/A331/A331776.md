# First define z(n) = A115004
z := proc(n)
    local a, b, r ;
    r := 0 ;
    for a from 1 to n do
    for b from 1 to n do
        if igcd(a, b) = 1 then
            r := r+(n+1-a)*(n+1-b);
        end if;
    end do:
    end do:
    r ;
end proc:
A331776 := n -> if n=1 then 4 else 4*z(n)+16*n^2 - 20*n; fi;

from sympy import totient
