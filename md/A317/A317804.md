## Python
```Python
from heapq import heappush, heappop
def sequence():
    pq = [1]
    seen = set(pq)
    while True:
        value = heappop(pq)
        yield value
        seen.remove(value)
        for x in 2 * value, 12 * value:
            if x not in seen:
                heappush(pq, x)
                seen.add(x)
seq = sequence()
finalsequence_list = [next(seq) for i in range(100)]  # _Dario Ch_, Sep 01 2018
```
