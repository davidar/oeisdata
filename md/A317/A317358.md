def g(n,p,q): # compute (-n + sum_{k=1,n} k^p)  mod q
    c = (-n) % q
    for k in range(1,n+1):
        c = (c+pow(k,p,q)) % q
    return c
def A317358(n):
    k = 2
    while g(n,k-1,k):
        k += 1
    return k # _Chai Wah Wu_, Jul 30 2018
