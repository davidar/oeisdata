## Python
```Python
from sympy.utilities.iterables import partitions
def A317085(n):
    c = 1
    for d in partitions(n,m=n*2//3):
        l = len(d)
        if l > 0:
            k = sorted(d.keys())
            for i in range(l//2):
                if d[k[i]] != d[k[l-i-1]]:
                    break
            else:
                c += 1
    return c # _Chai Wah Wu_, Jun 22 2020
```
