## Python

from sympy import factorint, primepi
A317087_list = [1]
for n in range(1,10**5):
    d = factorint(n)
    k, l = sorted(d.keys()), len(d)
    if l > 0 and l == primepi(max(d)):
        for i in range(l//2):
            if d[k[i]] != d[k[l-i-1]]:
                break
        else:
            A317087_list.append(n) # _Chai Wah Wu_, Jun 23 2020
