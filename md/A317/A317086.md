## Python
```Python
from sympy.utilities.iterables import partitions
from sympy import integer_nthroot, isprime
def A317086(n):
    if n > 3 and isprime(n):
        return 1
    else:
        c = 1
        for d in partitions(n,k=integer_nthroot(2*n,2)[0],m=n*2//3):
            l = len(d)
            if l > 0:
                k = max(d)
                if l == k:
                    for i in range(k//2):
                        if d[i+1] != d[k-i]:
                            break
                    else:
                        c += 1
        return c # _Chai Wah Wu_, Jun 22 2020
```
