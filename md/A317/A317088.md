## Python
```Python
from sympy.utilities.iterables import partitions
from sympy import integer_nthroot
def A317088(n):
    if n == 0:
        return 1
    c = 0
    for d in partitions(n,k=integer_nthroot(2*n,2)[0]):
        l = len(d)
        if l > 0 and l == max(d):
            v = set(d.values())
            if len(v) == max(v):
                c += 1
    return c # _Chai Wah Wu_, Jun 23 2020
```
