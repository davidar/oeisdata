from __future__ import division
from sympy import isprime
def A266146(n):
     return 4*n if (n==1 or n==2) else sum(1 for d in range(-7,3) for i in range(n) if isprime(7*(10**n-1)//9+d*10**i)) # _Chai Wah Wu_, Dec 27 2015
