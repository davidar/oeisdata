## Python
```Python
from itertools import count
from collections import Counter as Ctr
def a(n):
    r = n%3
    if r == 1: return 0
    s = str(n)
    return next(N for N in count(r, 3) if Ctr(str(n*N)) == Ctr(s+str(N)))
print([a(n) for n in range(1, 67)]) # _Michael S. Branicky_, Nov 15 2022
```
