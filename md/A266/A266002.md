## Python
```Python
from sympy import *
#pi, E, phi
pi_const = str(N(pi, 10000))
e_const = str(N(E, 10000))
phi_const = str(S.GoldenRatio.n(10000))
for x in range(10000):
    if pi_const[x] == e_const[x]:
        if e_const[x] == phi_const[x]:
            print(x)
```
