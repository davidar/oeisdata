def main():
    name = "b266295.txt"
    file = open(name, 'w')
    file.write('1' + ' ' + '1\n')
    file.write('2' + ' ' + '3\n')
    file.write('3' + ' ' + '5\n')
    file.write('4' + ' ' + '7\n')
    a, b, c, d = 1, 3, 5, 7
    for i in range(5,10001):
        x=a+b+c+d
        while x%2==0:
            x /= 2
        a, b, c, d = b, c, d, x
        file.write(str(i) + ' ' + str(int(d)) + '\n')
    file.close()
main()
