    x=str(a007623(n))[::-1]
    return 1 if n==0 or x.count("1")==0 else reduce(mul, [prime(i + 1) for i in range(len(x)) if x[i]=='1'])
def a257684(n):
    x=str(a007623(n))[:-1]
    y="".join(str(int(i) - 1) if int(i)>0 else '0' for i in x)[::-1]
    return 0 if n==1 else sum([int(y[i])*f(i + 1) for i in range(len(y))])
