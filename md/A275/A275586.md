def remove_duplicates(values):
    output = []
    seen = set()
    for value in values:
        if value not in seen:
            output.append(value)
            seen.add(value)
    return output
def Non_Unique_Catalan_Triangle(k):
    t = []
    t.append([])
    t[0].append(1)
    for h in range(1, k):
        t.append([])
        t[0].append(1)
    for i in range(1, k):
        for j in range(0, k):
            if i>j:
                t[i].append(0)
            else:
                t[i].append(t[i-1][j] + t[i][j-1])
    l = []
    for r in range(0, k):
        for s in range(0, k):
            l.append(t[r][s])
    non_unique = []
    for n in l:
        if  n <= k and n>1 and l.count(n) > 1:
            non_unique.append(n)
    non_unique = remove_duplicates(non_unique)
