from functools import cache
@cache
def T(n, k):
    if k == 0 or k == n: return 1
    return T(n-1, k-1) + (-1)**k*(2*k + 1)*T(n-1, k)
for n in range(10):
    print([T(n, k) for k in range(n+1)]) # _Peter Luschny_, Jul 22 2021
