## C
```C
#include <stdio.h>
#define START 1
#define END 1000
int main(){
  unsigned int R,N,M,power_cntr;
  int mod1,mod2;
  for(N=START;N<=END;N++){
    R=N;
    M=0;
    power_cntr=1;
    while(R!=0){
      mod1=R%10;
      if(mod1==0) break;
      mod2=N%mod1;
      M+=mod2*power_cntr;
      power_cntr*=10;
      R=R/10;}
    if(mod1!=0) printf("%u %u\n",N,M);}
  return 0;}
```

## Python
```Python
def f(k, digits): return int("".join(map(str, map(lambda x: k%x, digits))))
def aupton(terms):
    alst, k = [], 1
    while len(alst) < terms:
        s = str(k)
        if '0' not in s: alst.append(f(k, list(map(int, s))))
        k += 1
    return alst
print(aupton(73)) # _Michael S. Branicky_, Aug 22 2021
```
