## Python
```Python
def t(n):
    d = []
    while n:
        d.append(n%3)
        n //= 3
    return d
def dif(n1, n2):
    return sum(d1 != d2 for d1, d2 in zip(n1 + [0] * (len(n2)-len(n1)), n2))
a = [0]
for n in range(2000):
    if all(dif(t(n1), t(n)) >= 3 for n1 in a):
        a.append(n)
print(a) # _Andrey Zabolotskiy_, Sep 30 2021
```
