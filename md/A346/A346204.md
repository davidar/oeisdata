import math
bn = [1,1,1]
wn = [0,0,0]
kn = [1,1,1]
def summation(n):
    final = bn[n] - bn[n-1]
    for k in range(4,n+1):
        final -= wn[k-1]*bn[n-k]
    return final
def smallsum(n):
    final = bn[n-1]
    for k in range(4,n+1):
        final += wn[k-1]*bn[n-k]
    return final
def derrangement(n):
    finalsum = 0
    for i in range(n+1):
        if i%2 == 0:
            finalsum += math.factorial(n)*1//math.factorial(i)
        else:
            finalsum -= math.factorial(n)*1//math.factorial(i)
    if finalsum != 0:
        return finalsum
    else:
        return 1

def no_cycles(n):
    goal = n
    cycles = [0, 1]
    current = 2
    while current<= goal:
        new = 0
        k = 1
        while k<=current:
            new += (math.factorial(k-1)-cycles[k-1])*(math.factorial(current-k))
            k+=1
        cycles.append(new)
        current+=1
    return cycles
