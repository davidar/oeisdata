## Python
```Python
def sumRepUnits(n): # A014824
  return ((10**n-1)*10 - 9*n)//81
def a(n): # A346535
  d = 1 + (n-1)%9
  m = 1 + (n-1)//9
  return d*sumRepUnits(m)
for n in range(1,1000):
  print(n, a(n))
```
