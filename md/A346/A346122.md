## Python
```Python
from sympy import S
def aupton(terms):
    digits_of_pi = "0" + str(S.Pi.n(terms+1)).replace('.', '')
    return [n*int(digits_of_pi[n]) for n in range(1, terms+1)]
print(aupton(59)) # _Michael S. Branicky_, Jul 08 2021
```
