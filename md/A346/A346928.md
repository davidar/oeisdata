from math import factorial
def auptor(rows):
    alst, aset = [1], {1}
    for n in range(2, rows+1):
        fn = factorial(n)
        for m in range(n-1, 0, -1):
            fm = factorial(m)
            q, r = divmod(fn, factorial(m))
            if r == 0 and q not in aset:
                alst.append(q); aset.add(q)
    return alst
print(auptor(11)) # _Michael S. Branicky_, Oct 17 2021
