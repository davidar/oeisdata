from math import isqrt
def A346528(n):
    if n == 1: return 17
    a, b, k, k2, m, r, s = -6*(n+1)**2, (n+1)**4, 2, 4, 1, 0, 0
    while 2*m+a < 0 or m*(m+a)+b < 0:
        if isqrt(2*m) - isqrt(m-1) == n:
            r = m
        if s == 0 and isqrt(2*m+2)-isqrt(m) == n:
            s = m
        k += 1
        k2 += 2*k-1
        m = (k2-1)//2
    return r-s # _Chai Wah Wu_, Sep 28 2021
