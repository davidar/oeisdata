## Python
```Python
from sympy import primorial, primefactors, gcd
def A346970(n):
    c, ps = 2, primorial(n)
    while True:
        m = ps//gcd(ps,c)
        if m == 1:
            return c
        p = max(primefactors(m))
        for a in range(p,c,p):
            if a*(c-a) % m == 0:
                return c
        c += 1 # _Chai Wah Wu_, Oct 13 2021
```
