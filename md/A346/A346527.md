```Python
from math import isqrt
def A346527(n):
    a, b, k, k2, m, r = -6*(n+1)**2, (n+1)**4, 2, 4, 1, 0
    while 2*m+a < 0 or m*(m+a)+b < 0:
        if isqrt(2*m) - isqrt(m-1) == n:
            r = m
        k += 1
        k2 += 2*k-1
        m = (k2-1)//2
    return r # _Chai Wah Wu_, Sep 28 2021
```
