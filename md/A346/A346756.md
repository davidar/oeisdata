## Python
```Python
from sympy import isprime, nextprime
def aupton(terms):
    alst, p = [], 2
    while len(alst) < terms:
        revp = int(str(p)[::-1])
        if p < revp and isprime(revp):
            alst.append(revp - p)
        p = nextprime(p)
    return alst
print(aupton(49)) # _Michael S. Branicky_, Sep 08 2021
```
