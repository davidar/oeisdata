## Python
```Python
def power2(n): return n == 2**(n.bit_length()-1)
def a(n):
    if power2(n): return n.bit_length() - 1
    n_c, b = 2**32 - (2**32%n) - 1, 1
    while 2**b <= n_c * (n - 1 - ((2**b - 1)%n)): b += 1
    return b
print([a(n) for n in range(1, 67)]) # _Michael S. Branicky_, Aug 24 2021
```
