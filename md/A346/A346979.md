## Python
```Python
from sympy import isprime
def p_count(k):
    global ct; d = [2, 3, 5, 7] if k == 0 else [1, 3, 7, 9]
    for i in range(4):
        m = 10*k + d[i]
        if isprime(m): ct += 1; p_count(m)
    return ct
for n in range(100):
    ct = 0; print(p_count(n))
```
