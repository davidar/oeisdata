## Python
```Python
from sympy import isprime
def aupton(terms):
  n, alst = 1, []
  while len(alst) < terms:
    s = str(n)
    t = int('2'+s+'3'+s+'5'+s+'7'+s+'11')
    if isprime(t): alst.append(n)
    n += 1
  return alst
print(aupton(55)) # _Michael S. Branicky_, Apr 18 2021
```
