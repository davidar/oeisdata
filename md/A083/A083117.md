## Python

from itertools import count
def A083117(n):
    if not (n%10 and n%16 and n%25): return 0
    for l in count(1):
        k = (10**l-1)//9
        for a in range(1,10):
            b, c = divmod(a*k,n)
            if not c:
                return b # _Chai Wah Wu_, Jan 23 2024
