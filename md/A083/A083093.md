a083093 n k = a083093_tabl !! n !! k
a083093_row n = a083093_tabl !! n
a083093_tabl = iterate
   (\ws -> zipWith (\u v -> mod (u + v) 3) ([0] ++ ws) (ws ++ [0])) [1]
