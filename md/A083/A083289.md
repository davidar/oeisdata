## Python
```Python
from sympy import nextprime, factorint
def A083289(n):
    a, b = divmod(n,2)
    c, d = 10**n, 10**a
    if b == 0: return nextprime(d)**2-c
    k = 0
    while True:
        fs = factorint(c+k,multiple=True)
        if len(fs) == 2 and min(fs) >= d:
            return k
        k += 1 # _Chai Wah Wu_, Sep 28 2021
```
