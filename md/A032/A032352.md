## Python
```Python
from sympy import isprime
def aupto(limit):
  alst = []
  for d in range(2, limit+1):
    td = [10*d + j for j in [1, 3, 7, 9]]
    if not any(isprime(t) for t in td): alst.append(d)
  return alst
print(aupto(342)) # _Michael S. Branicky_, May 30 2021
```
