## Python
```Python
from sympy import isprime
def ok(n):
    s, fdigs, edigs = str(n), "123456789", "1379"
    return not any(isprime(int(f+s+e)) for f in fdigs for e in edigs)
print([k for k in range(10**5) if ok(k)]) # _Michael S. Branicky_, Sep 05 2022
```
