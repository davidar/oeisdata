from sympy import integer_nthroot
def A245631(n):
    m = 10*n
    if integer_nthroot(m,3)[1]: return 0
    a = 1
    while (k:=(integer_nthroot(a*(m+1)-1,3)[0]+1)**3-m*a)>=10*a:
        a *= 10
    return k # _Chai Wah Wu_, Feb 15 2023
