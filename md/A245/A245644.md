## Maple
```Maple
isArithPow := proc(n,e)
    local dvs,d ;
    dvs := numtheory[divisors](n^e) ;
    add(d,d=dvs)/nops(dvs) ;
    if type(%,'integer') then
        true;
    else
        false;
    end if;
end proc:
for n to 300 do
    if isArithPow(n,3) then
        printf("%d,",n) ;
    end if;
end do:
```
