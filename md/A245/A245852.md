## Python

from itertools import count, islice
def A245852_gen(): # generator of terms
    return filter(lambda n:not '0' in str(n),(1<<3*n for n in count(0)))
A245852_list = list(islice(A245852_gen(),10)) # _Chai Wah Wu_, Nov 10 2022
