## PFGW
```PFGW
SCRIPT
DIM n,0
DIMS t
OPENFILEOUT myf,a(n).txt
LABEL loop1
SET n,n+1
SETS t,%d,%d\,;n;p(n)
PRP 8*p(n)^2-1,t
IF ISPRP THEN GOTO a
GOTO loop1
LABEL a
WRITE myf,t
GOTO loop1
```
