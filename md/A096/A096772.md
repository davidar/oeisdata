from itertools import count, islice
def A096772_gen(): # generator of terms
    aset1, aset2, aset3, alist = set(), set(), set(), []
    for k in count(1):
        bset2, bset3 = {k<<1}, {3*k}
        if 3*k not in aset3:
            for d in aset1:
                if (m:=d+(k<<1)) in aset3:
                    break
                bset2.add(d+k)
                bset3.add(m)
            else:
                for d in aset2:
                    if (m:=d+k) in aset3:
                        break
                    bset3.add(m)
                else:
                    yield k
                    alist.append(k)
                    aset1.add(k)
                    aset2 |= bset2
                    aset3 |= bset3
A096772_list = list(islice(A096772_gen(),30)) # _Chai Wah Wu_, Sep 05 2023
