```Mathematica
b = 10; a = {4}; d = {2, 3, 5, 7};
For[n = 2, n <= 5, n++,
  p = Select[Range[b^(n - 1), b^n - 1], PrimeQ[#] &];
  ct = 0;
  For[i = 1, i <= Length[p], i++,
   c = IntegerDigits[p[[i]], b];
   For[j = 1, j <= n, j++,
    t = Delete[c, j];
    If[t[[1]] == 0, Continue[]];
    If[MemberQ[d, FromDigits[t, b]], AppendTo[d, p[[i]]]; ct++;
     Break[]]]];
  AppendTo[a, ct]];
```

## Python
```Python
from sympy import isprime
def ok(n, prevset):
    if not isprime(n): return False
    s = str(n)
    si = (s[:i]+s[i+1:] for i in range(len(s)))
    return any(t[0] != '0' and int(t) in prevset for t in si)
def afind(terms):
    s, snxt = {2, 3, 5, 7}, set()
    print(len(s), end=", ")
    for n in range(2, terms+1):
        for i in range(10**(n-1), 10**n):
            if ok(i, s):
                snxt.add(i)
        s, snxt = snxt, set()
        print(len(s), end=", ")
afind(6) # _Michael S. Branicky_, Jan 14 2022
```
