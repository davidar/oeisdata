## Python
```Python
import decimal
iters = int(input('Precision: '))
decimal.getcontext().prec = iters
D = decimal.Decimal
def agm(a, b):
    for x in range(iters):
        a, b = (a + b) / 2,(a * b).sqrt()
    return a
print(agm(1, (D(5).sqrt()+1)/4))
```
