A057630_list, dlist, p = [], [str(d)*d for d in range(10)], 2
while len(A057630_list) < 10000:
    if isprime(int(''.join(dlist[int(d)] for d in str(p)))):

    p = nextprime(p) # _Chai Wah Wu_, Dec 19 2019, corrected Jan 01 2022
