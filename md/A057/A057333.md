from sympy import isprime
def f(w,dir):
    if dir == 1:
        for s in w:
            for t in range(int(s[-1])+1,10):
                yield s+str(t)
    else:
        for s in w:
            for t in range(0,int(s[-1])):
                yield s+str(t)
def A057333(n):
    c = 0
    for d in '123456789':
        x = d
        for i in range(1,n):
            x = f(x,(-1)**i)
        c += sum(1 for p in x if isprime(int(p)))
        if n > 1:
            y = d
            for i in range(1,n):
                y = f(y,(-1)**(i+1))
            c += sum(1 for p in y if isprime(int(p)))
    return c # _Chai Wah Wu_, Apr 25 2021
