isA000032 := proc(n)
    local l1,l2 ;
    if n <= 0 then
        false;
    elif n <= 4 then
        true ;
    else
        l1 := 3 ; l2 := 4 ;
        while true do
            l := l1+l2 ;
            if l > n then
                return false;
            elif l = n then
                return true;
            else
                l1 := l2 ; l2 := l ;
            end if;
        end do:
    end if;
end proc:
isA057854 := proc(n)
    not isA000032(n) ;
end proc:

## Python

def A057854(n):
    def f(x):
        if x<=2: return n+2
        a, b, c = 1, 3, 0
        while b<=x:
            a, b = b, a+b
            c += 1
        return n+c+2
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m # _Chai Wah Wu_, Sep 10 2024
