## Python
```Python
from sympy import prime, primerange
def a(n):
    P = prime(n)
    x, plst, traj, seen = P, list(primerange(2, P)), [], set()
    while x not in seen:
        traj.append(x)
        seen.add(x)
        x = next((x//p for p in plst if x%p == 0), P*x+1)
    return traj.index(x)
print([a(n) for n in range(2, 82)]) # _Michael S. Branicky_, Dec 11 2023
```
