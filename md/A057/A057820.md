## Python
```Python
from sympy import primepi, integer_nthroot
def A057820(n):
    def f(x): return int(n+x-1-sum(primepi(integer_nthroot(x,k)[0]) for k in range(1,x.bit_length())))
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    r, k = m, f(m)+1
    while r != k: r, k = k, f(k)+1
    return r-m # _Chai Wah Wu_, Sep 12 2024
```
