## Python
```Python
from sympy import prime, primerange
def a(n):
    P = prime(n)
    x, plst, seen = P, list(primerange(2, P)), set()
    while x > 1 and x not in seen:
        seen.add(x)
        x = next((x//p for p in plst if x%p == 0), P*x+1)
    return max(seen)
print([a(n) for n in range(2, 32)]) # _Michael S. Branicky_, Dec 11 2023
```
