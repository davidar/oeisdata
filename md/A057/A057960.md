from functools import cache
@cache
def B(n, j):
    if not 0 <= j < 5:
        return 0
    if n == 0:
        return j == 0
    return B(n - 1, j - 1) + B(n - 1, j) + B(n - 1, j + 1)
def A057960(n):
    return sum(B(n, j) for j in range(5))
