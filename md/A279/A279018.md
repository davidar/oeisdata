def t(n):
    s=0
    for a in range(-n,n+1):
        for b in range(-n,n+1):
            for c in range(-n,n+1):
                for d in range(-n,n+1):
                   if (a!=b  and a!=d and b!=d and c!=a and c!=b and c!=d):
                        if (a*d+b*c)==a+d:
                            s+=1
    return s
