## Maple
```Maple
with(Bits):
n:= 100:
l:= []:
g:=[seq(0, i = 0..n-1)]:
for i from 1 to n by 1
do
a:= 0;
while (And(g[a + 1], i)) > 0
do
a++;
end do:
g[a + 1] += i;
l:= [op(l), a];
end do:
print(l); # _Reza K Ghazi_, Dec 29 2021
```

## Python

```Python
n = 100
g = n * [0]
for i in range(1, n + 1):
    a = 0
    while g[a] & i:
        a += 1
    g[a] += i
    print(a, end=', ') # _Reza K Ghazi_, Dec 29 2021
```
