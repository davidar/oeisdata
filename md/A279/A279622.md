## Python
```Python
from sympy import integer_log
def A279622(n):
    def f(x):
        c = n
        for i in range(integer_log(x,5)[0]+1):
            for j in range(integer_log(y:=x//5**i,3)[0]+1):
                c += (y//3**j).bit_length()
        return c
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m # _Chai Wah Wu_, Sep 16 2024
```
