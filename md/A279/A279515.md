## Python
```Python
def gcd(a, b):
    while b:
        a, b = b, a % b
    return a
def lcm(a, b):
    return a * b // gcd(a, b)
def c(*ar):
    return reduce(lcm, ar)
def a(n):
    if n==0:
        return 0
    x=bin(c(*range(1, n+1)))[2:]
    return x.count("0")
for i in range(0, 10001):
    print str(i)+" "+str(a(i))
```
