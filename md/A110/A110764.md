## Python
```Python
from sympy import factorint
def aupton(terms):
    alst, astr = [1], "1"
    for n in range(2, terms+1):
        an = len(factorint(int(astr)))
        alst.append(an)
        astr += str(an)
    return alst
print(aupton(45)) # _Michael S. Branicky_, Dec 02 2021
```
