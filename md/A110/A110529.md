## Python
```Python
from re import split
from sympy import isprime
def ternary (n):
    if n == 0:
        return '0'
    nums = []
    while n:
        n, r = divmod(n, 3)
        nums.append(str(r))
    return ''.join(reversed(nums))
seq_list, n = [],1
while len(seq_list) < 10000:
    for d in split('1+|2+', ternary(n)[1:]):
        if isprime(len(d)):
            seq_list.append(n)
    n += 1
```
