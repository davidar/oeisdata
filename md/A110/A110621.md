from sympy import S
pi = str(S.Pi.n(3*10**5)).replace('.', '') # alternatively
def aupton(terms):
    global pi
    alst, idx, seen = [1], 0, {0, 1}
    while len(alst) < terms:
        k = 1
        while int(pi[idx:idx+k]) in seen or pi[idx+k] == '0': k += 1
        diffn = int(digits_of_pi[idx:idx+k])
        alst.append(alst[-1] + diffn)
        seen |= {diffn, alst[-1]}
        idx += k
    return alst
