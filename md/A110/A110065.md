## Python

```Python
from sympy import isprime
def afind(limit):
  m, pow10 = 0, 1
  while m <= limit:
    if isprime(pow10 - m): print(m, end=", ")
    m, pow10 =  m + 1, pow10 * 10
afind(1000) # _Michael S. Branicky_, Mar 23 2021
```
