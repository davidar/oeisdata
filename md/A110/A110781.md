from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    s = ""
    while True:
        for d in "19":
            for k in count(1):
                if isprime(int(s+d*k)): break
            yield k
            s += d*k
print(list(islice(agen(), 7))) # _Michael S. Branicky_, Aug 24 2022
