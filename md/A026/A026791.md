## Maple
```Maple
T:= proc(n) local b, ll;
      b:= proc(n,l)
            if n=0 then ll:= ll, l[]
          else seq(b(n-i, [l[], i]), i=`if`(l=[],1,l[-1])..n)
            fi
          end;
      ll:= NULL; b(n, []); ll
    end:
seq(T(n), n=1..8);  # _Alois P. Heinz_, Jul 16 2011
```

```Python
t = [[[]]]
for n in range(1, 10):
    p = []
    for minp in range(1, n):
        p += [[minp] + pp for pp in t[n-minp] if min(pp) >= minp]
    t.append(p + [[n]])
print(t)
```
