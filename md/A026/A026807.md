## Python
```Python
from see_there import a231599_row  # A231599
from sympy.ntheory import npartitions  # A000041
def a026807(n, k):
    if k > n:
        return 0
    elif k > n/2:
        return 1
    else:
        vc = a231599_row(k-1)
        t = len(vc)
        vp_range = range(n-t, n+1)
        vp_range = vp_range[::-1]  # reverse
        r = 0
        for i in range(0, t):
            r += vc[i] * npartitions(vp_range[i])
        return r
# _Tilman Piesk_, Feb 21 2016
```
