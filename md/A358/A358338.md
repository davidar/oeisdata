## Python

from collections import Counter
def aupton(terms):
    alst, inventory = [0], Counter([0])
    for n in range(2, terms+1):
        c = abs(alst[-1] - inventory[alst[-1]])
        alst.append(c); inventory[c] += 1
    return alst
print(aupton(85)) # _Michael S. Branicky_, Nov 10 2022
