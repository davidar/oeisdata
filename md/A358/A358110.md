## Python

# using function harmonic from A001008
def A358110(n: int) -> int:
    if n == 0: return 0
    x, y = harmonic(1, n + 1)
    a = d = v = 0
    b = c = t = 1
    while True:
        m = a + c
        n = b + d
        if x * n == y * m:
            return v + t
        if x * n < y * m:
            c, d = m, n
        else:
            v, a, b = v + t, m, n
        t *= 2
print([A358110(n) for n in range(19)]) # (after _Rémy Sigrist_) _Peter Luschny_, Nov 08 2022
