## Python

from fractions import Fraction
from sympy import Matrix
def A358161(n): return Matrix(n,n,[Fraction(i*j,3).__ceil__() for i in range(1,n+1) for j in range(1,n+1)]).per() if n else 1 # _Chai Wah Wu_, Nov 02 2022
