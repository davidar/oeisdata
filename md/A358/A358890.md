from sympy import factorint
def A358890(n):
    m = 1
    gpf1 = 1
    k = 1
    while 1:
        while 1:
            gpf2 = max(factorint(m+k))
            if gpf2 < gpf1: break
            gpf1 = gpf2
            k += 1
        if k == n: return m
