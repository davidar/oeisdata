sequence=[0]

seq_for_digits=list(map(int, list(str(sequence[0]))))
for ii in range(length):
     sequence.append(seq_for_digits.count(max(list(map(int,list(str(sequence[-1])))))))
     seq_for_digits.extend(list(map(int, list(str(sequence[-1])))))
