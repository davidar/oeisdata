## Python
```Python
from sympy import mobius, integer_nthroot, divisor_count
def A358040(n):
    def f(x): return n+x-sum(mobius(k)*(x//k**3) for k in range(1, integer_nthroot(x,3)[0]+1))
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    return divisor_count(m) # _Chai Wah Wu_, Aug 06 2024
```
