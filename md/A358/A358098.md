## Python
```Python
from itertools import product
def a(n):
    s = str(n)
    r = sorted(set("1234567890") - set(s), reverse=True)
    if len(r) == 0: return -1
    if r == ["0"]: return 0
    for d in range(len(s), 0, -1):
        for p in product(r, repeat=d):
            m = int("".join(p))
            if m < n: return m
print([a(n) for n in range(1, 81)]) # _Michael S. Branicky_, Oct 29 2022
```
