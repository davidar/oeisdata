## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen():
    p, q, r = 2, 3, 5
    while True:
        if isprime(p**3 + q**3 - r**3): yield p
        p, q, r = q, r, nextprime(r)
print(list(islice(agen(), 56))) # _Michael S. Branicky_, Nov 29 2022
```
