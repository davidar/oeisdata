from math import prod

def A358061(n):
    f = factorint(n).items()
    d = prod(e+1 for p, e in f)
    return prod(pow(p,e-1,d)*((p-1)%d) for p, e in f) % d # _Chai Wah Wu_, Oct 29 2022
