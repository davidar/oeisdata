   x:= n;

     x:= 10*x+1;
     if not isprime(x) then return x fi;
   od

## Python
```Python
from sympy import isprime
def A358154(n):
    t = str(n)+'1'
    while isprime(int(t)):t=t+'1'
    return int(t)
print([A358154(i) for i in range(1, 100)])
```
