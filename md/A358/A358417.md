from itertools import islice, count
from sympy import integer_nthroot
def A358417_gen(): # generator of terms
    a = 0
    for n in count(1):
        c, d = integer_nthroot(((b:=n**2)-a<<3)+1,2)
        if d:
            yield c-1>>1
            a = b
