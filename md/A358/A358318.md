## Python
```Python
from sympy import isprime, prime
def a(n):
    s, c = str(prime(n)), 1
    while isprime(int(s[:-1] + '0'*c + s[-1])): c += 1
    return c
print([a(n) for n in range(5, 92)]) # _Michael S. Branicky_, Nov 09 2022
```
