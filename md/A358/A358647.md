## C++
```C++
int a(int n)
{
    int n2 = n;
    int size = 0; do { n2 /= 10; size++; } while (n2 != 0);
    int * nums = new int[size];
    for(int i = size - 1; i >= 0; i--)
    {
        nums[i] = n % 10;
        n /= 10;
    }
    int currentIndex = 0;
    for (int j = 0; j < size; j++)
    {
        currentIndex += nums[currentIndex];
        currentIndex %= size;
    }
    int returnVal = nums[currentIndex];
    delete[] nums;
    return returnVal;
}
```
