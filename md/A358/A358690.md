## Python
```Python
from sympy import primerange
def a(n):
    num=0
    for f in range(1,10,2):
        p=list(primerange(f*10**(n-1),(f+1)*10**(n-1)))
        num+=sum(1 for k in p if all(int(d) %2 for d in str(k)))
    return(num)
print ([a(n) for n in range(1,8)])
```
