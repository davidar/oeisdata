## Maple
```Maple
A358272 := proc(n)
    local a,pe,e,p ;
    a := 1;
    for pe in ifactors(n)[2] do
        e := op(2,pe) ;
        p := op(1,pe) ;
        a := a*(-1)^e*p^(2*floor(e/2)) ;
    end do:
    a ;
end proc:
seq(A358272(n),n=1..80) ; # _R. J. Mathar_, Jan 17 2023
```
