## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen():
    p, q, r = 2, 3, 5
    while True:
        if isprime(p+q-r): yield p
        p, q, r = q, r, nextprime(r)
print(list(islice(agen(), 61))) # _Michael S. Branicky_, Nov 29 2022
```
