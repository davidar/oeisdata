## Python
```Python
from itertools import count
from math import prod
from sympy import isprime, factorint
def A358199(n):
    for m in count(2):
        f = factorint(m).items()
        if all(map(isprime,(prod((p**((e+1)*i)-1)//(p**i-1) for p,e in f) - m**i for i in range(1,n+1)))):
            return m # _Chai Wah Wu_, Nov 15 2022
```
