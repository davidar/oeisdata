## Python
```Python
from itertools import count, product
def a(n):
    s = str(n)
    r = sorted(set("1234567890") - set(s))
    if len(r) == 0 or r == ["0"]: return -1
    for d in count(len(s)):
        for p in product(r, repeat=d):
            m = int("".join(p))
            if m > n: return m
print([a(n) for n in range(75)]) # _Michael S. Branicky_, Oct 29 2022
```
