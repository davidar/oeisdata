## Python

import math
primes = [2]*2
primes[1] = 3
def isprime(k):
  for p in primes:
    if k%p==0:  return 0
  primes.append(k)
  return 1
def rootTriangular(a):
    sr = 2**(int(math.log(a,2))+2)
    while a < sr*(sr+1)/2:
          sr>>=1
    b = sr>>1
    while b:
      s = sr+b
      if a >= s*(s+1)/2:
        sr = s
      b>>=1
    return sr
for i in range(1L<<10):
    k = 0
    for p in primes:

      r = rootTriangular(i - p)
      if r*(r+1)/2 == i-p:  k+=1

      if i<=3:  k += 1
      else:     k += isprime(i)
