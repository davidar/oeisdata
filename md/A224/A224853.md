# Program that generates all values of a(x) less than a given input n.
def a(n):
      seq=[0, 1]
      for x in range(2, n+1):
          c=0
          for y in seq:
              if (x+y)/2 not in seq:
                  if (x*y)**0.5 not in seq[1:]:
                      if (2*x*y)/(x+y) not in seq[1:]:
                          c+=1
          if c==len(seq):
              seq.append(x)
      return seq
