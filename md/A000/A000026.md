## Haskell

a000026 n = f a000040_list n 1 (0^(n-1)) 1 where
   f _  1 q e y  = y * e * q
   f ps'@(p:ps) x q e y
     | m == 0    = f ps' x' p (e+1) y
     | e > 0     = f ps x q 0 (y * e * q)
     | x < p * p = f ps' 1 x 1 y
     | otherwise = f ps x 1 0 y
