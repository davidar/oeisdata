# Generalized algorithm of L. Seidel
def A000108_list(n) :
    D = [0]*(n+1); D[1] = 1
    b = True; h = 1; R = []
    for i in range(2*n-1) :
        if b :
            for k in range(h,0,-1) : D[k] += D[k-1]
            h += 1; R.append(D[1])
        else :
            for k in range(1,h, 1) : D[k] += D[k+1]
        b = not b
    return R
A000108_list(31) # _Peter Luschny_, Jun 02 2012
