# Algorithm of L. Seidel (1877)
def A000111_list(n) :
    R = []; A = {-1:0, 0:1}; k = 0; e = 1
    for i in (0..n) :
        Am = 0; A[k + e] = 0; e = -e
        for j in (0..i) : Am += A[k]; A[k] = Am; k += e
        R.append(Am)
    return R
