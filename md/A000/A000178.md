## Ruby
```Ruby
def mono_choices(a,b,n)
    n - [a,b].max
end
def comm_mono_choices(n)
    accum =1
    0.upto(n-1) do |i|
        i.upto(n-1) do |j|
            accum = accum * mono_choices(i,j,n)
        end
    end
    accum
end
1.upto(12) do |k|
    puts comm_mono_choices(k)
end # _Chad Brewbaker_, Nov 03 2013
```
