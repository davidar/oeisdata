## Python
```Python
import numpy
from sympy import sieve as prime
aupto = 50
A000230 = np.zeros(aupto+1, dtype=object)
A000230[0], it = 2, 2
while all(A000230) == 0:
    gap = (prime[it+1] - prime[it]) // 2
    if gap <= aupto and A000230[gap] == 0: A000230[gap] = prime[it]
    it += 1
print(list(A000230)) # _Karl-Heinz Hofmann_, Jun 07 2023
```
