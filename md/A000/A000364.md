# Algorithm of L. Seidel (1877)
# n -> [a(0), a(1), ..., a(n-1)] for n > 0.
def A000364_list(len) :
    R = []; A = {-1:0, 0:1}; k = 0; e = 1
    for i in (0..2*len-1) :
        Am = 0; A[k + e] = 0; e = -e
        for j in (0..i) : Am += A[k]; A[k] = Am; k += e
        if e < 0 : R.append(A[-i//2])
    return R
