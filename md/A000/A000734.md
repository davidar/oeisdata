# Algorithm of L. Seidel (1877)
def A000734_list(n) :
    A = {-1:0, 0:1}; R = []
    k = 0; e = 1; Bm = 1
    for i in range(n) :
        Am = Bm
        A[k + e] = 0
        e = -e
        for j in (0..i) :
            Am += A[k]
            A[k] = Am
            k += e
        Bm += Bm
        R.append(A[e*i//2]/2)
    return R
