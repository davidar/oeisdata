# Algorithm of L. Seidel (1877)
def A000660_list(n) :
    R = []; A = {-1:0, 0:1}
    k = 0; e = 1
    for i in range(n) :
        Am = i
        A[k + e] = 0
        e = -e
        for j in (0..i) :
            Am += A[k]
            A[k] = Am
            k += e
        print([A[z] for z in (-i//2..i//2)])
        R.append(A[e*i//2])
    return R
