## Julia
```Julia
function tau(n)
    i = 2; num = 1
    while i * i <= n
        if rem(n, i) == 0
            e = 0
            while rem(n, i) == 0
                e += 1
                n = div(n, i)
            end
            num *= e + 1
        end
        i += 1
    end
    return n > 1 ? num + num : num
end
println([tau(n) for n in 1:104])  # _Peter Luschny_, Sep 03 2023
```
