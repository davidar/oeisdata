## Python
```Python
# Starts at a(3), i.e. computes n*(n+1)*(n+2)*(n+3)/24
# which is more in line with A000217 and A000292.
def A000332():
    x, y, z, u = 1, 1, 1, 1
    yield 0
    while True:
        yield x
        x, y, z, u = x + y + z + u + 1, y + z + u + 1, z + u + 1, u + 1
a = A000332(); print([next(a) for i in range(41)]) # _Peter Luschny_, Aug 03 2019
```
