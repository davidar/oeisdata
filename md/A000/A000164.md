## Maple
```Maple
A000164 := proc(n)
    local a,x,y,z2,z ;
    a := 0 ;
    for x from 0 do
        if 3*x^2 > n then
            return a;
        end if;
        for y from x do
            if x^2+2*y^2 > n then
                break;
            end if;
            z2 := n-x^2-y^2 ;
            if issqr(z2) then
                z := sqrt(z2) ;
                if z >= y then
                    a := a+1 ;
                end if;
            end if;
        end do:
    end do:
    a;
end proc: # _R. J. Mathar_, Feb 12 2017
```
