## Python
```Python
from sympy import isprime
def aupton(terms):
  alst, astr = [6], "6"
  for n in range(2, terms+1):
    an = alst[-1] + 1
    while not isprime(int(astr+str(an))): an += 1
    alst, astr = alst + [an], astr + str(an)
  return alst
print(aupton(62)) # _Michael S. Branicky_, Jun 07 2021
```
