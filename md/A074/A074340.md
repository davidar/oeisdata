## Python
```Python
from sympy import isprime
def aupton(terms):
  alst, astr = [5], "5"
  while len(alst) < terms:
    an = alst[-1] + 2
    while an%5 ==0 or not isprime(int(astr + str(an))): an += 2
    alst, astr = alst + [an], astr + str(an)
  return alst
print(aupton(48)) # _Michael S. Branicky_, May 09 2021
```
