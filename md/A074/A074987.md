## Maple
```Maple
N:= 1000: # to get a(n) for n <= N
todo:= N;
for n from 1 while todo > 0 do
  v:= numtheory:-phi(n);
  if assigned(R[v]) then
    if n <= N then
      A[n]:= R[v]; todo:= todo-1;
    fi;
    if R[v] <= N and not assigned(A[R[v]])  then
      A[R[v]]:= n; todo:= todo-1;
    fi;
  else
    R[v]:= n
  fi
od:
seq(A[n],n=1..N); # _Robert Israel_, Aug 12 2016
```
