from itertools import accumulate
def A074051_list(size):
    if size < 1: return []
    L, accu = [], [1]
    for n in range(size-1):
        accu = list(accumulate([-accu[-1]] + accu))
        L.append(-(-1)**n*accu[-2])
    return L
print(A074051_list(28)) # _Peter Luschny_, Apr 25 2016
