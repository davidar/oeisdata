## Python
```Python
from sympy import isprime
def aupton(terms):
    alst, astr = [10], "10"
    while len(alst) < terms:
        k = alst[-1] + 1 + (alst[-1]%2)
        while not isprime(int(astr+str(k))): k += 2
        alst.append(k)
        astr += str(k)
    return alst
print(aupton(46)) # _Michael S. Branicky_, Oct 13 2021
```
