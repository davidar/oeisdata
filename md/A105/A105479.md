from itertools import count, accumulate, islice
def A105479_gen(): # generator of terms
    yield from (0,0,1)
    blist, b, c = (1,), 1, 1
    for n in count(2):
        c += n
        blist = list(accumulate(blist, initial=(b:=blist[-1])))
        yield b*c
A105479_list = list(islice(A105479_gen(),20)) # _Chai Wah Wu_, Jun 22 2022
