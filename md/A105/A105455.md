## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen(): # generator of terms
    m, p, q, r = 1, 2, 3, 5
    while True:
        t = m*p + (m+1)*q + (m+2)*r
        if isprime(t): yield m
        m, p, q, r = m+1, q, r, nextprime(r)
print(list(islice(agen(), 58))) # _Michael S. Branicky_, May 17 2022
```
