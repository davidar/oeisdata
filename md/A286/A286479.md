## Python
```Python
from sympy import factorint, divisor_count
def P(n):
    f = factorint(n)
    return sorted([f[i] for i in f])
def a046523(n):
    x=1
    while True:
        if P(n) == P(x): return x
        else: x+=1
def a(n): return a046523(n + divisor_count(n)) # _Indranil Ghosh_, May 21 2017
```
