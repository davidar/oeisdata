## Ruby
```Ruby
def A(k, m, n)
  ary = [0]
  a, mod = k, m
  n.times{
    b = a % mod
    ary << b
    a = b ** m
    mod *= m
  }
  ary
end
def A286840(n)
  A(5, 13, n)
end
p A286840(100)
```

## Python
```Python
def A(k, m, n):
    ary=[0]
    a, mod = k, m
    for i in range(n):
          b=a%mod
          ary.append(b)
          a=b**m
          mod*=m
    return ary
def a286840(n):
    return A(5, 13, n)
print(a286840(100)) # _Indranil Ghosh_, Aug 03 2017, after Ruby
```
