## Python

from __future__ import division
def is_cubefree(s):
    l = len(s)
    for i in range(l-2):
        for j in range(1,(l-i)//3+1):
            if s[i:i+2*j] == s[i+j:i+3*j]:
                return False
    return True
A286261_list = [n for n in range(10**4) if not is_cubefree(bin(n)[2:])] # _Chai Wah Wu_, May 06 2017
