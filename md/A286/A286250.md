from sympy import factorint, prevprime
from operator import mul
def P(n):
    f = factorint(n)
    return sorted([f[i] for i in f])
def a046523(n):
    x=1
    while True:
        if P(n) == P(x): return x
        else: x+=1
def a064216(n):
    f=factorint(2*n - 1)
    return 1 if n==1 else reduce(mul, [prevprime(i)**f[i] for i in f])
