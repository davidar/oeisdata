def T(n, k):
    i=1
    if k==1: return 1
    while n%(k**i)==0:
        i+=1
    return i-1
for n in range(1, 21): print([T(n, k) for k in range(1, n + 1)]) # _Indranil Ghosh_, May 20 2017
