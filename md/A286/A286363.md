## Python

from sympy import factorint
from operator import mul
def P(n):
    f = factorint(n)
    return sorted([f[i] for i in f])
def a046523(n):
    x=1
    while True:
        if P(n) == P(x): return x
        else: x+=1
def a072436(n):
    f = factorint(n)
    return 1 if n == 1 else reduce(mul, [1 if i%4==3 else i**f[i] for i in f])
