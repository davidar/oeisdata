## Python
```Python
from __future__ import division
from sympy import prime
from gmpy2 import is_square
def A286328(n): # assumes n >= 2
    p, area = prime(n), 0
    k, q, kq = (p + 1)//2, (p**2 - 1)//2, (p - 1)*(p + 1)**2//4
    while True:
        area += kq
        if is_square(area):
            return k
        k += 1
        kq += q # _Chai Wah Wu_, May 15 2017
```
