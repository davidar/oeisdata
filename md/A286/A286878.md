## Ruby
```Ruby
def A(k, m, n)
  ary = [0]
  a, mod = k, m
  n.times{
    b = a % mod
    ary << b
    a = b ** m
    mod *= m
  }
  ary
end
def A286878(n)
  A(13, 17, n)
end
p A286878(100)
```
