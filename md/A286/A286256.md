from sympy import factorint
def T(n, m): return ((n + m)**2 - n - 3*m + 2)/2
def P(n):
    f = factorint(n)
    return sorted([f[i] for i in f])
def a046523(n):
    x=1
    while True:
        if P(n) == P(x): return x
        else: x+=1
def a(n): return T(a046523(n), a046523(n + 2)) # _Indranil Ghosh_, May 07 2017
