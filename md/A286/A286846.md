## Python
```Python
from itertools import permutations
def t2i(t): return int("".join(map(str, t)))
alst = [t2i(p) for p in permutations(range(1, 10)) if t2i(p[:3]) - t2i(p[3:6]) == t2i(p[6:])]
print(alst) # _Michael S. Branicky_, May 30 2022
```
