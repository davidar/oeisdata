## Ruby
```Ruby
def A(k, m, n)
  ary = [0]
  a, mod = k, m
  n.times{
    b = a % mod
    ary << b
    a = b ** m
    mod *= m
  }
  ary
end
def A286841(n)
  A(8, 13, n)
end
```

## Python
```Python
def A(k, m, n):
    ary=[0]
    a, mod = k, m
    for i in range(n):
          b=a%mod
          ary.append(b)
          a=b**m
          mod*=m
    return ary
def a286841(n):
    return A(8, 13, n)
```
