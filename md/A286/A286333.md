## Python
```Python
from itertools import product
from sympy import isprime
A286333_list = []
for l in range(14):
    for w in product('1379',repeat=l):
        for d in '0123456789':
            for t in '1379':
                s = ''.join(w)+d+t
                n = int(s)
                for i in range(l+1):
                    if not isprime(int(s)):
                        break
                    s = s[1:]+s[0]
                else:
                    if n > 10 and not isprime(int(s)):
                        A286333_list.append(n) # _Chai Wah Wu_, May 21 2017
```
