from sympy import factorint, gcd, divisor_sigma
def T(n, m): return ((n + m)**2 - n - 3*m + 2)/2
def P(n):
    f = factorint(n)
    return sorted([f[i] for i in f])
def a046523(n):
    x=1
    while True:
        if P(n) == P(x): return x
        else: x+=1
def a(n): return T(a046523(n), gcd(n, divisor_sigma(n))) # _Indranil Ghosh_, May 26 2017
