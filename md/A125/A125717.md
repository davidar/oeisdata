from itertools import count, islice
def agen(): # generator of terms
    an, aset = 0, {0}
    for n in count(1):
        yield an
        an = next(m for m in count(an%n, n) if m not in aset)
        aset.add(an)
print(list(islice(agen(), 70))) # _Michael S. Branicky_, Jun 07 2023
