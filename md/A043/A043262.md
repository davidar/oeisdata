## Python

from gmpy2 import digits
def A043262(n):
    if n == 1: return 0
    y = 3*(x:=3**(len(digits(n>>1, 3))-1))
    return int((s:=digits(n-x,3))[-1])+(sum(int(d) for d in s[:-1])<<1) if n<x+y else sum(map(int,digits(n-y,3)))<<1 # _Chai Wah Wu_, Jul 24 2024
