## Maple
```Maple
A043276 := proc(n)
    local a,rl,i ;
    if n > 0 then
        rl := 1 ;
    else
        rl := 0 ;
    end if;
    a := rl ;
    dgs := convert(n,base,2) ;
    for i from 2 to nops(dgs) do
        if op(i,dgs) = op(i-1,dgs) then
            rl := rl+1 ;
            a := max(a,rl) ;
        else
            a := max(a,rl) ;
            rl := 1;
        end if;
    end do:
    a ;
end proc:
```
