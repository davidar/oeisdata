## Maple
```Maple
isA034096 := proc(n)
    local dgs ;
    dgs := convert(n,base,10) ;
    for i from 2 to nops(dgs) do
        if op(i,dgs) = op(i-1,dgs) then
            return false;
        end if;
    end do:
    true ;
end proc:
for n from 0 to 150 do
    if isA034096(n) then
        printf("%d,",n) ;
    end if;
end do: # _R. J. Mathar_, Jun 26 2023
```
