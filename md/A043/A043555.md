```Maple
NRUNS := proc(L::list)
    local a,i;
    a := 1 ;
    for i from 2 to nops(L) do
        if op(i,L) <> op(i-1,L) then
            a := a+1 ;
        end if
    end do:
    a ;
end proc:
A043555 := proc(n)
    convert(n,base,3) ;
    NRUNS(%) ;
end proc:
```
