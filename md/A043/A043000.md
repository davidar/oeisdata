## Python
```Python
def count(n,b):
    c = 0
    while n > 0:
        n, c = n//b, c+1
    return c
n = 0
while n < 50:
    n = n+1
    a, b = 0, 1
    while b < n:
        b = b+1
        a = a + count(n,b)
    print(n,a) # _A.H.M. Smeets_, Dec 14 2019
```
