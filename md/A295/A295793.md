N:= 100: # to get a(0)..a(N)
A295520:= proc(n) local k;
  for k from 0 do if isprime(Bits:-Xor(k,n)) then return k fi od
end proc:
V:= Array(0..N,-1):
count:= 0:
for n from 0 while count < N+1 do
r:= A295520(n);
if r <= N and V[r]=-1 then
   count:= count+1; V[r]:= n
fi
od:
convert(V,list); # _Robert Israel_, Nov 27 2017
