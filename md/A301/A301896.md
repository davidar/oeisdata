b:= proc(n) option remember; `if`(n=0, [1, 0], b(n-1)+
     (l-> [add(1-i, i=l), add(i, i=l)])(Bits[Split](n)))
    end:
a:= n-> (l-> l[1]*l[2])(b(n)):
seq(a(n), n=0..50);  # _Alois P. Heinz_, Mar 01 2023
