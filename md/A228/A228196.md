def funcL(n):
   q = n**2
   return q
def funcR(n):
   q = 2**n
   return q
for n in range (1,9871):
   t=int((math.sqrt(8*n-7) - 1)/ 2)
   i=n-t*(t+1)/2-1
   j=(t*t+3*t+4)/2-n-1
   sum1=0
   sum2=0
   for m1 in range (1,i+1):
      sum1=sum1+funcR(m1)*binomial(i+j-m1-1,i-m1)
   for m2 in range (1,j+1):
      sum2=sum2+funcL(m2)*binomial(i+j-m2-1,j-m2)
   sum=sum1+sum2
