## Python

from gmpy2 import is_prime, next_prime
def isemirp(n,b=10): # check if n is emirp in base b
....x, y = n, 0
....while x >= b:
........x, r = divmod(x,b)
........y = y*b + r
....y = y*b + x
....return n != y and is_prime(y)
def A228768(n):
....m = 1
....while True:
........m = next_prime(m)
........for b in range(2,n+1):
............if not isemirp(m,b):
................break
........else:
............return m # _Chai Wah Wu_, Jun 11 2015
