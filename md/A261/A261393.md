## Python
```Python
def h(n):
  if(n == 0):     return 0
  elif(n%2 == 0): return h(n/2)
  else:           return h((n-1)/2) + 1
def a(n):
  if(n == 0):     return 0
  elif(n%2 == 0): return 2*a(n/2)
  else:           return a((n-1)/2) + 3**h((n-1)/2)
```
