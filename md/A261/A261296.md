def cube(z, p):
    iscube=False
    y=int(pow(z, 1/p)+0.01)
    if y**p==z:
        iscube=True
    return iscube
for n in range (1, 10**5):
    for m in range(n+1, 10**5):
        a=(m-n)*(m**2+m*n+n**2)
        b=(m-n)*(m+n)
        if cube(a, 2)==True and cube(b, 3)==True:
            print (n, m)
