## Python

```Python
from itertools import count, islice
def agen(): # generator of terms
    for d in count(1):
        for out in "123456789":
            for mid in "0123456789":
                if mid != out:
                    yield int(out*d + mid + out*d)
print(list(islice(agen(), 52))) # _Michael S. Branicky_, May 17 2022
```
