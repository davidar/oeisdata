## Julia
```Julia
function isA003136(n)
    n % 3 == 2 && return false
    n in [0, 1, 3] && return true
    M = Int(round(2*sqrt(n/3)))
    for y in 0:M, x in 0:y
        n == x^2 + y^2 + x*y && return true
    end
    return false
end
A003136list(upto) = [n for n in 0:upto if isA003136(n)]
```
