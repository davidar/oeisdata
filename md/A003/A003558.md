A003558 := proc(n)
    local m,mo ;
    if n = 0 then
        return 0 ;
    end if;
    for m from 1 do
        mo := modp(2^m,2*n+1) ;
        if mo in {1,2*n} then
            return m;
        end if;
    end do:
end proc:
seq(A003558(n),n=0..20) ; # _R. J. Mathar_, Dec 01 2014

def A003558(n):
