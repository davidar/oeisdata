## Python
```Python
from sympy import isprime, prime
def a(n):
    s = str(prime(n))
    ss = set(int(s[i:i+1+l]) for i in range(len(s)) for l in range(len(s)))
    return min(t for t in ss if isprime(t))
print([a(n) for n in range(1, 94)]) # _Michael S. Branicky_, Jun 29 2022
```
