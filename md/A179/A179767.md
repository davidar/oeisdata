## Python
```Python
from sympy import isprime
def iters(p):
  c = 0
  while isprime(4*p + 3): p, c = 4*p + 3, c + 1
  return c
def a(n):
  k = 0
  while True:
    p, k = 4*k + 3, k + 1
    if isprime(p) and iters(p) == n: return p
print([a(n) for n in range(9)]) # _Michael S. Branicky_, Mar 19 2021
```
