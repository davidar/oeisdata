def A061584_first(n):
    an = "1"
    a061584 = []
    while n > 1:
        a061584.append(int(an))
        newan = ""
        for i in an:
            newan += str(6*int(i))
        an = newan
        n -= 1
    a061584.append(int(an))
    return a061584 # _John Cerkan_, May 25 2018
