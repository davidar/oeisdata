## Python
```Python
from itertools import count, islice
def f(n): return int("".join(str(int(d)+1) for d in str(n)[::-1]))
def agen(): yield (an:=0); yield from (an:=f(an) for n in count(1))
print(list(islice(agen(), 41))) # _Michael S. Branicky_, Apr 14 2023
```
