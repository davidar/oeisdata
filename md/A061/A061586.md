def A061586_first(n):
    an = "1"
    cnt = 1
    a061586 = []
    while cnt < n:
        a061586.append(int(an))
        newan = ""
        for i in an:
            newan += str(8*int(i))
        an = newan
        cnt += 1
    a061586.append(int(an))
    return a061586 # _John Cerkan_, May 25 2018
