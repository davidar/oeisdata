## Python
```Python
def A061583_first(n):
    an = "1"
    a061583 = []
    while n > 1:
        a061583.append(int(an))
        newan = ""
        for i in an:
            newan += str(5*int(i))
        an = newan
        n -= 1
    a061583.append(int(an))
    return a061583 # _John Cerkan_, May 25 2018
```
