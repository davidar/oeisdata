## Python
```Python
from sympy import nextprime
def a053669(n):
    p=2
    while n%p==0: p=nextprime(p)
    return p
def a007978(n):
    p=2
    while n%p==0: p+=1
    return p
def a(n): return a053669(n) - a007978(n)
print([a(n) for n in range(1, 101)]) # _Indranil Ghosh_, Jul 29 2017
```
