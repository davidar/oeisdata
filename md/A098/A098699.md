from sympy import factorint
def A098699(n):
    if n < 2:
        return n+1
    for m in range(1,(n**2>>2)+1):
        if sum((m*e//p for p,e in factorint(m).items())) == n:
            return m
    return 0 # _Chai Wah Wu_, Sep 12 2022
