## Python
```Python
from itertools import count, islice
def cond(n):
    d = list(map(int, str(n)))
    return any(d[i+1] == d[i]+1 for i in range(len(d)-1))
def agen():
    yield from filter(cond, count(1))
print(list(islice(agen(), 54))) # _Michael S. Branicky_, Dec 23 2021
```
