from functools import cache
@cache
def A098474row(n: int) -> list[int]:
    if n == 0: return [1]
    a = A098474row(n - 1) + [0]
    row = [0] * (n + 1)
    row[0] = 1; row[1] = n
    for k in range(2, n + 1):
        row[k] = (a[k] * (n + k + 1) + a[k - 1] * (4 * k - 2)) // (n + 1)
    return row  # _Peter Luschny_, Jun 22 2023
