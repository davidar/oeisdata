## Python
```Python
from sympy import factorint
def semiprime(n): f = factorint(n); return sum(f[p] for p in f) == 2
def a(n):
  an = 10**n - 1
  while not semiprime(an): an -= 1
  return an
print([a(n) for n in range(1, 21)]) # _Michael S. Branicky_, Apr 10 2021
```
