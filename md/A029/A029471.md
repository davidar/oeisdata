from itertools import count
def a029471():
    total = 0
    power_of_two = 1
    index_of_two = 0
    length_of_string = 0
    for n in count(1):
        total += (n<<length_of_string)
        if n == power_of_two:
            power_of_two *= 2
            index_of_two += 1
        length_of_string += index_of_two
        if total % n == 0:
            yield n
