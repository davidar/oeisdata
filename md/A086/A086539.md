## Python
```Python
from math import gcd, lcm
from itertools import count, islice
def agen(): # generator of terms
    alst, an, LCM,  = [1], 1, 1
    for n in count(2):
        yield an
        LCM = lcm(*alst[-(n//2):])
        an = next(k for k in count(an+1) if gcd(k, LCM) == 1)
        alst.append(an)
print(list(islice(agen(), 62))) # _Michael S. Branicky_, Jun 25 2024
```
