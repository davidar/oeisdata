## Python
```Python
from sympy.ntheory import digits
def a(n):
    xn, b = n, 2
    while xn >= b: xn = sum(digits(xn, b)[1:]); b += 1
    return xn
print([a(n) for n in range(105)]) # _Michael S. Branicky_, Aug 10 2022
```
