## Python

def aupton(terms):
    a, n = [1], 0
    while len(a) < 3*terms: a, n = a + [(1<<n) + ai for ai in a] + a, n+1
    return a[:terms]
print(aupton(81)) # _Michael S. Branicky_, Jul 31 2022
