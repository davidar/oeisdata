## Python
```Python
from sympy.ntheory import digits
def a(n):
    b = 2
    while n != 0 and n%sum(digits(n, b)[1:]): b += 1
    return sum(digits(n, b)[1:])
print([a(n) for n in range(1, 88)]) # _Michael S. Branicky_, Aug 12 2022
```
