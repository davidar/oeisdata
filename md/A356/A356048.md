## Python
```Python
from sympy import isprime, nextprime, prime
def a(n):
    p = pn = prime(n)
    while not isprime(p*p - 2*pn*pn): p = nextprime(p)
    return p
print([a(n) for n in range(1, 64)]) # _Michael S. Branicky_, Jul 24 2022
```
