from itertools import islice
from sympy import nextprime
def A356421_gen():
    m = 1
    while True:
        n = m+1<<1
        k = m
        p = nextprime(k)
        while k+p>n:
            k -=1
            p = nextprime(k)
        if k+p==n:
            yield k
        m = n-1
A356421_list = list(islice(A356421_gen(),30)) # _Chai Wah Wu_, Sep 11 2022
