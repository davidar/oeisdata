## Maple
```Maple
A356491 := proc(n)
    local c ;
    if n =0 then
        return 1 ;
    end if;
    LinearAlgebra[ToeplitzMatrix]([seq(ithprime(c),c=1..n)],n,symmetric) ;
    LinearAlgebra[Permanent](%) ;
end proc:
seq(A356491(n),n=0..15) ; # _R. J. Mathar_, Jan 31 2023
```
