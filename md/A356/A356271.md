## Python
```Python
from sympy import nextprime; R = [2]; p0 = 2
while len(R) < 50:
    p1 = nextprime(p0); p = nextprime(p1); g1 = p - p1
    if g1 >= p1 - p0: R.append(p1)
    else:
        while p - p1 <= g1: p1 = p; p = nextprime(p)
    p0 = p1
print(*R, sep = ', ')
```
