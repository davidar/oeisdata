## Python
```Python
from sympy import divisor_sigma
def sigma(n): return divisor_sigma(n, 1)
def ok(n): return n > 1 and int(str(sigma(n-1)) + str(sigma(n+1)))%n == 0
print([k for k in range(10**5) if ok(k)])
```
