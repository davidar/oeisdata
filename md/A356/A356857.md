## Python
```Python
def A(d):
    A = [[0 for col in range(d)] for row in range(d)]
    for i in range(d):
        A[0][i] = 1
    for i in range(1, d):
        for j in range(i, d):
            A[i][j] = (A[i][j - 1] + A[i - 1][j - 1])*(i + 1)
    for i in range(d):
        for j in range(i, d):
            for k in range(i+3, j+3):
                A[i][j] *= k
    a = []
    for i in range(d):
        for j in range(i+1):
            a.append((-1)**(i+j)*A[j][i])
    return(a)
```
