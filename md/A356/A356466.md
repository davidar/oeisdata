from sympy import nextprime; R = []; p0 = 2
while len(R) < 60:
    p1 = nextprime(p0); M = [p1]; p = nextprime(p1); g1 = p - p1
    while g1 < p1 - p0 and p - p1 <= g1: M.append(p); p1 = p; p = nextprime(p)
    if len(M) == 2: R.extend(M)
    p0 = p1
print(*R, sep = ', ')
