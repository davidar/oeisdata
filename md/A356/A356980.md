## Python
```Python
from sympy import nextprime
from itertools import count, islice
def agen(): # generator of terms
    pk = 2
    for k in count(1):
        if set(str(pk)) <= set(str(k)): yield k
        pk = nextprime(pk)
print(list(islice(agen(), 46))) # _Michael S. Branicky_, Sep 09 2022
```
