## Python
```Python
from math import gcd
from itertools import groupby
def a(n):
    if n == 0: return 0 # by convention
    return gcd(*(len(list(g)) for k, g in groupby(bin(n)[2:])))
print([a(n) for n in range(87)]) # _Michael S. Branicky_, Oct 15 2022
```
