## Python
```Python
from functools import cache # requires Python 3.9
KNIGHT_MOVES = ((1, 2), (2, 1), (-1, 2), (-2, 1),
                (1, -2), (2, -1), (-1, -2), (-2, -1))
def manhattan(x, y):
    return abs(x) + abs(y)
@cache
def A356359(m, n):
    if (m, n) == (0, 0):
        return 1
    value = 0
    for move in KNIGHT_MOVES:
        new_m, new_n = m + move[0], n + move[1]
        if manhattan(new_m, new_n) < manhattan(m, n):
            value += A356359(new_m, new_n)
    return value
```
