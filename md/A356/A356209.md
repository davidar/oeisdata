from sympy.solvers.diophantine.diophantine import diop_DN
def A356209(n):
    for m in range(n**2<<1,0,-1):
        if n==max((a for a, b in diop_DN(-1,m)),default=0):
            return m # _Chai Wah Wu_, Sep 08 2022
