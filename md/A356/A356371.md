from math import gcd
from itertools import count
def A356371(n):
    for k in count(n,n):
        if len(set(gcd(i,j) for i in range(k,n+k+1) for j in range(i+1,n+k+1))) == n:
            return k # _Chai Wah Wu_, Oct 18 2022
