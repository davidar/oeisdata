## Python
```Python
from num2words import num2words
def n2w(n): return num2words(n).replace(" and", "").replace(chr(44), "")
def a(n):
    target, t = n2w(n), n+1
    while n2w(t) <= target: t += 1
    return t
print([a(n) for n in range(1, 64)]) # _Michael S. Branicky_, Jul 23 2022
```
