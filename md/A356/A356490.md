## Maple
```Maple
A356490 := proc(n)
    local T,c ;
    if n =0 then
        return 1 ;
    end if;
    T := LinearAlgebra[ToeplitzMatrix]([seq(ithprime(c),c=1..n)],n,symmetric) ;
    LinearAlgebra[Determinant](T) ;
end proc:
seq(A356490(n),n=0..15) ; # _R. J. Mathar_, Jan 31 2023
```
