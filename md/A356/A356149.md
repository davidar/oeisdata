## Python
```Python
def row(n):
    N = n.bit_length()
    c, s = ((1<<N)-1)^n, set()
    for i in range(N):
        for l in range(N-i):
            mask = ((2<<l)-1) << i
            s.add((mask&n) >> i)
            s.add((mask&c) >> i)
    return sorted(s - {0})
print([t for r in range(19) for t in row(r)]) # _Michael S. Branicky_, Jul 28 2022
```
