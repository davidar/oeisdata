from sympy import integer_nthroot
def A243092(n):
    m, a = 10*n, 10**(len(str(n))-1)
    while (k:=(integer_nthroot(a*(m+1)-1,3)[0]+1)**3-m*a)>=10*a or k<=n:
        a *= 10
    return k # _Chai Wah Wu_, Feb 20 2023
