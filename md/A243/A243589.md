## Python
```Python
for n in range(1, 100):
    BL = len(bin(n))-2
    x = (n>>1) + ((n&1) << (BL-1))   # A038572(n)
    x^= (n*2) - (1<<BL) + 1   # A006257(n)  for n>0
    print str(x)+',',
```
