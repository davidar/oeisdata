## Python
```Python
from itertools import count, islice
def noz(n): return int(str(n).replace("0", ""))
def agen(): # generator of terms
    yield (an:=1)
    yield from (an:=noz(n*an) for n in count(1))
print(list(islice(agen(), 32))) # _Michael S. Branicky_, Jul 02 2024
```
