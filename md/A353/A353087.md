```Maple
      while min((p-> seq(coeff(p, x, j), j=0..9))(add(
```

## Python
```Python
def A353087(n):
    k, m, r  = 1, 1, 10**(10*n-1)
    while m < r:
        k += 1
        m *= k
    s = str(m)
    while any(s.count(d) < n for d in '0123456789'):
        k += 1
        m *= k
        s = str(m)
    return k # _Chai Wah Wu_, Apr 22 2022
```
