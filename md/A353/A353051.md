    for n = 1:max_n
        m = n;
        while ~isprime(m) && m > 2

from sympy import factorint
def A353051(n):
    while n > 1 and len(f:=factorint(n,multiple=True)) > 1:
        n -= sum(f)
    return n  # _Chai Wah Wu_, May 19 2022
