## Python
```Python
def T(n, k):
    j, n2, c = 2, n*n, 0
    while c != k:
        if pow(j, n-1, n2) == 1: c += 1
        j += 1
    return j-1
def auptodiag(maxd):
    return [T(d+2-j, j) for d in range(1, maxd+1) for j in range(d, 0, -1)]
print(auptodiag(11)) # _Michael S. Branicky_, Apr 29 2022
```
