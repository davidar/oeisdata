from itertools import count
from sympy.ntheory.factor_ import primeomega
def A353172(n):
    a = primeomega(n)
    for k in count(2):
        if (m := n % k) > 0 and primeomega(m) == a:
            return k # _Chai Wah Wu_, Jun 20 2022
