## Python

from sympy.ntheory.residue_ntheory import nthroot_mod
from sympy import prime
def A353943(n): return 2**10+1 if n == 1 else int(nthroot_mod(1,(p:= prime(n))-1,p**10,True)[1]) # _Chai Wah Wu_, May 17 2022
