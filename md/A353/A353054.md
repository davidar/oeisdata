## Python
```Python
from itertools import count, islice
def A353054_gen(): # generator of terms
    for l in count(1):
        a, b = 10**l-2, 10**(l-1)-2
        for m in range(1,10):
            q, r = divmod(m*a-1,19)
            if r == 0 and b <= q - 2 <= a:
                yield 10*q+m
A353054_list = list(islice(A353054_gen(),20)) # _Chai Wah Wu_, Apr 23 2022
```
