## Python
```Python
from itertools import islice
def A353986_gen(): # generator of terms
        a, b, k, ah = 1, 1, 1, 1
        while True:
            if ah == (bh := b.bit_count()):
                yield k
            a, b, ah = b, a+b, bh
            k += 1
A353986_list = list(islice(A353986_gen(),30)) # _Chai Wah Wu_, May 13 2022
```
