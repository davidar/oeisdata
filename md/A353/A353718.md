from itertools import count, islice
def A353718_gen(): # generator of terms
    s, a, b, c, ab, k = {0,1}, 0, 1, 2, 1, 1
    yield from (1,1)
    while True:
        for n in count(c):
            if not (n & ab or n in s):
                a, b = b, n
                ab = a|b
                s.add(n)
                if c in s:
                    yield k
                    k = 0
                    while c in s:
                        c += 1
                k += 1
                break
A353718_list = list(islice(A353718_gen(),20)) # _Chai Wah Wu_, May 10 2022
