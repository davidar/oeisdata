from itertools import count, islice
def A353715_gen(): # generator of terms
    s, a, b, c, ab = {0,1}, 0, 1, 2, 1
    yield 1
    while True:
        for n in count(c):
            if not (n & ab or n in s):
                yield b+n
                a, b = b, n
                ab = a|b
                s.add(n)
                while c in s:
                    c += 1
                break
