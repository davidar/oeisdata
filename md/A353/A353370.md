## Python
```Python
from sympy import factorint, primepi
def a(n):
    v = [0, 0]
    for p, e in factorint(n).items(): v[primepi(p)%2] += e
    return int(v[0] == v[1])
print([a(n) for n in range(1, 123)]) # _Michael S. Branicky_, Apr 16 2022
```
