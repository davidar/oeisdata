from itertools import islice
def A353728_gen(): # generator of terms
    yield 1
    l1, s, b = 1, 2, set()
    while True:
        i = s
        while True:
            if i & l1 and not i in b:
                yield int(bin(i)[2:])
                l1 = i
                b.add(i)
                while s in b:
                    b.remove(s)
                    s += 1
                break
            i += 1
A353728_list = list(islice(A353728_gen(),30)) # _Chai Wah Wu_, May 13 2022
