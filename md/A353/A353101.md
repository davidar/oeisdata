## Python
```Python
from sympy import isprime, prime
def a(n, startb=2):
    pn = prime(n); pn2 = pn**2; b = startb
    while not isprime((b**pn2-1)//(b**pn-1)): b += 1
    return b
print([a(n) for n in range(1, 10)]) # _Michael S. Branicky_, Jun 21 2022
```
