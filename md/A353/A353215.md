## C++
```C++
#include <iostream>
using namespace std;
// Think of unsigned int as a natural number
unsigned int f(unsigned int n) {
    return (3*n - 1)/2;
}
unsigned int a(unsigned int pow, unsigned int n) {
    if (pow == 0) return n;
    else return a(pow-1, f(n));
}
int main() {
    for (unsigned int n(0); n <= 20; ++n)
        cout << a(n, n) << " ";
    return 0;
}
```
