from itertools import count, islice
from fractions import Fraction
from sympy.ntheory.continued_fraction import continued_fraction
def A353243_gen(): # generator of terms
    k, c = Fraction(), 0
    for n in count(1):
        k += Fraction(1,n)
        if c < (m := max(continued_fraction(k))):
            c = m
            yield n
