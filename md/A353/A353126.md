## Python
```Python
from sympy import primepi, sieve
sieve.extend(12*10**6)
def sod(n): return sum(map(int, str(n)))
def afind(limit):
    for k in range(1, limit+1):
        if k == primepi(k*sod(k)):
            print(k, end=", ")
afind(100363) # _Michael S. Branicky_, Apr 25 2022
```
