## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen(): # generator of terms
    p, q = 3, 5
    while True:
        if q == p+2:
            t = p*q*(p+q)+1
            if isprime(t):
                yield t
        p, q = q, nextprime(q)
print(list(islice(agen(), 25))) # _Michael S. Branicky_, Apr 19 2022
```
