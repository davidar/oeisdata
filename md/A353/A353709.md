from itertools import count, islice
def A353709_gen(): # generator of terms
    s, a, b, c, ab = {0,1}, 0, 1, 2, 1
    yield from (0,1)
    while True:
        for n in count(c):
            if not (n & ab or n in s):
                yield n
                a, b = b, n
                ab = a|b
                s.add(n)
                while c in s:
                    c += 1
                break
A353709_list = list(islice(A353709_gen(),20)) # _Chai Wah Wu_, May 07 2022
