## Python

from math import gcd
from itertools import count, islice
def A353990_gen(): # generator of terms
    yield 1
    a, s, b = 1, 2, set()
    while True:
        for i in count(s):
            if not (i == a+1 or i & a or gcd(i,a) > 1 or i in b):
                yield i
                a = i
                b.add(i)
                while s in b:
                    s += 1
                break
A353990_list = list(islice(A353990_gen(),30)) # _Chai Wah Wu_, May 24 2022
