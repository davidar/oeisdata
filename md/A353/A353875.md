from sympy import factorint
def s(n):
    f=factorint(n)
    d, q=(list(f.keys()), list(f.values()))
    (a, b, c, x)=(0, 1, 1, 0)
    if(d[0]==2):
        a, x=(0, 1)
        if q[0]>1:
             a=q[0]-1
    for p in range(x, len(d)):
        b*=(1+2*q[p])
        if d[p]%4==1:
            c*=(1+2*q[p])
    return((b-1)//2+a*b+(c-1)//2)
def a(n):
    max=0
    for i in range(1+10**(n-1), 10**n):
        if s(i)>max:
            k,max=(i,s(i))
    return(n,[k,max])
