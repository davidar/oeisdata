def isA333369(n):
    digits = list(map(int, str(n)))
    return all(digits.count(d)%2 == d%2 for d in set(digits))
def a(n): return sum(1 for i in range(10**(n-1), 10**n) if isA333369(i))
print([a(n) for n in range(1, 7)]) # _Michael S. Branicky_, May 06 2022
