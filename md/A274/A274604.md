## Python

from itertools import groupby
def A274604(n):
    if n&1: return (n>>1)+1
    c = 0
    for k, g in groupby(bin(m:=n>>1)[2:]):
        c = c+len(list(g)) if k == '1' else (-c if len(list(g))&1 else c)
    return m-c # _Chai Wah Wu_, Mar 03 2023
