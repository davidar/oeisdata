## Python
```Python
# see link for an alternate that searches in parallel to a limit
from sympy import nextprime
def a(n):
  psum, p = 2, 3
  while (n + psum)%p: psum, p = psum + p, nextprime(p)
  return p
for n in range(12):
  print(a(n), end=", ") # _Michael S. Branicky_, May 03 2021
```
