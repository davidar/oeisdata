    if is_even(n): return False
    if is_prime(n): return False
    for m in (3..(n+3)//3):

## Python

from sympy import isprime
A274967_list = []
for n in range(3,10**6,2):
    if not isprime(n):
        k = 3
        while k*(k+1) <= 2*n:
            if not (2*(k*(k-2)+n)) % (k*(k - 1)):
                break
            k += 1
        else:
            A274967_list.append(n) # _Chai Wah Wu_, Jul 28 2016
