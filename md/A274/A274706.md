def orbital_integral(n):
    if n == 0: return [1]
    S = [0]*(1+floor(n^2//4))
    for u in unit_orbitals(n):
        L = list(accumulate(accumulate(u)))
        S[abs(L[-1])] += 1
    return S
for n in (0..8): print orbital_integral(n)
