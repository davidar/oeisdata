## Python
```Python
from sympy import isprime
def afind(limit, startk=3):
    alst, pow4 = [], 4**startk
    for k in range(startk, limit+1):
        if isprime(pow4 - 27): print(k, end=", ")
        pow4 *= 4
afind(600) # _Michael S. Branicky_, Sep 22 2021
```
