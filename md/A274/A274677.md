## Python
```Python
from sympy import isprime
def afind(limit, startk=0):
    sevenpow10 = 7*10**startk
    for k in range(startk, limit+1):
        if isprime(sevenpow10 + 19):
            print(k, end=", ")
        k += 1
        sevenpow10 *= 10
afind(500) # _Michael S. Branicky_, Dec 31 2021
```
