```Maple
round(evalf(k*sin(sqrt(k)+j*Pi/2))); end;
```

## Python
```Python
from sympy import sin, sqrt, pi
def A274095(n, t = 1):
    k, j = divmod(n,t)
    return int((k*sin(sqrt(k)+j*pi/2)).round())
A274095_list = [A274095(n,1) for n in range(10001)] # _Chai Wah Wu_, Jun 11 2016
```
