## Python
```Python
def pod(n):
    p = 1
    for x in str(n):
        p *= int(x)
    return p
def sod(n):
    return (sum(int(d) for d in str(n))
def cube(z,p):
    iscube=False
    y=int(pow(z,1/p)+0.01)
    if y**p==z:
        iscube=True
    return iscube
for c in range (1, 10**8):
    aa,ab=c-pod(c,10),c-sod(c,10)
    if cube(aa,2)==True and cube(ab,2)==True and aa>0:
       print (c,aa,ab)
```
