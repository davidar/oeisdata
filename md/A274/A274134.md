## Python
```Python
from sympy import isprime
for n in range(3, 10000, 2):
    if not isprime(n): continue
    BL = len(bin(n))-2
    x = (n>>1) + ((n&1) << (BL-1))   # A038572(n)
    if not isprime(x): continue
    y = (n*2) - (1<<BL) + 1   # A006257(n)  for n>0
    if not isprime(y): continue
    print(str(n), end=', ')
```
