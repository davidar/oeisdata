from sympy import sin, sqrt, pi
def f(n, t = 1):
    k, j = divmod(n,t)
    return int((k*sin(sqrt(k)+j*pi/2)).round())
A274096_list = [f(n,2) for n in range(10001)] # _Chai Wah Wu_, Jun 11 2016
