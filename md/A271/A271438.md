## Python
```Python
from sympy import isprime, nextprime
def ispal(n): strn = str(n); return strn == strn[::-1]
def agen():
  np, p, pstr = 1, 11, "11"
  while True:
    if isprime(int(pstr)): yield np
    p = nextprime(p)
    while not ispal(p): p = nextprime(p)
    np += 1
    pstr += str(p)
for an in agen(): print(an, end=", ") # _Michael S. Branicky_, Jan 17 2021
```
