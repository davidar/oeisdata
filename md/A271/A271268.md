## Python
```Python
from functools import reduce
from operator import mul
def product(seq):
    return reduce(mul, seq, 1)
def conversion(n):
    n = str(n)
    return str(sum(int(i) for i in n)) + \
           str(product(int(i) for i in n))
def a271268(n):
    if n == 1:
        return 8
    else:
        r = 8
        while n > 1:
            r = conversion(r)
            n -= 1
        return int(r)
```
