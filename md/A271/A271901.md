from sympy import prime
def A271901(n):
    p = prime(n)
    i, a, b, c =  1, 1, 1, 2 % p
    while a != 1 or b != 1 or c != 1:
        i += 1
        a, b, c = b, c, (a+c) % p
    return i # _Chai Wah Wu_, Feb 26 2017
