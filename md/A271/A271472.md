## Python
```Python
from gmpy2 import c_divmod
u = ('0000','1000','0011','1011')
def A271472(n):
    if n == 0:
        return 0
    else:
        s, q = '', n
        while q:
            q, r = c_divmod(q, -4)
            s += u[r]
        return int(s[::-1]) # _Chai Wah Wu_, Apr 09 2016
```
