from itertools import product
def sortby(x): return (len(x), x.count('1'), x)
def agen(maxbindigs):
    for i in range(1, maxbindigs+1):
        for t in sorted([p for p in product("01", repeat=i)], key=sortby):
            yield int("".join(t), 2)
