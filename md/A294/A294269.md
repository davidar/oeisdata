R:= [1]:
Cands:= [$2..200]:
for n from 2 to 100 do
  found:= false;
  for i from 1 to nops(Cands) do
    x:= Cands[i];
    if nops(select(t -> igcd(t,x) > 1, R))::even then
       R:= [op(R),x];
       Cands:= subsop(i=NULL,Cands);
       found:= true;
       break
    fi
  od;
  if not found then break fi;
od:

## Python
```Python
from math import gcd
def getSeq(n):
    if n == 1:
        return [1]
    prev = getSeq(n-1)
    cand = 1
    while True:
        cand += 1
        if cand in prev:
            continue
        if len([n for n in prev if gcd(cand, n) > 1]) % 2 == 0:
            prev.append(cand)
            return prev
print(getSeq(100))
```
