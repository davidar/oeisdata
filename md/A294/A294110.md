def prime_factors(n):
    factors = {}
    i = 2
    while n != 1:
        while n % i == 0:
            n /= i
            if i in factors:
                factors[i] += 1
            else:
                factors[i] = 1
        i += 1
    return factors
def a(n):
    i = 1
    c = 0
    while c < n:
        i += 1
        p = prime_factors(i)
        if len(p) > 1 and list(p.keys())[1:]+[1] == list(p.values()):
            c +=1
    return i
