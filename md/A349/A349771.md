## Python
```Python
def auptod(maxdigits):
    alst = []
    for d in range(3, maxdigits+1, 2):
        mid = d//2
        for k in range(10**(d-1), 10**d):
            sk = str(k)
            m = int(sk[:mid] + sk[(mid+1):])
            if k%m == 0: alst.append(k)
    return alst
print(auptod(5)) # _Michael S. Branicky_, Nov 29 2021
```
