## Python
```Python
def R(k): c = 3*k+1; return c//(c&-c)
def A349954(n):
    if n == 1: return 0
    ct = 1; m = R(2*n-1); d = m - 2*n + 1
    while m > 1:
        if (R(m) - m)*d < 0: ct += 1; d = -d
        m = R(m)
    return ct
```
