## Python

def aupton(terms):
    alst, pset = [1, 1], set()
    for n in range(3, terms+1):
        p = p2 = alst[-1]*alst[-2]
        while p in pset: p += p2
        alst.append(p//p2); pset.add(p)
    return alst
print(aupton(86)) # _Michael S. Branicky_, Nov 12 2021
