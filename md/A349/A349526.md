## Python
```Python
def auptoj(maxj):
    alst = []
    for j in range(1, maxj+1):
        for i in range(1, j+1):
            if i != j: alst.extend([i, j])
            else: alst.append(i)
    return alst
print(auptoj(10)) # _Michael S. Branicky_, Nov 21 2021
```
