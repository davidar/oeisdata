## Python
```Python
def f(n):
    s = str(n)
    return int("".join(str(s.count(d)) for d in "9876543210").lstrip("0"))
def a(n):
    orbit, fn = {n}, f(n)
    while fn not in orbit:
        orbit.add(fn)
        n, fn = fn, f(fn)
    return len(orbit)
print([a(n) for n in range(70)]) # _Michael S. Branicky_, Nov 18 2021
```
