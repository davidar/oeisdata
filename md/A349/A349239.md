# Using functions NumToFib and RevFibToNum from A349238.
n, a = 0, 0
print(a + a, end = ", ")
while n < 65:
    n += 1
    print(n + RevFibToNum(NumToFib(n)), end = ", ") # _A.H.M. Smeets_, Nov 14 2021
