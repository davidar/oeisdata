## Python
```Python
import numpy as np
def pascals(n):
  a = np.ones(1)
  f = np.ones(2)
  triangle = [a]
  for i in range(n):
    a = np.convolve(a,f)
    triangle.append(a)
  return triangle
def test(n,tri):
  for i, element in enumerate(tri):
    for sub_e in element:
      if sub_e % n == 0:
        return i
tri = pascals(500)
for i in range(1,50):
  print(test(i,tri),end=',')
```
