## Python
```Python
def A349325(n):
    prevc = c = n
    h = 1
    while c > 1:
        if c % 2:
            c = (3*c+1) // 2
            if prevc < n and c > n: h += 1
        else:
            c //= 2
            if prevc > n and c < n: h += 1
        prevc = c
    return h
print([A349325(n) for n in range(1, 100)])
```
