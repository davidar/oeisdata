## Python

from sympy import divisor_count
terms = []
for n in range(1, 101):
    s, t = [1], True
    while t:
        for i in range(2, len(s)):
            if s[-i] == s[-1]:
                t = False
                terms.append(i - 1)
                break
        s.append(n*divisor_count(s[-1]))
print(terms) # _Gleb Ivanov_, Nov 17 2021
