## Python

from sympy import prevprime, factorint, prod
def f(p, e):
    return 0 if e > 1 else -1 if p == 2 else -prevprime(p)
def a(n):
    return prod(f(p, e) for p, e in factorint(n).items()) # _Sebastian Karlsson_, Nov 13 2021
