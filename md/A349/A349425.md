def A349425(n):
    if n % 10 == 0: return 0
    m, n1, n2 = n, 10**n, 10**(n-1)
    while (k := pow(n,m,n1)) != m: m = k
    return k//n2 # _Chai Wah Wu_, Dec 19 2021
