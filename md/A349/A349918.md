## Python
```Python
from sympy.ntheory.digits import digits
def fromdigits(d, b):
    n = 0
    for di in d: n *= b; n += di
    return n
def auptor(rows):
    alst = [1]
    for n in range(2, rows+1):
        alst.extend(digits(fromdigits(alst, n-1), n)[1:])
    return alst
print(auptor(8)) # _Michael S. Branicky_, Dec 05 2021
```
