## Python

def meandigval(n): s = str(n); return sum(map(int, s))/len(s)
def afind(limit):
    alst, k, t, record = [], 0, 0, -1
    while t <= limit:
        mdv = meandigval(t)
        if mdv > record:
            print(t, end=", ")
            record = mdv
        k += 1
        t += k
afind(10**14) # _Michael S. Branicky_, Dec 03 2021
