## Python
```Python
from sympy import isprime, nextprime
def aupto(limit):
    p, q, r, s, alst = 2, 3, 5, 7, []
    while p <= limit:
        if all(isprime(p+q+r+s+k) for k in [p, q, r, s]): alst.append(p)
        p, q, r, s = q, r, s, nextprime(s)
    return alst
print(aupto(500000)) # _Michael S. Branicky_, Nov 22 2021
```
