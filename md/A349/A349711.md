## Python
```Python
from itertools import product
from sympy import factorint
def A349711(n):
    f = factorint(n)
    plist, m = list(f.keys()), sum(f[p]*p for p in f)
    return sum((lambda x: x*(m-x))(sum(d[i]*p for i, p in enumerate(plist))) for d in product(*(list(range(f[p]+1)) for p in plist))) # _Chai Wah Wu_, Nov 27 2021
```
