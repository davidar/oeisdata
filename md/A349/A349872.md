## Python
```Python
def aupton(terms):
    alst, aset = [2], {2}
    for n in range(2, terms+1):
        k = 2
        while k in aset or any(k%j == 0 for j in alst[-k:]): k += 1
        alst.append(k); aset.add(k)
    return alst
print(aupton(73)) # _Michael S. Branicky_, Dec 03 2021
```
