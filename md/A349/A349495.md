## Python
```Python
from sympy import factorint
def ok(n):
    if n < 13: return False
    f = factorint(n)
    sig, p, q = list(f.values()), min(f), max(f)
    return sig == [2, 1] and (q-1)%p == 0 and (q-1)%p**2 != 0
print([m for m in range(1422) if ok(m)]) # _Michael S. Branicky_, Dec 16 2021
```
