from sympy import nextprime
p0 = 1
while p0 < 13000:
    p = nextprime(p0); s = len(str(p)); t = p; L = set()
    while t not in L: L.add(t); t = (t*t) % 10**s
    if t == p: print(p, end = ', ')
    p0 = p
