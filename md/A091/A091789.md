from itertools import count, islice
def agen():
    s, t = "1", "2"
    yield from [1, 2]
    while True:
        s, t = t, (s+t)[::-1]
        yield int(t)
print(list(islice(agen(), 11))) # _Michael S. Branicky_, Jul 01 2022
