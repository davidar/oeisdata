def k(s):
    maxk = 1
    for m in range(1, len(s)+1):
        i, y, kk = 1, s[-m:], len(s)//m
        if kk <= maxk: return maxk
        yY = [y, y[::-1]]
        while s[-(i+1)*m:-i*m] in yY: i += 1
        maxk = max(maxk, i)
def aupton(terms):
    alst = [1]
    for n in range(2, terms+1):
        alst.append(k(alst))
    return alst
print(aupton(105)) # _Michael S. Branicky_, Nov 05 2023
