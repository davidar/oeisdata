from functools import cache
@cache
def a_rec(n):
    if n == 0: return 1
    p = mul(s for s in map(lambda i: i + 1, divisors(n)) if is_prime(s))
    return p * a_rec(n - 1)
print([a_rec(n) for n in range(22)]) # _Peter Luschny_, Dec 12 2023
