## Python
```Python
from sympy import isprime
from sympy.utilities.iterables import multiset_permutations
def A091938(n):
    for i in range(n,-1,-1):
        q = 2**n
        for d in multiset_permutations('0'*(n-i)+'1'*i):
            p = q+int(''.join(d),2)
            if isprime(p):
                return p # _Chai Wah Wu_, Apr 08 2020
```
