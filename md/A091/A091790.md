## Python
```Python
from itertools import count, islice
def agen():
    s, t = "1", "2"
    while True:
        u = (s+t)[::-1]
        yield int(s+t+u)
        s, t = t, u
print(list(islice(agen(), 7))) # _Michael S. Branicky_, Jul 01 2022
```
