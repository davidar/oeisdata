from sympy import npartitions
def A091582(n):
    if n == 1: return 1
    y = 10*(x:=10**(len(str(n>>1))-1))
    return npartitions((c:=n-x)*x+int(str(c)[-2::-1]or 0) if n<x+y else (c:=n-y)*y+int(str(c)[-1::-1]or 0)) # _Chai Wah Wu_, Jun 14 2024
