from sympy.ntheory import digits
def ok(n):
    c = 0
    for b in range(2, n-1):
        d = digits(n, b)[1:]
        c += int(d == d[::-1])
        if c == 2: return True
    return c > 1
print([k for k in range(125) if ok(k)]) # _Michael S. Branicky_, Feb 05 2024
