## Maple

# requires Maple 12 or later
b:= proc(n) local t, L,i;
  L:= convert(n,base,2);
  t:= 0;
  for i from 1 to nops(L) do if L[i]=1 then
    t:= Bits:-Xor(t,i-1)
  fi od;
  t;
end proc:
seq(b(n),n=0..100); # _Robert Israel_, Dec 30 2014

## Python
```Python
def a(n):
    r = 0
    b = 0
    while n > 0:
        if (n & 1):
            r = r ^ b
        b = b + 1
        n = n >> 1
    return r
print([a(n) for n in range(20)])
```
