```Python
from sympy import isprime, primerange
def aupto(limit):
  alst = []
  for p in primerange(2, limit+1):
    if all(map(isprime, [p+2, p+12, p+14, p+24, p+26])): alst.append(p)
  return alst
print(aupto(3*10**6)) # _Michael S. Branicky_, May 17 2021
```
