## Python
```Python
def isqrt(a):
    sr = 1 << (int.bit_length(int(a)) >> 1)
    while a < sr*sr:  sr>>=1
    b = sr>>1
    while b:
        s = sr + b
        if a >= s*s:  sr = s
        b>>=1
    return sr
for n in range(1000):
    cube = n*n*n
    r = isqrt(cube)
    sqr = r**2
    if cube-sqr < n or sqr+2*r+1-cube < n:  print(str(n), end=',')
```
