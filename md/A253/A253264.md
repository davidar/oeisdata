## Python
```Python
from gmpy2 import is_prime, next_prime
A253264_list, p = [], 2
for _ in range(10**7):
    q = p**2 - 2
    if is_prime(q):
        r = q**2 -2
        if is_prime(r):
            s = r**2-2
            if is_prime(s) and is_prime(s**2-2):
                A253264_list.append(p)
    p = next_prime(p) # _Chai Wah Wu_, May 02 2015
```
