## Python
```Python
def A253412(n):
    c, fs = 0, '0'+str(n)+'b'
    for i in range(2**n):
        s = '01'+format(i,fs)+'10'
        for j in range(n):
            if s[j:j+4] == '0100' or s[j+1:j+5] == '0010' or s[j+1:j+4] == '000' or s[j+1:j+4] == '111':
                break
        else:
            c += 1
    return c # _Chai Wah Wu_, Jan 02 2015
```
