## Python
```Python
from sympy import prime, isprime
def reversedigits(n, b=10): # reverse digits of n in base b
    x, y = n, 0
    while x >= b:
        x, r = divmod(x, b)
        y = b*y + r
    return b*y + x
A253149_list = []
for n in range(1, 300):
    p = prime(n)
    if p > 255 and isprime(reversedigits(p,256)):
        A253149_list.append(p)
print(A253149_list)
```
