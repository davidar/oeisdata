## Python
```Python
def a(n):
    if n<2: return [1]
    l=[n, ]
    while True:
        if n%2==0: n//=2
        else: n = 3*n + 1
        if n not in l:
            l.append(n)
            if n<2: break
        else: break
    return l
print([n for n in range(1, 501) if max(a(n)) == n]) # _Indranil Ghosh_, Apr 14 2017
```
