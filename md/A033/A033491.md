## Python
```Python
import numpy
nupto = 62
A033491 = numpy.zeros(nupto, dtype=object)
k, counter = 1, 0
while counter < nupto:
    kk, n = k, 0
    while n <= nupto and kk != 1:
        if kk % 2 == 0:
            kk //= 2
        else:
            kk = (kk*3+1)//2
            n += 1
        n += 1
    if n < nupto and not A033491[n]:
        A033491[n] = k
        counter += 1
    k += 1
print(list(A033491)) # _Karl-Heinz Hofmann_, Feb 11 2023
```
