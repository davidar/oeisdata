## Maple
```Maple
read("transforms") :
isA033548 := proc(n)
    if isprime(n) and digsum(n) = digsum(numtheory[pi](n)) then
        true ;
    else
        false;
    end if;
end proc:
A033548 := proc(n)
    local p, k;
    if n = 1 then
        131;
    else
        p := nextprime(procname(n-1)) ;
        while true  do
            if isA033548(p) then
                return p;
            end if;
            p := nextprime(p) ;
        end do:
    end if;
end proc:
```
