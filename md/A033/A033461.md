## Python
```Python
from functools import cache
from sympy.core.power import isqrt
@cache
def b(n,i):
  # Code after _Alois P. Heinz_
  if n == 0: return 1
  if i == 0: return 0
  i2 = i*i
  return b(n, i-1) + (0 if i2 > n else b(n - i2, i-1))
a = lambda n: b(n, isqrt(n))
print([a(n) for n in range(1, 101)]) # _Darío Clavijo_, Nov 30 2023
```
