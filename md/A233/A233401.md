## Python
```Python
def isqrt(a):
    sr = 1L << (long.bit_length(long(a)) >> 1)
    while a < sr*sr:  sr>>=1
    b = sr>>1
    while b:
      s = sr+b
      if a >= s*s:  sr = s
      b>>=1
    return sr
def isTriangular(a):
  a+=a
  sr = isqrt(a)
  return (a==sr*(sr+1))
for n in range(1,79999):
  n3 = n*n*n
  b = isqrt(n3)
  if b*b==n3: b-=1
  if isTriangular(n3-b*b):  print str(n)+',',
```
