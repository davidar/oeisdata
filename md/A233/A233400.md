## Python
```Python
def isqrt(a):
    sr = 1 << (int.bit_length(int(a)) >> 1)
    while a < sr*sr:  sr>>=1
    b = sr>>1
    while b:
        s = sr+b
        if a >= s*s:  sr = s
        b>>=1
    return sr
def isTriangular(a):
    a+=a
    sr = isqrt(a)
    return (a==sr*(sr+1))
for n in range(77777):
    n3 = n*n*n
    a = isqrt(n3)+1
    if isTriangular(a*a-n3):  print(str(n), end=', ')
```
