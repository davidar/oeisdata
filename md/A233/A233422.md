## Python

def isqrt(a):
    sr = 1L << (long.bit_length(long(a)) >> 1)
    while a < sr*sr:  sr>>=1
    b = sr>>1
    while b:
      s = sr+b
      if a >= s*s:  sr = s
      b>>=1
    return sr
def isSquare(a):
  sr = isqrt(a)
  return (a==sr*sr)
