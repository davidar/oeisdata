# requires python code of A001597
class A233410() :
    def __init__(self) :
        self.a001597 = A001597()
    def perf_pows_up_to(self, n):
        idx = 1
        while True:
            if self.a001597.at(idx) > n :
                return idx-1
            idx += 1
    def at(self, n):
        k = 1
        while True :
            m = k*n
            if self.perf_pows_up_to(m) <= k:
                return m
            k += 1
