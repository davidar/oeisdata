## Python
```Python
from sympy import prime
from itertools import permutations
def zeroless_pans():
    for p in permutations("123456789"):
        yield int("".join(p))
def a(n):
    pn = prime(n)
    return sum(zlp%pn==0 for zlp in zeroless_pans())
print([a(n) for n in range(1, 43)]) # _Michael S. Branicky_, Dec 07 2020
```
