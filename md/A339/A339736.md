## Python
```Python
from sympy.ntheory.digits import digits
def okb(n, b):
    d = digits(n, b)[1:]
    return (len(d) == 1) or (0 in d) or (b-1 not in d)
def ok(n):
    return all(okb(n, b) for b in range(2, n+1))
print(list(filter(ok, range(1, 6301)))) # _Michael S. Branicky_, Sep 13 2021
```
