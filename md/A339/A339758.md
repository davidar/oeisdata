## Python
```Python
from itertools import count
from sympy import nthroot_mod, isprime
def A339758(n):
    m = (n<<1)+1
    r = 1<<m
    a = sorted(nthroot_mod(m,m,r,all_roots=True))
    for i in count(0):
        for k in a:
            if isprime(k+i*r):
                return int(k+i*r) # _Chai Wah Wu_, May 07 2024
```
