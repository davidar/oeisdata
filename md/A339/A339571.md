from math import gcd
from itertools import count, islice
def A339571_gen(): # generator of terms
    a, aset = 1, {1}
    yield 1
    for n in count(2):
        a = a+n+1 if (b:=gcd(a,n)) == 1 else a//b
        if a not in aset:
            aset.add(a)
            yield a
