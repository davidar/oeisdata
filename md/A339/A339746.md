```Maple
N:= 1000: # for terms <= N
R:= {}:
for k1 from 0 to floor(N/6) do
  for k0 in [1,5] do
    k:= k0 + 6*k1;
    for j from 0 while 3^j*k <= N do
      for i from (j mod 3) by 3 do
        x:= 2^i * 3^j * k;
        if x > N then break fi;
        R:= R union {x}
od od od od:
sort(convert(R,list)); # _Robert Israel_, Apr 08 2021
```

## Python
```Python
from sympy import factorint
def ok(n):
  f = factorint(n, limit=4)
  i, j = 0 if 2 not in f else f[2], 0 if 3 not in f else f[3]
  return (i-j)%3 == 0
def aupto(limit): return [m for m in range(1, limit+1) if ok(m)]
print(aupto(200)) # _Michael S. Branicky_, Mar 26 2021
```
