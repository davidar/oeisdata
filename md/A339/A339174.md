## Python
```Python
from sympy import isprime
A339174_list, a = [2], 2
while len(A339174_list) < 10:
    k, c, b = 1, 1, (a-1)*a
    while True:
        c += b
        if isprime(c):
            A339174_list.append(k)
            a = c
            break
        k += 1 # _Chai Wah Wu_, Dec 04 2020
```
