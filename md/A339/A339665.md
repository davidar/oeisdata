## Python
```Python
from itertools import combinations
from sympy import divisors
def A339665(n):
    ds = tuple(divisors(n, generator=True))
    return sum(sum(1 for d in combinations(ds,i) if n*i % sum(d) == 0) for i in range(1,len(ds)+1)) # _Chai Wah Wu_, Nov 09 2021
```
