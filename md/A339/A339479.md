## Python
```Python
from functools import cache
@cache
def r339479(n, k):
    if n == 0:
        return 1
    elif k == 0:
        return r339479(n - 1, 1)
    else:
        return r339479(n - 1, k + 1) + r339479(n, k // 2)
def a339479(n): return r339479(n,0)
print([a339479(n) for n in range(1, 100)])
```
