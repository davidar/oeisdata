## Python
```Python
from sympy import isprime
from sympy import prime
for i in range(1,201):
  j=prime(i)
  xor=2**len(bin(j).strip('0b'))-1
  p=(j^xor)//2
  if isprime(p):
   print(j,end=', ')
```
