## Python
```Python
from sympy import isprime
from sympy import prime
for i in range(1,50):
  d=2*prime(i)
  l=len(bin(d).lstrip('0b'))
  xor=2**l-1
  p=d^xor+2**l
  while not isprime(p):
   l+=1
   p+=2**l
  print(p,end=', ')
```
