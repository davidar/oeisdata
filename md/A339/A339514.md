## Python
```Python
from sympy import divisors
def test(n):
    if n<2: return n-1
    return len(divisors(n))
def a(n):
    tests = [test(i) for i in range(n+1)]
    return sum(2**tests.count(v)-1 for v in set(tests))
print([a(n) for n in range(49)]) # _Michael S. Branicky_, Dec 07 2020
```
