def aupton(terms):
  alst = [1, 10]
  for n in range(3, terms+1): alst.append(int(str(alst[-2])+str(alst[-1])))
  return alst[:terms]
print(aupton(10)) # _Michael S. Branicky_, Apr 24 2021
