## Python
```Python
from sympy import divisors
def s(k): return sum(d for d in divisors(k)[:-1])
def aupto(n):
  alst, aset = [1, 2], {1, 2}
  for k in range(2, n):
    ak = sanm1 = s(alst[-1])
    while ak in aset: ak += sanm1
    alst.append(ak); aset.add(ak)
  return alst     # use alst[n-1] for a(n)
print(aupto(478)) # _Michael S. Branicky_, Dec 29 2020
```
