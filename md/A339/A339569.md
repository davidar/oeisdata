## Python
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def b(n, sos, c):
  if n == 0:
    if c>0:
      if sos==c*c*c: return 1
    return 0
  return b(n-1, sos, c) + b(n-1, sos+n*n, c+1)
a = lambda n: b(n, 0, 0)
print([a(n) for n in range(1, 44)]) # _Michael S. Branicky_, Dec 10 2020
```
