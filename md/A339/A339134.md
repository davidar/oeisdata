from sympy import isprime
def search(n):
    k = 1
    while 1:
        i = k*n + 1
        while i < k*n + k:
            if isprime(i) == 1: return i
            i += 2
        k = 2*k
a_rec = 0
n = 1
while 1:
    m = search(n)
    a = m.bit_length() - n.bit_length()
    if a > a_rec:
        print(n)
        a_rec = a
    n += 1
