import math
a3 = a2 = a1 = 1
last_position = 0
run_lengths = []
for position in range(4, 20000):
    gcd = math.gcd(a1, a3)
    if gcd > 1:
        run_length = position - last_position - 1
        run_lengths.append(run_length)
        last_position = position
    a3, a2, a1 = a2, a1, (a1 + a3) // gcd
print(run_lengths)
