## Python
```Python
from sympy import primepi
a_1 = a_2 = 1
print(a_2, "\n", a_1)
list1 = [a_1, a_2]
for n in range(3, 1001):
    a = a_1 + a_2
    i = 2
    while i < len(list1):
        d = a - primepi(list1[i])
        if d >= 2 and d not in list1: a = d
        else: break
        i += 1
    list1.insert(0, a)
    a_2 = a_1
    a_1 = a
    print(a)
```
