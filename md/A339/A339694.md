## Python
```Python
def A014682(k):
    if k % 2 == 0:
        return k // 2
    else:
        return (3*k + 1) // 2
def x(i, k):
    while i > 0:
        k = A014682(k)
        i = i - 1
    return k % 2
def A(n, k):
    L = [x(i, k) * 2**i for i in range(0, n)]
    return sum(L)
```
