N:= 1000: # to get all terms <= N
R:= {1,2}:
p:= 2:
do
p:= nextprime(p);
if p > N then break fi;
if p mod 4 <> 1 then next fi;
R:= R union select(`<=`,map(`*`,R,p),N);
od:
sort(convert(R,list)); # _Robert Israel_, Aug 23 2017
