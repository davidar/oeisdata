## Python
```Python
from sympy import nextprime
n = x = L = 2
print('0', end=',')
while L < 27:
    nextn = nextprime(n)
    if (nextn ^ n) > n:  # if lengths of binary representations are different
        print(str(x), end=',')
        x = 0
        prevL = L
        L = len(bin(nextn))-2
        for j in range(prevL, L-1):  print('0', end=',')
    n = nextn
    x ^= n
```
