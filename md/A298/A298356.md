from functools import lru_cache
@lru_cache(maxsize=None)
def A298356(n):
    if n <= 2:
        return 1
    c, j = A298356(n-1)+A298356(n-2), 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*A298356(k1)
        j, k1 = j2, n//j2
    return c+n-j+1 # _Chai Wah Wu_, Mar 31 2021
