from functools import lru_cache
@lru_cache(maxsize=None)
def A298357(n):
    if n <= 2:
        return n+1
    c, j = A298357(n-1)+A298357(n-2), 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*A298357(k1)
        j, k1 = j2, n//j2
    return c+2*(n-j+1) # _Chai Wah Wu_, Mar 31 2021
