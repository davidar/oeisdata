from functools import lru_cache
@lru_cache(maxsize=None)
def A298369(n):
    if n <= 2:
        return 1
    c, j = A298369(n-1)+A298369(n-2), 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2*(j2-1)-j*(j-1))*A298369(k1)//2
        j, k1 = j2, n//j2
    return c+(n*(n+1)-j*(j-1))//2 # _Chai Wah Wu_, Mar 31 2021
