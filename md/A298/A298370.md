from functools import lru_cache
@lru_cache(maxsize=None)
def A298370(n):
    if n <= 2:
        return n+1
    c, j = A298370(n-1)+A298370(n-2), 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2*(j2-1)-j*(j-1))*A298370(k1)//2
