R:= NULL:
count:= 0: F:= 10;
for n from 4 while count < 100 do
  F:= F * (4*n-2)/n;
  if not isprime(n) then
     count:= count+1;
     R:= R, F mod (n^4);
  fi
od:
R; # _Robert Israel_, Feb 02 2018
