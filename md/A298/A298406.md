from functools import lru_cache
@lru_cache(maxsize=None)
def A298406(n):
    if n <= 2:
        return 1
    c, j = 2*A298406(n-1)-A298406(n-3), 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*A298406(k1)
        j, k1 = j2, n//j2
    return c+n-j+1 # _Chai Wah Wu_, Mar 31 2021
