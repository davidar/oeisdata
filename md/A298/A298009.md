## Python
```Python
# Generates all elements of the sequence smaller than last
last = 1000
p=[2]
c=1
for i in range(3,last+2,2):
    prime = True
    for j in p:
        if i%j == 0:
            prime=False;
            break
    if prime:
        p.append(i)
        c = c + 1
    ii = (i//100)*100
    if i-ii == 1:
        if prime:
            print(ii-100+c-1, end=',')
            c = 1
        else:
            print(ii-100+c, end=',')
            c = 0
```
