## Python
```Python
from fractions import Fraction
from itertools import count, islice
def agen(): # generator of terms
    Hn = Fraction(0, 1)
    for n in count(0):
        yield (-3*n + 2*(1+n)*Hn).numerator
        Hn += Fraction(1, n+1)
print(list(islice(agen(), 27))) # _Michael S. Branicky_, Apr 17 2022
```
