read("transforms"):
A093882 := proc(n)
    local a,dgs,d,dgsred ;
    a := 0 ;
    dgs := convert(n,base,10) ;
    for d from 1 to nops(dgs) do
        [op(1..d-1,dgs),op(d+1..nops(dgs),dgs)] ;
        a := a+digcatL(%) ;
    end do:
    a ;
end proc: # _R. J. Mathar_, May 06 2019
