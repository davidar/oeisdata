## Python

from math import prod
from sympy import factorint
def A093803(n):
    if n == 1: return 1
    f = factorint(n)
    m = min(f)
    return prod(p**(0 if p == 2 else e-1 if p == m else e) for p,e in f.items()) # _Chai Wah Wu_, Oct 27 2022
