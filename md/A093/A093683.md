## Python
```Python
from sympy import prime, sieve # use primerange for larger terms
def afind(terms):
  c, prevp = 0, 1
  for n in range(1, terms+1):
    for p in sieve.primerange(prevp+1, prime(10**n)+1):
      if prevp == p - 2: c += 1
      prevp = p
    print(c, end=", ")
afind(6) # _Michael S. Branicky_, Apr 25 2021
```
