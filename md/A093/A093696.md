## Python
```Python
from sympy import divisors, isprime
def c(n): return bin(n).count("1")&1
def ok(n): return n > 0 and all(c(d) for d in divisors(n, generator=True))
print([k for k in range(174) if ok(k)]) # _Michael S. Branicky_, Jul 24 2022
```
