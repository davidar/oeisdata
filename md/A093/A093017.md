## Python
```Python
def a(n):
    s = str(n)
    r = s[::-1]
    x = sum(int(d) for d in r[::2])
    x += sum(q if (q:=2*int(d)) < 10 else q-9 for d in r[1::2])
    return x
print([a(n) for n in range(87)]) # _Michael S. Branicky_, Jul 23 2024
```
