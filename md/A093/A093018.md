def a(n):
    s = str(n)
    r = s[::-1]
    x = sum(int(d) for d in r[1::2])
    x += sum(q if (q:=2*int(d)) < 10 else q-9 for d in r[::2])
    x = x%10
    c = str((10 - x) if x > 0 else 0)
    return int(s+c)
