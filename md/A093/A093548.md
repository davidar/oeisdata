## Python
```Python
from sympy import primefactors, primorial
def a(n):
  m = primorial(n)
  while True:
    if len(primefactors(m)) == n:
      if len(primefactors(m+1)) == n: return m
      else: m += 2
    else: m += 1
for n in range(1, 6):
  print(a(n), end=", ") # _Michael S. Branicky_, Feb 14 2021
```
