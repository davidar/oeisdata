## Python

def afind(terms):
  m, n, c = 0, 1, 0
  while n <= terms:
    p = m*(5*m-3)//2
    s = str(p)
    if len(s) == n:
       if s == s[::-1]: c += 1
    else:
      print(c, end=", ")
      n, c = n+1, int(s == s[::-1])
    m += 1
afind(14) # _Michael S. Branicky_, Mar 01 2021
