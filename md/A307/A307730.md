from itertools import islice
from collections import Counter
def A307730(): # generator of terms. Greedy algorithm
    c, b = Counter(), 1
    while True:
        k, kb = 1, b
        while c[kb] >= kb:
            k += 1
            kb += b
        c[kb] += 1
        b = k
        yield kb
