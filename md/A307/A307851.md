## Python
```Python
from math import prod
from sympy import nextprime
def aupton(terms):
  p, k, t = 2, 1, 0
  while t < terms:
    strp = str(p)
    if '0' not in strp and prod(int(d) for d in strp)%k == 0:
      t += 1; print(p, end=", ")
    p, k = nextprime(p), k+1
aupton(5) # _Michael S. Branicky_, Feb 17 2021
```
