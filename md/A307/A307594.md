from itertools import accumulate, count, islice
def A307594_gen(): # generator of terms
    blist, a, b = tuple(), 1, -1
    for n in count(1):
        yield (blist := tuple(accumulate(reversed(blist),initial=a)))[-1]
        a, b = a*n+b, -b
