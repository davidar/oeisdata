A053392 := proc(n)
    if n < 10 then
        0;
    else
        dgs := convert(n,base,10) ;
        dgsL := [op(1,dgs)+op(2,dgs)] ;
        for i from 3 to nops(dgs) do
            dgsL := [op(i,dgs)+op(i-1,dgs),op(dgsL)] ;
        end do:
        digcatL(dgsL) ;
