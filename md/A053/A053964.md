## Python
```Python
from itertools import product
from sympy import integer_nthroot
A053964_list = []
for l in range(1,21):
    for p in product(*['479']*l):
        a, b = integer_nthroot(int(''.join(p)),2)
        if b:
            A053964_list.append(a) # _Chai Wah Wu_, Apr 01 2021
```
