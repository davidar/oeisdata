A[1]:= 3;
for n from 2 to 100 do
d:= 10^(ilog10(A[n-1])+1);
for k from 1 do
  p:= A[n-1]+d*k;
  if isprime(p) then
    A[n]:= p;
    break
  fi

## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(an=3):
    while True:
        yield an
        pow10 = 10**len(str(an))
        for t in count(pow10+an, step=pow10):
            if isprime(t):
                an = t
                break
print(list(islice(agen(), 17))) # _Michael S. Branicky_, Jun 23 2022
```
