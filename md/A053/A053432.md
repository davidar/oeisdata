import Data.IntSet (fromList, deleteFindMin, union)
a053432 n = a053432_list !! (n-1)
a053432_list = 0 : f (fromList [1..9]) where
   f s = x : f (s' `union`
         fromList (map (+ 10 * x) $ dropWhile (/= mod x 10) digs))
     where (x, s') = deleteFindMin s
   digs = [8, 5, 4, 9, 1, 7, 6, 3, 2, 0]
