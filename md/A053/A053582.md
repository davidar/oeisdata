## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(an=1): # generator of terms
    while True:
        yield an
        pow10 = 10**len(str(an))
        for t in count(pow10+an, step=pow10):
            if isprime(t):
                an = t
                break
print(list(islice(agen(), 18))) # _Michael S. Branicky_, Jun 23 2022
```
