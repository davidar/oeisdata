prpr = 0
prev = 1
jac = [0]*10000
for n in range(10000):
    jac[n] = prpr
    curr = prpr*2 + prev
    prpr = prev
    prev = curr
prpr, prev = 0, 1
for n in range(1, 44):
    print(prpr, end=', ')
    b = c = 0
    while c<=prev:
        c = jac[b] - prpr
        b+=1
    prpr = prev
    prev = c
