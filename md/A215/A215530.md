## Python
```Python
TOP = 10000
a = [0]*TOP
b = [0]*TOP
a[1] = b[1] = 1
n = 2
while n*2+1<TOP:
  a[n:] = a[:n]
  a.append(0)
  b[n:] = b[:n]
  b.append(1)
  n += n+1
for i in range(n-1):
  print a[i],
  if b[i]!= 1 - a[i+1]:
    print 'error!'
    break
```
