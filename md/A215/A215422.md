## Python
```Python
TOP = 33
fib2m1 = [0]*TOP  # Fibonacci(2^n-1)
fib2   = [1]*TOP  # Fibonacci(2^n)
print(1, end=',')
for n in range(1,TOP):
    fib2[n]   = (2*fib2m1[n-1] + fib2[n-1])*fib2[n-1]
    fib2m1[n] = fib2m1[n-1]*fib2m1[n-1] + fib2[n-1]*fib2[n-1]
    print(len(bin(fib2[n]))-2, end=',')
```
