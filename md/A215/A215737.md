## Python

def A215737(n):
    a, s = 1, tuple(str(i)*n for i in range(10))
    while True:
        a *= 11
        t = str(a)
        for i, x in enumerate(s):
            if x in t:
                return i # _Chai Wah Wu_, Mar 30 2021
