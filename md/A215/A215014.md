## Python
```Python
# _Ely Golden_, Dec 26 2017
def consecutive(li):
  for i in range(len(li)-1):
    if(li[i+1]!=1+li[i]): return False
  return True
def sorted_digits(n):
  lst=[]
  while(n>0):
    lst+=[n%10] ; n//=10
  lst.sort() ; return lst
j=0
for i in range(1,10001):
  while(not consecutive(sorted_digits(j))): j+=1
  print(str(i)+" "+str(j)) ; j+=1
```
