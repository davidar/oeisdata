## Python
```Python
from __future__ import division
from sympy import divisors
def A232179(n):
    if n == 0:
        return 0
    t = 2*n**2
    ds = divisors(t)
    for i in range(len(ds)//2-1,-1,-1):
        x = ds[i]
        y = t//x
        a, b = divmod(y-x,2)
        if b:
            return a
    return -1 # _Chai Wah Wu_, Sep 12 2017
```
