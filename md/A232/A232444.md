## Python

from sympy import isprime, divisor_sigma
A232444_list = [2]+[n for n in (d**2 for d in range(1,10**4)) if isprime(divisor_sigma(n)) and isprime(divisor_sigma(n**2))] # _Chai Wah Wu_, Jul 23 2016
