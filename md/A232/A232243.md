## JavaScript
```JavaScript
function bitCount(n) {
   var i,c,s;
   c=0;
   s=n.toString(2);
   for (i=0;i<s.length;i++)
      if (s.charAt(i)==1)
         c++;
   return c;
}
for (i=0;i<100;i++) document.write(bitCount(i*i)-bitCount(i)+", ");
```
