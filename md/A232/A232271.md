```Python
TOP = 2**30  # if enough RAM
a = [1]*TOP
for n in range(1,TOP):
  a[n]=n
for n in range(1,TOP):
  if n+a[n]<TOP and n+a[n]>0: a[n+a[n]] -= a[n]
for n in range(1,1000):
```
