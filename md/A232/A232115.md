## Python
```Python
import math
def get_x_y(n):
  sr = int(math.sqrt(n-1))
  sr = sr-1+(sr&1)
  rm = n-sr*sr
  d = (sr+1)/2
  if rm<=sr+1:
     return -d+rm,d
  if rm<=sr*2+2:
     return d,d-(rm-(sr+1))
  if rm<=sr*3+3:
     return d-(rm-(sr*2+2)),-d
  return -d,-d+rm-(sr*3+3)
for n in range(1,333):
  x0,y0 = get_x_y(n)
  x1,y1 = get_x_y(n*(n+1)/2)
  print str(abs(x1-x0)+abs(y1-y0))+',',
```
