## Python
```Python
from __future__ import division
from sympy import divisors
def A232178(n):
    if n == 0:
        return 0
    t = n*(n+1)//2
    ds = divisors(t)
    l, m = divmod(len(ds),2)
    if m:
        return 0
    for i in range(l-1,-1,-1):
        x = ds[i]
        y = t//x
        a, b = divmod(y-x,2)
        if not b:
            return a
    return -1 # _Chai Wah Wu_, Sep 12 2017
```
