```Python
from __future__ import division
from sympy import divisors
def A232175(n):
    n3 = n**3
    ds = divisors(n3)
    for i in range(len(ds)//2-1,-1,-1):
        x = ds[i]
        y = n3//x
        a, b = divmod(y-x,2)
        if not b:
            return a
    return 0 # _Chai Wah Wu_, Sep 12 2017
```
