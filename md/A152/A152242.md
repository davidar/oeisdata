## Python
```Python
def ok(n):
    if isprime(n): return True
    s = str(n)
    return any(s[i]!="0" and isprime(int(s[:i])) and ok(int(s[i:])) for i in range(1, len(s)))
print([k for k in range(180) if ok(k)]) # _Michael S. Branicky_, Sep 01 2024
```
