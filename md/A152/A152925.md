## Python
```Python
def different(d):
    for i in range(len(d)):
        for j in range(i):
            if d[i] == d[j]:
                return False
    return True
def a(base):
    d = [0] * base
    n = 0
    while True:
        n += 1
        m = n
        while m > 0:
            d[m % base] += 1
            m //= base
        if different(d):
            break
    return n
```
