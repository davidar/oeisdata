from re import split
from sympy import isprime
A319302_list, n  = [], 1
while len(A319302_list) < 10000:
    for d in split('1+',bin(n)[2:]):
        if isprime(len(d)):
            A319302_list.append(n)
            break
    n += 1 # _Chai Wah Wu_, Oct 02 2018
