  L:= convert(n,base,10);
  add(L[-i]*10^(i-1),i=1..nops(L));
end proc:

## Python
```Python
nmax=10000
def is_prime(num):
    if num == 0 or num == 1: return(0)
    for k in range(2, num):
       if (num % k) == 0:
           return(0)
    return(1)
ris = ""
for i in range(nmax):
    if is_prime(i):
       r=int((str(i)[::-1]))
       t=pow(i,2)+r
       if is_prime(t):
          ris = ris+str(i)+","
print(ris)
```
