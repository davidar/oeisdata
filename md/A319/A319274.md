## Python
```Python
import itertools
def getData(a,b):
    dig = (itertools.permutations(str(a),b))
    for d in dig:
        yield d
for w in range(2,6):
    kk=int(w*'1')
    for i in range (kk,10**(w+3),kk):
        m=[]
        get = getData(i,w)
        while True:
            try:
                n = next(get)
                ee=int("".join((n)))
                if ee not in m:
                    m.append(ee)
            except StopIteration:
                if sum (m)==i and len(m)>1:
                      m.sort()
                    print (sum(m),len(m),m,i)
                break
```
