## Python
```Python
def a(n, base=3):
    c, b, w = 0, base, 0
    while True:
        d = n%b
        if d+1 < b and c:
            return (n+1)*b**w + ((c-1)%(b-1)+1)*b**((c-1)//(b-1))-1
        c += d; n //= b; w += 1
print([a(n) for n in range(1, 67)]) # _Michael S. Branicky_, Jul 10 2022 after _RÃ©my Sigrist_
```
