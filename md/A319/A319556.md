## Python
```Python
def alt(k):
    return sum(k[::2])-sum(k[1::2])
def alt_run(n):
    m = []
    m.append(n)
    for i in range (1, n):
        m.append(m[0]+i)
    return alt(m)
t=[]
for i in range (100):
    t.append(alt_run(i))
print(t)
```
