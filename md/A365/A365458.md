## Python
```Python
def A365458(n):
    kmin, kmax = 0, 1
    while 3**kmax <= n:
        kmax <<= 1
    while True:
        kmid = kmax+kmin>>1
        if 3**kmid > n:
            kmax = kmid
        else:
            kmin = kmid
        if kmax-kmin <= 1:
            break
    return 3**kmin # _Chai Wah Wu_, Sep 17 2023
```
