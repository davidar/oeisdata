from math import prod
from itertools import takewhile
from sympy import factorint, divisors
def A365049(n): return sum(1+(prod((e+(p&1)<<1)-1 for p, e in factorint(d).items())>>1)+(prod((e+(p&1)<<1)-1 for p, e in factorint(n//d).items())>>1 if d*d<n else 0) for d in takewhile(lambda d:d*d<=n, divisors(n))) # _Chai Wah Wu_, Aug 21 2023
