## Python
```Python
from itertools import count
from sympy import divisors
def A365314(n):
    a = set()
    for i in range(1,n+1):
        if not n%i:
            a.update(tuple(sorted((i,j))) for j in range(1,n+1) if j!=i)
        else:
            for j in count(0,i):
                if j > n:
                    break
                k = n-j
                for d in divisors(k):
                    if d>=i:
                        break
                    a.add((d,i))
    return len(a) # _Chai Wah Wu_, Sep 12 2023
```
