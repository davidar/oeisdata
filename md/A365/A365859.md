from sympy import totient, lucas, divisors
def A365859(n):
    m = n>>(~n&n-1).bit_length()
    return sum(totient(k)*lucas(m//k) for k in divisors(m,generator=True))//m # _Chai Wah Wu_, Sep 23 2023
