## Python
```Python
from itertools import count
from sympy import divisors
def A365315(n):
    a = set()
    for i in range(1,n+1):
        for j in count(i,i):
            if j >= n:
                break
            for d in divisors(n-j):
                if d>=i:
                    break
                a.add((d,i))
    return len(a) # _Chai Wah Wu_, Sep 13 2023
```
