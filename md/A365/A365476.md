from sympy import primepi, divisors
def A365476(n):
    if n == 1: return 4
    m, k = n, primepi(n) + n + (n>>1)
    while m != k:
        m, k = k, primepi(k) + n + (k>>1)
    return min(int(d).bit_count()*int(m//d).bit_count() for d in divisors(m,generator=True) if 1<d<m) # _Chai Wah Wu_, Aug 02 2024
