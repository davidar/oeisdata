## Python

from sympy.utilities.iterables import partitions
def A365659(n): return n>>1 if n&1 or n==0 else (m:=n>>1)+sum(1 for p in partitions(m) if max(p.values(),default=1)==1)-2 # _Chai Wah Wu_, Sep 18 2023
