## Python
```Python
from itertools import combinations
from sympy.utilities.iterables import partitions
def A365044(n):
    mlist = tuple({tuple(sorted(p.keys())) for p in partitions(m,k=m-1)} for m in range(1,n+1))
    return n+1+sum(1 for k in range(2,n+1) for w in combinations(range(1,n+1),k) if w[:-1] not in mlist[w[-1]-1]) # _Chai Wah Wu_, Nov 20 2023
```
