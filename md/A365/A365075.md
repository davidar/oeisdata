## Python

from itertools import count, islice
from math import gcd
def A365075_gen(): # generator of terms
    c = 1
    for n in count(2):
        for i in range(1,1+(n-1>>1)):
            if gcd(i,n-i)==1:
                c *= 10
                yield (i*c//(n-i))%10
A365075_list = list(islice(A365075_gen(),30)) # _Chai Wah Wu_, Aug 28 2023
