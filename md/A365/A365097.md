## Python
```Python
from itertools import count
from sympy.ntheory.factor_ import digits
def A365097(n):
    c, a, q, m = 1, 1, 0, 1
    for k in count(2):
        m += 1
        if m == n:
            m = 0
            q += 1
            a = digits(q,n).count(1)
        elif m==1:
            a += 1
        elif m==2:
            a -= 1
        c += a
        if c == k:
            return k # _Chai Wah Wu_, Sep 28 2023
```
