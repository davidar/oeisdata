## Python
```Python
from math import isqrt
from sympy.utilities.iterables import partitions
def A365311(n):
    a = {tuple(sorted(set(p))) for p in partitions(n)}
    return sum(1 for m in range(1,n+1) for b in partitions(m,m=isqrt(1+(n<<3))>>1) if max(b.values()) == 1 and any(set(d).issubset(set(b)) for d in a)) # _Chai Wah Wu_, Sep 13 2023
```
