## Python

```Python
def G(m: list[int], s: int) -> int:
    k = len(m)
    assert s >= k
    assert set(range(k)) == set(m), \
           f"Sequence m of length {k} should contain a permutation of all the " \
           f"numbers 0..{k-1} inclusive."
    s_init = list(range(s))
    n, s = 0, None
    while s != s_init:
        if n == 0:
            s = s_init
        n += 1
        s = sum((s[offset::k] for offset in m),
                start=[])
    return n
```
