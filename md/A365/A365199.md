## Python

from itertools import count,islice
def A365199_gen(): # generator of terms
    yield (a:=1)
    for n in count(2):
        if a<n:
            a += n
        elif a==n:
            a = n**2
        elif a%n:
            a -= n
        else:
            a //= n
        if a == 1:
            yield n
A365199_list = list(islice(A365199_gen(),10)) # _Chai Wah Wu_, Sep 23 2023
