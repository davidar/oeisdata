## Python
```Python
from bisect import bisect
from sympy import divisor_count
def A365485(n):
    m = 10**n
    plist, qlist, c = tuple(divisor_count(i) for i in range(1,m+1)), [0]*(m+1), 0
    for i in range(m):
        qlist[a:=bisect(qlist,plist[i],lo=1,hi=c+1,key=lambda x:plist[x])]=i
        c = max(c,a)
    return c
```
