from math import gcd
def A365877(n):
    if n == 1: return 1
    c = 1
    for m in range(2,n+1):
        for v in range(1,m+1>>1):
            u = m-(v<<1)
            if gcd(u,v)==1:
                v2, u2 = v*v, v*(u<<2)
                if v2+u2 >= 0:
                    c +=1
                if v2-u2 >= 0:
                    c +=1
    return c # _Chai Wah Wu_, Oct 05 2023
