## Python
```Python
from itertools import count, islice
def agen():
    cat = ""
    for i in count(1):
        s = str(i)
        if cat != "" and int(cat[-1])%2 != int(s[0])%2:
            yield int(cat)
            cat = ""
        cat += s
print(list(islice(agen(), 55))) # _Michael S. Branicky_, Nov 18 2023
```
