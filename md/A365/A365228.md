## Maple
```Maple
A365228 := proc(n)
   local j,k,s; s := 0;
   for j from 1 to n do
      for k from j to n do
         if gcd(j,k) = 1 then s := s + 1/(j*k);
         end if;
      end do;
   end do;
   denom(s);
end proc;
```
