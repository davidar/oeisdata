## Python
```Python
from sympy import isprime
def a(n):
    s = str(n)
    ss = (s[i:j] for i in range(len(s)) for j in range(i+1, len(s)+1))
    return sum(p for w in ss if w[0]!="0" and isprime(p:=int(w)))
print([a(n) for n in range(81)]) # _Michael S. Branicky_, Aug 30 2023
```
