from sympy import divisors
def agen(k=34): # generator of terms starting at k
    while True:
        yield k
        s = set()
        for d in divisors(k, generator=True):
            v, w = str(d), str(k//d)
            if "0" not in v and "0" not in w:
                s.add(int(v + "0" + w))
        if len(s) == 0: return
        k = min(s)
print(list(agen())) # _Michael S. Branicky_, Sep 26 2023
