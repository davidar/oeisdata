## Python
```Python
from itertools import islice
import hashlib
def g():
  c, vmax, m = 1, b"", b""
  while True:
    if (m:= hashlib.sha256(m).digest()) > vmax:
      vmax = m
      yield c
    c += 1
print(list(islice(g(),16)))
```
