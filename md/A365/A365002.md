## Python
```Python
from itertools import combinations
from collections import Counter
from sympy.utilities.iterables import partitions
def A365002(n):
    aset = Counter(tuple(sorted(set(p))) for p in partitions(n))
    return sum(sum(aset[t] for t in aset if set(t).issubset(set(q))) for l in range(1,n+1) for q in combinations(range(1,n+1),l) if sum(q)<=n) # _Chai Wah Wu_, Sep 20 2023
```
