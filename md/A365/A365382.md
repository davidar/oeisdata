## Python
```Python
from math import gcd
from sympy.utilities.iterables import partitions
def A365382(n):
    a = {tuple(sorted(set(p))) for p in partitions(n)}
    return sum(1 for m in range(1,n) for b in partitions(m) if gcd(*b.keys()) == 1 and not any(set(d).issubset(set(b)) for d in a)) # _Chai Wah Wu_, Sep 13 2023
```
