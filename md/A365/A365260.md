## Python
```Python
from sympy import divisors
def a(n):
    k, steps = n, 1
    while True:
        s = set()
        for d in divisors(k, generator=True):
            v, w = str(d), str(k//d)
            if "0" not in v and "0" not in w:
                s.add(int(v + "0" + w))
        if len(s) == 0: return steps
        k, steps = min(s), steps + 1
print([a(n) for n in range(1, 101)]) # _Michael S. Branicky_, Sep 26 2023
```
