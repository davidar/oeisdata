## Python
```Python
from bisect import bisect
from sympy import totient
def A365737(n):
    plist, qlist, c = tuple(-totient(i) for i in range(1,n+1)), [0]*(n+1), 0
    for i in range(n):
        qlist[a:=bisect(qlist,plist[i],lo=1,hi=c+1,key=lambda x:plist[x])]=i
        c = max(c,a)
    return c
```
