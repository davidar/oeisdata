## Python
```Python
from itertools import combinations
from sympy.utilities.iterables import partitions
def A365542(n):
    a = {tuple(sorted(set(p))) for p in partitions(n)}
    return sum(1 for m in range(1,n) for b in combinations(range(1,n),m) if any(set(d).issubset(set(b)) for d in a)) # _Chai Wah Wu_, Sep 12 2023
```
