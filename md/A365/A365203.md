A365203 := proc(n) option remember;
    if n = 1 then 1;
    elif procname(n - 1) < n then procname(n - 1) + n;
    elif n = procname(n - 1) then n*n;
    elif irem(procname(n - 1), n) = 0 then procname(n - 1)/n;
    else procname(n - 1) - n;
    end if;
end proc;

from itertools import count, islice
def A365203_gen(): # generator of terms
    yield (a:=1)
    for n in count(2):
        if a<n:
            b = a+n
        elif a==n:
            b = n**2
        elif a%n:
            b = a-n
        else:
            b = a//n
        yield (a:=b)
A365203_list = list(islice(A365203_gen(),30)) # _Chai Wah Wu_, Sep 23 2023
