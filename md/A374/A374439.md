```Python
from functools import cache
@cache
def T(n: int, k: int) -> int:
    if k > n: return 0
    if k < 2: return k + 1
    return T(n - 1, k) + T(n - 2, k - 2)
```

```Python
from math import comb as binomial
def T(n: int, k: int) -> int:
    o = k & 1
    return binomial(n - o - (k - o) // 2, (k - o) // 2) << o
```
