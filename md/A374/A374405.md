```Python
from itertools import count, islice
def agen(): # generator of terms
    s = ""
    for n in count(0):
        sn = str(n)
        s += sn
        locn = 1 + s.index(sn)
        s += str(locn)
        yield from [n, locn]
print(list(islice(agen(), 70))) # _Michael S. Branicky_, Jul 07 2024
```
