## Python

from itertools import count
from sympy.abc import x,y
from sympy.solvers.diophantine.diophantine import diop_quadratic
def A374090(n): return next(m for m in (3*k**2 if n&1 else k for k in count(0)) if sum(1 for d in diop_quadratic(x*(x+y)+y**2-m) if d[0]>0 and d[1]>0) == n) # _Chai Wah Wu_, Jun 28 2024
