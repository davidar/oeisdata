from itertools import count, islice
from math import gcd
def A374983_gen(): # generator of terms
    p, q = 0, 1
    for k in count(1):
        yield p
        m = q//(k*(q-p))+1
        p, q = p*k*m+q, k*m*q
        p //= (r:=gcd(p,q))
        q //= r
A374983_list = list(islice(A374983_gen(),11)) # _Chai Wah Wu_, Aug 28 2024
