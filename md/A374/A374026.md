# uses A023183() in A023183
from itertools import count
def a(n):
    if A023183(n) == -1:
        return -1
    f, g, s = 1, 2, str(n)
    pow10 = 10**len(s)
    for k in count(3):
        if g%pow10 == n:
            sfib = str(g)
            if g > n and sfib.startswith(s):
                return k
        f, g = g, f+g
print([a(n) for n in range(1, 51)]) # _Michael S. Branicky_, Jul 03 2024
