## Python
```Python
from itertools import count, islice
from sympy import isprime
def A374901_gen(): # generator of terms
    f = 1
    for k in count(1):
        if isprime((k**2+1)*f+1):
            yield k
        f *= k**2
A374901_list = list(islice(A374901_gen(),10)) # _Chai Wah Wu_, Oct 02 2024
```
