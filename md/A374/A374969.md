from math import prod
from sympy import factorint
def A374969(n):
    f = factorint(n).items()
    return (n+1)*prod(e+1 for p,e in f)-(prod((p**(e+1)-1)//(p-1) for p,e in f)<<1) # _Chai Wah Wu_, Jul 26 2024
