## C
```C
#include <stdio.h>
#define N    10000ULL
typedef unsigned long long ull_t;
ull_t Sums[2 * N * N];
int main() {
    ull_t sol = 0;
    for (ull_t i = 1; i < N; i++)
        for (ull_t j = i + 1; j <= N; j++)
            sol += Sums[i * i + j * j]++;
    printf("%llu \n", sol);
}
```

## Python
```Python
from itertools import count, islice
from collections import Counter
def A374020_gen(): # generator of terms
    c, s = 0, Counter()
    for n in count(1):
        n2 = n**2
        for i in range(1,n):
            c += s[m:=i**2+n2]
            s[m] += 1
        yield c
A374020_list = list(islice(A374020_gen(),20)) # _Chai Wah Wu_, Jul 18 2024
```
