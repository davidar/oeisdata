## Python
```Python
from itertools import islice
def z(n): return int(str(n).replace("0", ""))
def agen(): # generator of terms
    yield 1
    alst = [1, 1]
    an = 1
    while True:
        yield an
        an = max(max(z(ai+an) for ai in alst[:-1]), an)
        alst.append(an)
print(list(islice(agen(), 45))) # _Michael S. Branicky_, Jul 24 2024
```
