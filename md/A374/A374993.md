def A374993(n):
    if n<1: return 0
    cost,c = 0,comp(n)
    m = list(range(1,max(c)+1))
    for i in c:
        j = m.index(i)
        cost += j+1
        if j > 0:
            m.insert(j-1,m.pop(j))
