from itertools import count, islice
from math import gcd
def A374663_gen(): # generator of terms
    p, q = 0, 1
    for k in count(1):
        yield (m:=q//(k*(q-p))+1)
        p, q = p*k*m+q, k*m*q
        p //= (r:=gcd(p,q))
        q //= r
A374663_list = list(islice(A374663_gen(),11)) # _Chai Wah Wu_, Aug 28 2024
