from sympy import primefactors
def A374366(n): # based on multiplicative property of a(n)
    a = 0
    for p in primefactors(n>>(~n & n-1).bit_length()):
        if p&2:
            a += 1
        else:
            return 0
    return 1<<a # _Chai Wah Wu_, Jul 07 2024
