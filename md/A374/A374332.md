## Python

from fractions import Fraction
from itertools import count, islice
def A374332_gen(): # generator of terms
    a = Fraction(0,1)
    for n in count(1):
        yield a.numerator
        a = (2*a+Fraction(1,n)) % 1
A374332_list = list(islice(A374332_gen(),20)) # _Chai Wah Wu_, Jul 13 2024
