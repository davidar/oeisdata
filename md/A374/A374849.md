## Python
```Python
def a(n):
  e, B = "", bin(n)[2:]
  for i in range(0,len(B),2):
      if B[i:i+2] == "10": e += "1"
      if B[i:i+2] == "01": e += "0"
  if e == '': return 0
  return int(e,2)
print([a(n) for n in range(1,70)])
```
