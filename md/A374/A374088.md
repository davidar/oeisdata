from sympy.abc import x, y
from sympy.solvers.diophantine.diophantine import diop_quadratic
def A374088(n): return sum(1 for d in diop_quadratic(x*(x+y)+y**2-n) if d[0]>0 and d[1]>0) # _Chai Wah Wu_, Jun 28 2024
