## Python
```Python
from sympy.ntheory import digits, nextprime
def row(n):
    if n == 2: return [2]
    p, r, used = 2, [2], {2}
    while len(used) < n:
        while (ds:=set(digits(p:=nextprime(p), n)[1:])) <= used: pass
        r.append(p)
        used |= ds
    return r
print([an for b in range(2, 13) for an in row(b)]) # _Michael S. Branicky_, Jul 01 2024
```
