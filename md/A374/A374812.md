## Python
```Python
from functools import lru_cache
from math import isqrt
from sympy import mobius
def A374812(n):
    r = 10**n
    @lru_cache(maxsize=None)
    def g(x): return mobius(x)
    def f(x): return r+sum(g(k)*(x//k**2) for k in range(1, isqrt(x)+1))
    m, k = r, f(r)
    while m != k:
        m, k = k, f(k)
    return m
```
