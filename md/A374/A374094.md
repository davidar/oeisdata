## Maple
```Maple
N:= 10^6:
V:= Array(0..N):
for x from 1 to floor(sqrt(N/3)) do
  for y from x+1 do
     v:= x^2 + x*y + y^2;
     if v > N then break fi;
     V[v]:= V[v]+1;
od od:
W:= Array(0..10);
for i from 1 to N while count < 11 do
  v:= V[i];
  if W[v] = 0 then W[v]:= i; count:= count+1 fi
od:
0, seq(W[i],i=1..10); # _Robert Israel_, Jun 28 2024
```
