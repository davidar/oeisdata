## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    seen, an = {0}, 0
    for n in count(1):
        yield an
        an = 0 if n in seen else an + n
        seen.add(an)
print(list(islice(agen(), 71))) # _Michael S. Branicky_, Jul 25 2024
```
