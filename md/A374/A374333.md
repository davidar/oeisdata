## Python
```Python
from itertools import count, islice
from fractions import Fraction
def A374333_gen(): # generator of terms
    a = Fraction(0,1)
    for n in count(1):
        yield a.denominator
        a = (2*a+Fraction(1,n)) % 1
A374333_list = list(islice(A374333_gen(),20)) # _Chai Wah Wu_, Jul 13 2024
```
