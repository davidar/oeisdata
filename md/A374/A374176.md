## Python
```Python
def a(n):
    b, c, m = bin(n)[2:], 0, 0
    for i in range(len(b)-1):
        if b[i] != b[i+1]: c += 1
        else: m = max(m, c); c = 0
    return max(m, c)
print([a(n) for n in range(1, 89)]) # _Michael S. Branicky_, Jul 06 2024
```

from itertools import groupby
def a(n): return max((len(list(g)) for k, g in groupby(bin(n^(n>>1))[3:]) if k=="1"), default=0)
print([a(n) for n in range(1, 89)]) # _Michael S. Branicky_, Jul 06 2024
