def A374996(n,k):
    if k<1: return 0
    cost,c = 0,comp(k)
    m = list(range(1,max(c)+1))
    for i in c:
        j = m.index(i)
        cost += j+1
        jp = 0
        if j >= n:
            jp += j-n
        m.insert(jp,m.pop(j))
    return cost # _John Tyler Rascoe_, Aug 02 2024
