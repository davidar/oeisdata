## Python
```Python
# get n-th term in sequence
def isqrt(n):
  i=0
  while(i*i<=n):
    i+=1
  return i-1
for n in range (200):
  s = isqrt(3*n)//3
  a = s*3
  b = (a*a)//3
  c = n-b
  d = 4*(n*3+a+(c<s)+(c>4*s+1)+(c>5*s+1))+5
  e = isqrt(d)
  f = e-1-( (d-e*e) >> 1 )
  r = ((((8<<e)-(1<<f))//3)-1)//3
  print(r,end =", ") # _André Hallqvist_, Jul 25 2019
```
