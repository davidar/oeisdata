## Python
```Python
def A198376(n):
  ncopy, p, an = n, 1, ""
  for d in range(9, 1, -1):
    while ncopy%d == 0: ncopy//=d; p *= d; an += str(d)
  if p == n and len(an) <= n: return int(an+'1'*(n-len(an)))
  return 0
print([A198376(n) for n in range(1, 25)]) # _Michael S. Branicky_, Jan 21 2021
```
