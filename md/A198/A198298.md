from itertools import combinations, permutations
def agen():
    c = 0
    digits = list("0123456789")
    for f in digits[1:]:
        rest = digits[:]
        rest.remove(f)
        for p in permutations(rest):
            t = (f, ) + p
            s = "".join(t)
            if all(str(int(t[i])*int(t[i+1])) in s for i in range(9)):
                yield int(s)
afull = list(agen())
print(afull) # _Michael S. Branicky_, Oct 03 2024
