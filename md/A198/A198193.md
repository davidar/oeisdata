## MATLAB
```MATLAB
% n is number of terms to be computed, b is the base. The examples all use b=2:
function [V] = revAddition(n,b)
   for i = 0:n
      k = i;
      if (i > 0)
         l = floor(log(i)/log(b));
      end
      s = 0;
      while(k ~= 0)
         if ((i-l) >= 0)
            s = s + mod(k,b)*(i-l);
         end
         l = l - 1;
         k = (k - mod(k,b))/b;
      end
      V(i+1) = s;
   end
end
```
