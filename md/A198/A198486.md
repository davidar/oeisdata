## Python
```Python
def A198486():
    print('Numbers whose consecutive digits differ by 9')
    for i in range(1, 100001):
        b, n = True, i
        if n > 9:
            while n > 9:
                a = abs((n // 10) % 10 - n % 10)
                if a != 9: b = False
                n = n // 10
        if b: print(i, end=', ')
    return
```
