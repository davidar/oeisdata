## Python
```Python
def A141435(terms):
    seq = [1, 2]
    for n in range(3, terms):
        s = 0
        for m in seq:
            if (n - m) > 0:
                s += seq[n - m - 1] #fix for python indexing
        seq.append(s)
    return seq
print(A141435(40)) # _Andres Cruz y Corro A_, Jun 19 2019
```
