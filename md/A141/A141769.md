from itertools import count, islice
def agen(): # generator of terms
    h1, h2, h3, h4 = 1, 2, 3, 4
    while True:
        if h4 - h1 == 3: yield h1
        h1, h2, h3, h4, = h2, h3, h4, next(k for k in count(h4+1) if k%sum(map(int, str(k))) == 0)
print(list(islice(agen(), 40))) # _Michael S. Branicky_, Mar 17 2024
