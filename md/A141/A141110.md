## Python

from sympy.combinatorics import Permutation
def a(n):
    p = list(range(n, 0, -2)) + list(range(1+(n%2), n, 2))
    return Permutation([pi-1 for pi in p]).cycles
print([a(n) for n in range(1, 86)]) # _Michael S. Branicky_, Dec 27 2021
