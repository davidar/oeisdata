## Python
```Python
from math import isqrt
def agen(LIMIT): # generator of terms less than sqrt(LIMIT)
    fibs = set()
    f, g = 1, 2
    while f <= LIMIT:
        fibs.add("".join(sorted(str(f))))
        f, g = g, f+g
        r = s = 1
    r = s = 1
    while s <= LIMIT:
        if "".join(sorted(str(s))) in fibs: yield r
        r += 1
        s = r*r
print(list(agen(10**10))) # _Michael S. Branicky_, Feb 18 2024
```
