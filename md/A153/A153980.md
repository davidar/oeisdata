## Python
```Python
from itertools import combinations
def ok(n):
    s, ds, ss = str(n), set(), 0
    for d in range(len(s)-1, 0, -1):
        for c in combinations(s, d):
            t = int("".join(c))
            if t not in ds:
                ds.add(t)
                ss += t
                if ss > n:
                    return False
    return n and ss == n
print([k for k in range(10**6) if ok(k)]) # _Michael S. Branicky_, Nov 07 2022
```
