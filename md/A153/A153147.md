## Python
```Python
from sympy import mobius, integer_nthroot
def A153147(n):
    def f(x): return int(n+1-sum(mobius(k)*(integer_nthroot(x,k)[0]-1) for k in range(2,x.bit_length())))
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m**3
```
