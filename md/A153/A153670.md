## Python
```Python
from itertools import count, islice
def A153670gen(): # generator of terms
    k10, k11 = 100, 101
    for k in count(1):
        if (k11 % k10)*k < k10:
            yield k
        k10 *= 100
        k11 *= 101
A153670_list = list(islice(A153670gen(),16)) # _Chai Wah Wu_, Dec 23 2021
```
