## Python
```Python
from sympy import isprime
def afind(limit, startk=2):
    k = startk + (startk%2)
    t = int("1" + "51"*(k//2))
    for k in range(startk, limit+1, 2):
        if isprime(t): print(k, end=", ")
        t *= 100
        t += 51
afind(600) # _Michael S. Branicky_, Dec 13 2021
```
