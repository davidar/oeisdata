A153679_list, m, n, k, q = [], 1, 1024, 1000, 0
while m < 10**4:
    r = n % k
    if r > q:
        q = r
        A153679_list.append(m)
    m += 1
    n *= 1024
    k *= 1000
    q *= 1000 # _Chai Wah Wu_, May 16 2020
