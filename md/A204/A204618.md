## Python

from itertools import count, accumulate, islice
def A204618_gen(): # generator of terms
    yield 0
    blist, b = (1,), 1
    for n in count(1):
        blist = list(accumulate(blist, initial=(b:=blist[-1])))
        yield b*n**2
A204618_list = list(islice(A204618_gen(),20)) # _Chai Wah Wu_, Jun 22 2022
