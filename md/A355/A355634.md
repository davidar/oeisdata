## Python
```Python
from sympy import divisors
def row(n):
    s = bin(n)[2:]
    return sorted(d for d in divisors(n, generator=True) if bin(d)[2:] in s)
def table(r): return [i for n in range(1, r+1) for i in row(n)]
print(table(25)) # _Michael S. Branicky_, Jul 11 2022
```
