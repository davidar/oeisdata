## Python
```Python
from sympy import divisor_count
from collections import Counter
from itertools import count, islice
def agen():
    an, c = 1, Counter()
    for n in count(2):
        yield an; c[an] += 1
        an = divisor_count(an) if c[an] == 1 else c[an]*an
print(list(islice(agen(), 75))) # _Michael S. Branicky_, Jul 06 2022
```
