## Python
```Python
from sympy import divisors
def row(n):
    s = str(n)
    return sorted(d for d in divisors(n, generator=True) if str(d) in s)
def table(r): return [i for n in range(1, r+1) for i in row(n)]
print(table(44)) # _Michael S. Branicky_, Jul 11 2022
```
