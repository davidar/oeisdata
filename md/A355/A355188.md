## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen():
    p = 2
    while True:
        t = 2**p+p**2
        if t%3 == 0 and isprime(t//3):
            yield p
        p = nextprime(p)
print(list(islice(agen(), 10))) # _Michael S. Branicky_, Jun 23 2022
```
