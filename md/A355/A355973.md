## Python
```Python
from sympy import divisors
def ok(n):
    if not n&1: return False
    t = bin(n)[2:][::-1]
    return any(t==bin(d)[2:]+bin(n//d)[2:] for d in divisors(n, generator=True))
print([k for k in range(10**6) if ok(k)]) # _Michael S. Branicky_, Apr 13 2024
```
