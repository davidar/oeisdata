from collections import Counter
def aupton(terms):
    num, alst, inventory = 0, [0, 0], Counter([0, 0])
    for n in range(3, 2*terms):
        c = [inventory[num], num]
        num = 0 if c[0] == 0 else num + 1
        alst.extend(c)
        inventory.update(c)
    return alst[:2*terms:2]
print(aupton(82)) # _Michael S. Branicky_, Sep 25 2022
