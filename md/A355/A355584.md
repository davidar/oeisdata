## Python
```Python
from sympy import multiplicity as v
def a(n): return (2**(v(2, n)+1)-1) * (3**(v(3, n)+1)-1) * (5**(v(5, n)+1)-1) // 8
print([a(n) for n in range(1, 77)]) # _Michael S. Branicky_, Jul 08 2022
```
