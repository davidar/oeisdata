## Python
```Python
from sympy import divisors
from itertools import count, islice
def agen():
    s = {tuple()}
    for n in count(1):
        yield len(s)
        s = set(tuple(sorted(t+(d,))) for t in s for d in divisors(n))
print(list(islice(agen(), 16))) # _Michael S. Branicky_, Aug 03 2022
```
