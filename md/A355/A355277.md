from itertools import product
def A355277(n):
    for a in '7531':
        for b in product('97531',repeat=n-1):
            m = int(a+''.join(b))
            if set(str(m*(m+1)>>1)) <= {'1', '3', '5', '7', '9'}:
                return m # _Chai Wah Wu_, Sep 08 2022
