from itertools import count, islice
from sympy import nextprime
def A355329_gen(): # generator of terms
    p = 2
    for m in count(6,6):
        while q:=(p-1)%m:
            p = nextprime(p+m-q-1)
        yield p
        p = nextprime(p)
A355329_list = list(islice(A355329_gen(),30)) # _Chai Wah Wu_, Jun 30 2022
