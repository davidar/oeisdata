from math import gcd
from itertools import islice
def agen():
    a = [1, 2]
    while True: g = gcd(*a); yield g; a = [a[-1], g + sum(a)//g]
print(list(islice(agen(), 85))) # _Michael S. Branicky_, Sep 19 2022
