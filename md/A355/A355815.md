## Python
```Python
from math import gcd
from sympy import nextprime, divisor_sigma
def A355815(n):
    m, p, c = 1, 2, n
    while c:
        c, a = divmod(c,p)
        m *= p**a
        p = nextprime(p)
    return gcd(m,n//gcd(n, divisor_sigma(n)-1)) # _Chai Wah Wu_, Jul 18 2022
```
