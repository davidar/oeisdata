## Python
```Python
from sympy import isprime
from sympy.ntheory.digits import digits
def s(n, b): return sum(digits(n, b)[1:])
def a(n):
    b = 2
    while not isprime(s(n, b)): b += 1
    return s(n, b)
print([a(n) for n in range(2, 89)]) # _Michael S. Branicky_, Jun 16 2022
```
