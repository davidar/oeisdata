```Maple
isgpn := k -> ormap(n -> 0 = 8*k-(n+irem(n,2))*(3*n+2-irem(n,2)), [$0..k]);
issum := k -> ormap(p -> member(k - p, P), P);
```

## Python
```Python
def A355774_list(upto: int) -> list[int]:
    P: list[int] = []
    for k in range(upto + 1):
        if any(
            k == ((n + n % 2) * (3 * n + 2 - n % 2)) >> 3
            for n in range(k + 1)
        ) or not any([(k - p) in P for p in P]):
            P.append(k)
    return P
print(A355774_list(448))
```
