## Python
```Python
from math import prod, gcd
from sympy import nextprime, factorint
def A355820(n):
    k = prod(nextprime(p)**e for p, e in factorint(n).items())
    m, p = 1, 2
    while n:
        n, a = divmod(n,p)
        m *= p**a
        p = nextprime(p)
    return int(gcd(k,m)==1) # _Chai Wah Wu_, Jul 18 2022
```
