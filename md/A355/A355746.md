## Python
```Python
from sympy import factorint
from itertools import count, islice
def agen():
    s = {(1,)}
    for n in count(2):
        yield len(s)
        s = set(tuple(sorted(t+(d,))) for t in s for d in factorint(n))
print(list(islice(agen(), 53))) # _Michael S. Branicky_, Aug 03 2022
```
