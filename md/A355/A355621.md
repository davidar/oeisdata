## Python
```Python
from itertools import count, islice
def agen():
    an, alst = 1, [1]
    for n in count(2):
        yield an
        an = sum(1 for k in alst if k&an)
        alst.append(an)
print(list(islice(agen(), 79))) # _Michael S. Branicky_, Jul 10 2022
```
