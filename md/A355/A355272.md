## Python
```Python
from sympy import nextprime
from itertools import islice
def agen():
    p, q = 2, [3, 1]
    while True:
        if (p + q[int(p%3 == 1)])%4 != 0: yield p
        p, q = q[0], [nextprime(q[0]), p]
print(list(islice(agen(), 51))) # _Michael S. Branicky_, Jun 26 2022
```
