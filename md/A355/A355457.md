## Python
```Python
from itertools import count, islice
def agen():
    an, seen = 1, {1}
    for n in count(2):
        t = an - n
        if t not in seen and t >= 0: an = t
        else: an *= n; yield n
        seen.add(an)
print(list(islice(agen(), 25))) # _Michael S. Branicky_, Jul 02 2022
```
