from collections import Counter
from itertools import count, islice
def agen():
    num, inventory = 0, Counter([0, 0])
    yield 0
    for k in count(2):
        c = [inventory[num], num]
        if c[0] == 0: yield num; num = 0
        else: num = num + 1
        inventory.update(c)
print(list(islice(agen(), 63))) # _Michael S. Branicky_, Sep 25 2022
