## Python
```Python
from sympy import prime
def f(n):
  S = 0
  for x in range(1, n + 1):
    for y in range(x + 1 , n + 1):
      if ((pow(x, y, n) == pow(y, x, n))):
        S += 2
  return S + n
def a(n): return f(prime(n))
```
