from sympy import isprime
A161354_list, i, j = [], 0, 0
while j < 10**15:
    p = int(str(j)[::-1])
    if isprime(p):
        A161354_list.append(j)
    j += 3*i*(i+1)+1
    i += 1
A161354_list = sorted(A161354_list) # _Chai Wah Wu_, Dec 20 2015
