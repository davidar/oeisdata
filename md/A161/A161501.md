def A161501(n):
    s = bin(n)[2:]
    if s == s[::-1]:
        return n
    for i in range(1,len(s)):
        if s[i:] == s[-1:i-1:-1]:
            return int(s+s[i-1::-1],2) # _Chai Wah Wu_, Aug 27 2021
