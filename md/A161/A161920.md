def A161920(n):
    a, b = 1+(m:=n-1).bit_length(), 1
    for i, j in enumerate(bin(m)[:1:-1], 1):
        if int(j):
            a += i-b
            b += 1
    return a # _Chai Wah Wu_, Jul 26 2023
