## Python
```Python
from sympy import nextprime
def a(n):
    if n == 1: return 10123457689
    p = 2
    while not(len(set(str(p**n))) == 10): p = nextprime(p)
    return p
print([a(n) for n in range(1, 83)]) # _Michael S. Branicky_, Jul 04 2021
```
