## Python
```Python
from fractions import Fraction
from sympy import lcm
def A112822(n):
    k, l, h = 1, 1, Fraction(1,1)
    while l != h.denominator*(2*n-1):
        k += 1
        l = lcm(l,k)
        h += Fraction(1,k)
    return k # _Chai Wah Wu_, Mar 06 2021
```
