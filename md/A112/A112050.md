## Python

from sympy import jacobi_symbol as J
def a112046(n):
    i=1
    while True:
        if J(i, 2*n + 1)!=1: return i
        else: i+=1
def a(n): return a112046(n) - 1
print([a(n) for n in range(1, 103)]) # _Indranil Ghosh_, May 24 2017
