## Python
```Python
from sympy import factorint
def aupton(terms):
    alst, k, p = [], 1, 1
    while len(alst) < terms:
        if sum(factorint(k).values()) == 2:
            p *= k
            alst.append(p)
        k += 1
    return alst
print(aupton(18)) # _Michael S. Branicky_, Aug 31 2021
```
