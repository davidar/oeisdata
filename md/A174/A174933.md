from math import isqrt
from sympy import divisor_count, divisors
def A174933(n): return sum(isqrt(d)**(c+2) if (c:=divisor_count(d)) & 1 else d**(c//2+1) for d in divisors(n,generator=True)) # _Chai Wah Wu_, Jun 25 2022
