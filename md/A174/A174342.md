## Python

from sympy import bernoulli, Rational
def B(n):
    return bernoulli(n) if n != 1 else -bernoulli(n)
def a(n):
    return (B(n) + Rational(1, n + 1)).as_numer_denom()[1]
[a(n) for n in range(61)] # _Indranil Ghosh_, Jun 19 2017
