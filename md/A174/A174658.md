## Python
```Python
def a(n):
    s=0
    x=0
    while n>0:
        x=n%3
        n //= 3
        if x==2:
            x=-1
            n+=1
        s+=x
    return s
print([n for n in range(301) if a(n)==0]) # _Indranil Ghosh_, Jun 07 2017
```
