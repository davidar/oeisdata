## Python
```Python
from sympy import isprime
def a(n):
  if n < 2: return list([2, 5])[n]
  n5s, i, pow10, end_digits = int('5'*n), 1, 1, 0
  while True:
    i = 1
    while i < pow10:
      istr = str(i)
      if istr[0] == '5' and len(istr) == end_digits:
        i += 2 if pow10 <= 10 else pow10 // 10
      else:
        t = n5s * pow10 + i
        if isprime(t): return t
        i += 2
    pow10 *= 10; end_digits += 1
print([a(n) for n in range(20)]) # _Michael S. Branicky_, Feb 05 2021, corrected Mar 03 2021
```
