## Python
```Python
from sympy import E
digits_of_e = str(E.n(13*10**4))[2:] # or load data from file
def afind():
    kint = 0
    for k in range(1, len(digits_of_e)):
        kint *= 10
        kint += int(digits_of_e[k-1])
        if kint%k == 0: print(k, end=", ")
afind() # _Michael S. Branicky_, Dec 28 2022
```
