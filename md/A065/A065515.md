N:= 100: # to get a(1) to a(N)
L:= Vector(N):
L[1]:= 1:
p:= 1:
while p < N do
  p:= nextprime(p);
  for k from 1 to floor(log[p](N)) do
    L[p^k] := 1;
  od
od:
ListTools:-PartialSums(convert(L,list)); # _Robert Israel_, May 03 2015
