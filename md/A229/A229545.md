## Python
```Python
def ispal(n):
..r = ''
..for i in str(n):
....r = i + r
..return n == int(r)
def DS(n):
..s = 0
..for i in str(n):
....s += int(i)
..return s
{print(n,end=', ') for n in range(10**3) if ispal(n+DS(n))}
```
