## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen():
    p, q = 2, 3
    while True:
        t = p*q-30
        if isprime(t):
            yield t
        p, q = q, nextprime(q)
print(list(islice(agen(), 40))) # _Michael S. Branicky_, Jun 25 2022
```
