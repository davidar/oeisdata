import math
def isTriangular(a):
    a+=a
    sr = int(math.sqrt(a))
    return (a==sr*(sr+1))
for n in range(1,1000000000L):
    tn = n*(n+1)/2  # = x+y = distance between squares
    if tn&1:
        k = tn>>1
        k*= k       # square below t
        a = int(math.sqrt(k*2))
        t = a*(a+1)/2
        if t <= k:
            a+=1
            t+=a
        ktn = k+tn   # square above t
