c.sort(function(a,b) {return a-b;});
i=-1;

from gmpy2 import is_square
A229134_list = []
for i in range(10**3):
    m, m2, j, k = 2, 4, 4*i**2+1, 2*i**2
    while k >= m2 + m:
        if is_square(j-m2):
            A229134_list.append(i**2)
            break
        m2 += 2*m+1
        m += 1 # _Chai Wah Wu_, Feb 25 2016
