## Python

from itertools import islice
def A229629(): # generator of terms
    n = 1
    while True:
        s, sn = str(n**n), str(n)
        l, ln = len(s), len(sn)
        if (ln-l) % 2 == 0 and s[l//2-ln//2:l//2+(ln+1)//2] == sn: yield n
        n += 1
A229629_list = list(islice(A229629(),5)) # _Chai Wah Wu_, Nov 21 2021
