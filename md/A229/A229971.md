## Python
```Python
from sympy import divisors
def PD(n):
  p = 1
  for i in divisors(n):
    if i != n:
      p *= i
  return p
def pal(n):
  r = ''
  for i in str(n):
    r = i + r
  return r == str(n)
{print(n,end=', ') for n in range(1,10**6) if pal(n) and pal(PD(n)) and (PD(n)-1) and PD(n)-n}
```
