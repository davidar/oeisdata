## Python
```Python
def A229294(n):
    ndict = {}
    n2 = 2*n
    for i in range(n2):
        i3 = pow(i,2,n2)
        for j in range(i+1):
            j3 = pow(j,2,n2)
            m = (i3+j3) % n2
            if m in ndict:
                if i == j:
                    ndict[m] += 1
                else:
                    ndict[m] += 2
            else:
                if i == j:
                    ndict[m] = 1
                else:
                    ndict[m] = 2
    count = 0
    for i in ndict:
        j = (n-i) % n2
        if j in ndict:
            count += ndict[i]*ndict[j]
    return count # _Chai Wah Wu_, Jun 07 2017
```
