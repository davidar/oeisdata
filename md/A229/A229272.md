with(numtheory); P:=proc(q) local a,n,p; for n from 1 to q do
a:=n*add(op(2,p)/op(1,p),p=ifactors(n)[2]);
if isprime(a+n) and isprime(a-n) then print(n); fi;
od; end: P(10^5);
