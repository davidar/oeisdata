## Python
```Python
def rev(n):
    return int(''.join(reversed(str(n))))
def DP(n):
    p = 1
    for i in str(n):
        p *= int(i)
    return p
{print(n, end=', ') for n in range(10**3) if n-DP(n)==rev(n-DP(n))}
```
