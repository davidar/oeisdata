import sympy
from sympy import isprime
def DP(n):
..p = 1
..for i in str(n):
....p *= int(i)
..return p
{print(n,end=', ') for n in range(10**4) if DP(n) and isprime(n) and isprime(n+DP(n)) and isprime(n-DP(n))}
