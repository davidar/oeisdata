## Python
```Python
from sympy.utilities.iterables import connected_components
def bin_i(n): #binary indices
    return([(i+1) for i, x in enumerate(bin(n)[2:][::-1]) if x =='1'])
def A326753(n):
    E,a = [],[bin_i(k) for k in bin_i(n)]
    m = len(a)
    for i in range(m):
        for j in a[i]:
            for k in range(m):
                if j in a[k]:
                    E.append((i,k))
    return(len(connected_components((list(range(m)),E)))) # _John Tyler Rascoe_, Jul 16 2024
```
