isA004431 := proc(n)
    local a,b ;
    for a from 2 do
        if a^2>= n then
            return false;
        end if;
        b := n -a^2 ;
        if b < 1 then
            return false ;
        end if;
        if issqr(b) then
            if ( sqrt(b) <> a ) then
                return true;
            end if;
        end if;
    end do:
    return false;
end proc:

## Python

def aupto(limit):
  s = [i*i for i in range(1, int(limit**.5)+2) if i*i < limit]
  s2 = set(a+b for i, a in enumerate(s) for b in s[i+1:] if a+b <= limit)
  return sorted(s2)
