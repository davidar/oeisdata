## Python
```Python
from math import gcd
from itertools import count, islice
def agen(): # generator of terms
    an, aset = 1, {1}
    for n in count(2):
        yield an
        k = 1
        while k in aset or any(gcd(t, k) != 1 for t in [n, an]): k+= 1
        aset.add(k)
        an = k
print(list(islice(agen(), 69))) # _Michael S. Branicky_, Apr 13 2022
```
