from itertools import count, islice
def agen(): # generator of terms
    an = 1
    for n in count(1):
        yield an
        if n&1: an += (n if n&3 == 1 else -n)
        else: an *= n
print(list(islice(agen(), 30))) # _Michael S. Branicky_, Aug 09 2022
