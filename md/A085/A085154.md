## Python
```Python
from sympy import factorint
def aupton(terms):
    alst, wn, Wn, k = [1], 0, 0, 1
    while len(alst) < terms:
        while True:
            k += 1
            pf = factorint(k, multiple=True)
            wk, Wk = len(pf), len(set(pf))
            if wk > wn or Wk > Wn:
                break
        alst.append(k)
        wn, Wn = wk, Wk
    return alst
print(aupton(58)) # _Michael S. Branicky_, Oct 07 2021
```
