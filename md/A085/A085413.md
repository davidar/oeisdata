## Python
```Python
from itertools import count, islice
from sympy import isprime, primerange
def agen(): # generator of terms
    for d in count(1):
        for f in [1, 3, 7, 9]:
            for p in primerange(f*10**d, (f+1)*10**d):
                if isprime(10*p+f):
                    yield p
print(list(islice(agen(), 50))) # _Michael S. Branicky_, Mar 24 2023
```
