from fractions import Fraction
def A085068(n):
    c, x, m = 1, Fraction(4*n,3), Fraction(4,3)
    while x.denominator > 1:
        x = m*x.__ceil__()
        c += 1
    return c # _Chai Wah Wu_, Mar 01 2021
