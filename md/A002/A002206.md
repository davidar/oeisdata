## Sage
```Sage
from functools import cache
@cache
def h(n):
    return (-sum((-1)**k * h(n - k) / (k + 1) for k in range(1, n + 1))
            + (-1)**n * n / (2*(n + 1)*(n + 2)))
def a(n): return h(n).numer() if n > 0 else 1
print([a(n) for n in range(-1, 20)]). # _Peter Luschny_, Dec 12 2023
```
