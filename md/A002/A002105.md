# Algorithm of L. Seidel (1877)
# n -> [a(1), ..., a(n)] for n >= 1.
def A002105_list(n) :
    D = [0]*(n+2); D[1] = 1
    R = []; z = 1/2; b = True
    for i in(0..2*n-1) :
        h = i//2 + 1
        if b :
            for k in range(h-1, 0, -1) : D[k] += D[k+1]
            z *= 2
        else :
            for k in range(1, h+1, 1) :  D[k] += D[k-1]
        b = not b
        if b : R.append(D[h]*z/h)
    return R
