A002324 := proc(n)
    local a,pe,p,e;
    a :=1 ;
    for pe in ifactors(n)[2] do
        p := op(1,pe) ;
        e := op(2,pe) ;
        if p = 3 then
            ;
        elif modp(p,3) = 1 then
            a := a*(e+1) ;
        else
            a := a*(1+(-1)^e)/2 ;
        end if;
    end do:
    a ;
end proc:
