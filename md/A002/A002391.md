## Python
```Python
# Use some guard digits when computing.
# BBP formula P(1, 4, 2, (1, 0)).
from decimal import Decimal as dec, getcontext
def BBPlog3(n: int) -> dec:
    getcontext().prec = n
    s = dec(0); f = dec(1); g = dec(4)
    for k in range(2 * n):
        s += f / dec(2 * k + 1)
        f /= g
    return s
print(BBPlog3(200))  # _Peter Luschny_, Nov 03 2023
```
