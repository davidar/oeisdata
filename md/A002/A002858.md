ulam :: Int -> Integer -> [Integer] -> [Integer]
ulam n u us = u' : ulam (n + 1) u' us where
   u' = f 0 (u+1) us'
   f 2 z _                         = f 0 (z + 1) us'
   f e z (v:vs) | z - v <= v       = if e == 1 then z else f 0 (z + 1) us'
                | z - v `elem` us' = f (e + 1) z vs
                | otherwise        = f e z vs
   us' = take n us

def isUlam(n, h, u, r):
    if h == 2: return False
    hu = u[0]; hr = r[0]
    if hr <= hu: return h == 1
    if hr + hu > n: r = r[1:]
    elif hr + hu < n: u = u[1:]
    else: h += 1; r = r[1:]; u = u[1:]
    return isUlam(n, h, u, r)
def UlamList(length):
    u = [1, 2]; r = [2, 1]; n = 2
    while len(u) < length:
        n += 1
        if isUlam(n, 0, u[:], r[:]):
            u.append(n); r.insert(0, n)
    return u

function isUlam(u, n, h, i, r)
    h == 2 && return false
    ur = u[r]; ui = u[i]
    ur <= ui && return h == 1
    if ur + ui > n
        r -= 1
    elseif ur + ui < n
        i += 1
    else
        h += 1; i += 1; r -= 1
    end
    isUlam(u, n, h, i, r)
end
function UlamList(len)
    u = Array{Int, 1}(undef, len)
    u[1] = 1; u[2] = 2; i = 2; n = 2
    while i < len
        n += 1
        if isUlam(u, n, 0, 1, i)
            i += 1
            u[i] = n
        end
    end
    return u
end
