from itertools import islice
def A366585_gen(): # generator of terms
    a = 1
    while True:
        c = a+int(str(a)[0])
        yield int(''.join(str(i) for i in range(a,c)))
        a = c
A366585_list = list(islice(A366585_gen(),20)) # _Chai Wah Wu_, Nov 04 2023
