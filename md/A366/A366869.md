from itertools import count, islice
from sympy import nextprime
def A366869_gen(): # generator of terms
    a, aset, p = 1, {0,1}, 2
    for i in count(3):
        for b in count(a,p):
            if b not in aset:
                aset.add(b)
                c = b%(p:=nextprime(p))
                if c > a:
                    yield c
                a = c
                break
A366869_list = list(islice(A366869_gen(), 20))
