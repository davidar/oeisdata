from itertools import permutations
def afull(): return [int(s) for s in ("".join(c) for c in permutations("0123456789") if c[0]!="0") if all(str(sum(map(int, s[i:i+2]))) in s for i in range(len(s)-1))] # _Michael S. Branicky_, Oct 15 2023
