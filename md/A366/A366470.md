## Python
```Python
from itertools import count, islice
from sympy import nextprime
def A366470_gen(): # generator of terms
    a, aset, p = 1, {0,1}, 2
    while True:
        yield a
        for b in count(a,p):
            if b not in aset:
                aset.add(b)
                a = b%(p:=nextprime(p))
                break
A366470_list = list(islice(A366470_gen(),30)) # _Chai Wah Wu_, Oct 22 2023
```
