from itertools import count
from sympy import nextprime
def A366913(n):
    a, aset, p, c = 1, {0,1}, 2, 0
    for i in count(1):
        if c == n:
            return i
        k, b = divmod(a,p)
        for j in count(-k):
            if b not in aset:
                aset.add(b)
                a, p = b, nextprime(p)
                c += j
                break
            b += p # _Chai Wah Wu_, Oct 27 2023
