## Python
```Python
from itertools import combinations
from sympy.utilities.iterables import partitions
def A366130(n):
    a = tuple(set(p.keys()) for p in partitions(n+1,k=n) if max(p.values(),default=0)==1)
    return sum(1 for k in range(2,n+1) for w in (set(d) for d in combinations(range(1,n+1),k)) if any(s<=w for s in a)) # _Chai Wah Wu_, Nov 24 2023
```
