from itertools import count, islice
from sympy import nextprime
def A366868_gen(): # generator of terms
    a, aset, p = 1, {0,1}, 2
    for i in count(3):
        for b in count(a,p):
            if b not in aset:
                aset.add(b)
                c = b%(p:=nextprime(p))
                if c > a:
                    yield a
                a = c
                break
A366868_list = list(islice(A366868_gen(),20))
