from itertools import count, islice
from sympy import nextprime
def A366911_gen(): # generator of terms
    a, aset, p = 1, {0,1}, 2
    while True:
        k, b = divmod(a,p)
        for i in count(-k):
            if b not in aset:
                aset.add(b)
                a, p = b, nextprime(p)
                yield i
                break
            b += p
A366911_list = list(islice(A366911_gen(),30)) # _Chai Wah Wu_, Oct 27 2023
