## Python
```Python
from sympy import prime
def A366275(n):
    if n:
        k, c, m = int(bin(n>>(r:=(~n & n-1).bit_length()))[:1:-1],2)<<r, 0, 1
        while k:
            c += 1
            m *= prime(c)**(s:=(~k&k-1).bit_length())
            k >>= s+1
        return m*prime(c)
    return 1 # _Chai Wah Wu_, Oct 08 2023
```
