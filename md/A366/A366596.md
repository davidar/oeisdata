def A366596(n):
    digitlen, digit = (n+12)//14*6, (n+12)%14-4
    if digit < 1: digitlen += digit - 1; digit = 7
    return 10**digitlen // 9 * digit # _Karl-Heinz Hofmann_, Dec 04 2023
