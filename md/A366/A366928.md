## Python

from itertools import count
from sympy import perfect_power
def A366928(n): return next(m for m in count(0) if next(k for k in count(0) if perfect_power(m+k) or perfect_power(m-k) or m-k==1 or m+k==1) == n) # _Chai Wah Wu_, Nov 12 2023
