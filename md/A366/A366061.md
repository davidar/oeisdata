## Python
```Python
from itertools import islice
import hashlib
def g():
  c, vmin, m = 1, b"\xFF" * 32, b""
  while True:
    if (m:= hashlib.sha256(m).digest()) < vmin:
      vmin = m
      yield c
    c += 1
print(list(islice(g(), 16)))
```
