from itertools import count
from sympy.ntheory.primetest import is_square
def A366091(n):
    c = 0
    for k in count(0):
        if (a:=3*k**2)>n:
            break
        for j in count(0):
            if (b:=a+(j**2<<1))>n:
                break
            if is_square(n-b):
                c += 1
    return c # _Chai Wah Wu_, Sep 29 2023
