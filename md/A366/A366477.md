from itertools import count
from sympy import nextprime
def A366477(n):
    a, aset, p = 1, {0,1}, 1
    for i in count(2):
        p = nextprime(p)
        b = a%p
        for j in count(0):
            if b not in aset:
                aset.add(b)
                a = b
                break
            b += p
        if j>=n:
            return i # _Chai Wah Wu_, Oct 27 2023
