## Python
```Python
from itertools import count, takewhile
from sympy import divisors
def A366165(n):
    a, l1, l2 = 10**((n<<1)-1), 10**(n-1), 10**n
    for k in count(1):
        b = a-k
        if any(l1<=d<l2 and d*l2>b for d in takewhile(lambda m:m*m<=b, divisors(b))):
            return k # _Chai Wah Wu_, Oct 05 2023
```
