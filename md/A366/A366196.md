## Mathematica
```Mathematica
intPowCountPos[n_] := Module[{m, F, i, t},
  m = n (GCD @@ FactorInteger[n][[All, 2]]);
  t = 0;
  While[Mod[m, 2] == 0,
   t++;
   m = m/2];
  t = t + 1;
  F = FactorInteger[m][[All, 2]];
  If[m > 1,
   For[i = 1, i <= Length[F], i++,
     t = t (F[[i]] + 1)];
   ];
  Return[t]]
```
