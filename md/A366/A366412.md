## Python
```Python
import math
LEN = 79
carr=[]
for base in range(2, LEN):
    k = int(2 * math.log(base - 1) / math.log(2) + 2) + 1
    k = max(k, 5)
    I = (base ** k - 1) // (base - 1) - 1
    count = 0
    for b in range(2, base):
        for c_k in range(1, b):
            c = b * I + c_k
            a = b * c // (b * base - (base - 1) * c_k)
            if a < base ** (k - 1):
                continue
            if (a * base + b) * c == a * (b * base**k + c):
                count += 1
    carr.append(count)
print(carr)
```
