
## Python
```Python
def agen(): # generator of terms
    yield 0
    magic, biggermagic, digits = list(range(1, 10)), [], 2
    while len(magic) > 0:
        yield from magic
        for i in magic:
            for d in range(10):
                t = 10*i + d
                if t%digits == 0:
                    biggermagic.append(t)
        magic, biggermagic, digits = biggermagic, [], digits+1
print([an for an in agen()][:70]) # _Michael S. Branicky_, Feb 07 2022
```
