## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    an, p = 1, 2
    while True:
        yield an
        an = next(k for k in count(an+1) if isprime(p*k+1))
        p *= an
print(list(islice(agen(), 58))) # _Michael S. Branicky_, Jan 13 2023
```
