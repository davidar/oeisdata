## Python
```Python
from sympy import isprime
def composite(n): return n >= 4 and not isprime(n)
def aupton(nn):
  alst = [4]
  for n in range(1, nn+1):
    an = max(alst[-1] + 1, n + 4)
    while not (composite(an) and composite(an-n)): an += 1
    alst.append(an)
  return alst
print(aupton(68)) # _Michael S. Branicky_, Apr 09 2021
```
