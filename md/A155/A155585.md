def A155585_list(n): # Akiyama-Tanigawa algorithm
    A = [0]*(n+1); R = []
    for m in range(n+1) :
        d = divmod(m+3, 4)
        A[m] = 0 if d[1] == 0 else (-1)^d[0]/2^(m//2)
        for j in range(m, 0, -1) :
            A[j - 1] = j * (A[j - 1] - A[j])
        R.append(A[0])
    return R
