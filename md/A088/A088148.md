## Python
```Python
from itertools import count
from sympy import isprime
def A088148(n):
    if n == 1: return 2
    for p in count((1<<n)-1):
        if p.bit_count() >= n:
            m = p.bit_length()
            l = 1<<m-1
            k, cset, q = l-1, set(), p
            for _ in range(m):
                if p not in cset and isprime(p):
                    cset.add(p)
                p = bool(p&l)+((p&k)<<1)
            if len(cset) == n:
                return q # _Chai Wah Wu_, Jan 23 2023
```
