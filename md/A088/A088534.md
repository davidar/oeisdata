## Julia

```Julia
function A088534(n)
    n % 3 == 2 && return 0
    M = Int(round(2*sqrt(n/3)))
    count = 0
    for y in 0:M, x in 0:y
        n == x^2 + y^2 + x*y && (count += 1)
    end
    return count
end
```

## Python
```Python
def A088534(n):
    c = 0
    for y in range(n+1):
        if y**2 > n:
            break
        for x in range(y+1):
            z = x*(x+y)+y**2
            if z > n:
                break
            elif z == n:
                c += 1
    return c # _Chai Wah Wu_, May 16 2022
```
