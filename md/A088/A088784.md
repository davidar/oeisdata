## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def agen(): # generator of terms
    p, pstr = 2, "2"
    while True:
        q = nextprime(p)
        qstr = str(q)
        t = int(qstr + pstr)
        if isprime(t):
            yield t
        p, pstr = q, qstr
print(list(islice(agen(), 32))) # _Michael S. Branicky_, Jan 05 2022
```
