## Python
```Python
from sympy import factorint, isprime
def sopfr(n):
    f=factorint(n)
    return sum([i*f[i] for i in f])
l=[]
record=-1
for n in range(1, 501):
    if not isprime(n):
        x=sopfr(n)
        if x>record:
            record=x
            l.append(n)
print(l) # _Indranil Ghosh_, Jun 29 2017
```
