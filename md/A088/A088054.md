## Python
```Python
from itertools import count, islice
from sympy import isprime
def A088054_gen(): # generator of terms
    f = 1
    for k in count(1):
        f *= k
        if isprime(f-1):
            yield f-1
        if isprime(f+1):
            yield f+1
A088054_list = list(islice(A088054_gen(),10)) # _Chai Wah Wu_, Feb 18 2022
```
