A[1]:= 1: A[2]:= 1: S:= {1}:
for n from 3 to 100 do
  Sp:= select(type,map(s -> s/A[n-1],S),integer);
  if nops(Sp) = Sp[-1] then A[n]:= Sp[-1]+1
  else A[n]:= min({$1..Sp[-1]} minus Sp)
  fi;
  S:= S union {A[n-1]*A[n]};
od:

from itertools import islice
def A088177(): # generator of terms
    yield 1
    yield 1
    p, a = {1}, 1
    while True:
        n = 1
        while n*a in p:
            n += 1
        p.add(n*a)
        a = n
        yield n
