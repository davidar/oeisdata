## Python
```Python
def ispal(n):
    return n==int(str(n)[::-1])
def A088001(n):
    s=0
    for i in range(1, n+1):
        if n%i==0 and not ispal(i):
             s+=i
    return s # _Indranil Ghosh_, Feb 10 2017
```
