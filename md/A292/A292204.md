from itertools import count, islice
from sympy import isprime
def A292204_gen(): # generator of terms
    for n in count(0):
        k, m = n, n>>1
        while m > 0:
            k ^= m
            m >>= 1
        if isprime(k):
            yield k
A292204_list = list(islice(A292204_gen(),30)) # _Chai Wah Wu_, Jun 29 2022
