## Python
```Python
def nextAntidiagonal(d):
  if(d[0]==0): return [d[1]+1,0]
  return [d[0]-1,d[1]+1]
def fibonacciM(m,n):
  f0,f1=0,1
  if(n<0):
    for _ in range(-n): f1,f0=f0,f1-f0
  else:
    for _ in range(n): f0,f1=f1,f0+f1
  return f1*m+f0
d=[0,0]
for i in range(10001):
  print(str(i)+" "+str(fibonacciM(d[0],d[1])))
  d=nextAntidiagonal(d)
```
