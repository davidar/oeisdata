def A292729(n):
....s_in = set([(1,)*n])
....count=-1
....while len(s_in) > 0:
........s_out = set()
........for s in s_in:
............last = -1 ; idx = 0
............while (idx+1) < len(s):
................h = s[idx]
................if h!=last and s[idx+1]==h:
....................for q in range(1,h+1):
........................lst = list(s[:idx]) + list(s[idx+2:])
........................lst += [2*h] if h==q else [ h-q, h+q]
........................t = tuple(sorted(lst))
........................if not t in s_out:
............................s_out.add(t)
................last = s[idx] ; idx += 1
........count += 1
........s_in = s_out
....return count
