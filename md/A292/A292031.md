def smallestSeq(n):
  if(n<0): return []
  if(n==0): return [0,0]
  j,r0,r1=0,0,1
  while(r1<=n): r0,r1=r1,r0+r1 ; j+=1
  while(r1>1):
    if(n%r1==r0): return [n//r1,j]
    r1,r0=r0,r1-r0
    j-=1
  return [n-1, j]
for i in range(10001):
  print(str(i)+" "+str(smallestSeq(i)[0]))
