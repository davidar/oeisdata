## Python
```Python
from sympy import isprime
def aupto(limit):
  alst, is2 = [], False
  for k in range(1, limit+1):
    is1, is2 = is2, isprime(k**2 + (k+1)**2)
    if is1 and is2: alst.append(k)
  return alst
print(aupto(2060)) # _Michael S. Branicky_, Apr 25 2021
```
