## Python
```Python
from sympy import divisors
def A109883(n):
    if n == 1: return 0
    s = n
    for d in divisors(n)[:-1]:
        if s < d: break
        s -= d
    return s
print([A109883(n) for n in range(1, 80)]) # _Michael S. Branicky_, Mar 31 2024
```
