## Python
```Python
from sympy import prime, primerange
def aupton(nn):
    alst, prevp, prev_prod = [], 2, 6
    for p in primerange(3, prime(nn+2)+1):
        cur_prod = prevp * p
        alst.append(cur_prod - prev_prod)
        prevp = p
        prev_prod = cur_prod
    return alst[1:]
print(aupton(48)) # _Michael S. Branicky_, Sep 20 2021
```
