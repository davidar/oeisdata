## Python
```Python
from math import isqrt
def a(n):
    sn = str(n)
    k = 1 if n == 0 else isqrt(int(sn*n))
    while True:
        if str(k*k).count(sn) == n:
            return k*k
        k += 1
print([a(n) for n in range(10)]) # _Michael S. Branicky_, Feb 02 2022
```
