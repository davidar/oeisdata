from sympy import prime, isprime
def A263879(n):
    if n <= 2: return 7-n
    p, c = prime(n), 1
    while isprime(p:=(p<<1)+(-1 if p%3==1 else 1)):
        c += 1
    return c # _Chai Wah Wu_, Jul 07 2023
