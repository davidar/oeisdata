## Python
```Python
def getNumber(n):
    y = int(math.ceil(math.log(n, 2)))
    smallIncrement = 0.5 ** y
    marker = 0
    nMarker = 0
    toReturn = 0
    while marker < 1:
        newMarker = marker + smallIncrement
        if (nMarker + 1) / n < newMarker:
            nMarker += 1
        if nMarker / n <= marker and (nMarker + 1) / n >= newMarker:
            toReturn += 1
        marker = newMarker
    return toReturn
```
