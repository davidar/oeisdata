number_of_terms=38
def Cn(X):
    l=len(X)
    cn=1
    for i in range(1,int(l/2)+1):
        j=i
        while(X[l-j-1]==X[l-j-1+i]):
            j=j+1
            if j>=l:
                break
        candidate=int(j/i)
        if candidate>cn:
            cn=candidate
    return cn
