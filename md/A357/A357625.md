## Python
```Python
from itertools import count, islice
def comp(n): #row n of A066099 after Franklin T. Adams-Watters
    v,k = [],0
    while n > 0:
        k += 1
        if n%2 == 1:
            v.append(k)
            k = 0
        n = n//2
    return(v[::-1])
def a_gen():
    for n in count(0):
        c = comp(n)
        x = sum(c[i]*(-1)**(i//2) for i in range(len(c)))
        if x == 0:
            yield(n)
A357625_list = list(islice(a_gen(), 60)) # _John Tyler Rascoe_, Jun 01 2024
```
