## Maple
```Maple
f:= proc(n) local d,L,i,t;
  t:= 0;
  for d in numtheory:-divisors(n) do
    L:= convert(d,base,10);
    if `and`(seq(L[i]>L[i+1],i=1..nops(L)-1)) then t:= t+1 fi
  od;
  t
end proc:
map(f, [$1..100]); # _Robert Israel_, Sep 16 2022
```

## Python
```Python
from sympy import divisors
def c(n): s = str(n); return s == "".join(sorted(set(s)))
def a(n): return sum(1 for d in divisors(n, generator=True) if c(d))
print([a(n) for n in range(1, 101)]) # _Michael S. Branicky_, Sep 16 2022
```
