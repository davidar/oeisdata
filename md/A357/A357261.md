## Python
```Python
def A357261_list(maxn):
    """Returns a list of the first maxn terms"""
    A = []
    g = 3
    c = 0
    for n in range(1,maxn+1):
        if (n + c)%g == 0:
            A.append(g)
            g += 1
            c = 0
        else:
            A.append((n + c)%g)
            c = A[-1]
    return A
```
