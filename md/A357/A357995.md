## Python
```Python
def A357995(n):
    a, b = set([0]), set(range(1,n**2))
    for m in [n+k**2 for k in range(n+1)]:
        d=m
        while d < n**2:
            c2 = set([x for x in b if x-d in a])
            a |= c2 ; b -= c2 ; d*=2
    return max(b) # _Bert Dobbelaere_, Oct 30 2022
```
