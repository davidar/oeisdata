## Python
```Python
from sympy import nextprime
from itertools import islice
def agen():
    p, q = 2, 3
    while True:
        if pow(p, q-1, q*q) == 1 or pow(q, p-1, p*p) == 1: yield q
        p, q = q, nextprime(q)
print(list(islice(agen(), 5))) # _Michael S. Branicky_, Sep 30 2022
```
