## Python

def ok(n):
    s = str(n)
    if len(s) < 2 or len(s) > 9: return False
    return len(s) == int(s[0]) and len(set(s)) == int(s[1])
print([k for k in range(4394) if ok(k)]) # _Michael S. Branicky_, Oct 17 2022
