from sympy import sieve
A357744, A357745, aupto = [], [], 82
for n in range (1, sieve[aupto]**2):
    A357745.append(((n+3)**2 >> 4) + 1 if n % 8 != 1 else (n+3)**2 >> 4)
for p in sieve[1:aupto + 1]:
    k = 1
    while (p*k) not in A357745: k += 1
    A357744.append(k)
print(A357744)
