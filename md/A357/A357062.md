from sympy.utilities.iterables import combinations_with_replacement
from math import prod
def A357062(n): return sum(max(1,3*(len(set(d))-1)) for d in combinations_with_replacement(range(1,n+1),3) if prod(d)+sum(d) == n) # _Chai Wah Wu_, Oct 21 2022
