```Python
from gmpy2 import *
def fermat(n):
    a, rem = isqrt_rem(n)
    b2 = -rem
    c0 = (a << 1) + 1
    c = c0
    while not is_square(b2):
        b2 += c
        c += 2
    return (c-c0) >> 1
def A357928(n):
  if is_square(n):
      return 0
  elif ((n-2) % 4) != 0:
      return fermat(n)
  else:
      return -1
```
