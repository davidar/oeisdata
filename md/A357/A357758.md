## Python
```Python
def ok(n):
    b = bin(n)[2:]
    if "00" in b and "11" in b: return False
    for l in range(3, len(b)):
        h = set(b[i:i+l].count("1") for i in range(len(b)-l+1))
        if max(h) - min(h) > 1: return False
    return True
print([k for k in range(112) if ok(k)]) # _Michael S. Branicky_, Oct 12 2022
```
