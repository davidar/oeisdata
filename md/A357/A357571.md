D(n,a,b)=prod(j=0,a+b-1,n-j)*sum(i=0,b, binomial(b,i)*C(i)*H(n,b-i,a,b))*P(n-a-b)

C(n)=if(n<2, n==0, (n-1)*(C(n-1)+15*C(n-2)))

from math import factorial, comb
