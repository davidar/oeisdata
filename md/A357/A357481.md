## Python
```Python
from sympy import divisors
from sympy.ntheory import digits
def ok(n, b):
    target, s = digits(n, b)[1:], []
    if target[0] != 1: return False
    for d in divisors(n):
        s += digits(d, b)[1:]
        if len(s) >= len(target): return s == target
def a(n):
    if n == 1: return 2
    for b in range(2, n):
        if ok(n, b): return b
    return -1
print([a(n) for n in range(1, 76)]) # _Michael S. Branicky_, Oct 05 2022
```
