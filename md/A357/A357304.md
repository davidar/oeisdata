from itertools import count, islice
def A357304_gen(): # generator of terms
    c = -1
    for n in count(0):
        if (m := (n**2).bit_count() if sys.version_info >= (3,10) else bin(n**2).count('1'))>c:
            yield (c:=m)
A357304_list = list(islice(A357304_gen(),20)) # _Chai Wah Wu_, Oct 01 2022
