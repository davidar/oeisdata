def flag(p, n):
    return all(p.count(d) == n for d in "0123456789")
def a(n):
    num=0
    for i in range(10**10-1, 3*int(10**(10-1/n)/3), -3):
        if flag(str(i**n), n):
            num+=1
    return(num)
