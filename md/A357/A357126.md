## Python
```Python
from sympy import factorint
to_s_exp = lambda n: tuple(i[1] for i in sorted(factorint(n).items()))
terms = []
for i in range(2, 100):
    k = i+1;t = to_s_exp(i)
    while t != to_s_exp(k):k+=1
    terms.append(k)
print(terms)
```
