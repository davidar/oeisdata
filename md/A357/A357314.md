pal = lambda s: s == s[::-1]
up_to = 10
terms = [1, ]
for i in range(up_to-1):
    c, r = ''.join(map(str, terms)), 0
    for j in range(len(str(terms[-1])), len(c)+1):
        found, p  = False, int(c[:j][::-1])
        if p > terms[-1] and pal(c + c[:j][::-1]):
            r+=1
            if r == 2:
                terms.append(p);found = True;break
    if found: continue
    j = 0
    while 1:
        j+=1
        r+=1
        if r == 2:
            terms.append(int(str(j) + c[::-1]))
            break
print(terms)
