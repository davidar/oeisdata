def A357048_upto(N):
    a=0; b=1; used={1}; unused=2; S=set()
    sa=set(str(a))
    for k in range(N):
        sb=set(str(b))
        if sa & sb:
            a,sa,b = b,sb,unused
        else:
            a,sa,b = b,sb,a+b
            S |= {b}
        used |= {b}
        while unused in used: used -= {unused-1}; unused += 1
    return sorted(x for x in S if x < 2*unused+2)
