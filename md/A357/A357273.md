```Python
from itertools import count, islice
from sympy import divisors
def A357273_gen(): # generator of terms
    yield 1
    for n in count(1):
        r = str(n)
        if n&1 or r.startswith('2'):
            m = 10**(c:=len(r))+n
            s, sm = '', str(m)
            for d in divisors(m):
                s += str(d)
                if len(s) >= c+1:
                    break
            if s.startswith(sm):
                yield m
A357273_list = list(islice(A357273_gen(),10)) # _Chai Wah Wu_, Sep 23 2022
```
