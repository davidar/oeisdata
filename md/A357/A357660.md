## Python
```Python
from math import isqrt
def A357660(n):
    c = 0
    for k in range(isqrt((1<<n)-1)+1,isqrt((1<<n+1)-1)+1):
        if (d:=(k**2).bit_count()) > c:
            c, m = d, k
        elif d == c:
            m = max(m,k)
    return m # _Chai Wah Wu_, Oct 14 2022
```
