## Python
```Python
def ok(n):
    s = str(n); sk = int(s[0])
    for k in range(len(s)-1):
        if n%sk != 0: return False
        sk += int(s[k+1])
    return n%sk == 0
print([k for k in range(1, 401) if ok(k)]) # _Michael S. Branicky_, Oct 12 2022
```
