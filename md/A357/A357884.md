function a = A357884( max_n )
    a = 0;
    for n = 2:max_n
        sa = num2str(a(n-1));
        s = intersect(sa,num2str(n-2));
        if isempty(s)
            a(n) = a(n-1)+(n-2);
        else
            sd = ['0' regexprep(sa,['[' s ']'],'')];
            a(n) = str2double(sd);
        end
    end
    a = a(2:end);
end % _Thomas Scheuerle_, Oct 18 2022
