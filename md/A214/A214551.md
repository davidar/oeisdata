## Perl

use bignum;
my @seq = (1, 1, 1);
print "1 1\n2 1\n3 1\n";
for ( my $i = 3; $i < 400; $i++ )
{
    my $next = ( $seq[$i-1] + $seq[$i-3] ) /
        gcd( $seq[$i-1], $seq[$i-3] );
    my $ind = $i+1;
    print "$ind $next\n";
    push( @seq, $next );
}
sub gcd {
    my ($x, $y) = @_;
    ($x, $y) = ($y, $x % $y) while $y;
    return $x;
}

## Python

from math import gcd
def aupton(nn):
    alst = [1, 1, 1]
    for n in range(3, nn+1):
        alst.append((alst[n-1] + alst[n-3])//gcd(alst[n-1], alst[n-3]))
    return alst
print(aupton(64)) # _Michael S. Branicky_, Mar 28 2022
