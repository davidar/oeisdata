f:= proc(a,b)
  local s;
  s:= map(t -> rhs(op(t)), [msolve(x^2=1, 8*a)]);
  min(select(`>`, map(t -> (t^2-1)/(8*a), s), b))
end proc:
A[0]:= 1: A[1]:= 1:
