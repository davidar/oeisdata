## Python
```Python
def aupto(n):
  alst, remaining = [0, 1], [0, 1]
  for i in range(2, n+1):
    an = remaining.pop(0) + remaining[0]
    alst.append(an)
    remaining.extend(list(map(int, str(an))))
  return alst    # use alst[n] for a(n)
print(aupto(89)) # _Michael S. Branicky_, Dec 14 2020
```
