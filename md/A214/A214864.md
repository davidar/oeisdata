## Magma
```Magma
XOR := func<a, b | Seqint([ (adigs[i] + bdigs[i]) mod 2 : i in [1..n]], 2)
    where adigs := Intseq(a, 2, n)
    where bdigs := Intseq(b, 2, n)
    where n := 1 + Ilog2(Max([a, b, 1]))>;
m:=10;
for n in [1 .. 250] do
if (XOR(n, m) eq n-m) then n; end if;
end for;
```
