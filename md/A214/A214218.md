## Python
```Python
from itertools import count, islice
from sympy.utilities.iterables import multiset_permutations as mp
def agen():
    for d in count(2, 2):
        for s in mp("1"*(d//2) + "2"*(d//2), d):
            yield int("".join(s))
print(list(islice(agen(), 33))) # _Michael S. Branicky_, Dec 21 2021
```
