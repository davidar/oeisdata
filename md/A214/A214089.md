A214089 := proc(n)
     local m,k,p;
   m:= 2*mul(ithprime(j),j=1..n);
   for k from 1 do
     p:= sqrt(m*k+1);
     if type(p,integer) and isprime(p) then return(p)
     end if
   end do
end proc;
