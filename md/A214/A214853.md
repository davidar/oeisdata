## Python
```Python
def count0(x):
    c = 0
    while x:
        c+= 1 - (x&1)
        if c>1:
            return 2
        x>>=1
    return c
prpr, prev = 0,1
TOP = 1<<12
print(0, end=',')
for i in range(1,TOP):
    if count0(prpr)==1:
        print(prpr, end=',')
    if (i&4095)==0:
        print('.', end=',')
    prpr, prev = prev, prpr+prev
```
