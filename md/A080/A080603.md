## Python
```Python
from sympy import isprime
def ok(n):
    if n < 10 or not isprime(n): return False
    s = str(n)
    si = (s[:i]+s[i+1:] for i in range(len(s)))
    return any(isprime(int(t)) for t in si)
print([k for k in range(368) if ok(k)]) # _Michael S. Branicky_, Jan 28 2023
```
