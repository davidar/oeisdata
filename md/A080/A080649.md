## Python
```Python
from sympy import isprime
def sigma(n):
    return sum(i for i in range(1,n+1) if n % i == 0)
def sopf(n):
    return sum(i for i in range(1,n+1) if n % i == 0 and isprime(i))
def A080649(n):
    return sopf(sigma(n))
for i in range(2,101):
    print(str(i)+" "+str(A080649(i))) # _Indranil Ghosh_, Jan 15 2017
```
