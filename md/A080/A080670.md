ifsSorted := proc(n)
        local fs,L,p ;
        fs := sort(convert(numtheory[factorset](n),list)) ;
        L := [] ;
        for p in fs do
                L := [op(L),[p,padic[ordp](n,p)]] ;
        end do;
        L ;
end proc:
A080670 := proc(n)
        local a,p ;
        if n = 1 then
                return 1;
        end if;
        a := 0 ;
        for p in ifsSorted(n) do
                a := digcat2(a,op(1,p)) ;
                if op(2,p) > 1 then
                        a := digcat2(a,op(2,p)) ;
                end if;
        end do:
        a ;
end proc: # _R. J. Mathar_, Oct 02 2011
