## Python

```Python
from sympy import mobius, integer_nthroot, primepi
def A080769(n):
    def f(x): return int(n-1+x+sum(mobius(k)*(integer_nthroot(x,k)[0]-1) for k in range(2,x.bit_length())))
    def bisection(f,kmin=0,kmax=1):
        while f(kmax) > kmax: kmax <<= 1
        while kmax-kmin > 1:
            kmid = kmax+kmin>>1
            if f(kmid) <= kmid:
                kmax = kmid
            else:
                kmin = kmid
        return kmax
    return int(-primepi(a:=bisection(f,n,n))+primepi(bisection(lambda x:f(x)+1,a,a))) # _Chai Wah Wu_, Sep 09 2024
```
