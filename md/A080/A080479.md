## Python
```Python
def a(n):
  s = "".join(sorted("".join(map(str, range((n-1)*n//2+1, n*(n+1)//2+1)))))
  if '0' not in s: return int(s)
  rz = s.rfind('0')
  return int(s[rz+1] + s[:rz+1] + s[rz+2:])
print([a(n) for n in range(1, 15)]) # _Michael S. Branicky_, Jan 23 2021
```
