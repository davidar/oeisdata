## Python
```Python
primes = [2]*2
primes[1] = 3
def addPrime(k):
  for p in primes:
    if k%p==0:  return
    if p*p > k:  break
  primes.append(k)
for n in range(5, 1000000, 6):
  addPrime(n)
  addPrime(n+2)
for n in range(1,99):
  p = 1
  j = n
  i = 0
  while j>1:
    e = 0
    while j % primes[i] == 0:
      j /= primes[i]
      e+=1
    if e:
      e = ((e-1)^1) + 1
      p*= primes[i^1]**e
    i += 1
  print(str(p), end=', ')
```
