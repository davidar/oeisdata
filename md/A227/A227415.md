for n in range(2, 77):
    prOdd = prEven = 1
    for i in range(1, n, 2): prOdd *= i
    for i in range(2, n, 2): prEven *= i
    if n&1: print str(prEven % prOdd)+',',
    else:   print str(prOdd % prEven)+',',
