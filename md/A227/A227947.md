## Python
```Python
from sympy import divisors
def pal(n):
  r = ''
  for i in str(n):
    r = i + r
  return r == str(n)
{print(n,end=', ') for n in range(1,10**7) if pal(n) and pal(sum(divisors(n))-n) and len(divisors(n)) > 2}
```
