## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    ip, sp, hp = 0, 0, "0"
    for i in count(1):
        s = i*i
        h = "".join(sorted(str(s)))
        if h == hp: yield sp
        ip, sp, hp = i, s, h
print(list(islice(agen(), 23))) # _Michael S. Branicky_, Feb 18 2024
```
