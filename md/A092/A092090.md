## Python
```Python
from itertools import accumulate, islice
def A092090_gen(): # generator of terms
    blist, a, b = tuple(), 1, 2
    while True:
        yield (blist := tuple(accumulate(reversed(blist),initial=a)))[-1]
        a, b = b, a+b
A092090_list = list(islice(A092090_gen(),40)) # _Chai Wah Wu_, Jun 12 2022
```
