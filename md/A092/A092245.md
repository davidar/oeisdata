## Maple
```Maple
for n from 1 to 100 do
  r:= 10^(n-1);
  p:= nextprime(r); q:= nextprime(p);
  while q - p > 2 do
    p:= q; q:= nextprime(p);
  od;
  A[n]:= p;
od:
seq(A[n],n=1..100); # _Robert Israel_, Aug 04 2014
```

```Python
        p=sympy.nextprime(p)
```
