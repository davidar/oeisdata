A049502 := proc(n)
    local a,ndgs,p ;
    a := 0 ;
    ndgs := convert(n,base,2) ;
    for p from 1 to nops(ndgs)-1 do
        if op(p,ndgs)- op(p+1,ndgs) = 1 then
            a := a+p ;
        end if;
    end do:
    a ;
end proc: # _R. J. Mathar_, Oct 17 2012
