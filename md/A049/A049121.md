## Python
```Python
from itertools import count, islice
from sympy import isprime, factorint
def agen():
    yield from [0, 1]
    an = 11
    for n in count(3):
        yield an
        if isprime(an):
            an = 10*an + 1
        else:
            an = int("".join(str(p) for p in factorint(an, multiple=True)))
print(list(islice(agen(), 21))) # _Michael S. Branicky_, Apr 04 2022
```
