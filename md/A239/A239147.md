## Maple
```Maple
isA239147 := proc(n)
    local k ;
    for k from 1 do
        if n-k <= 1 then
            return false;
        end if;
        if isprime(n+k) and isprime(n-k) and isprime(n^2+k)
            and isprime(n^2-k) and isprime(n^3+k) and isprime(n^3-k)            then
            return true;
        end if;
    end do;
end proc:
for n from 1 to 800 do
    if isA239147(n) then
        printf("%d,",n) ;
    end if;
end do: # _R. J. Mathar_, Mar 12 2014
```
