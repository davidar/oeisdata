## Maple
```Maple
A239146 := proc(n)
    local k ;
    for k from 1 do
        if n-k <= 1 then
            return 0;
        end if;
        if isprime(n+k) and isprime(n-k) and isprime(n^2+k)
            and isprime(n^2-k) then
            return k;
        end if;
    end do;
end proc:
seq(A239146(n),n=1..80) ; # _R. J. Mathar_, Mar 12 2014
```
