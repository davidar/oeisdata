## Python

from sympy import isprime, nextprime
A239638_list, p = [], 5
while p < 10**6:
    if (p % 6) == 5:
        n = (p-1)//2
        if pow(2,n,p) == 1 and isprime((2**n-1)//p):
            A239638_list.append(n)
    p = nextprime(p) # _Chai Wah Wu_, Jun 05 2019
