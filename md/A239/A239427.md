## Python
```Python
from math import prod
def A031286(n):
    ap = 0
    while n > 9: n, ap = sum(map(int, str(n))), ap+1
    return ap
def A031346(n):
    mp = 0
    while n > 9: n, mp = prod(map(int, str(n))), mp+1
    return mp
def ok(n): return A031286(n) == A031346(n)
print([k for k in range(107) if ok(k)]) # _Michael S. Branicky_, Sep 17 2022
```
