## Python
```Python
from collections import Counter
from itertools import count, islice
def agen(): # generator of terms
    an, a, c = 0, dict(), Counter()
    for n in count(1):
        yield an
        a[n] = an
        c[an] += 1
        an = a[c[an]] + 1
print(list(islice(agen(), 100))) # _Michael S. Branicky_, Jul 12 2024
```
