use strict;
use warnings;
use feature 'say';
use Math::Prime::XS qw(is_prime);
use Memoize;
use Math::BigInt;
memoize('factorial');
use Data::Dumper;
my @primes = ();
for (2 .. 200) {
        if(is_prime($_)) {
                push @primes, $_;
        }
}
for (1 .. $#primes) {
        say factorial($primes[$_]) - factorial($primes[$_ - 1]);
}
sub factorial {
        my $x = Math::BigInt->new(shift);
        return $x if $x == 1;
        return factorial($x - 1) * $x;
}
