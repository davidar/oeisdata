## Python
```Python
from itertools import combinations
A239103_list = []
for n in range(16):
    for k in range(n, -1, -1):
        c, d0 = 0, ['0']*(n+k)
        for x in combinations(range(n+k), n):
            d = list(d0)
            for i in x:
                d[i] = '1'
            if not '1011101' in ''.join(d):
                c += 1
        A239103_list.append(c) # _Chai Wah Wu_, Sep 12 2014
```
