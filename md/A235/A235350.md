## Python
```Python
a = [0, 1]
for n in range(20):
    m = len(a)
    d = 0
    for i in range (1, m):
        for j in range (1, m):
            if (i+j)%(m-1) == 0 and (i+j) < m:
                d += a[i]*a[j]
    f = 0
    for i in range (1, m):
        for j in range (1, m):
            if (i+j)%m == 0 and (i+j) <= m:
                f += a[i]*a[j]
    g = 0
    for i in range (1, m):
        for j in range (1, m):
            for k in range (1, m):
                if (i+j+k)%m == 0 and (i+j+k) <= m:
                    g += a[i]*a[j]*a[k]
    y = g + 2*f - d
    a.append(y)
print(a[1:]) # Edited by _Andrey Zabolotskiy_, Sep 04 2024
```
