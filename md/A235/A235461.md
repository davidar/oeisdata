## Python
```Python
from itertools import islice
from sympy import nextprime, isprime
def A235461_gen(): # generator of terms
    p = 1
    while (p:=nextprime(p)):
        if isprime(m:=int(bin(p)[2:],4)):
            yield m
A235461_list = list(islice(A235461_gen(),20)) # _Chai Wah Wu_, Aug 21 2023
```
