## Python
```Python
from functools import cache
@cache
def u(n, r):
    if r < 0 or r > 9: return 0
    if n == 1: return (r > 0)
    return u(n-1, r-1) + u(n-1, r) + u(n-1, r+1)
def a(n): return sum(u(n, r) for r in range(10))
print([a(n) for n in range(1, 28)]) # _Michael S. Branicky_, Sep 26 2021
```
