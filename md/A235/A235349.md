## Python
```Python
a = [0, 1]
for n in range(20):
    m = len(a)
    d = 0
    for i in range (1, m):
        for j in range (1, m):
            if (i+j)%m == 0 and (i+j) <= m:
                d += a[i]*a[j]
    g = 0
    for i in range (1, m-1):
        for j in range (1, m-1):
            for k in range (1, m-1):
                if (i+j+k)%m == 0 and (i+j+k) <= m:
                    g += a[i]*a[j]*a[k]
    y = 2*g + d - a[m-1]
    a.append(y)
print(a)
```
