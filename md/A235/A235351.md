## Python
```Python
a = [0, 1]
for n in range(20):
    m = len(a)
    d = 0
    for i in range (1, m):
        for j in range (1, m):
            if (i+j)%m ==0 and (i+j) <= m:
                d = d + a[i]*a[j]
    g = 0
    for i in range (1, m):
        for j in range (1, m):
            for k in range (1, m):
                if (i+j+k)%m ==0 and (i+j+k) <= m:
                    g = g + a[i]*a[j]*a[k]
    y = 2*g + 3*d - a[m-1]
    a.append(y)
print(a)
```
