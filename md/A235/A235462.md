## Python
```Python
from itertools import islice
from sympy import isprime, nextprime
def A235462_gen(): # generator of terms
    p = 1
    while (p:=nextprime(p)):
        if isprime(m:=int(bin(p)[2:],5)):
            yield m
A235462_list = list(islice(A235462_gen(),20)) # _Chai Wah Wu_, Aug 21 2023
```
