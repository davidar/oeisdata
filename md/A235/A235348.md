class A235348() :
    def __init__(self) :
        self.a = [1, 2, 12, 82, 636, 5266]
    def at(self, n):
        if n <= len(self.a):
            return self.a[n-1]
        else:
            rhs = -3*(n-1)*(160*n-237)*self.at(n-1) \
            +3*(-422*n**2+1721*n-1713)*self.at(n-2) \
            +2*(-67*n**2+388*n-552)*self.at(n-3) \
            +(137*n**2-1352*n+3279)*self.at(n-4) \
            +(7*n-37)*(n-6)*self.at(n-5) -(n-6)*(n-7)*self.at(n-6)
            rhs //= (-54*n*(n-1))
            self.a.append(rhs)
            return self.a[-1]
