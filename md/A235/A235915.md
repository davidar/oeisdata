## Python
```Python
def adddigits(i):
  s = str(i)
  t=0
  for j in s:
    t = t+int(j)
  return t
n = 1
a = [1]
for i in range(0, 100):
  r = adddigits(n)%5+1
  n = n+r
  a.append(n)
print(a)
```
