## Python
```Python
from math import isqrt
def aupto(limit):
  alst, rootlimit = [], isqrt(limit)
  for k in range(1, rootlimit+1):
    if set(str(k*k)) <= set("2357"): alst.append(k*k)
  return alst
print(aupto(7333353225)) # _Michael S. Branicky_, May 15 2021
```
