## Python
```Python
from sympy import isprime, primerange
from functools import cache
@cache
def A000586(n, k=None): # after _Charles R Greathouse IV_
    if k == None: k = n
    if n < 1: return int(n == 0)
    return sum(A000586(n-p, p-1) for p in primerange(1, k+1))
def a(n): return A000586(n) - isprime(n)
print([a(n) for n in range(1, 83)]) # _Michael S. Branicky_, Sep 03 2021
```
