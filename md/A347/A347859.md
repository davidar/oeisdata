## Python
```Python
from sympy import isprime
def a(n):
    if n%2 == 0 and isprime(n//2): return n//2
    if (n-1)%2 == 0 and isprime((n-1)//2): return 3*(n-1)//2
    return n
print([a(n) for n in range(71)]) # _Michael S. Branicky_, Feb 21 2022
```
