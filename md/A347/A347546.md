def b(n):
    if (0<=n<=3):
        return 1
    if (n==4):
        return 2
    if (n%2==1):
      t=0
      for k in range(1, ((n+1)//2)):
          t+=b(2*k-2)*b(n-2*k)
      return t
    else:
        s=0
        for j in range(round(n/4), (n//2)):
            s+=b(4*j-n)*b(n-2*j-1)
        return b(n-1)+s
for i in range(30):
    print(str(i)+': '+str(b(i)))
