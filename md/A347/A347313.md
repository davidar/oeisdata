from math import gcd
from sympy import prime
def A347313(n):
    p = prime(n)
    i, j, nset, m = 1, 2, {1}, 2
    while True:
        k = m
        i += 1
        while k == j or gcd(k,j) == 1 or k in nset:
            k += 1
        if k == p:
            return i
        j = k+1
        nset.add(k)
        while m in nset:
            m += 1 # _Chai Wah Wu_, Sep 06 2021
