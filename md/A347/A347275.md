## Maple
```Maple
A347275 := proc(n)
    local a,i,j ;
    a := 0 ;
    for i from 0 to n do
    for j from 0 to n-i do
        if i*j <= n then
            a := a+1 ;
        end if;
    end do:
    end do:
    a ;
end proc:
seq(A347275(n),n=0..40) ; # _R. J. Mathar_, Sep 15 2021
```

## C
```C
int T(int n) { int cnt = 0; for (int a = 0; a <= n; ++a) for (int b = 0; b <= n - a; ++b) if (a * b <= n) ++cnt; return cnt; }
```
