from functools import lru_cache
@lru_cache(maxsize=None)
def A347030(n):
    if n <= 1:
        return n
    c, j = 1, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j&1)*(-1 if j&1 else 1)*A347030(k1)
        j, k1 = j2, n//j2
    return c+(n+1-j&1)*(-1 if j&1 else 1) # _Chai Wah Wu_, Apr 04 2023
