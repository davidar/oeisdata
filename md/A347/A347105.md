## Python
```Python
def drs(x, m):
  if x == 0:
    return 0
  return ((x - 1) % m) + 1
def a(n):
  DRS = [drs(x, 9) for x in range(0, n + 1)]
  for a in range(2, n):
    da = DRS[a]
    for b in range(2, 1 + (n // a)):
      ab = a * b
      if ab > n:
        break
      else:
        x = da + DRS[b]
        if DRS[ab] < x:
          DRS[ab] = x
  return DRS[1: n + 1][-1]
```
