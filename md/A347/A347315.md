## Python
```Python
from collections import Counter
def aupton(nn):
    num, inventory, rowsum, alst = 0, Counter([0]), 0, [0]
    while len(alst) <= nn:
        c = inventory[num]
        num += 1
        rowsum += c
        inventory.update([c])
        if c == 0:
            alst.append(rowsum)
            num = rowsum = 0
    return alst
print(aupton(52)) # _Michael S. Branicky_, Nov 14 2021
```
