## Python
```Python
from sympy import factorint, isprime
def aupto(limit):
    alst, k, s = [], 1, 0
    for k in range(1, limit+1):
        if sum(factorint(k).values()) == 2:
            s += k
            if s > limit: break
            if isprime(s): alst.append(s)
    return alst
print(aupto(320000)) # _Michael S. Branicky_, Aug 29 2021
```
