## Python
```Python
from sympy import divisors, factorint
def a(n):
    npf = len(factorint(n, multiple=True))
    for d in divisors(n)[1:-1]:
        if 2*len(factorint(d, multiple=True)) == npf: return d
    return 1
print([a(n) for n in range(1, 88)]) # _Michael S. Branicky_, Aug 18 2021
```

```Python
from math import prod
from sympy import factorint
def A347045(n):
    fs = factorint(n,multiple=True)
    q, r = divmod(len(fs),2)
    return 1 if r else prod(fs[:q]) # _Chai Wah Wu_, Aug 20 2021
```
