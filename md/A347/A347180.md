## Python
```Python
def aupton(terms):
    alst, aset = [1], {1}
    while len(alst) < terms:
        an, target = 2, str(alst[-1])[0]
        while an in aset or target not in str(alst[-1]*an): an += 1
        alst.append(an); aset.add(an)
    return alst
print(aupton(200)) # _Michael S. Branicky_, Aug 21 2021
```
