## Python
```Python
from sympy import isprime
route = set({})
nums = [i*(10**j) for i in range(1, 10) for j in range(2)]
def addnum(a):
    global route
    for j in nums:
        b = int("{}{}".format(a, j))
        if isprime(b):
            if b not in route:
                route.add(b)
                addnum(b)
    for j in nums:
        b = int("{}{}".format(j, a))
        if isprime(b):
            if b not in route:
                route.add(b)
                addnum(b)
def run():
    for i in nums:
        if isprime(i):
            addnum(i)
```
