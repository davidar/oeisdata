## Python
```Python
from sympy import factorint
def aupton(terms):
    alst, i, n, s, p = [], 1, 0, 0, 1
    while n < terms:
        if sum(factorint(i).values()) == 2:
            n += 1; s += i; p *= i
            alst.append(p%s)
        i += 1
    return alst
print(aupton(68)) # _Michael S. Branicky_, Sep 01 2021
```
