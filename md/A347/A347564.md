from collections import Counter
def aupton(terms):
    num, alst, inventory = 0, [0], Counter([0])
    for n in range(2, terms+1):
        if num == 0:
            c = len(inventory)
        else:
            c = sum(inventory[i] == num for i in inventory)
        num = 0 if c == 0 else num + 1
        alst.append(c)
        inventory.update([c])
    return alst
print(aupton(83)) # _Michael S. Branicky_, Oct 06 2021
