from collections import Counter
def aupton(terms):
    num, A342585lst, inventory, alst, rowlen = 0, [0], Counter([0]), [1], 0
    while len(alst) < terms:
        rowlen += 1
        c = inventory[num]
        if c == 0:
            num = 0
            alst.append(rowlen)
            rowlen = 0
        else:
            num += 1
        A342585lst.append(c)
        inventory.update([c])
    return alst
