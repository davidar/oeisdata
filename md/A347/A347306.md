from math import gcd
def A347306(n):
    if n == 1:
        return 1
    i, j, nset, m = 1, 2, {1}, 2
    while True:
        k = m
        i += 1
        while k == j or gcd(k,j) == 1 or k in nset:
            k += 1
        if k == n:
            return i
        j = k+1
        nset.add(k)
