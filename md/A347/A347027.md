from collections import deque
from itertools import islice
def A347027_gen(): # generator of terms
    aqueue, f, b, a = deque([3]), True, 1, 3
    yield from (1, 3)
    while True:
        a += 2*b
        yield a
        aqueue.append(a)
        if f: b = aqueue.popleft()
        f = not f
