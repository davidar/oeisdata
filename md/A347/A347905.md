# Note that in the (rare) case when m and n are coprime but there are no primes in the Fibonacci-like sequence, this function will go into an infinite loop.
from sympy import isprime,gcd
def A347905(m,n):
    if gcd(m,n) != 1:
        return 0
    m,n = n,m+n
    k=2
    while not isprime(n):
        m,n = n,m+n
        k += 1
    return k
