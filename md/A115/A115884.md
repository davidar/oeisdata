## Python
```Python
from sympy import nextprime
def ispal(n): s = str(n); return s == s[::-1]
def agen(): # generator of terms
    k, pk = 1, 2
    while True:
        if ispal(k+pk): yield k
        k, pk = k+1, nextprime(pk)
g = agen()
print([next(g) for n in range(1, 51)]) # _Michael S. Branicky_, Jun 15 2022
```
