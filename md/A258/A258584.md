def pod(n, m):
    kk = 1
    while n > 0:
        kk= kk*(n%m)
        n =int(n//m)
    return kk
for b in range(0, 6):
    dd, bb=0, (b-1)//2+2
    j=10**bb
    for c in range (10*j, 100*j):
        d, a, ca=0, 0, pod(c, j)
        while ca>0:
            d, a=d+ca, a+1
            if ca<j:
                ca=j
            ca=pod(ca, j)
            if c==d and ca==0:
