from sympy import divisors
from gmpy2 import is_prime, isqrt, isqrt_rem, is_square
A258660_list = []
for l in range(1,17):
....if not is_prime(l):
........fs = divisors(l)
........a, b = isqrt_rem(10**(l-1))
........if b > 0:
............a += 1
........for n in range(a,isqrt(10**l-1)+1):
............n2 = n**2
............ns = str(n2)
............for g in fs:
................y = 0
................for h in range(0,l,g):
....................y += int(ns[h:h+g])
................if not is_square(y):
....................break
............else:
................A258660_list.append(n2) # _Chai Wah Wu_, Jun 08 2015
