def A258603(n):
    c, m = 0, prime(n)**6
    for y1 in range(1,integer_nthroot(m,11)[0]+1):
        if all(d<=1 for d in factorint(y1).values()):
            for y2 in range(1,integer_nthroot(z2:=m//y1**11,10)[0]+1):
