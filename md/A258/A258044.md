## Python
```Python
from sympy import nextprime as np
from sympy import prevprime as pp
f1=1
f2=1
f3=f1+f2
while f3>0:
    if f3%2==0 and f3>3:
        i=f3/2
        p=pp(i); q=np(p)
        if p+q==f3:
            print(f3)
    f1=f2; f2=f3
    f3=f1+f2
```
