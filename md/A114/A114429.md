```Python
    p=sympy.prevprime(10**i)

        p=sympy.prevprime(p)
```

```Python
from sympy import prevprime
def a(n):
    p = prevprime(10**n); pp = prevprime(p)
    while p - pp != 2: p, pp = pp, prevprime(pp)
    return p
print([a(n) for n in range(1, 19)]) # _Michael S. Branicky_, Jan 17 2022
```
