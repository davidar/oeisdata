import Data.List (delete)
a252022 n = a252022_list !! (n-1)
a252022_list = 1 : f [1] (drop 2 a031298_tabf) where
   f xs zss = g zss where
     g (ds:dss) = if all (<= 9) $ zipWith (+) xs ds
       then (foldr (\d v -> 10 * v + d) 0 ds) : f ds (delete ds zss)
       else g dss
