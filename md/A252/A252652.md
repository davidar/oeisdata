import re
def A252652(n):
    if n == 0:
        return 0
    f, i, s = 1, 0, re.compile('[0-9]*[1-9]0{'+str(n)+'}[1-9][0-9]*')
    while s.match(str(f)) == None:
        i += 1
        f *= i
    return i # _Chai Wah Wu_, Dec 29 2015
