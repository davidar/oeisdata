## Python
```Python
from sympy import factorint, prevprime
from operator import mul
def a064989(n):
    f = factorint(n)
    return 1 if n==1 else reduce(mul, [1 if i==2 else prevprime(i)**f[i] for i in f])
def a(n): return 1 if n==1 else n//2 if n%2==0 else a064989(n)
```
