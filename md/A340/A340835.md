## Python
```Python
def A340835(n):
    if n == 0:
        return 0
    s = str(n)
    for i, x in enumerate(s):
        if x != '9':
            break
    else:
        return n
    s1, s2 = s[:i+1], s[i+1:]
    if s2 == '':
        if s1[-1] == '0':
            return int(str(n+1)[::-1])
        else:
            return int(s[::-1])
    if int(s2) <= 1:
        return int('1'+s2[-2::-1]+s1[::-1])
    else:
        return int('1'+'0'*(len(s2)-1)+str(int(s1)+1)[::-1]) # _Chai Wah Wu_, Mar 14 2021
```
