def aupto(n):
  alst, aset = [], set()
  for k in range(1, n+1):
    ak = 1
    while True:
      while ak in aset: ak += 1
      binakplusk = bin(ak+k)[2:]
      if binakplusk.count("0")==binakplusk.count("1"): break
      ak += 1
    alst.append(ak)
    aset.add(ak)
  return alst
print(aupto(66)) # _Michael S. Branicky_, Jan 02 2021
