```Python
from sympy import factorint
def fromdigits(d, b):
  n = 0
  for di in d: n *= b; n += di
  return n
def a(n):
  f = sorted(factorint(n, multiple=True))
  return fromdigits(f, f[-1]+1)
print([a(n) for n in range(2, 76)]) # _Michael S. Branicky_, Jan 06 2021
```
