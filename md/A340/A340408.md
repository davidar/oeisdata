## Python
```Python
from sympy import isprime, nextprime
def aupton(terms):
  p, q, r, s, alst = 2, 3, 5, 7, []
  while len(alst) < terms:
    if isprime(p*q + q*r + r*s): alst.append(p)
    p, q, r, s = q, r, s, nextprime(s)
  return alst
print(aupton(55)) # _Michael S. Branicky_, Mar 17 2021
```
