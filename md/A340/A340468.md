## Python
```Python
from sympy import isprime, nextprime, prime
def a(n):
  prodpnpm = pm = prime(n)
  while not isprime(2+prodpnpm): pm = nextprime(pm); prodpnpm *= pm
  return 2+prodpnpm
print([a(n) for n in range(2, 12)]) # _Michael S. Branicky_, Jan 08 2021
```
