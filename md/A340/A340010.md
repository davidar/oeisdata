## Python
```Python
import networkx as nx
def T(n): #A014682
    return n//2 if n%2 == 0 else (3*n+1)//2
def a(n):
    G = nx.Graph()
    G.add_nodes_from(range(1, n+1))
    G.add_edges_from([(m,T(m)) for m in range(1, n+1) if T(m) <= n])
    return len(max(nx.connected_components(G)))
for n in range(1, 70):
    print(a(n), end=", ")
```
