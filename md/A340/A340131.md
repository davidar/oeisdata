## Python
```Python
def digits(n, b):
  out = []
  while n >= b:
    out.append(n % b)
    n //= b
  return [n] + out[::-1]
def ok(n):
  t = digits(n, 3)
  if t.count(1) != t.count(2): return False
  return all(t[:i].count(1) >= t[:i].count(2) for i in range(1, len(t)))
print([n for n in range(750) if ok(n)]) # _Michael S. Branicky_, Dec 29 2020
```
