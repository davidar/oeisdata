## Python

def aupton(terms):
    alst, chapnk, k = [], [1], 1
    for n in range(1, terms+1):
        chapn = chapnk.pop(0)
        while chapn not in chapnk:
            k += 1
            chapnk.extend(list(map(int, str(k))))
        alst.append(chapnk.index(chapn) + 1)
    return alst
print(aupton(74)) # _Michael S. Branicky_, Sep 13 2021
