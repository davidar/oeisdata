## Python
```Python
from itertools import count
from sympy import prime
def A106298(n):
    a = b = (5%(p:=prime(n)),1%p,7%p,3%p,15%p)
    s = sum(b) % p
    for m in count(1):
        b, s = b[1:] + (s,), (s+s-b[0]) % p
        if a == b:
            return m # _Chai Wah Wu_, Feb 22-27 2022
```
