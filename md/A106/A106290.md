## Python
```Python
from itertools import count,product
def A106290(n):
    bset, tset = set(), set()
    for t in product(range(n),repeat=5):
        t2 = t
        for c in count(1):
            t2 = t2[1:] + (sum(t2)%n,)
            if t == t2:
                bset.add(c)
                tset.add(t)
                break
            if t2 in tset:
                tset.add(t)
                break
    return len(bset) # _Chai Wah Wu_, Feb 22 2022
```
