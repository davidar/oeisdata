## Python
```Python
from itertools import count
from sympy import prime
def A106302(n):
    a = b = (0,)*2+(1 % (p:= prime(n)),)
    for m in count(1):
        b = b[1:] + (sum(b) % p,)
        if a == b:
            return m # _Chai Wah Wu_, Feb 27 2022
```
