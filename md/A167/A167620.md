## Python
```Python
A167620 = []
for k in range(1,511176):
    dprod, k_str = 1, str(k)
    for d in range(0,len(k_str)): dprod *= int(k_str[d])
    if dprod != 0 and k % dprod == 0 and str(dprod) == k_str[-(len(str(dprod))):]:
        A167620.append(k)
print(A167620) # _Karl-Heinz Hofmann_, Jan 26 2024
```
