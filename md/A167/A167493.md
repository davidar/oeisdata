## Python
```Python
from math import gcd
from itertools import count, islice
def agen(): # generator of terms
    an = 2
    for n in count(2):
        yield an
        an = an + gcd(n, an) if n&1 else an + gcd(n-2, an)
print(list(islice(agen(), 66))) # _Michael S. Branicky_, Jan 22 2023
```
