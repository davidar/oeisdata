## Python
```Python
A167619 = []
for k in range(1,7736):
    dprod, k_str = 1, str(k)
    for d in range(0,len(k_str)): dprod *= int(k_str[d])
    if dprod != 0 and str(dprod) == k_str[-(len(str(dprod))):]: A167619.append(k)
```
