## Python
```Python
from sympy import divisors
from itertools import count
def a(n):
    d = divisors(n)
    return next(k for k in count(1) if len(set(di%k for di in d))==len(d))
print([a(n) for n in range(1, 93)]) # _Michael S. Branicky_, Jan 30 2023
```
