## Python
```Python
def euclid_steps(a,b):
    step_count = 0
    while(b != 0):
        a , b = b , a % b
        step_count += 1
    return step_count
for n in range(1,1001):
    l = 0
    for i in range(n): l = max(l,euclid_steps(n,i))
    print(str(n)+" "+str(l)) # _Ely Golden_, May 18 2020
```
