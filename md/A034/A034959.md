## Python
```Python
from itertools import islice
from sympy import nextprime
def A034959_gen(): # generator of terms
    a, p = 0, 2
    while True:
        yield p*((a<<1)+p-1)
        a, p = a+p, nextprime(p)
A034959_list = list(islice(A034959_gen(),20)) # _Chai Wah Wu_, Mar 22 2023
```
