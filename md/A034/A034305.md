## Python
```Python
from sympy import isprime
def ok(n):
    if n < 10 or isprime(n): return False
    s = str(n)
    return "0" not in s and not any(isprime(int(s[:i]+s[i+1:])) for i in range(len(s)))
print([k for k in range(229) if ok(k)]) # _Michael S. Branicky_, Jan 15 2023
```
