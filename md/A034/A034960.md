## Python
```Python
from itertools import islice
from sympy import nextprime
def A034960_gen(): # generator of terms
    a, p = 0, 2
    while True:
        yield p*((a<<1)+p)
        a, p = a+p, nextprime(p)
A034960_list = list(islice(A034960_gen(),20)) # _Chai Wah Wu_, Mar 22 2023
```
