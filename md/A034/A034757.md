from itertools import count, islice
def A034757_gen(): # generator of terms
    aset1, aset2, alist = set(), set(), []
    for k in count(1,2):
        bset2 = {k<<1}
        if (k<<1) not in aset2:
            for d in aset1:
                if (m:=d+k) in aset2:
                    break
                bset2.add(m)
            else:
                yield k
                alist.append(k)
                aset1.add(k)
                aset2.update(bset2)
A034757_list = list(islice(A034757_gen(),30)) # _Chai Wah Wu_, Sep 05 2023
