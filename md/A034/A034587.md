## Python

def ok(n):
    f, g = n, n+1
    while g < 10**9:
        if g > 123456788 and "".join(sorted(str(g))) == "123456789":
            return True
        f, g = g, f+g
    return False
print([k for k in range(10**4) if ok(k)]) # _Michael S. Branicky_, Feb 18 2024
