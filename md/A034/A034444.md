# returns the number of unitary divisors of n and a list of them
f:=proc(n)
local ct,i,t1,ans;
ct:=0; ans:=[];
t1:=divisors(n);
for i from 1 to nops(t1) do
d:=t1[i];
if igcd(d,n/d)=1 then ct:=ct+1; ans:=[op(ans),d]; fi;
od:
RETURN([ct,ans]);
end;
# _N. J. A. Sloane_, May 01 2013
