## Python
```Python
from math import sqrt
def a(n):
  k, s, pow10 = 3, sqrt(2), 10**len(str(n))
  while int(pow10*(s-int(s))) != n: k, s = k+1, sqrt(k)
  return k-1
print([a(n) for n in range(1, 67)]) # _Michael S. Branicky_, Mar 15 2021
```
