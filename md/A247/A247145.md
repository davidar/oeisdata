## Python
```Python
from functools import reduce
from operator import mul
def divs(n):
    for i in range(1, int(n / 2 + 1)):
        if n % i == 0:
            yield i
    yield n
g = []
for a in range(2, 100):
    q = list(divs(a))[0:-1]
    if reduce(mul, q, 1) % sum(q) == 0 and len(q) != 1:
        g.append(a)
print(g)
```
