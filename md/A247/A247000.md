```Python
def A247000(n):
....maxcount = 0
....for i in range(2**(n-1),2**n):
........s = format(i,'0'+str(n)+'b')
........s, plist = s+s[:-1], []
........for j in range(n):
............for k in range(n):
................t = s[j:j+k+1]
................if t == t[::-1] and not t in plist:
....................plist.append(t)
........if len(plist) > maxcount:
............maxcount = len(plist)
....return maxcount # _Chai Wah Wu_, Sep 16 2014
```
