## Python
```Python
from itertools import count, islice
from sympy import divisors
def A094519_gen(): # generator of terms
    for n in count(1):
        for i in range(1,len(d:=divisors(n))):
            di = d[i]
            for j in range(i):
                if n % (di+d[j]) == 0:
                    yield n
                    break
            else:
                continue
            break
A094519_list = list(islice(A094519_gen(),20)) # _Chai Wah Wu_, Dec 26 2021
```
