```Maple
# The objective of this implementation is efficiency.
# m -> [a(0), a(1), ..., a(m-1)] for m > 0.
A094577_list := proc(m)
   local A, R, M, n, k, j;
   M := m+m-1; A := array(1..M);
   j := 1; R := 1; A[1] := 1;
   for n from 2 to M do
      A[n] := A[1];
      for k from n by -1 to 2 do
         A[k-1] := A[k-1] + A[k]
      od;
      if is(n,odd) then
         j := j+1; R := R,A[j] fi
   od;
[R] end:
A094577_list(100); # example call - _Peter Luschny_, Jan 17 2011
```
