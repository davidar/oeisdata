## Python
```Python
from sympy import fibonacci
def a(n):
    k=0
    x=0
    while n>0:
        k=0
        while fibonacci(k)<=n: k+=1
        x+=10**(k - 3)
        n-=fibonacci(k - 1)
    return x
def ok(n):
    x=str(a(n))
    return x==x[::-1]
print([n for n in range(1101) if ok(n)]) # _Indranil Ghosh_, Jun 07 2017
```
