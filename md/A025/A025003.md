## Maple
```Maple
N:= 50: # to get a(0)..a(N)
V:= Array(0..N):
V[0]:= 1: V[1]:= 2:
m:= 2: p:= 3: g:= 1: n:= 1:
do
  if g+p-m-1 >= V[n] then
    m:= V[n]+m-g;
    n:= n+1;
    V[n]:= m;
    if n = N then break fi;
    g:= V[n-1];
  else
    g:= g+p-m;
    m:= p+1;
    p:= nextprime(m);
  fi;
od;
convert(V, list); # _Robert Israel_, Sep 08 2020
```
