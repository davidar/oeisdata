from sympy import prime, integer_nthroot, factorint
def A025474(n):
    if n == 1: return 0
    def f(x): return int(n+x-1-sum(primepi(integer_nthroot(x,k)[0]) for k in range(1,x.bit_length())))
    m, k = n, f(n)
    while m != k:
        m, k = k, f(k)
    return list(factorint(m).values())[0] # _Chai Wah Wu_, Aug 15 2024
