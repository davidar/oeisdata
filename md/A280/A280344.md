## Python
```Python
def t(n):
    s=0
    for a in range(0,n+1):
        for b in range(0,n+1):
            for c in range(0,n+1):
                for d in range(0,n+1):
                    if (a*d-b*c)==(a*d+b*c)**n:
                        s+=1
    return s
for i in range(0,41):
    print str(i)+" "+str(t(i))
```
