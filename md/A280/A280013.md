## Python
```Python
from sympy.ntheory.factor_ import core
from sympy import divisors
def s(n): return sum(list(filter(lambda i: core(i) == i, divisors(n))))
def ok(n):
    m=1
    while m<n:
        if not s(n)>s(m): return False
        m+=1
    return True # _Indranil Ghosh_, Apr 16 2017
```
