def t(n):
    s=0
    for a in range(n+1):
        for b in range(n+1):
            for c in range(n+1):
                for d in range(n+1):
                    if (a+b+c+d)==(a*d-b*c):
                        s+=1
    return s
for i in range(51):
    print(str(i)+" "+str(t(i)))
