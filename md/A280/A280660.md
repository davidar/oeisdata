a:= proc(n) local m, t, k, c, h; m, t:= 10^n, 2048;
      for k from 12 do t:= 2*t mod m; h, c:= t, 0;
        while h>0 do if irem(h, 10, 'h')=9 then c:= c+2 fi od;
        if c >= n then return k fi
      od
    end:

## Python
```Python
def A280660(n):
    m, k, l = 10**n,1,2
    while True:
        if 2*str(l).count('9') >= n:
            return k
        k += 1
        l = (l*2) % m # _Chai Wah Wu_, Jan 07 2017
```
