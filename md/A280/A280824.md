isA280824 := proc(n)
    if n < 10 then
        return false;
    end if;
    dgs := convert(n,base,10) ;
    if type(nops(dgs),'even') then
        type(nops(convert(dgs,set)),'even') ;
    else
        false;
    end if;
end proc: # _R. J. Mathar_, Jan 17 2017
