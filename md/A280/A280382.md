## Python
```Python
from sympy import primeomega
def aupto(limit):
  prv, cur, nxt, alst = 1, 1, 2, []
  for n in range(3, limit+1):
    if prv == nxt: alst.append(n)
    prv, cur, nxt = cur, nxt, primeomega(n+2)
  return alst
print(aupto(282)) # _Michael S. Branicky_, May 20 2021
```
