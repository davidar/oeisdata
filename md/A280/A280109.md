def qr_distribution(N):
    QR = []
    QN = []
    for i in range(N):
        t = (i*i)%N
        if t not in QR: QR.append(t)
    for i in range(N):
        if i not in QR: QN.append(i)
    out = 0
    for i in range(0,N):
        out *= 2
        if i in QR: out += 1
    return out
