## Python
```Python
def sigma(n):
    s=0
    for i in range(1,n+1):
        if n%i==0:
            s+=i
    return s
def C(n):
    s=""
    for i in range(n,0,-1):
        s+=str(i)
    return int(s)
for i in range(1,10001):
    print str(i)+" "+str(C(i)%sigma(i))
```
