from itertools import islice
from collections import deque
def A102378_gen(): # generator of terms
    aqueue, f, b, a = deque([2]), True, 1, 2
    yield from (1, 3)
    while True:
        a += b
        yield 2*a - 1
        aqueue.append(a)
        if f: b = aqueue.popleft()
        f = not f
