N:= 100:
H:= ListTools:-PartialSums([seq(1/n,n=1..N)]):
S:= ListTools:-PartialSums(map(numtheory:-tau,[$1..N])):
seq(floor(n*H[n])-S[n],n=1..N); # _Robert Israel_, Mar 20 2016
