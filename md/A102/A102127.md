x = y = direction = 0
cells, a = {}, []
for n in range(1, 1000):
    c = cells.get((x, y), 0)
    cells[(x, y)] = (c + 1) % 4
    direction += [1, 1, -1, -1][c]
    (dx, dy) = [(1, 0), (0, 1), (-1, 0), (0, -1)][direction%4]
    x += dx;  y += dy
    if (x, y) == (0, 0):
        a.append(n)
print(a) # A102110
print([x-y for x, y in zip(a, [0]+a)]) # this sequence -  _Andrey Zabolotskiy_, Jan 06 2023
