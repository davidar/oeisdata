from itertools import islice
def A172404_gen(): # generator of terms
    a, b, c, l = 3, 4, 1, 0
    while True:
        if a<=c:
            if c<b:
                yield l
            a *= 10
            b *= 10
        c <<= 1
        l += 1
A172404_list = list(islice(A172404_gen(),30)) # _Chai Wah Wu_, Nov 13 2023
