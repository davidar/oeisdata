from sympy import divisors
def A116137_list(max_n):
    A,z = [1],0
    while len(A) <= max_n:
        A.extend((A[-1]+d) for d in divisors(A[z])); z += 1
    return(A[:max_n+1]) # _John Tyler Rascoe_, Nov 26 2023
