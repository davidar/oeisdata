from sympy import divisors
def A(maxn):
    A = []
    for n in range(1,maxn+1):
        d = divisors(n)
        for j in range(0,len(d)):
            if d[j] > len(A): break
            if A.count(d[j]) < A[d[j]-1]: break
        A.append(d[j])
    return(A) # _John Tyler Rascoe_, Mar 04 2023
