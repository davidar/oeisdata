## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    yield from [1, 3]
    for k in count(2):
        t = 3*(10**(k+1) + 1)
        yield from (t//i for i in range(300, 30, -1) if t%i == 0)
print(list(islice(agen(), 33))) # _Michael S. Branicky_, Mar 26 2023
```
