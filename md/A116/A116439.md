## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    yield from [1, 2, 4]
    for k in count(2):
        t = 4*(10**(k+1) + 1)
        yield from (t//i for i in range(400, 40, -1) if t%i == 0)
print(list(islice(agen(), 38))) # _Michael S. Branicky_, Mar 26 2023
```
