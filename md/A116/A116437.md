## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    yield from [1, 2]
    for k in count(2):
        t = 2*(10**(k+1) + 1)
        yield from (t//i for i in range(200, 20, -1) if t%i == 0)
print(list(islice(agen(), 33))) # _Michael S. Branicky_, Mar 26 2023
```
