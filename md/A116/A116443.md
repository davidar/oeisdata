## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    yield from [1, 2, 4, 8]
    for k in count(2):
        t = 8*(10**(k+1) + 1)
        yield from (t//i for i in range(800, 80, -1) if t%i == 0)
print(list(islice(agen(), 41))) # _Michael S. Branicky_, Mar 26 2023
```
