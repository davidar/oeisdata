## Python
```Python
import sympy
def a(n):
  k = 2
  count = 0
  while k**n > 2**k:
    if sympy.isprime(k**n-2**k):
      count += 1
    k += 1
  return count
n = 1
while n < 1000:
  if not a(n):
    print(n, end=', ')
  n += 1 # _Derek Orr_, Aug 02 2014
```
