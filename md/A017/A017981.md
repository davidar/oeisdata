## Python
```Python
def icbrt(a):
    sr = 1 << (int.bit_length(int(a)) >> 1)
    while a < sr*sr*sr:  sr>>=1
    b = sr>>1
    while b:
      s = sr+b
      if a >= s*s*s:  sr = s
      b>>=1
    return sr
for n in range(100):  print(1+icbrt(2**n-1), end=",")
# _Alex Ratushnyak_, Jan 04 2014
```
