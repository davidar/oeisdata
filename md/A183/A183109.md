## Python
```Python
import math
f = math.factorial
def C(n,r): return f(n)//f(r)//f(n - r)
def T(n,m):
    return sum((-1)**j*C(m,j)*(2**(m - j) - 1)**n for j in range (m+1))
i=1
for n in range(1,21):
    for m in range(1, n+1):
        print(str(i)+" "+str(T(n, m)))
        i+=1 # _Indranil Ghosh_, Mar 14 2017
```
