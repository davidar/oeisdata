```Maple
N:= 3000: # for terms <= N
P:= select(isprime, [2,seq(i,i=3..N/2,2)]): nP:= nops(P):
R:= NULL:
for i from 1 to nP do
  p2:= P[i]^2;
  for j from i to nP do
    x:= p2*P[j];
    if x > N then break fi;
    R:= R, x
od od:
sort([R]);
```

## Python
```Python
from sympy import primepi, primerange, integer_nthroot
def A337877(n):
    def f(x): return int(n+x-sum(primepi(x//k**2)-a for a,k in enumerate(primerange(integer_nthroot(x,3)[0]+1))))
    def bisection(f,kmin=0,kmax=1):
        while f(kmax) > kmax: kmax <<= 1
        while kmax-kmin > 1:
            kmid = kmax+kmin>>1
            if f(kmid) <= kmid:
                kmax = kmid
            else:
                kmin = kmid
        return kmax
    return bisection(f) # _Chai Wah Wu_, Aug 29 2024
```
