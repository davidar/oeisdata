## Python
```Python
a=[]
a.append('1')
def bnot(string):
    nstring=''
    for letter in string:
        nstring+=str(-int(letter)+1)
    return nstring
for i in range(10):
    a.append(a[i]+'1'+bnot(a[i])[::-1])
print([int(i,2) for i in a])
```
