## Python
```Python
def A337856(n):
    k, n1, n2, pset = 0, 10**(n-1)//2-18, 10**n//2-18, set()
    while 50*k**2+60*k < n2:
        a, b = divmod(n1-30*k,50*k+30)
        m = max(k,a+int(b>0))
        r = 50*k*m+30*(k+m)
        while r < n2:
            pset.add(r)
            m += 1
            r += 50*k+30
        k += 1
    return len(pset) # _Chai Wah Wu_, Sep 26 2021
```
