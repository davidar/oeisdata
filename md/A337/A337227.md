from itertools import count
def A337227(n):
    s1 = tuple(int(d) for d in str(n))
    s2 = s1
    for i, s in enumerate(int(d) for k in count(n+1) for d in str(k)):
        s2 = s2[1:]+(s,)
        if s2 == s1:
            return i+1 # _Chai Wah Wu_, Feb 18 2022
