import sympy,itertools
def A337083(n):
  p=tuple(itertools.permutations(range(n)))
  m=len(p)
  q={p[i]:i for i in range(m)}
  Q=sympy.diag(*[n-1]*m)
  for i in range(m):
    for k in range(n-1):
      Q[i,q[p[i][:k]+tuple(reversed(p[i][k:k+2]))+p[i][k+2:]]]=-1
  return Q[:m-1,:m-1].det() # _Pontus von Br√∂mssen_, Jan 18 2021
