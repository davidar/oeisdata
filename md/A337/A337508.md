from sympy import nextprime, isprime
A337508_list, p = [], 11
while p < 10**6:
    s = str(p)
    l = len(s)//2
    if not (isprime(int(s[:l])) or isprime(int(s[-l:]))):
        A337508_list.append(p)
    p = nextprime(p) # _Chai Wah Wu_, Sep 14 2020
