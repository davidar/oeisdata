## Python

def A337212(n):
    x, y, k, r, m = (3**n-3)//2, (3**n-3)//2, (n-1)%3, 3**(n-1), 0
    while True:
        m += 1
        a, b = divmod(x,3)
        x, k = a+k*r, (k+k-b)%3
        if y == x:
            return m # _Chai Wah Wu_, Sep 14 2020
