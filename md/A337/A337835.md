## Python
```Python
from itertools import islice
from collections import Counter
def agen(): # generator of terms
    an, k = 0, Counter()
    while True:
        yield an
        k[an] += 1
        an = k[an] - an
print(list(islice(agen(), 86))) # _Michael S. Branicky_, Nov 12 2022
```
