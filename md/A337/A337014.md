## Python
```Python
from itertools import islice
from collections import Counter
def agen(): # generator of terms
    an, k, sign = 0, Counter(), -1
    while True:
        yield an
        k[an] += 1
        sign *= -1
        an = (k[an] - an)*sign
print(list(islice(agen(), 83))) # _Michael S. Branicky_, Nov 12 2022
```
