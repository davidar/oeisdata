from sympy import isprime, prime, primepi
a_1 = 1
a0 = 1
for n in range(1, 101):
    a = a_1 + a0
    m = primepi(a_1)
    b = 0
    while not isprime(a + b):
        m = m - 1
        b = b + prime(m)
    a = a + b
    print(a)
    a_1 = a0
    a0 = a
