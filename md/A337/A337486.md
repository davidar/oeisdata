## Python

from itertools import count, islice
def A337486_gen(): # generator of terms
    c = 1
    for n in count(1):
        a, b = divmod(c,n)
        if not b:
            c = a
            yield n
        else:
            c *= n
A337486_list = list(islice(A337486_gen(),30)) # _Chai Wah Wu_, Apr 11 2024
