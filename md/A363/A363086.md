## Python
```Python
from itertools import islice
from collections import Counter
def agen(): # generator of terms
    an, c = 1, Counter([1, 1])
    yield from [1, 1]
    while True:
        an = c[an]-an if c[an] < abs(an) else c[an]
        c[an] += 1
        yield an
print(list(islice(agen(), 80))) # _Michael S. Branicky_, May 19 2023
```
