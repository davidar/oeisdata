## Python
```Python
from itertools import count
from math import prod
from sympy import factorint
def A363285(n):
    m = 1<<n
    for r in count(1):
        s = prod((p**(e+1 if p&1 else e+n+1)-1)//(p-1) for p, e in factorint(r).items())*((m<<1)-1 if r&1 else 1)
        if abs(s-(r*m<<1))==m:
            return r*m # _Chai Wah Wu_, Jul 17 2023
```
