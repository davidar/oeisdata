from itertools import islice
def A363584_generator():
    alist = []
    while True:
        x = 1
        while True:
            for i,a in enumerate(alist):
                if a <= x <= 2*a and i <= a-1:
                    x = 2*a+1
                    break
                if x == a:
                    x += 1
                    break
            else: break
        yield x
        alist.insert(0,x)
def A363584_list(nterms):
    return list(islice(A363584_generator(),nterms)) # _Pontus von BrÃ¶mssen_, Aug 19 2023
