## Python
```Python
from functools import cache
@cache
def T(n: int, k: int) -> int:
    if k == 0: return 1
    if k == n:
        p = k % 2
        return p - sum(T(n, j) for j in range(p, n - 1, 2))
    return (T(n - 1, k) * n) // (n - k)
for n in range(10): print([T(n, k) for k in range(n + 1)])
```
