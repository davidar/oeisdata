## Python
```Python
from itertools import product
from fractions import Fraction
def is_A363802(n):
    s = [f"Fraction({d}, 1)" for d in str(n)]
    for ops in product("+-*/", repeat=len(s)-1):
        try: v = eval("".join(sum(zip(ops, s[1:]), (s[0],))))
        except: v = None
        if v == 10: return True
    return False
```
