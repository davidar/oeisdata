from sympy import sieve as prime
def A363544(n):
    if n == 0: return -1
    k = 2
    while (prime[k] + prime[k+1]) < 2*n and (prime[k] + prime[k+1]) // 2 != n: k += 1
    if (prime[k] + prime[k+1]) // 2 == n: return prime[k]
    k = 2
    while (prime[k+1] - prime[k]) // 2 != n: k += 1
    return prime[k]
print([A363544(n) for n in range(0,50)])
