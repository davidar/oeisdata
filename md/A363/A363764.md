## Python
```Python
from itertools import count,islice
from collections import defaultdict
def A363764_gen():
    x = 1
    freq = defaultdict(int)
    freq[x] = f0 = 1
    freqfreq = defaultdict(int)
    freqfreq[1] = 1
    for n in count(1):
        yield x
        x = n-f0*freqfreq[f0]
        freq[x] = f0 = freq[x]+1
        if f0 != 1: freqfreq[f0-1] -= 1
        freqfreq[f0] += 1
def A363764_list(nmax):
    return list(islice(A363764_gen(), nmax)) # _Pontus von Br√∂mssen_, Jul 01 2023 (after an idea by _Kevin Ryde_)
```
