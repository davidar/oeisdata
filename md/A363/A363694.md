```Python
# Inefficient but works
import sympy
m = 100
dict1 = {}
for n in range(1,m):
    edges = 0
    for i in sympy.primerange(n):
        for j in sympy.primerange(n):
            if i != j and i + j <= n:
                edges += 1
    dict1[n] = int(edges/2)
print(dict1.values())
```

from sympy import primepi, nextprime
def A363694(n):
    c, m, p = 0, 1, 2
    while p<<1 < n:
        c += primepi(n-p)-m
        p = nextprime(p)
        m += 1
    return c # _Chai Wah Wu_, Aug 05 2023
