## Python
```Python
# Gives 14 correct digits
from mpmath import stieltjes,fac
def limgen(n):
    terms = []
    for y in range(3, n):
        for x in range(y, n):
            terms.append((((-1)**y)*stieltjes(x)*(x-(y-1))**(y-2))/(fac(x-(y-2))*fac(y-2)))
    n,o_sum = 2,0
    while True:
        n_term = 1/((n-1)**(n+1))
        n_sum = o_sum + n_term
        if o_sum == n_sum:
            break
        o_sum = n_sum
        n += 1
    return sum(terms) + 0.5 - stieltjes(0) + n_sum
print(str(limgen(60))[:-1])
```
