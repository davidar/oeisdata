## Python
```Python
from sympy import divisor_sigma as s1
def bp(n): return (b:=bin(n)[2:])==b[::-1]
def ok(n): return n and bp(n) and bp(sum(map(int,str(n)))) and bp(s1(n)-n)
print([k for k in range(10**6) if ok(k)]) # _Michael S. Branicky_, Jul 24 2023
```
