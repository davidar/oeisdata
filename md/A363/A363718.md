def A363718_rowlist(root,row):
    A = [[root]]
    for i in range(0,row):
        A.append([])
        for j in range(0,len(A[i])):
            if A[i][j] != 1:
                A[i+1].append(A[i][j]-1)
            A[i+1].append(A[i][j]+1)
    return(A)
