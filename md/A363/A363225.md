## Python
```Python
from collections import Counter
from itertools import combinations_with_replacement
from sympy.utilities.iterables import partitions
def A363225(n): return sum(1 for p in partitions(n) if any(q[0]+q[1]==q[2] for q in combinations_with_replacement(sorted(Counter(p).elements()),3))) # _Chai Wah Wu_, Sep 21 2023
```
