## Python
```Python
from sympy import factorint
def A363761(n):
    for k in range(n>>1,3*n):
        c = 0
        for m in range(k**2+1,(k+1)**2):
            if all(p==2 or p&3==1 or e&1^1 for p, e in factorint(m).items()):
                c += 1
                if c>n:
                    break
        if c==n:
            return k
    return -1 # _Chai Wah Wu_, Jun 22 2023
```
