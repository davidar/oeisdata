## Python
```Python
from math import isqrt
from sympy import mobius, primepi
def A363597(n):
    if n==1: return 1
    def f(x): return n-1+sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1))-primepi(x)
    m, k = n-1, f(n-1)
    while m != k:
        m, k = k, f(k)
    return m # _Chai Wah Wu_, Aug 02 2024
```
