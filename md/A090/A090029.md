## Python
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def A090029(n):
    if n == 0:
        return 0
    c, j = 1, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*A090029(k1)
        j, k1 = j2, n//j2
    return (n+1)**7-c+127*(j-n-1) # _Chai Wah Wu_, Mar 30 2021
```
