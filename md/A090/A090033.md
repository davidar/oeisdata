# alst(), moves(), swap() in A089473
def T(j, k):  # chr(45) is '-'
    start, shape = "".join(chr(45+i) for i in range(j*k)), (j, k)
    return len(alst(start, shape))-1
for j in range(1, 5):
    for k in range(1, j+1):
        print(T(j,k), end=", ") # _Michael S. Branicky_, Aug 02 2021
