## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    m = 1
    for n in count(1):
        while not isprime(int(str(m)+str(n)+str(m))):
            m += 2
        yield m
        m += 2
print(list(islice(agen(), 46))) # _Michael S. Branicky_, Aug 10 2022
```
