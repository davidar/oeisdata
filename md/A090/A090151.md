## Python
```Python
from sympy import isprime
def afind(terms):
  print(3, end=", ")
  digits, n = 1, 1
  while n < terms:
    for id in "17":
      t = int('3' + id*digits)
      if isprime(t): print(t, end=", "); n += 1
    digits += 1
afind(12) # _Michael S. Branicky_, Mar 31 2021
```
