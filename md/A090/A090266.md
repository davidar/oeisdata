## Python
```Python
from sympy import prime, isprime
def a(n):
    spn = str(prime(n)); k = 1; sk = str(k)
    while not isprime(int(sk + spn + sk)): k += 1; sk = str(k)
    return k
print([a(n) for n in range(1, 97)]) # _Michael S. Branicky_, Jul 01 2021
```
