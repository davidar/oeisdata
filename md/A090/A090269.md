## Python
```Python
from sympy import prime, isprime
def a(n):
    if n == 5: return 0
    spn, digits = str(prime(n)), 1
    while True:
        for sk in [d*digits for d in "1379"]:
            if isprime(int(sk + spn + sk)): return int(sk)
        digits += 1
print([a(n) for n in range(1, 81)]) # _Michael S. Branicky_, Jul 01 2021
```
