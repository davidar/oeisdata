## Python

from sympy import isprime
def afind(limit):
  k = 5; pow2 = 2**(2*k-7)
  while k < limit:
    while not isprime(pow2 - 7) and k <= limit: k += 1; pow2 *= 4
    if k > limit: return
    print(k, end=", "); k += 1; pow2 *= 4
afind(2000) # _Michael S. Branicky_, Apr 23 2021
