from sympy import prime
def A138182(n):
    m, tlist = prime(n), [1,2]
    while tlist[-1]+tlist[-2] <= m:
        tlist.append(tlist[-1]+tlist[-2])
    for d in tlist[::-1]:
        if d == m:
            return d
        elif d < m:
            m -= d # _Chai Wah Wu_, Jun 14 2018
