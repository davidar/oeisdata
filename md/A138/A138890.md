## Python

def A138890(n):
    def f(x):
        if x<=1: return n+1
        a, b, c, d = 1, 1, 1, 0
        while c<=x:
            a, b, c = b, c, a+b
            d += 1
        return n+d-1
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m # _Chai Wah Wu_, Sep 10 2024
