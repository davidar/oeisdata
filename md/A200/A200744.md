## Python
```Python
from math import prod, factorial
from itertools import combinations
def A200744(n):
    m = factorial(n)
    return min((abs((p:=prod(d))-m//p),max(p,m//p)) for l in range(n,n//2,-1) for d in combinations(range(1,n+1),l))[1] # _Chai Wah Wu_, Apr 07 2022
```
