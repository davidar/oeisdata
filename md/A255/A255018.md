## Python
```Python
TOP = 6000000
a = [0]*TOP
for b in range(TOP):
  b3 = b**3
  if b3*3>=TOP: break
  for c in range(b,TOP):
    c3 = b3 + c**3
    if c3>=TOP: break
    for d in range(c,TOP):
      res = c3 + d**3
      if res>=TOP: break
      a[res] += 1
m = max(a)
r = [-1] * (m+1)
for i in range(TOP):
    if r[a[i]]==-1:  r[a[i]]=i
print(r)
```
