## Python
```Python
from itertools import count
from sympy import primerange
def A255901(n):
    for b in count(1):
        if n == sum(1 for p in primerange(2,b+1) if pow(b,p-1,p**2) == 1):
            return b # _Chai Wah Wu_, May 18 2022
```
