## Python
```Python
from __future__ import division
from sympy import isprime
def A241100(n):
    for i in range(1,10):
        x = i*(10**n-1)//9
        for j in range(n-1,-1,-1):
            for k in range(i,-1,-1):
                if j < n-1 or k < i:
                    y = x-k*(10**j)
                    if isprime(y):
                        return y
        for j in range(n):
            for k in range(1,9-i+1):
                y = x+k*(10**j)
                if isprime(y):
                    return y # _Chai Wah Wu_, Dec 29 2015
```
