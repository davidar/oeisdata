## Python
```Python
from sympy import isprime
def a(n):
    base = (10**n-1)
    for j in range(1, n+1):
        t = base - 8*10**(n-j)
        if isprime(t):
            return j
    return 0
print([a(n) for n in range(2, 78)]) # _Michael S. Branicky_, Jun 02 2024
```
