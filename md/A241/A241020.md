## Python
```Python
from sympy import isprime
def a(n):
    if (1+7*(n-1))%3 == 0:
        return 0
    base = (10**n-1)//9*7
    for j in range(1, n+1):
        t = base - 6*10**(n-j)
        if isprime(t):
            return j
    return 0
print([a(n) for n in range(2, 81)]) # _Michael S. Branicky_, Jun 02 2024
```
