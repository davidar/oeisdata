def nextpal(n,b): # returns the palindromic successor of n in base b
....m, pl = n+1, 0
....while m > 0:
........m, pl = m//b, pl+1
....if n+1 == b**pl:
........pl = pl+1
....n = (n//(b**(pl//2))+1)//(b**(pl%2))
....m = n
....while n > 0:
........m, n = m*b+n%b, n//b
....return m
n, a3, a5 = 0, 0, 0
