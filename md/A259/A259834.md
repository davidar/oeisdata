## Python
```Python
def A259834_list(len):
    L, u, x, y = [0], 1, 0, 0
    for n in range(len):
        y, x, u = x, x*n + u, -u
        L.append(y + 2*x)
    L[1] = 0
    return L
print(A259834_list(23)) # _Peter Luschny_, Oct 04 2017
```
