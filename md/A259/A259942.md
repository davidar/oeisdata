## Maple
```Maple
isA259942 := proc(n)
    local ifa;
    ifa := ifactors(n)[2] ;
    return (n >= add( op(2,p)*op(1,p)^3,p=ifa)) ;
end proc:
for n from 0 to 1000 do
    if isA259942(n) then
        printf("%d,",n);
    end if;
end do: # _R. J. Mathar_, Nov 27 2015
```

## Python
```Python
from sympy import factorint
for j in range(1, 1001):
    k =  factorint(j)
    it = k.keys()
    va = k.values()
    alfa = 0
    for l in range(0,len(k)):
        alfa = alfa + va[l]*(it[l]**3)
    if alfa <=j:
        print j
```
