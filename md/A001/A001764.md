def A001764_list(n) :
    D = [0]*(n+1); D[1] = 1
    R = []; b = false; h = 1
    for i in range(2*n) :
        for k in (1..h) : D[k] += D[k-1]
        if not b : R.append(D[h])
        else : h += 1
        b = not b
    return R
