A001359 := proc(n)
   option remember;
   if n = 1
      then 3;
   else
      p := nextprime(procname(n-1)) ;
      while not isprime(p+2) do
         p := nextprime(p) ;
      end do:
      p ;
   end if;
end proc: # _R. J. Mathar_, Sep 03 2011
