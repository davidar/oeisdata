```Maple
A001097 := proc(n)
    option remember;
    if n = 1 then
        3;
    else
        for a from procname(n-1)+1 do
            if isprime(a) and ( isprime(a-2) or isprime(a+2) ) then
                return a;
            end if;
        end do:
    end if;
end proc: # _R. J. Mathar_, Feb 19 2015
```

## Python
```Python
from sympy import nextprime
from itertools import islice
def agen(): # generator of terms
    yield 3
    p, q = 5, 7
    while True:
        if q - p == 2: yield from [p, q]
        p, q = q, nextprime(q)
print(list(islice(agen(), 58))) # _Michael S. Branicky_, Apr 30 2022
```
