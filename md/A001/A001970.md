from sympy.core.cache import cacheit
from sympy import npartitions, divisors
@cacheit
def a(n): return 1 if n == 0 else sum([sum([d*npartitions(d) for d in divisors(j)])*a(n - j) for j in range(1, n + 1)]) / n
