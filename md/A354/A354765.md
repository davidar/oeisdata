## Python
```Python
from math import gcd, lcm
from itertools import count, islice
from collections import deque
from sympy import primepi, primefactors
def A354765_gen(): # generator of terms
    aset, aqueue, c, b, f = {1}, deque([1]), 2, 1, True
    yield 0
    while True:
        for m in count(c):
            if m not in aset and gcd(m,b) == 1:
                yield sum(2**(primepi(p)-1) for p in primefactors(b))
                aset.add(m)
                aqueue.append(m)
                if f: aqueue.popleft()
                b = lcm(*aqueue)
                f = not f
                while c in aset:
                    c += 1
                break
A354765_list = list(islice(A354765_gen(),20)) # _Chai Wah Wu_, Jun 18 2022
```
