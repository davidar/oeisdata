## Python
```Python
from sympy import primefactors
def rad(num):
    primes = primefactors(num)
    value = 1
    for p in primes:
        value *= p
    return value
a = [1, 1]
for n in range(2, 1000):
    a += [(rad(31*a[n-1]) + rad(31*a[n-2])) // 31]
```
