def A354422(n):
    from sympy import isprime; ct = 0
    for m in range(1, 13):
        d_max = 31 if m in {1, 3, 5, 7, 8, 10, 12} else 30 if m in {4, 6, 9, 11} else 28 if (n%4 or (n%400 and not n%100)) else 29
        for d in range(1, d_max + 1, 2):
            if isprime(n*10000 + m*100 + d): ct += 1
    return ct
