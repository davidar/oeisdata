from itertools import count, islice
from math import prod, gcd, lcm
from collections import deque
from sympy import primefactors
def A354764_gen(): # generator of terms
    aset, aqueue, c, b, f = {1}, deque([1]), 2, 1, True
    yield 1
    while True:
        for m in count(c):
            if m not in aset and gcd(m,b) == 1:
                yield prod(primefactors(m))
                aset.add(m)
                aqueue.append(m)
                if f: aqueue.popleft()
                b = lcm(*aqueue)
                f = not f
                while c in aset:
                    c += 1
                break
A354764_list = list(islice(A354764_gen(),20)) # _Chai Wah Wu_, Jun 18 2022
