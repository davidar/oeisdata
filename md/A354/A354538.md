from itertools import count
def A354538(n):
    for k in count(1):
        c, m = 0, k
        while not (a:=divmod(m,3))[1]:
            c += 1
            m = a[0]
        if not (m==2 or m%3==1):
            c += 1
            m = (m+1)//3-2
            while (a:=divmod(m,3))[1]:
                c += 1
                m = a[0]
        if c == n: return k # _Chai Wah Wu_, Oct 15 2022
