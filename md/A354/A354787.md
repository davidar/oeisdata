## Python
```Python
from math import comb as binomial
def T(n, k):
    if k == 0: return k ** n
    if k == 1: return 1
    return sum(binomial(n - i - 1, k - 1) * binomial(i - 1, k // 2 - 1)
           * (-1) ** (i + k // 2) for i in range(k // 2, n - k + 1))
for n in range(11): print([T(n, k) for k in range(1 + (2 * n + 1) // 3)])
```
