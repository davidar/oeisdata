## Python
```Python
from sympy import divisors
def binwt(n): return bin(n).count("1")
def ok(n):
    binwts, divs = set(), 0
    for d in divisors(n, generator=True):
        b = binwt(d)
        if b in binwts: return False
        binwts.add(b)
        divs += 1
    return divs > 2
print([k for k in range(415) if ok(k)]) # _Michael S. Branicky_, Jun 04 2022
```
