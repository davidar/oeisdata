## Python
```Python
from math import gcd, prod
from itertools import combinations
from sympy import factorint
from sympy.ntheory.modular import crt
def A354989(n):
    if n == 1:
        return 0
    plist = tuple(p**q for p, q in factorint(n).items())
    return 0 if len(plist) == 1 else int(gcd(n,s:=int(min(min(crt((m, n//m), (0, -1))[0], crt((n//m, m), (0, -1))[0]) for m in (prod(d) for l in range(1, len(plist)//2+1) for d in combinations(plist, l))))) > gcd(n,s+1)) # _Chai Wah Wu_, Jun 16 2022
```
