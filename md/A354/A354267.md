## Python
```Python
from functools import cache
@cache
def A354267row(n):
    if n == 0: return [1]
    if n == 1: return [0, 1]
    row = A354267row(n - 1) + [1]
    s = row[1]
    for k in range(n-1, 0, -1):
        row[k] += row[k - 1]
    row[0] = s
    return row
for n in range(10): print(A354267row(n))
```
