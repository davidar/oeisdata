## Python
```Python
from functools import cache
@cache
def t(n, k, m):
    if k < 0 or n < 0: return 0
    if k == 0: return n ** k
    return m * t(n, k - 1, m) + t(n - 1, k, m + 1)
def A354794(n, k): return t(k - 1, n - k, n - k) if n != k else 1
for n in range(9): print([A354794(n, k) for k in range(n + 1)])
```
