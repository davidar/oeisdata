## Python
```Python
def ok(k, n):
    if k == n: return False
    sk, sn = str(k), str(n)
    return all(sk.count(d) >= sn.count(d) for d in set(sn))
def a(n):
    k = 0
    while not ok(k, n): k += 1
    return k
print([a(n) for n in range(71)]) # _Michael S. Branicky_, May 23 2022
```
