from math import factorial
from sympy import divisors
def A354862(n):
    f = factorial(n)
    return sum(f*(a := factorial(n//d))//(b:= factorial(d)) + (f*b//a if d**2 < n else 0) for d in divisors(n,generator=True) if d**2 <= n) # _Chai Wah Wu_, Jun 09 2022
