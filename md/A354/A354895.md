```Maple
N:= 100: # for a(1)..a(N)
H:= ListTools:-PartialSums([seq(1/i,i=1..2*N-1)]):
f:= n -> denom(binomial(2*n-1,n-1)*(H[2*n-1]-H[n-1])):
f(1):= 1:
map(f, [$1..N]); # _Robert Israel_, Jul 10 2023
```
