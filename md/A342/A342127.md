## Maple
```Maple
filter:= proc(n) local L,d,Lp,r,i;
  L:= convert(n,base,10);
  d:= nops(L);
  r:= add(L[-i]*10^(i-1),i=1..d);
  Lp:= convert(n*r,base,10);
  ormap(t -> Lp[t..t+d-1] = L, [$1..nops(Lp)+1-d])
end proc:
select(filter, [$0..120000]); # _Robert Israel_, Mar 24 2024
```
