## Python
```Python
from sympy import isprime, nextprime
def aupto(limit):
  p, q, num, alst = 2, 3, 7, []
  while  p <= limit:
    if num%3 == 0 and isprime(num//3): alst.append(p)
    p, q = q, nextprime(q)
    num = p**2 - p*q + q**2
  return alst
print(aupto(2819)) # _Michael S. Branicky_, Mar 18 2021
```
