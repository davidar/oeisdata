## Python
```Python
def a(n):
  b = bin(n)[2:]
  for k in range(len(b), -1, -1):
    for i in range(len(b)-k):
      for j in range(i+1, len(b)-k+1):
        if b[i:i+k] == b[j:j+k]: return k
print([a(n) for n in range(87)]) # _Michael S. Branicky_, Mar 07 2021
```
