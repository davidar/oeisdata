from itertools import count, islice
def agen():
    b = "1"
    yield 1
    for k in count(1):
        bk = bin(k)[2:]
        idx = b.rfind(bk)
        if idx == -1:
            yield 0
            b += "0"
        else:
            yield len(b) - idx
            b += bin(len(b) - idx)[2:]
print(list(islice(agen(), 79))) # _Michael S. Branicky_, Mar 18 2022
