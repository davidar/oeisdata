from sympy import primepi
def A342068(n):
    k, a, b, c = 2,0,primepi(n),primepi(2*n)
    while a+c <= 2*b:
        k += 1
        a, b, c = b, c, primepi(k*n)
    return k # _Chai Wah Wu_, Mar 25 2021
