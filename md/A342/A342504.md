## Python
```Python
from sympy import isprime, nextprime
def aupton(terms):
  alst, p, q, r, s = [], 2, 3, 5, 7
  while len(alst) < terms:
    if isprime((p*r + q*s)//2): alst.append(p)
    p, q, r, s = q, r, s, nextprime(s)
  return alst
print(aupton(54)) # _Michael S. Branicky_, Mar 14 2021
```
