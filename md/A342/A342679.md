import sympy
N=int(input())
A342679_n=[]
for n in range(2,N+1):
    n_0=n
    steps=0
    while not sympy.isprime(n) :
        exponents=list(sympy.factorint(n).values())
        m=""
        for i in exponents:
            m=m+str(i)
        n=int(m)
        if n==n_0:
            break
        steps+=1
    A342679_n.append(steps+1)
print(A342679_n)
