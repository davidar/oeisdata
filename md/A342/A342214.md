## Python

from sympy import isprime
def aupto(limit):
  c, t, alst = 6, 689, []
  while t < limit:
    t = int("".join(map(str, [c, c+2, c+3])))
    if isprime(c+1) and not isprime(c+3) and isprime(t): alst.append(t)
    c += [6, 4, 2, 2, 2][(c%10)//2]
  return alst
print(aupto(109610981099)) # _Michael S. Branicky_, Mar 05 2021
