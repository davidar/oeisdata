## Python
```Python
from sympy import primefactors
def aupton(terms):
  alst, aset = [1, 2], {1, 2}
  while len(alst) < terms:
    an = int("".join(map(str, primefactors(alst[-1]))))
    if an in aset:
      an = 1
      while an in aset: an += 1
    alst.append(an); aset.add(an)
  return alst[:terms]
print(aupton(100)) # _Michael S. Branicky_, Apr 02 2021
```
