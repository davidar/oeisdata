## Python
```Python
def agen():
  alst, aset = [0], {0}
  yield 0
  while True:
    an = 1
    while True:
      while an in aset: an += 1
      stran, t = str(an), str(alst[-1] + an)
      if len(stran) == len(set(stran)) and len(t) == len(set(t)):
        alst.append(an); aset.add(an); yield an; break
      an += 1
g = agen()
print([next(g) for n in range(77)]) # _Michael S. Branicky_, Mar 11 2021
```
