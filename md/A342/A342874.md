## Python
```Python
from sympy import nextprime
from itertools import count, islice
def agen():
    p, pdigs, q, qdigs = 2, ["2"], 3, ["3"]
    for k in count(1):
        if pdigs == qdigs: yield k
        p, q = q, nextprime(q)
        pdigs, qdigs = qdigs, sorted(str(q))
print(list(islice(agen(), 43))) # _Michael S. Branicky_, Nov 27 2022
```
