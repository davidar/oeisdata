## Python
```Python
def a(n):
  s, eset = str(n), set()
  for i in range(len(s)):
    for j in range(i+1, len(s)+1):
      if s[j-1] in "13579" and j-i < len(s): # odd and proper substring
        eset.add(int(s[i:j]))
  return len(eset)
print([a(n) for n in range(1, 105)]) # _Michael S. Branicky_, Mar 24 2021
```
