from numba import njit
@njit # works with 64 bits through a(14)
def digits0(n, b):
  count0 = 0
  while n >= b:
    n, r = divmod(n, b)
    count0 += (r==0)
  return count0 + (n==0)
from sympy import integer_nthroot
def a(n):
  an = integer_nthroot(n**n, 2)[0]
  while digits0(an*an, n) != n: an += 1
  return an
print([a(n) for n in range(2, 13)]) # _Michael S. Branicky_, Apr 07 2021

from itertools import product
from functools import reduce
from sympy.utilities.iterables import multiset_permutations
from sympy import integer_nthroot
def A342545(n):
    for a in range(1,n):
        p, q = integer_nthroot(a*n**n,2)
        if q: return p
    l = 1
    while True:
        cmax = n**(l+n+1)
        for a in range(1,n):
            c = cmax
            for b in product(range(1,n),repeat=l):
                for d in multiset_permutations((0,)*n+b):
                    p, q = integer_nthroot(reduce(lambda c, y: c*n+y, [a]+d),2)
                    if q: c = min(c,p)
            if c < cmax:
                return c
        l += 1 # _Chai Wah Wu_, Apr 07 2021
