## Python

from sympy import factorint
def A342025(n):
    f = factorint(n)
    return int(sum(b for a, b in f.items() if a % 4 == 3) == sum(b for a, b in f.items() if a % 4 == 1)) # _Chai Wah Wu_, Mar 09 2021
