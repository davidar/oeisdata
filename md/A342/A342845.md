## Python
```Python
def a(n):
  s, eset = str(n), set()
  for i in range(len(s)):
    for j in range(i+1, len(s)+1):
      if s[j-1] in "02468" and j-i < len(s): # even and proper substring
        eset.add(int(s[i:j]))
  return len(eset)
print([a(n) for n in range(105)]) # _Michael S. Branicky_, Mar 23 2021
```
