i=0
j=1
while j<=300:
    if bin(i)[2:].count("1")%2 == 0 == bin(i)[2:].count("0")%2:
        print(str(j)+" "+str(i))
        j+=1
    i+=1 # _Indranil Ghosh_, Feb 10 2017
