from sympy import isprime, nextprime
from itertools import islice
def agen(): # generator of terms
    p = 2
    while True:
        sp = str(p)
        if all(isprime(int(d+sp+d)) for d in "1379"):
            yield p
        p = nextprime(p)
print(list(islice(agen(), 40))) # _Michael S. Branicky_, Feb 23 2023
