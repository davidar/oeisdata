from sympy.ntheory.factor_ import digits
def bump(n,b):
  s=digits(n,b)[1:]
  l=len(s)
  return sum(s[i]*(b+1)**bump(l-i-1,b) for i in range(l) if s[i])
def A059936(n):
  for i in range(2,7):
    n=bump(n,i)-1
  return n # _Pontus von Br√∂mssen_, Sep 19 2020
