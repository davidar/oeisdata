def A059260_row(n):
    @cached_function
    def prec(n, k):
        if k==n: return 1
        if k==0: return 0
        return -prec(n-1,k-1)-sum(prec(n,k+i-1) for i in (2..n-k+1))
    return [(-1)^(n-k+1)*prec(n+1, n-k+1) for k in (1..n)]
