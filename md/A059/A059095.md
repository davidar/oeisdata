## Python
```Python
def b3(n):
    if n == 0: return []
    carry, trailing = [(0, 0), (0, 1), (1, -1)][n % 3]
    return b3(n//3 + carry) + [trailing]
t = []
for n in range(50):
    t += b3(n)
print(t)
# _Andrey Zabolotskiy_, Nov 10 2019
```
