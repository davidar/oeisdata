## Python
```Python
from sympy import isprime
from itertools import islice
def agen():
    an, bit, p = 2, 1, 0
    while True:
        while an&bit or not isprime(an+bit): bit <<= 1; p += 1
        yield p
        an, bit, p = an+bit, 1, 0
print(list(islice(agen(), 26))) # _Michael S. Branicky_, Oct 01 2022
```
