## Python

from sympy import isprime
A059758_list = []
for l in range(1,300):
    for a in '1379':
        for b in '0123456789':
            if a != b:
                p = int((a+b)*l+a)
                if isprime(p):
                    A059758_list.append(p) # _Chai Wah Wu_, Dec 21 2014
