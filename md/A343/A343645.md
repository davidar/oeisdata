## Python
```Python
def aupto(limit):
  f, alst = [0 for i in range(10)], []
  for n in range(1, limit+1):
    p = 1
    for d in list(map(int, str(n))): f[d] += 1; p *= f[d]
    alst.append(p)
  return alst
print(aupto(66)) # _Michael S. Branicky_, Apr 23 2021
```
