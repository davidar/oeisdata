## Python
```Python
from sympy import nextprime
def aupton(terms):
  alst = [1]
  for n in range(2, terms+1):
    alst.append(nextprime(alst[-1] + n) - alst[-1] - n)
  return alst
print(aupton(87)) # _Michael S. Branicky_, Apr 04 2021
```
