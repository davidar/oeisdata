## Python
```Python
from sympy import nextprime
def a(n):
    p = nextprime(10**n)
    q = nextprime(p); r = nextprime(q); s = nextprime(r); t = nextprime(s)
    while p < 10**(n+1):
        if t - p == 12: return p - 10**n
        p, q, r, s, t = q, r, s, t, nextprime(t)
    return 0
print([a(n) for n in range(14)]) # _Michael S. Branicky_, Jul 24 2021
```
