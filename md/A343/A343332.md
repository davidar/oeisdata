def A343332_list(n_max):
  a=0
  a_list=[0]
  count=[]
  for i in range(n_max-1):
    if a==len(count): count.append(0)
    else: count[a]+=1
    a=(a+count[a])//2
    a_list.append(a)
  return a_list
