## Python
```Python
from math import isqrt
def A343740(n):
    i = isqrt(n)
    if i**2 == n:
        return -1
    m, dset, pos = n, set(), 0
    for d in (int(s) for s in str(i)):
        dset.add(d)
        pos += 1
        if len(dset) == 10:
            return pos
    while len(dset) < 10:
        m *= 100
        d = isqrt(m) % 10
        dset.add(d)
        pos += 1
    return pos # _Chai Wah Wu_, Jul 07 2021
```
