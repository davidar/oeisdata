## Python
```Python
def aupto(limit):
  f, alst = [0 for i in range(10)], []
  for n in range(1, limit+1):
    s = 0
    for d in list(map(int, str(n))): f[d] += 1; s += f[d]
    alst.append(s)
  return alst
print(aupto(71)) # _Michael S. Branicky_, Apr 23 2021
```
