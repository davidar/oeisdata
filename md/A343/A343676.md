def f(w):
    for s in w:
        for t in range(int(s[-1])+1,10,2):
            yield s+str(t)
def g(w):
    for s in w:
        for t in range(1-int(s[-1])%2,int(s[-1]),2):
            yield s+str(t)
def A343676(n):
    c = 0
    for d in '123456789':
        x = d
        for i in range(1,n):
            x = g(x) if i % 2 else f(x)
        c += sum(1 for p in x if isprime(int(p)))
        if n > 1:
            y = d
            for i in range(1,n):
                y = f(y) if i % 2 else g(y)
            c += sum(1 for p in y if isprime(int(p)))
    return c
