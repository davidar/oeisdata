from sympy import isprime
def f(w):
    for s in w:
        for t in range(int(s[-1])+1,10,2):
            yield s+str(t)
def g(w):
    for s in w:
        for t in range(1-int(s[-1])%2,int(s[-1]),2):
            yield s+str(t)
def A343677(n):

        x = d

        c += sum(1 for p in x if isprime(int(p+p[-2::-1])))
