## Python
```Python
from sympy import S, isprime
pi_digits = str(S.Pi.n(10**5+1)).replace(".", "")[:-1]
def a(n):
    s, k = pi_digits[n-1], 1
    while not isprime(int(s)):
        s, k = s + pi_digits[n-1+k], k + 1
    return len(str(int(s)))
print([a(n) for n in range(1, 19)]) # _Michael S. Branicky_, Aug 21 2021
```
