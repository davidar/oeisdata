def ham(m, n):
  s, t = str(min(m, n)), str(max(m, n))
  s = '0'*(len(t)-len(s)) + s
  return sum(s[i] != t[i] for i in range(len(t)))
def aupton(terms):
  alst = [0]
  for n in range(2, terms+1):
    an = alst[-1] + 1
    while any(ham(an, alst[-i]) < 2 for i in range(1, len(alst)+1)): an += 1
    alst.append(an)
  return alst
print(aupton(66)) # _Michael S. Branicky_, Apr 15 2021
