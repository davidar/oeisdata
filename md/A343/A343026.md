```Python
#generates first x digits of sequence
def seq(x):
   a = []
   excluded = {}
   for n in range(x):
      if n == 0 or n == 1:
         a.append(1)
      else:
         m = 1
         while m in excluded[n]:
            m += 1
         a.append(m)
      for i in range(n):
         j = 2 * n - i
         k = a[i] * a[n]
         if j in excluded and k not in excluded[j]:
            excluded[j].append(k)
         else:
            excluded[j] = [k]
   return a
```
