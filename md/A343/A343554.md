## Python

def aupton(terms):
  n, alst, locs2 = 1, [0], dict()
  while n < terms:
    an = alst[-1]
    if an in locs2:
      if len(locs2[an]) > 1:
        i = min(locs2[an]); locs2[an].discard(i); anp1 = n - i
      else: anp1 = 0
    else: locs2[an] = set(); anp1 = 0
    alst.append(anp1); locs2[an].add(n); n += 1
  return alst
print(aupton(82)) # _Michael S. Branicky_, Apr 24 2021
