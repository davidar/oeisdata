## Python
```Python
from sympy import *
def b(n):
    f=factorint(n)
    l=sorted(f)
    return 1 if n==1 else int("".join(str(i)*f[i] for i in l))
# print([b(n) for n in range(1, 101)])
for n in range(1,200):
    if isprime(b(n)):
        print (n)
```
