## Python
```Python
from sympy import isprime, nextprime
def aupto(limit):
  p, q, r, s, alst = 2, 3, 5, 7, []
  while p <= limit:
    if isprime(p + q*(r+s)): alst.append(p)
    p, q, r, s = q, r, s, nextprime(s)
  return alst
print(aupto(2161)) # _Michael S. Branicky_, Apr 15 2021
```
