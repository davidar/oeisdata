## Python
```Python
def A343780(n):
    q = 1
    while True:
        s, c = [1]*n+[0]*n, 0
        for i in range(n):
            c = (c+q)%(2*n-i)
            if s[c]: break
            s = s[:c]+s[c+1:]
        else:
            return q+1
        q += 1 # _Chai Wah Wu_, Apr 30 2021
```
