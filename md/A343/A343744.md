## Python
```Python
def proddigit(n):
    p = 1
    while n > 0:
        n, p = n//10, p*(n%10)
    return p
n, a = 1, 1
while n > 0:
    aa, pa = a, proddigit(a)
    while pa > 1 and aa%pa == 0 and aa > 1:
        aa = aa//pa
        pa = proddigit(aa)
    if aa == 1:
        print(n,a)
        n = n+1
    a = a+1 # _A.H.M. Smeets_, Apr 29 2021
```
