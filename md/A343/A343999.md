from sympy.ntheory.modular import crt
from sympy import factorint
from itertools import product
def A343999(n):
    fs = factorint(2*n)
    plist = [p**fs[p] for p in fs]
    return int(min(k for k in (crt(plist,d)[0] for d in product([0,-1],repeat=len(plist))) if k > 0) % 2) # _Chai Wah Wu_, Jun 01 2021
