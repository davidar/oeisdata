N:= 50: # for a(0)..a(N)
V:= Array(0..N): count:=0: t:= numtheory:-tau(1):
for m from 1 while count < N+1 do
  s:= numtheory:-tau(m+1); v:= t - s;
  if v >= 0 and v <= N and V[v] = 0 then
    count:= count+1; V[v]:= m;
  fi;
  t:= s;
od:
convert(V,list); # _Robert Israel_, Jul 03 2024
