## Python
```Python
from sympy import isprime, nextprime
def aupto(limit):
  p, q, r, s, alst = 2, 3, 5, 7, []
  t = p + q*(r+s)
  while t <= limit:
    if isprime(t): alst.append(t)
    p, q, r, s = q, r, s, nextprime(s)
    t = p + q*(r+s)
  return alst
print(aupto(2067529)) # _Michael S. Branicky_, Apr 15 2021
```
