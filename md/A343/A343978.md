## Python
```Python
from labmath import mobius
def A343978(n): return sum(mobius(k)*(n//k)**6 for k in range(1, n+1))
```
```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def A343978(n):
    if n == 0:
        return 0
    c, j, k1 = 1, 2, n//2
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2-j)*A343978(k1)
        j, k1 = j2, n//j2
    return n*(n**5-1)-c+j # _Chai Wah Wu_, May 17 2021
```
