from functools import reduce
from operator import mul
from sympy import factorint
c, A343731_list = 0, [0]
for n in range(2,10**5):
    x = reduce(mul,(n*d+1 for d in factorint(n).values()))
    if x > c:
        c = x
        A343731_list.append(n) # _Chai Wah Wu_, Jun 03 2021
