def a(n):
    if n==0: return 0
    s=[]
    x=0
    while n>0:
        x=n%3
        n=n//3
        if x==2:
            x=-1
            n+=1
        s.append(x)
    l=s[::-1]
    return sum(l[i]*3**i for i in range(len(l)))
print([a(n) for n in range(101)]) # _Indranil Ghosh_, Jun 10 2017
