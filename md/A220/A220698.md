def rootTriangular(a):
    sr = 1<<33
    while a < sr*(sr+1)//2:
      sr>>=1
    b = sr>>1
    while b:
        s = sr+b
        if a >= s*(s+1)//2:
          sr = s
        b>>=1
    return sr
for i in range(1<<12):
