from sympy import isprime; P = [2, 3]; B = {2, 3, 5}
while P[-1] < 300:
    S = set()
    for i in B:
        if isprime(i) and i > P[-1]: P.append(i); break
    for j in B: S.add(j+P[-1])
    B = B|S
