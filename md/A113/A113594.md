from math import gcd
def A113594_list(nmax):
    A = [1,1]
    x,y = A[0],A[1]
    for n in range(3,nmax+1):
        z = (n-2)//2
        if n%2 != 0 and n > 4:
            x += A[z]
            y -= A[z]
        A.append(gcd(x,y))
        y += A[-1]
    return A # _John Tyler Rascoe_, Apr 07 2023
