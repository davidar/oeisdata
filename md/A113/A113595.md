## Python

def a(n):
    k, concat = 1, 12
    while concat <= n or concat%n:
        k += 1
        concat = int(str(k)+str(k+1))
    return concat
print([a(n) for n in range(1, 50)]) # _Michael S. Branicky_, Dec 12 2021
