## Python
```Python
def T(n): return n*(n+1)//2
def ok(n):
    if n < 10: return False
    s = str(n)
    splits = ((int(s[:i]), int(s[i:])) for i in range(1, len(s)))
    return any(n == T(x) + T(y) for x, y in splits)
print([k for k in range(10**6) if ok(k)]) # _Michael S. Branicky_, Jan 22 2022
```
