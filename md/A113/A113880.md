from itertools import count, islice
def A113880(): # generator of terms
    aset, an = {0, 1}, 1
    yield from [0, 1]
    for n in count(2):
        q, r = divmod(an, n)
        if r == 0 and q not in aset: an = q
        elif (d:=an-n) > 0 and d not in aset: an = d
        elif (s:=an+n) not in aset: an = s
        elif (m:=an*n) not in aset: an = m
        else: return
        yield an
        aset.add(an)
