## Python
```Python
from math import isqrt
from sympy import mobius
def A113850(n):
    def f(x): return int(n+x+1-sum(mobius(k)*(x//k**2) for k in range(1, isqrt(x)+1)))
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m**5 # _Chai Wah Wu_, Sep 13 2024
```
