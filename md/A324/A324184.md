from sympy import nextprime
def A324184(n):
    if n:
        c, p = 1, 1
        while n:
            c *= ((p:=nextprime(p))**(s:=(~n&n-1).bit_length()+1)-1)//(p-1)
            n >>= s
        return c*(p**(s+1)-1)//(p**s-1)
