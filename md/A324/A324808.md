b:= proc(n, i) option remember; `if`(n=0 and i=0, 1, `if`(i<1, 0,
      add(b(n-j, i-1)*binomial(n, j), j=0..min(8, n))))
    end:
a:= n-> b(n$2):
seq(a(n), n=0..20);  # _Alois P. Heinz_, Apr 01 2019
