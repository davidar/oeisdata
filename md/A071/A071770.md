## Python
```Python
from sympy.ntheory import digits
def a(n):
    d = digits(n, 3)[1:]
    return int(str(d[0]) + "".join(str((d[i]+d[i-1])%3) for i in range(1, len(d))), 3)
print([a(n) for n in range(75)]) # _Michael S. Branicky_, Aug 07 2024
```
