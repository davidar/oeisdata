from sympy import primepi
def A071904(n):
    if n == 1: return 9
    m, k = n, primepi(n) + n + (n>>1)
    while m != k:
        m, k = k, primepi(k) + n + (k>>1)
    return m # _Chai Wah Wu_, Jul 31 2024
