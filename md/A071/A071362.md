## Python
```Python
from sympy import nextprime
def inc(n):
  s = str(n); return len(s)==1 or all(d>pd for pd, d in zip(s[:-1], s[1:]))
def a(n):
  p = nextprime(10**(n-1))
  while not inc(p): p = nextprime(p)
  return p
for n in range(1, 9):
  print(a(n), end=", ") # _Michael S. Branicky_, Feb 13 2021
```
