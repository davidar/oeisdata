## Python

from math import isqrt
def A071177(n):
    m = 10*n
    if (k:=isqrt(m))**2!=m:
        a = 1
        while (k:=isqrt(a*(m+1)-1)+1)**2-m*a>=10*a:
            a *= 10
    return k # _Chai Wah Wu_, Feb 15 2023
