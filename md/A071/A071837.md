## Python
```Python
from sympy import factorint, isprime
A071837 = []
for n in range(1,10**5):
    f = factorint(n)
    fp, fe = list(f.keys()),list(f.values())
    if sum(fp) == sum(fe) and isprime(sum(fe)) and all([isprime(e) for e in fe]):
        A071837.append(n)
```
