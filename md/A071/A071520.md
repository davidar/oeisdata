from sympy import integer_log
def A071520(n):
    c = 0
    for i in range(integer_log(n,5)[0]+1):
        for j in range(integer_log(m:=n//5**i,3)[0]+1):
            c += (m//3**j).bit_length()
    return c # _Chai Wah Wu_, Sep 16 2024
