## Python
```Python
from itertools import count, islice
from sympy import isprime
def A071620_gen(): # generator of terms
    c, l = 0, 0
    for n in count(1):
        for d in str(n):
            c = 10*c+int(d)
            l += 1
            if isprime(c):
                yield l
A071620_list = list(islice(A071620_gen(),5)) # _Chai Wah Wu_, Feb 27 2023
```
