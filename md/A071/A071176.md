## Python
```Python
from math import isqrt
from sympy.ntheory.primetest import is_square
def A071176(n):
    m = 10*n
    if is_square(m): return 0
    a = 1
    while (k:=(isqrt(a*(m+1)-1)+1)**2-m*a)>=10*a:
        a *= 10
    return k # _Chai Wah Wu_, Feb 15 2023
```
