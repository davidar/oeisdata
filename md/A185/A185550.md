from sympy import integer_nthroot
def A185550(n):
    def f(x): return n+integer_nthroot(x**2,3)[0]
    def iterfun(f,n=0):
        m, k = n, f(n)
        while m != k: m, k = k, f(k)
        return m
    return iterfun(f,n) # _Chai Wah Wu_, Sep 09 2024
