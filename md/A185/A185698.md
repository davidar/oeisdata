## Python
```Python
from sympy import nextprime
def A185698(n):
    p, k, m = 2, 67**n, 10
    q, m2 = p % k, m % k
    while True:
        p = nextprime(p)
        while p >= m:
            m *= 10
            m2 = m % k
        q = (q*m2 + p) % k
        if q == 0:
            return p # _Chai Wah Wu_, May 01 2020
```
