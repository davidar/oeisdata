## Python
```Python
# WARNING: this prints numbers in the sequence, but not in increasing order.
def moda(n,a):
    kk = 0
    while n > 0:
        kk= kk+(n%10)**a
        n =int(n//10)
    return kk
def sod(n):
    kk = 0
    while n > 0:
        kk= kk+(n%10)
        n =int(n//10)
    return kk
for a in range (1, 10):
    for c in range (1, 10**8):
        if c==sod(c)**2*moda(c,a):
            print(c, end=",")
```
