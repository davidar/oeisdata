## Python
```Python
from sympy import factorial as f
def a007623(n, p=2): return n if n<p else a007623(n//p, p+1)*10 + n%p
def a(n):
    x=str(a007623(n))[1:][::-1]
    return sum(int(x[i])*f(i + 1) for i in range(len(x)))
print([a(n) for n in range(201)]) # _Indranil Ghosh_, Jun 21 2017
```
