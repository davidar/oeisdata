def a(n):
    s=[]
    l=[]
    x=0
    while n>0:
        x=n%3
        n//=3
        if x==2:
            x=-1
            n+=1
        s.append(x)
        l.append(-x)
    return [s, l]
print([-n for n in range(1, 25001) if a(n**2)[0].count(-1)==2*a(n)[1].count(-1) and a(n**2)[0].count(1)==2*a(n)[1].count(1) and a(n**2)[0].count(0)==2*a(n)[1].count(0)]) # _Indranil Ghosh_, Jun 07 2017
