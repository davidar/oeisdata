## Haskell
```Haskell
import Data.List (nub); import Data.List.Ordered (member)
a257860 n = a257860_list !! (n-1)
a257860_list = 1 : filter f [1..] where
   f x = any (\d -> member (x - q + d) $ ps d) $ filter (> 1) $ nub ds
         where q = sum ds; ds = (map (read . return) . show) x
   ps x = iterate (* x) (x ^ 2)
```
