## Python
```Python
import math
print("Start")
list =[]
for i in range(1, 100000):
   a = i*i
   b = str(a)
   l = len(b)
   for j in range(1, l):
      a_1 = b[:j]
      a_2 = b[j:]
      c = int(a_1)*int(a_1)+int(a_2)*int(a_2)
      sqrt_c = int(math.sqrt(int(c)))
      if (sqrt_c * sqrt_c == c) and (int(a_2[:1]) > 0):
         if not a in list:
            list.append(a)
print(list)
print("End")
```
