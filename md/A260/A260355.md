## Python
```Python
from itertools import permutations, combinations_with_replacement
def A260355(n,k): # compute T(n,k)
    if k == 1:
        return n*(n+1)//2
    ntuple, count = tuple(range(1,n+1)), n**(k+1)
    for s in combinations_with_replacement(permutations(ntuple,n),k-2):
        t = list(ntuple)
        for d in s:
            for i in range(n):
                t[i] *= d[i]
        t.sort()
        v = 0
        for i in range(n):
            v += (n-i)*t[i]
        if v < count:
            count = v
    return count
```
