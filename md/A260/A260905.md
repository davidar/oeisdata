## Python
```Python
from sympy import factorint, totient
def isBlum(n):
    fn = factorint(n)
    return len(fn) == sum(fn.values()) == 2 and all(f%4 == 3 for f in fn)
print([totient(k) for k in range(790) if isBlum(k)]) # _Michael S. Branicky_, Dec 20 2021
```
