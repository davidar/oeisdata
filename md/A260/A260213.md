## Python

def pod(n,m,a):
    kk = 1
    while n > 0:
        kk= kk*(n%m+a)
        n =int(n//m)
    return kk
for c in range (1,10000):
    for a in range (1,100):
        if c==pod(c,100,a):
            print (c)
