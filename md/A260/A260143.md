## Python
```Python
from sympy import factorint
def aupto(limit):
    aset, prevsig = {2}, [1]
    for k in range(3, limit+2):
        sig = sorted(factorint(k).values())
        if sig == prevsig: aset.update([k - 1, k])
        prevsig = sig
    return sorted(aset)
print(aupto(250)) # _Michael S. Branicky_, Sep 20 2021
```
