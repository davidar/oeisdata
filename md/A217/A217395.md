## Python
```Python
def agen():
  yield 3
  digits, adder = 1, 30
  while True:
    for i in range(10**digits): yield adder + i
    digits, adder = digits+1, adder*10
g = agen()
print([next(g) for i in range(54)]) # _Michael S. Branicky_, Mar 30 2021
```
