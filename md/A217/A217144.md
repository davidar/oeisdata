## Python
```Python
from itertools import accumulate, islice
def A217144_gen(): # generator of terms
    yield 1
    blist, b, c, f = (1,), 1, 1, 1
    while True:
        blist = list(accumulate(blist, initial=(b:=blist[-1])))
        yield (f:=-f)*(c := c+f*b**2)
A217144_list = list(islice(A217144_gen(),20)) # _Chai Wah Wu_, Jun 22 2022
```
