from itertools import accumulate, islice
def A217143_gen(): # generator of terms
    yield 1
    blist, b, c = (1,), 1, 1
    while True:
        blist = list(accumulate(blist, initial=(b:=blist[-1])))
        yield (c := c+b**2)
A217143_list = list(islice(A217143_gen(),20)) # _Chai Wah Wu_, Jun 22 2022
