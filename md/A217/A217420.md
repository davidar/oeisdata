with(numtheory):
b:= proc(n) option remember; `if`(n<=1, n,
      (add(add(d*b(d), d=divisors(j))*b(n-j), j=1..n-1))/(n-1))
    end:
a:= proc(n) option remember; (add(b(k)*b(n-1-k), k=0..n-1)-
      `if`(irem(n, 2, 'r')=1, b(r), 0))/2
    end:
seq(a(n), n=1..50); #  _Alois P. Heinz_, May 16 2013
