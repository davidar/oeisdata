def A052072(n):
    a, b, c = 0, 0, 0
    for i in range(10**9):
        s = str(c)
        for d in set(s):
            if s.count(d) != n:
                break
        else:
            return c
        c += a + b + 1
        b += 2*a + 3
        a += 3
    return 'search limit reached.'  # _Chai Wah Wu_, Aug 11 2015
