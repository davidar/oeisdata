## Python
```Python
from sympy import isprime
from itertools import combinations
def agen(): # generator of terms
    for d in range(1, 9):
        for c in combinations("123456789", d):
            if isprime(t:=int("".join(c))):
                yield t
print(list(agen())) # _Michael S. Branicky_, Dec 13 2023
```
