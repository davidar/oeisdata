```Python
    finalans.append(floor(ans*i))
```

```Python
from fractions import Fraction
from itertools import count, islice
def agen():
    Hn = 0
    for n in count(1):
        Hn += Fraction(1, n)
        yield (n*Hn.numerator)//Hn.denominator
print(list(islice(agen(), 60))) # _Michael S. Branicky_, Aug 10 2022
```
