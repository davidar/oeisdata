def A047749_list(n) :
    D = [0]*n; D[1] = 1
    R = []; b = False; h = 1
    for i in range(n) :
        for k in (1..h) :
            D[k] = D[k] + D[k-1]
        R.append(D[h])
        if b : h += 1
        b = not b
    return R
