## Python
```Python
from sympy import S, isprime
pi_digits = str(S.Pi.n(10**5))[2:-1]
def afind():
    kint = 0
    for k in range(len(pi_digits)):
        kint = 10*kint + int(pi_digits[k])
        if isprime(kint):
            print(k+1, end=", ")
afind() # _Michael S. Branicky_, Jan 29 2023
```
