## Python
```Python
def a(n: int) -> int:
    if n < 0: raise Exception("n must be a nonnegative integer")
    i = 0
    if n == 0: return i
    visited = {0}
    slab = 1
    while True:
        i += 1
        if slab == n: return i
        visited.add(slab)
        label = 1 + (slab >> 1)
        if not slab - label in visited:
            slab -= label  # jumping backwards
        else:
            slab += label  # jumping forwards
            if slab in visited: raise Exception(f"blocked at slab {slab}")
```
