## Python
```Python
def aupto(limit):
    c=[k**3 for k in range(1,limit) if k**3<=limit]
    s=set()
    for i in range(len(c)):
        for j in range(i+1,len(c)):
            t=(c[i]+c[j])
            for r in range(1, limit//t+1) :
                s.add(r*t)
    return(sorted(s))
print(aupto(500))
```
