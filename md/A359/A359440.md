import sympy
offset = 1
N = 100
l = []
for n in range(offset,N+1):
    j = 0
    first_sum = sympy.prime(n-j)+sympy.prime(n+j+1)
    while (n-j) > 1:
        j += 1
        sum = sympy.prime(n-j)+sympy.prime(n+j+1)
        if sum != first_sum:
            break
    l.append(max(0,j-1))
print(l)
