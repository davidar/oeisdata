## Python
```Python
def f(n): return 1 if n == 0 else (m:=3*n)^((1 << m.bit_length())-1)
def ok(n):
    i, fi, m = 0, n, n
    while fi != 0 and m <= n: i, fi, m = i+1, f(fi), max(m, fi)
    return m <= n
print([k for k in range(580) if ok(k)]) # _Michael S. Branicky_, Dec 20 2022
```
