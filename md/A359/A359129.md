from sympy import primepi, integer_nthroot
def A359129(n):
    def f(x): return int(n-2+x-sum(primepi(integer_nthroot(x,k)[0]) for k in range(1,x.bit_length())))
    kmin, kmax = 1,2
    while f(kmax) >= kmax:
        kmax <<= 1
    while True:
        kmid = kmax+kmin>>1
        if f(kmid) < kmid:
            kmax = kmid
        else:
            kmin = kmid
        if kmax-kmin <= 1:
            break
    return ((m:=kmax**2)*(m*(m*(m*(m*(m*(m*(m-1)+1)-2)+2)-2)+1)-1)+1)*m**6 # _Chai Wah Wu_, Aug 20 2024
