## Python
```Python
def f(n): return 1 if n == 0 else (m:=3*n)^((1 << m.bit_length())-1)
def a(n):
    i, fi, m, mi = 0, n, n, 0
    while fi != 0:
        i, fi = i+1, f(fi)
        if fi > m: m, mi = fi, i
    return mi
print([a(n) for n in range(64)]) # _Michael S. Branicky_, Dec 22 2022
```
