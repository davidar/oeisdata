```Python
def a(n: int) -> int:
    if n < 0: raise Exception("n must be a nonnegative integer")
    if n == 0: return 0
    visited = {0}
    slab = 1
    for i in range(1, n):
        label = 1 + (slab >> 1)
        if not slab - label in visited:
            slab -= label  # moving backwards
        else:
            slab += label  # moving forwards
            if slab in visited: raise Exception(f"blocked at slab {slab}")
        visited.add(slab)
    return slab
```
