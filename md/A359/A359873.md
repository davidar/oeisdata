## Python
```Python
def f(n):
    ans = 0
    for i in range(1, n):
        if n % i ==0:
            ans += i
    return ans
l=[]
for i in range(1, 100000):
    if f(i) - i > -1:
        l.append(i)
def listing(k, f):
    for i in range(len(l)-k):
        if l[i+k]-l[i] > f:
            return (l[i], l[i+k])
def a(n):
    a = 0
    while not listing(n, a) == None:
        a += 1
    return a
```
