import math
def is_valid_sequence_number(n):
    num_str = str(n)
    length = len(num_str)
    for count in range(math.ceil(length / 2), length):
        if num_str in str(
            math.perm(int(num_str[:count]), int(num_str[-(length - count) :]))
        ):
            return True
    return False
A359012 = []
for num in range(10, 10**4):
    if is_valid_sequence_number(num):
        A359012.append(num)
