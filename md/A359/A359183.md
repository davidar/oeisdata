from math import floor, log
def a(n):
    arr = []
    p = 0
    while True:
        for m in range(1, n):
            for b in range(2, max(3, n)):
                    k = m*b**p
                    if k in arr:
                        continue
                    arr.append(k)
                    q = 1
                    for b in range(3, n+1):
                        if floor(k/b**floor(log(k)/log(b))) != b-1:
                            q = 0
                            break
                    if q:
                        return k
        p += 1
