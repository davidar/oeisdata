## Python
```Python
from sympy import isprime
from itertools import  product
def a(n):
    c=3
    if n==1:return(c)
    x=[[1,7],[1,3,7,9],[3,9],'2468','02468']
    for k in range(2,n+1):
        for f in x[3]:
            for m in product(x[4], repeat=k-2):
                s = int(f+"".join(m))*10
                t=s%3
                for last in x[t]:
                    if isprime(s+last):
                        c+= 1
    return(c)
print([a(n) for n in range(1,7)])
```
