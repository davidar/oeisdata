from sympy import Matrix, I
def A359562(n): return Matrix(n,n,[(n+j-i if i>j else j-i-n) if i!=j else n*I for i in range(n) for j in range(n)]).per()*(1,-I,-1,I)[n&3] if n else 1 # _Chai Wah Wu_, Jan 25 2023
