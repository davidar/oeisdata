## Python
```Python
numbers = []
for i in range(2,880):
    numbers.append(i)
gaps = []
step = 4
current_step = 1
while current_step <= step:
    prime = numbers[0]
    new_numbers = []
    gaps = []
    gap = 0
    for i in range(1,len(numbers)):
        gap += 1
        if numbers[i] % prime != 0:
            new_numbers.append(numbers[i])
        else:
            gaps.append(gap)
            gap = 0
    current_step += 1
    numbers = new_numbers
print(gaps)
```
