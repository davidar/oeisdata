```Python
from functools import lru_cache
@lru_cache(maxsize=None)
def A359399(n):
    if n <= 1:
        return 1
    c, j = 0, 2
    k1 = n//j
    while k1 > 1:
        j2 = n//k1 + 1
        c += (j2*(j2-1)-j*(j-1)>>1)*A359399(k1)
        j, k1 = j2, n//j2
    return c+(n*(n+1)-(j-1)*j>>1) # _Chai Wah Wu_, Mar 31 2023
```
