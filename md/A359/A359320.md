## Python
```Python
from collections import Counter
def A359320(n):
    c = {0:1,1:1}
    for i in range(2,n+1):
        j, d = i**4, Counter(c)
        for k in c:
            d[k+j] += c[k]
        c = d
    return max(c.values()) # _Chai Wah Wu_, Jan 31 2024
```
