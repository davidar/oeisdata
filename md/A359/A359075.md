## Python
```Python
from sympy import divisors
def sod(n): return sum(map(int, str(n)))
def ok(n):
    s = set()
    for d in divisors(n, generator=True):
        sd = sod(d)
        if sd in s: return False
        s.add(sd)
    return True
print([k for k in range(1, 98) if ok(k)]) # _Michael S. Branicky_, Dec 15 2022
```
