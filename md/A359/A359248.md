## Python
```Python
from math import prod
from itertools import islice
def sp(n): d = list(map(int, str(n))); return sum(d) + prod(d)
def agen(increment=10**6):
    S, L, U, adict, rl, n = set(), 1, increment, dict(), 0, 1
    while True:
        S |= set(i + sp(i) for i in range(L, U))
        for i in range(L, U):
            if i in S: rl += 1
            else:
                if rl not in adict:
                    adict[rl] = i - rl
                    while n in adict: yield adict[n]; n += 1
                rl = 0
        S -= set(range(L, U))
        L, U = U, U + increment
print(list(islice(agen(), 34))) # _Michael S. Branicky_, Dec 23 2022
```
