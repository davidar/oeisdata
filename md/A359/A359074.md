## Python
```Python
from sympy import divisors
def sod(n): return sum(map(int, str(n)))
def ok(n):
    s = set()
    for d in divisors(n, generator=True):
        sd = sod(d)
        if sd in s: return True
        s.add(sd)
    return False
print([k for k in range(166) if ok(k)]) # _Michael S. Branicky_, Dec 15 2022
```
