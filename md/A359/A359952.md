## Python
```Python
from sympy import nextprime
from gmpy2 import powmod
max_n = 45
a = 2
seq = [a]
for i in range(2, max_n+1):
    p = 2
    while True:
        p_squared = p*p
        if powmod(a, p-1, p_squared) == 1 and (a-1) % p_squared != 0 and (a+1) % p_squared != 0:
            seq.append(p)
            a = p
            break
        else:
            p = nextprime(p)
print(seq)
```
