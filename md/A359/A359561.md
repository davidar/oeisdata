## Maple
```Maple
A359561 := proc(n)
    local T,c,r ;
    if n =0 then
        return 1 ;
    end if;
    T := Matrix(n,n) ;
    T[1,1] := n ;
    for c from 2 to n do
        T[1,c] := (n-c+1)*I ;
    end do:
    for r from 2 to n do
        for c from 1 to r-1 do
            T[r,c] := -T[c,r] ;
        end do:
        for c from r to n do
            T[r,c] := T[r-1,c-1] ;
        end do:
    end do:
    LinearAlgebra[Determinant](T) ;
    simplify(%) ;
end proc:
```
