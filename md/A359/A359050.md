def A359050(n):
    f, g, k = 0, 1, 0
    while f+g-n:
        k += 1
        m, a = k+1, [1,0]
        while m:
            a[m&1] = sum(a)
            m >>=1
        f, g = g, a[1]
    return k # _Chai Wah Wu_, Dec 16 2022
