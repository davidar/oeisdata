## Python
```Python
import networkx as nx
def F(n): return nx.DiGraph(((0,0),(0,1),(1,0))) if n == 1 else nx.line_graph(F(n-1))
def A359994(n): return nx.max_weight_clique(nx.complement(nx.Graph(F(n))),weight=None)[1]
```
