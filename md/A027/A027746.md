## Python
```Python
def factors(n: int) -> list[int]:
    p = n
    L:list[int] = []
    for f in range(2, p + 1):
        if f * f > p: break
        while True:
            q, r = divmod(p, f)
            if r != 0: break
            L.append(f)
            p = q
            if p == 1: return L
    L.append(p)
    return L  # _Peter Luschny_, Jul 18 2024
```
