## Maple
```Maple
A027384 := proc(n)
    local L,i,j ;
    L := {};
    for i from 0 to n do
        for j from i to n do
            L := L union {i*j};
        end do:
    end do:
    nops(L);
end proc: # _R. J. Mathar_, May 06 2016
```
