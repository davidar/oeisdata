## Maple
```Maple
A027427 := proc(n)
    local L, i, j ;
    L := {};
    for i from 0 to n do
        for j from i+1 to n do
            L := L union {i*j};
        end do:
    end do:
    nops(L);
end proc:  # _R. J. Mathar_, Jun 09 2016
```
