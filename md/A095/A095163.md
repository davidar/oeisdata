## Python
```Python
from sympy import divisors
from collections import Counter
from itertools import count, islice
def agen():
    c = Counter()
    for n in count(1):
        an = next(d for d in divisors(n) if c[d] < d)
        c[an] += 1
        yield an
print(list(islice(agen(), 83))) # _Michael S. Branicky_, Oct 14 2022
```
