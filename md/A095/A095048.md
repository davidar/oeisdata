## Python
```Python
from sympy import divisors
def a(n):
    s = set("1"+str(n))
    if len(s) == 10: return 10
    for d in divisors(n, generator=True):
        s |= set(str(d))
        if len(s) == 10: return 10
    return len(s)
print([a(n) for n in range(1, 99)]) # _Michael S. Branicky_, Nov 16 2022
```
