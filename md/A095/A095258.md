## Python
```Python
from itertools import islice
from sympy import divisors
def A095258_gen(): # generator of terms
    bset, s = {1}, 3
    yield 1
    while True:
        for d in divisors(s):
            if d not in bset:
                yield d
                bset.add(d)
                s += d
                break
A095258_list = list(islice(A095258_gen(),30)) # _Chai Wah Wu_, Jan 25 2022
```
