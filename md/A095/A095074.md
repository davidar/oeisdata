## Python
```Python
from sympy import isprime
i=1
j=1
while j<=25000:
    if isprime(i) and bin(i)[2:].count("0")<=bin(i)[2:].count("1"):
        print(str(j)+" "+str(i))
        j+=1
    i+=1 # _Indranil Ghosh_, Feb 03 2017
```
