## Python
```Python
from sympy import isprime
i = 1
j = 1
while j <= 2000:
    bi = bin(i)[2:]
    if isprime(i) and bi.count("1") > 1 + bi.count("0"):
        print(str(j) + " " + str(i))
        j += 1
    i += 1 # _Indranil Ghosh_, Feb 07 2017
```
