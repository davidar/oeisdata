## Python
```Python
from sympy import fibonacci, primerange
def a(n):
    k=0
    x=0
    while n>0:
        k=0
        while fibonacci(k)<=n: k+=1
        x+=10**(k - 3)
        n-=fibonacci(k - 1)
    return x
def ok(n): return str(a(n))[-3:]=="101"
print([n for n in primerange(1, 2001) if ok(n)]) # _Indranil Ghosh_, Jun 08 2017
```
