A095972 := proc(n)
    local a,q;
    a := 0 ;
    for q from 0 to n-1 do
        if numtheory[quadres](q,n) = -1 then
            a := a+1 ;
        end if;
    end do;
    a ;
end proc: # _R. J. Mathar_, Nov 05 2012
