from math import isqrt
def A095907_gen(): # generator of terms
    k = 1
    for n in count(3):
        m = (n+isqrt(5*n**2)>>1)
        for _ in range(m-k-1):
            yield 0
        yield 1
        k = m
A095907_list = list(islice(A095907_gen(),30)) # _Chai Wah Wu_, Aug 17 2022
