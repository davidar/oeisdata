## Python
```Python
from sympy import sign
from sympy.ntheory.factor_ import digits
def a(n): return sum([sign(sum(digits(5*k)[1:]) - sum(digits(k)[1:])) for k in range(n + 1)])
print([a(n) for n in range(51)]) # _Indranil Ghosh_, Aug 02 2017
```
