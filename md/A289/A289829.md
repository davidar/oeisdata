## Python

```Python
from __future__ import division
from sympy import divisors, isprime, prevprime, nextprime
A289829_list = []
for n in range(10**4):
    m = n**2-1
    for d in divisors(m):
        if d*d >= m:
            break
        r = m//d
        if not r % 2:
            r = r//2
            if not isprime(r):
                p, q = prevprime(r), nextprime(r)
                if m == (q-p)*(q+p):
                    A289829_list.append(n**2)
                    break # _Chai Wah Wu_, Jul 15 2017
```
