## Python
```Python
from sympy.ntheory.factor_ import digits
from sympy import sign
def a(n):
    d=digits(n, 3)[1:]
    return int(''.join(str(sign(i)) for i in d), 2)
print([a(n) for n in range(101)]) # _Indranil Ghosh_, Aug 03 2017
```
