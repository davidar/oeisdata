## Python

from sympy import binomial, factorial as f
def A(n, k): return f(n)*sum(binomial(n, i)*k**i/f(i) for i in range(n + 1))
for n in range(13): print([A(k, n - k) for k in range(n + 1)]) # _Indranil Ghosh_, Jun 28 2017
