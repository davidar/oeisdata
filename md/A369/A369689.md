## Python
```Python
from itertools import count
from sympy import sqrt_mod
def A369689(n):
    for j in count(0):
        b = 10**j
        a = b*10+1
        for k in sorted(sqrt_mod(n,a,all_roots=True)):
            m = (k**2-n)//a
            if m>0 and b <= m+n < a-1:
                return k # _Chai Wah Wu_, Feb 18 2024
```
