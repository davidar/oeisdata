def lis(n):
    table = [[0]*(n+1) for _ in range(n+1)]
    table[1][1] = 2
    for i in range(1, n+1) :
        table[i][0] = 3**(i-1)
    for i in range(1, n+1) :
        for j in range(1, n+1) :
            if (i == 1 and j == 1) :
                continue
            table[i][j] = table[i][j-1] + table[i-1][j] + 3*table[i-1][j-1]
    return [int(table[i][i]) for i in range(1, n+1)]
