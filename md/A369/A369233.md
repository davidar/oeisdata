## Python
```Python
from itertools import count
from sympy.ntheory.factor_ import digits
def A369233(n):
    m = 1<<n
    return next(b for b in count(2) if (s := digits(m,b)[1:])[:(t:=len(s)+1>>1)]==s[:-t-1:-1]) # _Chai Wah Wu_, Jan 17 2024
```
