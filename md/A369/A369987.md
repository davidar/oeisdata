## Python
```Python
from collections import Counter
def A369987(n):
    c = {0:1}
    for k in range(1,n+1):
        m, b = k**3, Counter(c)
        for j in c:
            b[j+m] -= c[j]
        c = b
    return max(map(abs,c.values())) # _Chai Wah Wu_, Feb 07 2024
```
