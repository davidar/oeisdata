## Python
```Python
from collections import Counter
def A369879(n):
    c = {0:1}
    for k in range(1,n+1):
        d = Counter(c)
        for j in c:
            d[j+k] -= c[j]
        c = d
    return len(set(c.values()))+int(max(c)+1>len(c)) # _Chai Wah Wu_, Feb 04 2024
```
