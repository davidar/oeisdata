from math import isqrt
fib = [1,1]
terms = []
while len(terms) < 1000:
    deg = 0
    target = 0
    test = 0
    while target == test:
        target = (10**deg+isqrt(5*10**(2*deg)))//2
        test = (10**deg*(fib[-1]))//fib[-2]
        deg += 1
    terms.append(deg-1)
    fib.append(fib[-1]+fib[-2])
print(terms)
