## Python

import sympy # for integer_nthroot (A048766), multiplicity (A122840)
def A369860(n: int):
    while True:
        C = sympy.integer_nthroot(n, 3)[0]; L = A055642(n-C**3)
        if sympy.multiplicity(10, C)*3 == L: return C//10**(L//3)
        n += C**3 * (10**L - 1)
