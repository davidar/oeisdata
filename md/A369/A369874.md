## Python
```Python
from collections import Counter
from sympy import divisors
def A369874(n):
    c = {0:1}
    for d in divisors(n,generator=True):
        b = Counter(c)
        for j in c:
            a = c[j]
            b[j+d] += a
            b[j-d] += a
        c = b
    return c[0] # _Chai Wah Wu_, Feb 05 2024
```
