## Python
```Python
from collections import Counter
def A369767(n):
    c = {j:1 for j in range(n+1)}
    for i in range(2,n+1):
        d = Counter()
        for k in c:
            for j in range(0,i*n+1,i):
                d[j+k] += c[k]
        c = d
    return max(c.values()) # _Chai Wah Wu_, Jan 31 2024
```
