## Python
```Python
from itertools import islice
def agen(): # generator of terms
    an, digits = 10, [0]
    while True:
        yield an
        an = 9 + digits.pop(0)
        digits += list(map(int, str(an)))
print(list(islice(agen(), 84))) # _Michael S. Branicky_, Jan 27 2024
```
