## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    an, digits = 10, [0]
    for n in count(2):
        yield an
        an = n + 8 + digits.pop(0)
        digits += list(map(int, str(an)))
print(list(islice(agen(), 79))) # _Michael S. Branicky_, Jan 27 2024
```
