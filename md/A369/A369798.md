## Python
```Python
from itertools import islice
from collections import deque
def agen(): # generator of terms
    S = deque([24])
    yield from [0, 1, 12]
    while True:
        an = S.popleft()
        yield an
        S.extend(an*d for d in map(int, str(an)))
print(list(islice(agen(), 54))) # _Michael S. Branicky_, Feb 01 2024
```
