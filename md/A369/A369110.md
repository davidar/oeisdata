from sympy import divisors
def A369110(n):
    c = {n}
    while n<4 or n>5:
        c.add(n:=(d:=divisors(n))[((l:=len(d))-1)>>1]+d[l>>1])
    if n==5:
        c.add(6)
    return len(c) # _Chai Wah Wu_, Apr 25 2024
