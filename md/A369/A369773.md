## Python
```Python
from collections import Counter
def A369773(n):
    c = {0:1}
    for k in range(1,n+1):
        d = Counter(c)
        for j in c:
            a = c[j]
            for i in range(1,k+1):
                d[j+i] += (a if i&1 else -a)
        c = d
    return max(c.values()) # _Chai Wah Wu_, Feb 01 2024
```
