from itertools import count, islice
from collections import Counter
def A369628_gen(): # generator of terms
    ccount = Counter({0:1})
    yield 0
    for i in count(1):
        bcount = Counter(ccount)
        for a in ccount:
            bcount[a+i] += ccount[a]
            bcount[a-i] += ccount[a]
        ccount = bcount
        yield(ccount[1])
A369628_list = list(islice(A369628_gen(),20)) # _Chai Wah Wu_, Jan 29 2024
