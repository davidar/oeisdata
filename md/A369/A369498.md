## Python
```Python
from math import isqrt
def A369498_list(n):
    return sorted([
        a**2 + b**2
        for a in range(1, isqrt(n) + 1)
        for b in range(1, a)
        for c in range(1, isqrt(n) + 1)
        for d in range(1, c)
        if a != c and a != d
        and a**2 + b**2 == c**2 + d**2
        and a + b == 3 * (c - d)
        and a**2 + b**2 <= n
    ])
print(A369498_list(18500))
```
