## Python
```Python
from itertools import count
def A193994(n):
    a = b = (0,)*4+(1 % n,)
    c, s = 0, 1 % n
    for m in count(1):
        b, s = b[1:] + (s,), (s+s-b[0])% n
        c += int(s==0)
        if a == b:
            return c # _Chai Wah Wu_, Feb 22-27 2022
```
