## Maple
```Maple
S:= NULL:
for x from 2 to 3^10 do
   L:= convert(x, base, 3):
   if numboccur(1,L) mod 3 <> 2 then next fi;
   L1:= subs(2=3,L);
   L2:= subs(1=2,L1);
   for LL in [L1,L2] do
     y:= add(LL[i]*10^(i-1), i=1..nops(L1));
     if isprime(y) then
      good:= true;
      for j from 1 to nops(LL) do
         yp:= y + 2*LL[j]*10^(j-1);
         if not isprime(yp) then
            good:= false;
            break
         fi
      od:
      if good then S:= S, y fi;
     fi;
   od
od:
sort([S]); # _Robert Israel_, Mar 07 2016
```
