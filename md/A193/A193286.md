## Haskell

```Haskell
-- See Theorem 5 in John Derbyshire link.
a193286 n = p n [] where
   p 0 ks       = product ks
   p n []       = p (n-1) [1]
   p n (k:ks)
    | n < 0     = 0
    | otherwise = max (p (n-1) ((k+1):ks)) (p (n-3) (1:k:ks))
-- _Reinhard Zumkeller_, Jul 22 2011, Jul 21 2011
```

## Python
```Python
def a(n):
    if n<8: return n
    elif n==8: return 9
    elif n>8 and n<=27:
        c=max(1, ((n + 3)//6))
        r=(n + 2)%c
        q=((n + 2)//c) - 2
        return q**(c - r)*(q + 1)**r
    else:
        q=((n + 2)//6)
        r=(n + 2)%6
        return 4**(q - r)*5**r
print([a(n) for n in range(1, 101)]) # _Indranil Ghosh_, Jun 27 2017, after Mathematica code
```
