## Python
```Python
from fractions import Fraction
def aupton(nn):
  Hnm1, alst = Fraction(1, 1), []
  for n in range(2, nn+1):
    Hn = Hnm1 + Fraction(1, n)
    alst.append((Hn/Hnm1).denominator)
    Hnm1 = Hn
  return alst
print(aupton(30)) # _Michael S. Branicky_, Feb 09 2021
```
