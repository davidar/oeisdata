def A193434(n):
    a = 1<<(m:=n+1).bit_length()-2
    s = bin(a|(m&a-1))[2:]
    return 45*int(s+(s[::-1] if a&m else s[-2::-1])) # _Chai Wah Wu_, Jul 23 2024
