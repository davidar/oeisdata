def a(n):
    c=(n + 6)//8
    if n<25:
        if n<10: return n
        r=(n + 3)%c
        m=(n + 3 - 3*c)//c
        return m**(c - r)*(m + 1)**r
    elif n==25: return 256
    else:
        r=(n + 6)%8
        b=max(0, 3 - r)
        d=max(0, r - 3)
        return 4**b*5**(c - (b + d))*6**d
print([a(n) for n in range(1, 101)]) # _Indranil Ghosh_, Jun 27 2017
