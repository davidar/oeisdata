```Maple
isA097393 := proc(n)
    local R ;
    R := digrev(n) ;
    if R <> n then
        if numtheory[bigomega](R) = 2 and numtheory[bigomega](n) = 2 then
            return true;
        else
            false;
        end if;
      else
        false;
    end if;
end proc:
for n from 1 to 500 do
    if isA097393(n) then
        printf("%d,",n) ;
    end if;
end do: # _R. J. Mathar_, Apr 05 2012
```
