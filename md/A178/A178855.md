from itertools import islice
from collections import deque
def A178855_gen(): # generator of terms
    aqueue, f, b, a = deque([2]), True, 1, 2
    while True:
        a += b
        aqueue.append(a)
        if f:
            yield (a-1)//2
            b = aqueue.popleft()
        f = not f
