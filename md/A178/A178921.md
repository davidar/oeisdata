## Python
```Python
from sympy import primerange
primes = list(primerange(2,500))
for n in range(1,100):
    d = n
    prev = 0
    product = 1
    for p in primes:
        if d%p==0:
            if prev:
                product *= p-prev
            while d%p==0:
                d//=p
            if d==1:
                break
            prev = p
    if prev==0:
        product = 0
    print(product, end=',')
```
