## Python

from itertools import count, islice, product
from sympy import isprime
def A178316_gen():
    yield from (2,5)
    r = ''.maketrans('69','96')
    for l in count(1):
        for a in '125689':
            for d in product('0125689',repeat=l):
                s = a+''.join(d)
                m = int(s)
                if isprime(m) and isprime(int(s[::-1].translate(r))):
                    yield m
A178316_list = list(islice(A178316_gen(),40)) # _Chai Wah Wu_, Apr 09 2024
