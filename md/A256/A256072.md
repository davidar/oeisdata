## Python
```Python
import sympy
from sympy import isprime
TOP = 10000
a = [0]*TOP
for y in range(2, TOP//2):
  for x in range(y, TOP//2):
    k = x*y + x + y
    if k>=TOP: break
    a[k]+=1
print([n for n in range(TOP) if a[n]==0 and isprime(n)])
```
