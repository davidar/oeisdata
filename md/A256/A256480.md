from gmpy2 import digits, mpz, is_prime
def A256480(n,limit=2000):
....sn = str(n)
....if not (n % 2 and n % 5):
........return 0
....for i in range(1,limit+1):
........for j in range(1,10):
............si = digits(j,10)*i
............p = mpz(si+sn)
............if is_prime(p):
................return int(p)
....else:
........return 'search limit reached.'
