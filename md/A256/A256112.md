def dgen(n,b):
    if n == 1:
        t = list(range(b))
        for i in range(1,b):
            u = list(t)
            u.remove(i)
            yield i, u
    else:
        for d, v in dgen(n-1,b):
            for g in v:
                k = d*b+g
                if not k % n:
                    u = list(v)
                    u.remove(g)
                    yield k, u
