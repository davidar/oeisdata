## Python
```Python
from sympy import composite
from sympy.ntheory.residue_ntheory import nthroot_mod
def A256517(n):
    z = nthroot_mod(1,(c := composite(n))-1,c**2,True)
    return int(z[0]+c**2 if len(z) == 1 else z[1]) # _Chai Wah Wu_, May 18 2022
```
