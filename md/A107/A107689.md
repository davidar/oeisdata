## Python
```Python
from sympy import isprime
def agen():
  digits = 0
  while True:
    for i in range(digits+1):
      t = int("1"*(digits-i) + "3" + "1"*i)
      if isprime(t): yield t
    digits += 1
g = agen()
print([next(g) for i in range(25)]) # _Michael S. Branicky_, Mar 13 2021
```
