## Python
```Python
from sympy import isprime
def auptod(maxdigits):
    alst = []
    for d in range(1, maxdigits+1):
        if d%3 == 0: continue
        for i in range(d):
            t = int('1'*(d-1-i) + '7' + '1'*i)
            if isprime(t): alst.append(t)
    return alst
print(auptod(20))  # _Michael S. Branicky_, Jul 12 2021
```
