## Python
```Python
w = [[[(0, 0)]]]
for n in range(1, 15):
    nw = []
    for walk in w[-1]:
        (x, t) = walk[-1]
        nss = [(x-1, t), (x+1, t), (x, (t+1)%3), (x, (t-1)%3)]
        for ns in nss:
            if ns not in walk:
                nw.append(walk[:] + [ns])
    w.append(nw)
print([len(x) for x in w])
```
