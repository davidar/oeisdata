## Python
```Python
from sympy import isprime
from itertools import count, islice
def agen(): # generator of terms
    an = 0
    while True:
        an = next(k for k in count(an+1) if isprime(k + int(str(k)[0])))
        yield an
print(list(islice(agen(), 58))) # _Michael S. Branicky_, Oct 03 2024
```
