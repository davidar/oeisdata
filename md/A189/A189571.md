## Python
```Python
from sympy import isprime, nextprime
def aupto(limit):
  plst, alst = [3, 5, 7, 11, 13, 17, 19, 23, 29], []
  while plst[0] <= limit:
    if isprime(sum(plst)): alst.append(plst[0])
    plst = plst[1:] + [nextprime(plst[-1])]
  return alst
print(aupto(1200)) # _Michael S. Branicky_, Mar 29 2021
```
