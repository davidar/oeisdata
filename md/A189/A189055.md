## Python
```Python
from itertools import count
primes = Primes()
def diff_pow(i, n):
  return (i+1)**n - i**n
def diff_pow_primes(n, limit=range(1000)):
  pows = [diff_pow(i, n) for i in limit]
  return [p for p in pows if p in primes]
diff_pow_primes(11, count())
```
