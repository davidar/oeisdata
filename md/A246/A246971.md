## Python
```Python
from itertools import combinations
A246971_list = []
for n in range(10):
    for k in range(n, -1, -1):
        c, d0 = 0, ['0']*(n+k)
        for x in combinations(range(n+k), n):
            d = list(d0)
            for i in x:
                d[i] = '1'
            if not '0100010' in ''.join(d):
                c += 1
        A246971_list.append(c) # _Chai Wah Wu_, Sep 12 2014
```
