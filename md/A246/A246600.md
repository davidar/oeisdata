## Maple
```Maple
A246600:=proc(n)
    local a, d, s, t, i, sw;
    a:=0;
    s:=convert(n, base, 2);
    for d in numtheory[divisors](n) do
        sw:= false;
        t:=convert(d, base, 2);
        for i from 1 to nops(t) do
            if t[i]>s[i] then
                sw:= true;
            fi;
        od:
        if not sw then
            a:=a+1;
        fi;
    od;
    a;
end;
seq(A246600(n), n=1..100);
```
