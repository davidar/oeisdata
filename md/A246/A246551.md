## Python
```Python
from sympy import primepi, integer_nthroot
def A246551(n):
    def f(x): return int(n-1+x-sum(primepi(integer_nthroot(x,k)[0])for k in range(1,x.bit_length(),2)))
    kmin, kmax = 1,2
    while f(kmax) >= kmax:
        kmax <<= 1
    while True:
        kmid = kmax+kmin>>1
        if f(kmid) < kmid:
            kmax = kmid
        else:
            kmin = kmid
        if kmax-kmin <= 1:
            break
    return kmax # _Chai Wah Wu_, Aug 13 2024
```
