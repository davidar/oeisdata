from itertools import combinations
def A246593(n):
....if n <= 1:
........return n
....else:
........s, y = bin(n)[2:], n
........for i in combinations(range(len(s)),2):
............s2 = int(s[:i[0]]+s[i[1]]+s[i[0]+1:i[1]]+s[i[0]]+s[i[1]+1:],2)
............if s2 > y:
................y = s2
........return y
