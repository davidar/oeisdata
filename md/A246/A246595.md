   if out1 = 1 and t1[i] = 1 then out1:=0; c:=c+1;
   elif out1 = 0 and t1[i] = 1 then c:=c+1;
   elif out1 = 1 and t1[i] = 0 then c:=c;
   elif out1 = 0 and t1[i] = 0 then lis:=[c, op(lis)]; out1:=1; c:=0;
   fi;
   if i = L1 and c>0 then lis:=[c, op(lis)]; fi;

ans:=[op(ans), a];
od:
