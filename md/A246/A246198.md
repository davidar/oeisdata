## Maple
```Maple
filter:= proc(n) local L,s,t,nL,B,j,k;
   L:= numtheory:-divisors(n) minus {n};
   s:= convert(L,`+`);
   if s::odd then return false fi;
   t:= s/2;
   nL:= nops(L);
   B:= Array(0..t,1..nL);
   B[0,1]:= 1;
   B[L[1],1]:= 1;
   for j from 2 to nL do
      B[..,j]:= B[..,j-1];
      for k from L[j] to t do
         B[k,j]:= B[k,j] + B[k-L[j],j-1]
      od:
      if B[t,j] > 0 then return true fi;
   od:
   false
end:
select(filter, [$2..300]); # _Robert Israel_, Aug 19 2014
```

## Mathematica
```Mathematica
filterQ[n_] := Module[{L, s, t, nL, B, j, k},
  L = Most[Divisors[n]];
  s = Total[L];
  If[OddQ[s], Return[False]];
  t = s/2;
  nL = Length[L];
  B[_, _] = 0;
  B[0, 1] = 1;
  B[L[[1]], 1] = 1;
  For[j = 2, j <= nL, j++,
    Do[B[k, j] = B[k, j-1], {k, 0, t}];
    For[k = L[[j]], k <= t, k++,
      B[k, j] = B[k, j] + B[k-L[[j]], j-1]
    ];
    If[ B[t, j] > 0, Return[True]];
  ];
  False
];
Select[Range[2, 300], filterQ] (* _Jean-FranÃ§ois Alcover_, Mar 04 2019, after _Robert Israel_ *)
```
