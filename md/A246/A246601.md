## Maple
```Maple
with(numtheory);
sd:=proc(n) local a,d,s,t,i,sw;
s:=convert(n,base,2);
a:=0;
for d in divisors(n) do
sw:=-1;
t:=convert(d,base,2);
for i from 1 to nops(t) do if t[i]>s[i] then sw:=1; fi; od:
if sw<0 then a:=a+d; fi;
od;
a;
end;
[seq(sd(n),n=1..100)];
```
