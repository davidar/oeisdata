def A246016(n):
    a, b = 0, 0
    for i, j in enumerate(bin(n)[:1:-1]):
        if int(j):
            a ^= (i&1)^b
            b ^= 1
    return -1 if a else 1 # _Chai Wah Wu_, Jul 26 2023
