from itertools import combinations
def A246591(n):
....if n <= 1:
........return n
....else:
........s = bin(n)[2:]
........l = len(s)
........y = 2**l-1
........for i in combinations(range(l),2):
............s2 = int(s[:i[0]]+s[i[1]]+s[i[0]+1:i[1]]+s[i[0]]+s[i[1]+1:],2)
............if s2 < y:
................y = s2
........return y
