```Python
from sympy import primerange
m = 10**5
A246655 = []
for p in primerange(1,m):
    pe = p
    while pe < m:
        A246655.append(pe)
        pe *= p
A246655 = sorted(A246655) # _Chai Wah Wu_, Sep 04 2014
```

```Python
from sympy import primepi, integer_nthroot
def A246655(n):
    def f(x): return int(n-1+x-sum(primepi(integer_nthroot(x,k)[0]) for k in range(1,x.bit_length())))
    kmin, kmax = 1,2
    while f(kmax) >= kmax:
        kmax <<= 1
    while True:
        kmid = kmax+kmin>>1
        if f(kmid) < kmid:
            kmax = kmid
        else:
            kmin = kmid
        if kmax-kmin <= 1:
            break
    return kmax # _Chai Wah Wu_, Aug 20 2024
```
