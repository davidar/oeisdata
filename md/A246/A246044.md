## Python
```Python
from itertools import permutations
from sympy import prime, isprime
A246044 = []
for n in range(1,10**6):
    p = prime(n)
    for x in permutations(str(p)):
        if x[0] != '0':
            p2 = int(''.join(x))
            if p2 != p and isprime(p2):
                break
    else:
        A246044.append(p) # _Chai Wah Wu_, Aug 27 2014
```
