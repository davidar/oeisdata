def dgen(n, b):
    if n == 1:
        t = list(range(b))
        for i in range(1, b):
            u = list(t)
            u.remove(i)
            yield i, u
    else:
        for d, v in dgen(n-1, b):
            for g in v:
                k = d*b+g
                if not k % n:
                    u = list(v)
                    u.remove(g)
                    yield k, u
print([a for n in range(2, 15, 2) for a, b in dgen(n, n)]) # Chai Wah Wu, Jun 07 2015
