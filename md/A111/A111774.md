## Python

from sympy import primepi
def A111774(n):
    def f(x): return int(n+(0 if x<=1 else primepi(x)-1)+x.bit_length())
    m, k = n, f(n)
    while m != k: m, k = k, f(k)
    return m # _Chai Wah Wu_, Sep 19 2024
