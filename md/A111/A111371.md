## Python
```Python
from sympy import factorint
def aupto(limit):
  alst = []
  for k in range(4, limit+1):
    f = factorint(k)
    # if max(f[p] for p in f) in f: alst.append(k)
    if set(f[p] for p in f) & set(f) != set(): alst.append(k)
  return alst
print(aupto(270)) # _Michael S. Branicky_, Apr 12 2021
```
