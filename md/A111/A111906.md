## Python

from itertools import count, islice
from sympy import sieve, factorint
def a_gen():
    for n in count(3):
        f = [sieve.search(i)[0] for i in factorint(n)]
        if len(f) < (f[-1]+1)//2:
            yield n
A111906_list = list(islice(a_gen(), 100)) # _John Tyler Rascoe_, Jun 22 2024
