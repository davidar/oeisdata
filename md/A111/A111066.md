## Python
```Python
from itertools import count, islice
def agen():
    for i in count(1):
        s = bin(i+1)[3:].replace('1', '2').replace('0', '1')
        if 0 < s.count('1') < len(s):
            yield int(s)
print(list(islice(agen(), 42))) # _Michael S. Branicky_, Dec 21 2021
```
