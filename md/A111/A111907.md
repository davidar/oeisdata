## Python
```Python
from itertools import count, islice
from sympy import sieve, factorint
def a_gen():
    yield 1
    for k in count(3):
        f = [sieve.search(i)[0] for i in factorint(k)]
        if 2*len(f) == f[-1]:
            yield k
A111907_list = list(islice(a_gen(), 100)) # _John Tyler Rascoe_, Jun 20 2024
```
