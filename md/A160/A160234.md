## Python
```Python
from itertools import count, islice
def agen(): # generator of terms
    s = ""
    for digits in count(1):
        c = 0
        for k in range(10**(digits-1), 10**digits):
            sk = str(k)
            if sk in s: c += 1
            s += sk
        yield c
print(list(islice(agen(), 5))) # _Michael S. Branicky_, Mar 17 2022
```
