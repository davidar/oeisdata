## Python
```Python
from sympy import mobius, integer_nthroot
def A160112(n): return sum(mobius(k)*(10**n//k**3) for k in range(1, integer_nthroot(10**n,3)[0]+1)) # _Chai Wah Wu_, Aug 06 2024
```
```Python
from bitarray import bitarray
from sympy import integer_nthroot
def A160112(n): # faster program
    q = 10**n
    m = integer_nthroot(q,3)[0]+1
    a, b = bitarray(m), bitarray(m)
    a[1], p, i, c = 1, 2, 4, q-sum(q//k**3 for k in range(2,m))
    while i < m:
        j = 2
        while i < m:
            if j==p:
                c -= (b[i]^1 if a[i] else -1)*(q//i**3)
                j, a[i], b[i] = 0, 1, 1
            else:
                t1, t2 = a[i], b[i]
                if (t1&t2)^1:
                    a[i], b[i] = (t1^1)&t2, ((t1^1)&t2)^1
                    c += (t2 if t1 else 2)*(q//i**3) if (t1^1)&t2 else (t2-2 if t1 else 0)*(q//i**3)
            i += p
            j += 1
        p += 1
        while a[p]|b[p]:
            p += 1
        i = p<<1
    return c # _Chai Wah Wu_, Aug 06 2024
```
