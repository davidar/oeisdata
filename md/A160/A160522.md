def A160522(n):
    if n == 1: return 5
    m, k = n, primepi(n) + n + (n>>1)
    while m != k:
        m, k = k, primepi(k) + n + (k>>1)
    return m-(n+1<<1) # _Chai Wah Wu_, Aug 01 2024
