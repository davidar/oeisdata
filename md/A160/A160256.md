A160256 := proc(n)
    local r,m,a,fnd,i ;
    option remember;
    if n <=2 then
        n;
    else
        r := procname(n-2)/igcd(procname(n-1),procname(n-2)) ;
        for m from 1 do
            a := m*r ;
            fnd := false;

                if procname(i) = a then
                    fnd := true;
                    break;
                end if;
            end do:
            if not fnd then
