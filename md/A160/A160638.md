import Data.Bits (testBit, setBit)
import Data.Word (Word8)
a160638 :: Word8 -> Word8
a160638 n = rev 0 0 where
   rev 8 y = y
   rev i y = rev (i + 1) (if testBit n i then setBit y (7 - i) else y)
