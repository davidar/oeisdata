## C
```C
#include <stdio.h>
int main()
{
    int n, d1, d2; int a[101]; a[1] = 1;
    printf ("%d, ", a[1]);
    for (n=2; n<101; n++)
    {
        if (n % 2==0) d1 =4;
        else d1 = 1;
        if (n%4==0) d2 = 15;
        else d2=0;
        a[n] = a[n-1] + d1 + d2;
        printf ("%d, ", a[n]);
    }
    printf("\n");
    return 0;
}
```
