from math import gcd
def aupton(terms):
    alst, aset = [1, 10], {1, 10}
    for n in range(2, terms+1):
        k = 2
        while True:
            while k in aset: k += 1
            if gcd(alst[-1], k) != 1: alst.append(k); aset.add(k); break
            k += 1
    return alst[1:]
print(aupton(73)) # _Michael S. Branicky_, Oct 02 2021
