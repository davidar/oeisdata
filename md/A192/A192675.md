## Python
```Python
from math import isqrt
from itertools import count, islice
def A192675_gen(): # generator of terms
    yield from (0,1,0)
    a, c = 0, 1
    for n in count(1):
        yield isqrt(a:=(c:=c*((n<<2)+2)//(n+2))-a>>1)
A192675_list = list(islice(A192675_gen(),20)) # _Chai Wah Wu_, Apr 26 2023
```
