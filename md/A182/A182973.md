## Pascal
```Pascal
program a182973;
var
  num,den,n: longint;
function gcd(i,j: longint):longint;
begin
  repeat
    if i>j then i:=i mod j else j:=j mod i;
  until (i=0) or (j=0);
  if i=0 then gcd:=j else gcd:=i;
end;
begin
  num:=1; den:=1; n:=0;
  repeat
    repeat
      inc(num); dec(den);
      if num>=den then
      begin
        inc(den,num); num:=1;
      end;
    until gcd(num,den)=1;
    inc(n); writeln(n,' ',den);
  until n=100000;
end.
```
