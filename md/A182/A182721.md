from sympy import isprime
def is_emirp(n):
    if not isprime(n): return False
    revn = int(str(n)[::-1])
    return n != revn and isprime(revn)
print([k for k in A134941() if is_emirp(k)]) # _Michael S. Branicky_, Dec 31 2021
