## Python
```Python
from sympy import divisors
def a(n):
    k = 10**n - 1
    divs = -1
    while divs != 4:
      k -= 1
      divs = 0
      for d in divisors(k, generator=True):
        divs += 1
        if divs > 4: break
    return k
print([a(n) for n in range(1, 21)]) # _Michael S. Branicky_, Jun 10 2021
```
