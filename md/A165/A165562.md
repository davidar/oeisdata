## Maple
```Maple
with(numtheory);
P:= proc(n)
local a,i,p,pfs;
for i from 1 to n do
pfs:=ifactors(i)[2]; a:=i*add(op(2,p)/op(1,p),p=pfs);
if isprime(a+i) then print(i); fi;
od;
end:
P(1000);
# alternative
isA165562 := proc(n)
    isprime(A129283(n)) ;
end proc:
for n from 1 to 1000 do
    if isA165562(n) then
        printf("%d,",n) ;
    end if;
end do: # _R. J. Mathar_, Feb 04 2022
```
