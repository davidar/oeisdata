from collections import Counter
def A165956_list(nmax):
    A,C = [1],Counter()
    for n in range(1,nmax+1):
        b = bin(n)[2:]
        C.update({bin(A[-1])[2:]})
        A.append(sum(C[i] for i in C if b.find(i) != -1))
    return A # _John Tyler Rascoe_, Mar 11 2023
