## Python
```Python
from sympy import primepi
def A165955(n):
    if n == 1: return 1
    m, k = n-1, primepi(n) + n - 1 + (n>>1)
    while m != k:
        m, k = k, primepi(k) + n - 1 + (k>>1)
    return m+(n<<1) # _Chai Wah Wu_, Aug 01 2024
```
