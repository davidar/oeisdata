## Python
```Python
from sympy import divisors
def ds(n): return sum(map(int, str(n)))
def ok(n):
    dsn = ds(n)
    return all(dsn%ds(d) == 0 for d in divisors(n, generator=True))
print([k for k in range(1, 174) if ok(k)]) # _Michael S. Branicky_, Jun 25 2022
```
