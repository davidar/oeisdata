def a(n): # returns term by construction
    if n == 1: return 1
    q, r = divmod(n+1, 2)
    s = "".join("1"*i + "0"*(n+1-i) for i in range(1, q+1))
    if r == 0: s = s.rstrip("0")
    return int(s, 2)
