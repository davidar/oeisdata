def aupton(terms):
  alst, deck = [1], list(range(1, 2*terms+1))
  for n in range(1, terms+1):
    first, next = deck[:n], deck[n:2*n]
    deck[0:2*n:2] = next
    deck[1:2*n:2] = first
    alst.append(deck[0])
  return alst
print(aupton(83)) # _Michael S. Branicky_, Feb 01 2021
