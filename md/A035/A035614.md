from sympy import fibonacci
def a122840(n): return len(str(n)) - len(str(int(str(n)[::-1])))
def a014417(n):
    k=0
    x=0
    while n>0:
        k=0
        while fibonacci(k)<=n: k+=1
        x+=10**(k - 3)
        n-=fibonacci(k - 1)
    return x
