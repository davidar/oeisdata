## Python
```Python
def reversedigits(n, b=10): # reverse digits of n in base b
    x, y = n, 0
    while x >= b:
        x, r = divmod(x, b)
        y = b*y + r
    return b*y + x
A035523_list, l = [1], 1
for _ in range(50):
    l += reversedigits(l, 3)
    A035523_list.append(l)
```
