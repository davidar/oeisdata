```Maple
with(numtheory):
b:= proc(n, k) option remember;
      `if`(n>k, 0, 1) +`if`(isprime(n), 0,
      add(`if`(d>k, 0, b(n/d, d-1)), d=divisors(n) minus {1, n}))
    end:
a:= n-> b(n$2):
seq(a(n), n=1..120);  # _Alois P. Heinz_, May 26 2013
```

## Python
```Python
from sympy.core.cache import cacheit
from sympy import divisors, isprime
@cacheit
def b(n, k): return (0 if n>k else 1) + (0 if isprime(n) else sum(0 if d>k else b(n//d, d - 1) for d in divisors(n)[1:-1]))
def a(n): return b(n, n)
print([a(n) for n in range(1, 121)]) # _Indranil Ghosh_, Aug 19 2017, after Maple code
```
