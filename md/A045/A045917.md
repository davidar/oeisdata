A045917 := proc(n)
    local a,i ;
    a := 0 ;
    for i from 1 to n do
        if isprime(i) and isprime(2*n-i) then
            a := a+1 ;
        end if;
    end do:
    a ;
end proc: # _R. J. Mathar_, Jul 01 2013

    x = 0

            x += 1
