## Python
```Python
from math import isqrt
def aupto(limit):
    alst, d, end = [], 1, 0
    while end < limit:
        start, end = isqrt(6*10**d) + 1, isqrt(7*10**d-1)
        alst.extend([an for an in list(range(start, end+1)) if an <= limit])
        d += 1
    return alst
print(aupto(799)) # _Michael S. Branicky_, Aug 25 2021
```
