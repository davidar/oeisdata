from itertools import combinations
def a(n):
  s, eset = str(n), set()
  for i in range(len(s)):
    for j in range(i+1, len(s)+1):
      if s[j-1] in "02468":
        if len(s[i:j]) <= 2 and j-i < len(s):
          eset.add(int(s[i:j]))
        else:
          middle = s[i+1:j-1]
          for k in range(len(middle)+1):
            for c in combinations(middle, k):
              t = s[i] + "".join(c) + s[j-1]
              if len(t) < len(s):
                eset.add(int(t))
  return len(eset)
print([a(n) for n in range(105)]) # _Michael S. Branicky_, Mar 24 2021
