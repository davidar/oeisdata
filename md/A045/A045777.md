## Python
```Python
from itertools import islice
from collections import deque
def agen(): # generator of terms
    a = deque([1, 2])
    while True:
        a.extend(list(map(int, str(a[0]*a[1]))))
        yield a.popleft()
print(list(islice(agen(), 105))) # _Michael S. Branicky_, Feb 15 2024
```
