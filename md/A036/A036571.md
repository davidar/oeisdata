from itertools import count, islice
from math import isqrt
def A036571_gen(): # generator of terms
    c = 0
    for n in count(1):
        yield c
        c += 1<<(m:=n<<1)-(k:=isqrt(m))-int((m<<2)>(k<<2)*(k+1)+1)-1
A036571_list = list(islice(A036571_gen(),25)) # _Chai Wah Wu_, Jul 26 2022
