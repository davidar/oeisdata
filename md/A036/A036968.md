```Sage
def A036968_list(len):
    e, f, R, C = 4, 1, [], [1]+[0]*(len-1)
    for n in (2..len-1):
        for k in range(n, 0, -1):
            C[k] = C[k-1] / (k+1)
        C[0] = -sum(C[k] for k in (1..n))
        R.append((2-e)*f*C[0])
        f *= n; e *= 2
    return R
print(A036968_list(34)) # _Peter Luschny_, Feb 22 2016
```
