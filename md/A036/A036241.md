def A036241_gen(): # generator of terms
    aset2, aset3 = {3,4,5}, {6}
    yield from (alist:=[1,2,3])
    for k in count(4):
        bset2, bset3 = set(), set()
        for a in alist:
            if (b2:=a+k) in aset2:

            bset2.add(b2)

                if (b3:=a2+k) in aset3:
                    break
                bset3.add(b3)

                aset2.update(bset2)
                aset3.update(bset3)
