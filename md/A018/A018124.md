from gmpy2 import iroot_rem
def A018124(n):
    i, j = iroot_rem(1<<(n<<1),5)
    return int(i)+int(j<<5>=10*i*((i*((i*(i+1)<<1)+1)<<2)+1)+1) # _Chai Wah Wu_, Jun 20 2024
