a:= proc(n) local k,d,x;
  if issqr(n) then return n
  else for d from 1 do
    for k from 0 to 10^d-1 do
    x:= 10^d*n+k;
    if issqr(x) then return x fi
    od
    od
  fi
end proc:

```Python
from gmpy2 import isqrt
def A018796(n):
    if n == 0:
        return 0
    else:
        d, nd = 1, n
        while True:
            x = (isqrt(nd-1)+1)**2
            if x < nd+d:
                return int(x)
            d *= 10
            nd *= 10 # _Chai Wah Wu_, May 23 2016
```
