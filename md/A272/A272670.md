## Maple
```Maple
isPal := proc(L::list)
    local i;
    for i from 1 to nops(L)/2 do
        if op(i,L) <> op(-i,L) then
            return false;
        end if;
    end do:
    true ;
end proc:
isA272670 := proc(n)
    local bdgs ;
    bdgs := convert(n,base,2) ;
    if isPal(bdgs) then
        return false;
    else
        A000265(n) ;
        bdgs := convert(%,base,2) ;
        isPal(bdgs) ;
    end if;
end proc:
for n from 1 to 500 do
    if isA272670(n) then
        printf("%d,",n) ;
    end if;
end do: # _R. J. Mathar_, May 20 2016
```
