## Python
```Python
from gmpy2 import isqrt
def A272680(n):
    if n == 0:
        return 0
    else:
        d, nd = 1, n
        while True:
            x = (isqrt(nd-1)+1)**2
            if x < nd+d:
                return int(x)
            d *= 2
            nd *= 2 # _Chai Wah Wu_, May 22 2016
```
