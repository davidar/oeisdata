## MATLAB
```MATLAB
function count = A272661(N)
  C = zeros(0,N);
  count = 0;
  V = zeros(1,N);
  L = -floor(N/2) + [0:N-1];
  for x = 0:2^(N^2)-1;
    r = dec2bin(x+2^(N^2))-'0';
    A = reshape(r(2:end),N,N);
    rowcounts = sum(A,2);
    colcounts = sum(A,1);
    if ~issorted(rowcounts)|| rowcounts(N) < max(colcounts)
      continue
    end
    for i = 1:N
        V(i) = round(det(A - L(i)*eye(N)));
    end
    if ~ismember(V, C, 'rows')
      count = count+1;
      C(count,:) = V;
    end
  end
end  % _Robert Israel_, Aug 18 2016
```
