## Python
```Python
from gmpy2 import isqrt
def A272679(n):
    if n == 0:
        return 0
    else:
        d, nd = 1, n
        while True:
            x = isqrt(nd-1)+1
            if x**2 < nd+d:
                return int(x)
            d *= 2
            nd *= 2 # _Chai Wah Wu_, May 22 2016
```
