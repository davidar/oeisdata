from sympy import divisors, isprime
A272383_list = []
for i in range(78, 10**6, 78):
    for d in divisors(i):
        if d not in (1,2,6,78) and isprime(d+1):
            break
    else:
        A272383_list.append(i) # _Chai Wah Wu_, May 02 2016
