## Python
```Python
from functools import cache
@cache
def a(n):
    if n < 0: return 0
    if n < 5: return [0, 5, 9, 4, 6][n]
    return a(n - a(n-1)) + a(n - a(n-2))
print([a(n) for n in range(1, 73)]) # _Michael S. Branicky_, Sep 20 2021
```
